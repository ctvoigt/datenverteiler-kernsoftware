22  package de.bsvrz.sys.funclib.filelock;
23  
24  import java.io.File;
25  import java.io.IOException;
26  
27  /**
28   * Diese Klasse stellt einen Mechanismus zur Verfügung, mit dem geprüft werden kann, ob sich eine Datei im Zugriff einer anderen Applikation befindet, die
29   * ebenfalls diesen Mechansimus verwendet.
30   * <p/>
31   * Damit oben beschriebene Aufgabe realisiert werden kann, wird im Konstruktor der Klasse eine Datei angegeben, die gegen unerlaubten Zugriff geschützt werden
32   * soll. Soll die übergebene Datei geschützt werden, wird eine entsprechende Methode aufgerufen. Dieser Methodenaufruf prüft, ob eine Datei mit der Endung
33   * ".lock" existiert, der restliche Pfad ist gleich der zu schützenden Datei.
34   * <p/>
35   * Ist so eine Datei vorhanden, wird eine Exception geworfen, da sich zu schützende Datei bereits im Zugriff befindet.
36   * <p/>
37   * Kann eine Datei angelegt werden, wird der Methodenaufruf keine weiteren Auswirkungen habe. Legt eine zweite Applikation eine Instanz dieser Klasse an und
38   * übergibt die gleiche Datei im Konstruktor und ruft die Methode zum sperren der Datei auf, wird automatisch eine Exception geworfen.
39   * <p/>
40   * Die Klasse enthält ebenfalls eine Methode zum löschen der Datei mit der Endung ".lock". Wurde die Datei gelöscht, können andere Applikationen die Datei
41   * wieder sperren.
42   *
43   * @author Kappich Systemberatung
44   * @version $Revision: 5010 $
45   */
46  public class FileLock {
47  
48  	/** Datei, die über den lock-Mechanismus geschützt werden soll. */
49  	private final File _protectedFile;
50  
51  	private final File _lockFile;
52  
53  	/** Endung, die an die Datei gehängt wird, die anzeigt, dass sich die zu schützende Datei im Zugriff befindet. */
54  	private static final String _lockFileEnding = ".lock";
55  
56  	/**
57  	 * Legt ein Objekt an, die übergebene Datei wird nicht gegen den Zugriff gesichert. Dies muss expliziet mit dem Aufruf von {@link #lock()} geschehen.
58  	 *
59  	 * @param protectedFile Datei, die gesperrt werden soll.
60  	 */
61  	public FileLock(final File protectedFile) {
62  		_protectedFile = protectedFile;
63  		_lockFile = new File(_protectedFile.getParent(), (_protectedFile.getName() + _lockFileEnding));
64  	}
65  
66  	/**
67  	 * Zeigt an, dass die im Konstruktor übergebene Datei im Zugriff einer Applikation ist. Jeder weitere Aufruf ohne vorher {@link #unlock()} aufzurufen wird zu
68  	 * einer IllegalStateException führen.
69  	 *
70  	 * @throws IllegalStateException Die Datei wurde bereits gesperrt und befindet sich somit im Zugriff.
71  	 */
72  	public void lock() throws IOException {
73  		if(!_lockFile.createNewFile())
74  		{
75  			// Die Datei war bereits vorhanden. Also wird die Datei bereits durch diesen Mechansimus geschützt und befindet sich im Zugriff.
76  			throw new IllegalStateException("Die Datei " + _protectedFile.getAbsolutePath() + " ist bereits gesperrt worden und kann nicht erneut gesperrt werden.");
77  		}else
78  		{
79  //			System.out.println("Erzeuge Lock: " + _lockFile.getAbsolutePath());
80  		}
81  	}
82  
83  	/** Gibt die Datei, die im Konstruktor übergeben wurde wieder frei. Nach Aufruf dieser Methode kann die Methode {@link #lock()} wieder aufgerufen werden. */
84  	public void unlock() {
85  		_lockFile.delete();
86  	}
87  }
