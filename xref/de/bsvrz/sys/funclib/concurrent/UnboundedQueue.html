22  package de.bsvrz.sys.funclib.concurrent;
23  
24  import java.util.LinkedList;
25  
26  /**
27   * Queue zum Austausch von Nachrichten zwischen Threads. Mit der Methode {@link #put} können beliebige Objekte
28   * (Nachrichten) in die Queue eingetragen (gesendet) werden und i.a. von einem anderen Thread mit den Methoden {@link
29   * #take} oder {@link #poll} aus der Queue entnommen (empfangen) werden. Die generische Queue kann durch Angabe eines
30   * Typs auf diesen bestimmten Objekttypen arbeiten. Die Anzahl der Nachrichten in der Queue ist nicht beschränkt. Die
31   * Nachrichten werden in der Reihenfolge empfangen in der sie versendet wurden (first in first out, FIFO).
32   *
33   * @author Kappich Systemberatung
34   * @version $Revision: 5006 $
35   */
36  public class UnboundedQueue <E> {
37  
38  	private final LinkedList<E> _list = new LinkedList<E>();
39  
40  	/**
41  	 * Sendet eine Nachricht an die Queue.
42  	 *
43  	 * @param message Zu versendende Nachricht.
44  	 */
45  	public void put(E message) {
46  		synchronized (_list) {
47  			_list.addFirst(message);
48  			_list.notify();
49  		}
50  	}
51  
52  	/**
53  	 * Empfängt und entfernt eine Nachricht aus der Queue. Wenn die Queue leer ist, dann wartet die Methode, bis eine
54  	 * Nachricht in die Queue gesendet wurde. Wenn mehrere Nachrichten in der Queue vorhanden sind, wird die Nachricht
55  	 * empfangen und entfernt, die als erstes in die Queue gesendet wurde.
56  	 *
57  	 * @return Empfangene Nachricht.
58  	 * @throws InterruptedException Wenn der Thread unterbrochen wurde während auf Nachrichten gewartet wurde.
59  	 */
60  	public E take() throws InterruptedException {
61  		synchronized (_list) {
62  			while (_list.isEmpty()) _list.wait();
63  			return _list.removeLast();
64  		}
65  	}
66  
67  	/**
68  	 * Empfängt und entfernt eine Nachricht aus der Queue. Wenn die Queue leer ist, dann wartet die Methode, bis eine
69  	 * Nachricht in die Queue gesendet wurde oder bis die angegebene Zeit verstrichen ist. Wenn mehrere Nachrichten in der
70  	 * Queue vorhanden sind, wird diejenige Nachricht empfangen und entfernt, die als erstes in die Queue gesendet wurde.
71  	 *
72  	 * @param timeout Zeit in Millisekunden, für die bei leerer Queue auf neue Nachrichten gewartet wird.
73  	 * @return Empfangene Nachricht oder <code>null</code>, wenn nach Ablauf des Timeouts immer noch keine Nachricht in der
74  	 *         Queue vorhanden ist.
75  	 * @throws InterruptedException Wenn der Thread unterbrochen wurde während auf Nachrichten gewartet wurde.
76  	 */
77  	public E poll(long timeout) throws java.lang.InterruptedException {
78  		long maxWaitTime = System.currentTimeMillis() + timeout;
79  		synchronized (_list) {
80  			long maxSleep = maxWaitTime - System.currentTimeMillis();
81  			while (_list.isEmpty() && maxSleep > 0) {
82  				_list.wait(maxSleep);
83  				maxSleep = maxWaitTime - System.currentTimeMillis();
84  			}
85  			if (_list.isEmpty()) return null;
86  			return _list.removeLast();
87  		}
88  	}
89  
90  	/**
91  	 * Bestimmt die Größe dieser Queue.
92  	 *
93  	 * @return Anzahl verbleibender Elemente in dieser Queue.
94  	 */
95  	public int size() {
96  		synchronized (_list) {
97  			return _list.size();
98  		}
99  	}
100 }
