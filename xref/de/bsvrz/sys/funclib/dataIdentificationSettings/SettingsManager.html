166 		atg = configuration.getAttributeGroup("atg.konfigurationsSchreibAnfrage");
167 		if(atg != null) _excludedAttributeGroups.add(atg);
168 		atg = configuration.getAttributeGroup("atg.konfigurationsAntwort");
169 		if(atg != null) _excludedAttributeGroups.add(atg);
170 		atg = configuration.getAttributeGroup("atg.konfigurationsAnfrage");
171 		if(atg != null) _excludedAttributeGroups.add(atg);
172 		atg = configuration.getAttributeGroup("atg.archivAnfrageSchnittstelle");
173 		if(atg != null) _excludedAttributeGroups.add(atg);
174 		atg = configuration.getAttributeGroup("atg.archivAntwort");
175 		if(atg != null) _excludedAttributeGroups.add(atg);

230 	/**
231 	 * Meldet die im Konstruktor übergebene Datenidentifikation an und startet damit auch die Verarbeitung und Weitergabe der alten und neuen Einstellungen pro
232 	 * Datenidentifikation aus erhaltenen Parameterdatensätzen an die angemeldeten Beobachter.
233 	 *
234 	 * @throws IllegalStateException Wenn der Manager bereits gestartet wurde.
235 	 */
236 	public void start() {
237 		if(_receiver != null) throw new IllegalStateException("Ist bereits gestartet");
238 		_changeNotifier.reset();
239 		ClientReceiverInterface receiver = new AsyncReceiver(new Receiver());
240 		_settingsTable = new HashMap<DataIdentification, Data>();
241 		_connection.subscribeReceiver(
242 				receiver, _parameterIdentification.getObject(), _parameterIdentification.getDataDescription(), ReceiveOptions.normal(), ReceiverRole.receiver()
243 		);
244 		_receiver = receiver;
245 		final Thread notificationThread = new Thread(_changeNotifier, "SettingsManager.ChangeNotifier");
246 		notificationThread.setDaemon(true);
247 		notificationThread.start();
248 	}

251 	 * Meldet die im Konstruktor übergebene Datenidentifikation wieder ab und beendet damit auch die Verarbeitung und Weitergabe der alten und neuen Einstellungen
252 	 * pro Datenidentifikation aus erhaltenen Parameterdatensätzen an die angemeldeten Beobachter.
253 	 */
254 	public void stop() {
255 		if(_receiver == null) return;
256 		_changeNotifier.notifyTermination();
257 		_connection.unsubscribeReceiver(_receiver, _parameterIdentification.getObject(), _parameterIdentification.getDataDescription());
258 		_receiver = null;
259 		registerDynamicObjects(new HashSet<DynamicObject>());
260 		registerDynamicObjectTypes(new HashSet<DynamicObjectType>());
261 		registerMutableSets(new HashSet<MutableSet>());
262 	}

264 	/**
265 	 * Aktualisiert die Tabelle mit den Einstellungen je Datenidentifikation mit den jeweiligen Einstellungen aus dem übergebenen Datensatz.
266 	 *
267 	 * @param settingsData
268 	 * @param newSettings  die neuen Einstellungen
269 	 */
270 	private void extractSettings(final Data settingsData, Map<DataIdentification, Data> newSettings) {
271 		// Objekt-Typen werden hier gespeichert. Diese werden am Ende der Methode weitergereicht, damit sich der SettingsManager sich auf Änderungen
272 		// an den Objekt-Typen anmeldet. Gleiches gilt für dynamische Objekte und änderbare Mengen.
273 		final Set<DynamicObjectType> dynamicObjectTypes = new HashSet<DynamicObjectType>();
274 		final Set<DynamicObject> dynamicObjects = new HashSet<DynamicObject>();
275 		final Set<MutableSet> mutableSets = new HashSet<MutableSet>();
276 
277 		DataModel configuration = _connection.getDataModel();
278 		Aspect parameterAspect = null;
279 		if(!_aspectUsed) parameterAspect = configuration.getAspect("asp.parameterVorgabe");
280 		_debug.finer("Parametrierte Daten = " + settingsData);
281 		if(settingsData != null) {
282 			boolean refreshParams = false;
283 			final Data modifiableDataCopy = settingsData.createModifiableCopy();
284 			Data parameterArray = modifiableDataCopy.getItem("ParameterSatz");
285 			//Schleife über die einzelnen Elemente des Arrays ParameterSatz
286 			for(Iterator iterator = parameterArray.iterator(); iterator.hasNext();) {
287 				final Data parameter = (Data)iterator.next();
288 				Data settings = parameter.getItem("Einstellungen");
289 				_debug.finer("parameter = " + parameter);
290 				_debug.finer("settings = " + settings);
291 				_debug.finer("parameter.getArray(\"Bereich\").getLength() = " + parameter.getArray("Bereich").getLength());
292 				final Data.ReferenceArray referenceArrayArea = parameter.getReferenceArray("Bereich");
293 
294 				SystemObject[] specifiedConfigAreas = referenceArrayArea.getSystemObjectArray();
295 				refreshParams |= fixOldReferences(referenceArrayArea, specifiedConfigAreas);
296 				final boolean configAreasEmpty = specifiedConfigAreas.length == 0;
297 				specifiedConfigAreas = removeNullReferences(specifiedConfigAreas);
298 
299 				for(Iterator dataSpecificationIterator = parameter.getItem("DatenSpezifikation").iterator(); dataSpecificationIterator.hasNext();) {
300 					final Data dataSpec = (Data)dataSpecificationIterator.next();
301 					final Data.ReferenceArray referenceArrayObject = dataSpec.getReferenceArray("Objekt");
302 
303 					SystemObject[] specifiedObjects = referenceArrayObject.getSystemObjectArray();
304 					refreshParams |= fixOldReferences(referenceArrayObject, specifiedObjects);
305 					final boolean objectsEmpty = specifiedObjects.length == 0;
306 					specifiedObjects = removeNullReferences(specifiedObjects);
307 
308 					final Data.ReferenceArray referenceArrayAttributeGroup = dataSpec.getReferenceArray("AttributGruppe");
309 
310 					SystemObject[] specifiedAtgs = referenceArrayAttributeGroup.getSystemObjectArray();
311 					refreshParams |= fixOldReferences(referenceArrayAttributeGroup, specifiedAtgs);
312 					final boolean attributeGroupsEmpty = specifiedAtgs.length == 0;
313 					specifiedAtgs = removeNullReferences(specifiedAtgs);
314 
315 					// Sortierung ist notwendig für das spätere binäre Suchen in diesem Array s.u.
316 					Arrays.sort(specifiedAtgs);
317 					final HashSet<AttributeGroup> specifiedAtgsHash = new HashSet<AttributeGroup>();
318 					for(SystemObject specifiedAtg : specifiedAtgs) {
319 						if(specifiedAtg instanceof AttributeGroup) {
320 							AttributeGroup attributeGroup = (AttributeGroup)specifiedAtg;
321 							specifiedAtgsHash.add(attributeGroup);
322 						}
323 					}
324 					SystemObject[] specifiedAspects;
325 					final boolean aspectsEmpty;
326 					if(_aspectUsed) {
327 
328 						final Data.ReferenceArray referenceArrayAspect = dataSpec.getReferenceArray("Aspekt");
329 						specifiedAspects = referenceArrayAspect.getSystemObjectArray();
330 						refreshParams |= fixOldReferences(referenceArrayAspect, specifiedAspects);
331 						aspectsEmpty = specifiedAspects.length == 0;
332 						specifiedAspects = removeNullReferences(specifiedAspects);
333 
334 						// Sortierung ist notwendig für das spätere binäre Suchen in diesem Array s.u.
335 						Arrays.sort(specifiedAspects);
336 					}
337 					else {
338 						specifiedAspects = new SystemObject[1];
339 						specifiedAspects[0] = parameterAspect;
340 						aspectsEmpty = false;
341 					}
342 
343 					final HashSet<Aspect> specifiedAspectsHash = new HashSet<Aspect>();
344 					for(SystemObject specifiedAspect : specifiedAspects) {
345 						if(specifiedAspect instanceof Aspect) {
346 							Aspect aspect = (Aspect)specifiedAspect;
347 							specifiedAspectsHash.add(aspect);
348 						}
349 					}
350 					short simulationVariant = -1;
351 					if(_simulationVariantUsed) {
352 						simulationVariant = dataSpec.getScaledValue("SimulationsVariante").shortValue();
353 					}
354 
355 					List<SystemObject> objectList;
356 					if(objectsEmpty) {
357 						// keine Objekte sind alle Objekte
358 						// Anmeldung auf relevante dynamische Typen
359 						registerDynamicObjectTypeOrSubTypes(
360 								dynamicObjectTypes, (DynamicObjectType)configuration.getType(Pid.Type.DYNAMIC_OBJECT), specifiedAtgsHash, specifiedAspectsHash
361 						);
362 //						dynamicObjectTypes.add((DynamicObjectType)configuration.getType(Pid.Type.DYNAMIC_OBJECT));
363 						// d.h. alle dynamischen Objekte und alle Konfigurationsobjekte
364 						List<SystemObject> dynObjects = ((MutableCollection)configuration.getType(Pid.Type.DYNAMIC_OBJECT)).getElements(
365 								_simulationVariantForTypeAndSetWildcardExpansion
366 						);
367 						List<SystemObject> configurationObjects = configuration.getType(Pid.Type.CONFIGURATION_OBJECT).getObjects();
368 						objectList = new ArrayList<SystemObject>(dynObjects.size() + configurationObjects.size());
369 						objectList.addAll(dynObjects);
370 						objectList.addAll(configurationObjects);
371 					}
372 					else {
373 						objectList = new ArrayList<SystemObject>(specifiedObjects.length);
374 						for(SystemObject object : specifiedObjects) {
375 							if(object instanceof DynamicObjectType) {
376 								final DynamicObjectType dynamicObjectType = (DynamicObjectType)object;
377 								// auf alle relevanten dynamischen Typen anmelden, falls neue Objekte hinzukommen oder welche ungültig werden
378 								registerDynamicObjectTypeOrSubTypes(dynamicObjectTypes, dynamicObjectType, specifiedAtgsHash, specifiedAspectsHash);
379 //								dynamicObjectTypes.add(dynamicObjectType);
380 								objectList.addAll(dynamicObjectType.getElements(_simulationVariantForTypeAndSetWildcardExpansion));
381 							}
382 							else if(object instanceof MutableSet) {
383 								final MutableSet mutableSet = (MutableSet)object;
384 								// änderbare Mengen merken, falls sich dort etwas ändert
385 								mutableSets.add(mutableSet);
386 								// alle Elemente der Menge berücksichtigen
387 								objectList.addAll(mutableSet.getElements(_simulationVariantForTypeAndSetWildcardExpansion));
388 							}
389 							else if(object instanceof SystemObjectCollection) {
390 								// ein SystemObjectType ist auch eine SystemObjectCollection
391 								SystemObjectCollection collection = (SystemObjectCollection)object;
392 								// Typen und Mengen werden durch die jeweils enthaltenen Elemente ersetzt
393 								objectList.addAll(collection.getElements());
394 							}
395 							else {
396 								objectList.add(object);
397 							}
398 						}
399 					}
400 
401 					// Einschränkung auf ausgewählte Konfigurationsbereiche
402 					if(!configAreasEmpty) {
403 						_debug.finest("specifiedConfigAreas.length = " + specifiedConfigAreas.length);
404 						List<SystemObject> specifiedConfigAreaList = Arrays.asList(specifiedConfigAreas);
405 						List<SystemObject> newObjectList = new LinkedList<SystemObject>();
406 						for(final SystemObject systemObject : objectList) {
407 							ConfigurationArea configurationArea = systemObject.getConfigurationArea();
408 							if(specifiedConfigAreaList.contains(configurationArea)) {
409 								newObjectList.add(systemObject);
410 							} // alle anderen werden aus der Objektliste gelöscht
411 						}
412 						objectList.clear();
413 						objectList.addAll(newObjectList);
414 					}
415 					// Folgende Map Enthält als Schlüssel die zu betrachtenden Typen und als Wert jeweils ein Set mit den zu
416 					// betrachtenden Objekten des Typs:
417 					Map<SystemObjectType, Collection<SystemObject>> type2objectSetMap = new TreeMap<SystemObjectType, Collection<SystemObject>>();
418 
419 					// Erzeugen der Map, d.h. sortieren nach Typen und sicherstellen, dass jedes Objekt nur einmal enthalten ist.
420 					for(final SystemObject object : objectList) {
421 						SystemObjectType type = object.getType();
422 						// wenn das Objekt nicht gültig ist, nicht weiter betrachten
423 						
424 
425 
426 						if(!object.isValid()) continue;
427 						Collection<SystemObject> objectCollection = type2objectSetMap.get(type);
428 						if(objectCollection == null) {
429 							objectCollection = new LinkedList<SystemObject>();
430 							type2objectSetMap.put(type, objectCollection);
431 						}
432 						objectCollection.add(object);
433 					}
434 
435 					objectList.clear();
436 					objectList = null;
437 					// FIXME: Riesige Funktion, in Methoden auslagern?
438 
439 					// Schleife über Map mit Typ/ObjektSet Paaren
440 					for(Map.Entry<SystemObjectType, Collection<SystemObject>> entry : type2objectSetMap.entrySet()) {
441 						SystemObjectType type = entry.getKey();
442 						Collection<SystemObject> objectCollection = entry.getValue();
443 						// Im folgenden werden alle Attributgruppen des Typs betrachtet und diese eventuell weiter
444 						// eingeschränkt, dadurch werden unsinnige Spezifikation ignoriert.
445 						List<AttributeGroup> typeAtgs = type.getAttributeGroups();
446 						for(AttributeGroup typeAtg : typeAtgs) {
447 							// Wenn keine Attributgruppen spezifiziert wurden, dann werden alle des Typs weiter betrachtet,
448 							// ansonsten werden nur die Attributgruppen des Typs weiter betrachtet, die auch explizit
449 							// spezifiziert wurden.
450 							if(!attributeGroupsEmpty && Arrays.binarySearch(specifiedAtgs, typeAtg) < 0) continue;
451 
452 							// Wenn die betrachtete Attributgruppe in der Liste der nicht erlaubten Attributgruppen ist, wird
453 							// diese nicht weiter betrachtet.
454 							if(_excludedAttributeGroups.contains(typeAtg)) continue;
455 
456 							Collection<Aspect> atgAspects;
457 
458 							atgAspects = typeAtg.getAspects();
459 
460 							for(Aspect atgAspect : atgAspects) {
461 								// Wenn kein Aspekt spezifiziert wurde, dann werden alle Aspekte der Attributgruppe weiter betrachtet,
462 								// ansonsten werden nur die Aspekte weiter betrachtet, die auch explizit
463 								// spezifiziert wurden.
464 								if(!aspectsEmpty && Arrays.binarySearch(specifiedAspects, atgAspect) < 0) continue;
465 
466 								// Konfigurierende Attributgruppenverwendungen werden ignoriert
467 								if(typeAtg.getAttributeGroupUsage(atgAspect).isConfigurating()) continue;
468 
469 								//Für alle spezifizierten Objekte des betrachteten Typs die spezifizierte Einstellung merken
470 								for(SystemObject object : objectCollection) {
471 									// alle dynamischen Objekte speichern, deren Typ nicht bereits für Änderungen gespeichert wird,
472 									// damit sich der SettingsManager darauf anmelden kann
473 									if(object instanceof DynamicObject && !dynamicObjectTypes.contains(object.getType())) {
474 										dynamicObjects.add((DynamicObject)object);
475 									}
476 									final DataDescription dataDescription;
477 									if(_simulationVariantUsed) {
478 										dataDescription = new DataDescription(typeAtg, atgAspect, simulationVariant);
479 									}
480 									else {
481 										dataDescription = new DataDescription(typeAtg, atgAspect);
482 									}
483 									DataIdentification key = new DataIdentification(object, dataDescription);
484 									newSettings.put(key, settings);
485 								}
486 							}
487 						}
488 					}
489 					
490 
491 				}
492 			}
493 			if(refreshParams){
494 				sendNewParams(modifiableDataCopy);
495 			}
496 		}
497 		// Anmelden auf Änderungen der dynamischen Objekt-Typen (hier können zur Laufzeit Objekte angelegt werden)
498 		registerDynamicObjectTypes(dynamicObjectTypes);
499 		registerDynamicObjects(dynamicObjects);
500 		registerMutableSets(mutableSets);
501 	}

638 	 * @param toBeRegisteredType Zu betrachtender dynamischer Typ.
639 	 * @param specifiedAtgs Relevante Attributgruppen. Ein leeres Set wird als Wildcard für alle Attributgruppen berücksichtigt.
640 	 * @param specifiedAspects Relevante Aspekte. Ein leeres Set wird als Wildcard für alle Aspekte berücksichtigt.

644 			final DynamicObjectType toBeRegisteredType, final Set<AttributeGroup> specifiedAtgs, final Set<Aspect> specifiedAspects) {
645 		final List<AttributeGroup> attributeGroups = toBeRegisteredType.getDirectAttributeGroups();
646 		for(AttributeGroup attributeGroup : attributeGroups) {
647 			if(specifiedAtgs.isEmpty() || specifiedAtgs.contains(attributeGroup)) {
648 				final Collection<AttributeGroupUsage> attributeGroupUsages = attributeGroup.getAttributeGroupUsages();
649 				for(AttributeGroupUsage attributeGroupUsage : attributeGroupUsages) {
650 					if(!attributeGroupUsage.isConfigurating() && (specifiedAspects.isEmpty() || specifiedAspects.contains(attributeGroupUsage.getAspect()))) {
651 						// Es gibt also am Typ eine passende Attributgruppe mit einem passenden Aspekt
652 						return true;
653 					}
654 				}
655 			}
656 		}
657 		return false;

747 	 * Iteriert über die Tabelle mit den Einstellungen je Datenidentifikation und informiert die angemeldeten Beobachter über Änderungen an den Einstellungen je
748 	 * Datenidentifikation. Die übergebenen Einstellungen werden als aktuelle Einstellungen übernommen.
749 	 *
750 	 * @param newSettingsTable Map mit Einstellungen je Datenidentifikation
751 	 */
752 	private void activateSettings(Map<DataIdentification, Data> newSettingsTable) {
753 		// Schleife über Zeilen der neuen Tabelle
754 		// entsprechender Eintrag in alter Tabelle wird gelöscht
755 		// Benachrichtigung an alle Beobachter
756 		long startTime = System.currentTimeMillis();
757 		_debug.fine("+++aktiviere " + newSettingsTable.size() + " neue Einstellungen");
758 		for(Iterator<Map.Entry<DataIdentification,Data>> iterator = newSettingsTable.entrySet().iterator(); iterator.hasNext();) {
759 			Map.Entry<DataIdentification, Data> entry = iterator.next();
760 			DataIdentification dataIdentification = entry.getKey();
761 			Data newSettings = entry.getValue();
762 			Data oldSettings = (Data)_settingsTable.remove(dataIdentification);
763 			notifySettings(dataIdentification, oldSettings, newSettings);
764 		}
765 		long duration = System.currentTimeMillis() - startTime;
766 		_debug.finer(
767 				"+++Ende Aktivierung " + newSettingsTable.size() + " neue Einstellungen (Zeit: " + duration + " ms [pro Parametersatz "
768 				+ ((double)duration / (double)((newSettingsTable.size() == 0) ? 1 : (newSettingsTable.size()))) + " ms]"
769 		);
770 
771 		_debug.fine("---deaktiviere verbleibende " + _settingsTable.size() + " alte Einstellungen");
772 		startTime = System.currentTimeMillis();
773 		// Schleife über die verbleibenden Einträge in der alten Tabelle, das sind die Einträge, zu denen es keine
774 		// neuen Einträge mehr gibt. Jeweils entsprechende Benachrichtigung der angemeldeten Beobachter.
775 		for(Iterator<Map.Entry<DataIdentification,Data>> iterator = _settingsTable.entrySet().iterator(); iterator.hasNext();) {
776 			Map.Entry<DataIdentification, Data> entry = iterator.next();
777 			DataIdentification dataIdentification = entry.getKey();
778 			Data oldSettings = (Data)_settingsTable.get(dataIdentification);
779 			notifySettings(dataIdentification, oldSettings, null);
780 		}
781 		duration = System.currentTimeMillis() - startTime;
782 		_debug.finer(
783 				"+++Ende Deaktivierung " + _settingsTable.size() + " alte Einstellungen (Zeit: " + duration + "ms [pro Parametersatz "
784 				+ ((double)duration / (double)((_settingsTable.size() == 0) ? 1 : (_settingsTable.size()))) + " ms]"
785 		);
786 		// Neue Einstellungstabelle wird als aktuelle übernommen
787 		_settingsTable.clear();
788 		_settingsTable = newSettingsTable;
789 		notifyEndOfSettings();
790 	}

822 	private class Receiver implements ClientReceiverInterface {
823 
824 		/** Flag, dass dafür sorgt, dass initiale leere Datensätze ignoriert werden */
825 		private boolean _seenNonEmptyData = false;
826 
827 		/**
828 		 * Der Datensatzindex des zuletzt verarbeiteten Parameterdatensatz wird als Workaround benutzt um doppelte Datensätze zu erkennen und zu ignorieren.
829 		 * <p/>
830 		 * Doppelte Datensätze können bei veränderten AnmeldeOptionen einer zweiten Anmeldung auf diese Datenidentifikation auftreten.
831 		 */
832 		private long _lastDataIndex = 0;
833 
834 		/**
835 		 * Aktualisierungsmethode, die nach Empfang eines angemeldeten Datensatzes von den Datenverteiler-Applikationsfunktionen aufgerufen wird. Diese Methode muss
836 		 * von der Applikation zur Verarbeitung der empfangenen Datensätze implementiert werden.
837 		 *
838 		 * @param results Feld mit den empfangenen Ergebnisdatensätzen.
839 		 */
840 		public void update(ResultData results[]) {
841 			try {
842 				boolean gotNewData = false;
843 				for(int i = 0; i < results.length; i++) {
844 					ResultData result = results[i];
845 					if(result.hasData()) {
846 						_seenNonEmptyData = true;
847 					}
848 					long resultDataIndex = result.getDataIndex();
849 					if(resultDataIndex != _lastDataIndex) {
850 						_lastDataIndex = resultDataIndex;
851 						gotNewData = true;
852 					}
853 				}
854 				if(_seenNonEmptyData && gotNewData) _changeNotifier.notifyData(results[results.length - 1].getData());
855 			}
856 			catch(Exception e) {
857 				e.printStackTrace();
858 			}
859 		}
860 	}

862 	class ChangeNotifier implements Runnable {
863 
864 		private LinkedList<NotificationObject> _notificationQueue = new LinkedList<NotificationObject>();
865 
866 		public void reset() {
867 			synchronized(_notificationQueue) {
868 				_notificationQueue.clear();
869 			}
870 		}
871 
872 		public void notifyDynamicChange() {
873 			notify(new NotificationObject(NotificationType.DYNAMIC_CHANGE, null));
874 		}
875 
876 		public void notifyData(Data data) {
877 			notify(new NotificationObject(NotificationType.NEW_DATA, data));
878 		}
879 
880 		public void notifyTermination() {
881 			notify(new NotificationObject(NotificationType.TERMINATION, null));
882 		}
883 
884 		private void notify(final NotificationObject notificationObject) {
885 			synchronized(_notificationQueue) {
886 				_notificationQueue.addFirst(notificationObject);
887 				_notificationQueue.notifyAll();
888 			}
889 		}
890 
891 		public void run() {
892 			try {
893 				_debug.fine("SettingsManager.ChangeNotifier wurde gestartet");
894 				while(!Thread.interrupted()) {
895 					Data lastDataGot = null;
896 					boolean gotNewData = false;
897 					boolean gotDynamicUpdate= false;
898 					synchronized(_notificationQueue) {
899 						// Warten auf neue Aufträge
900 						while(_notificationQueue.isEmpty()) _notificationQueue.wait();
901 
902 						// Verzögerung solange, bis mindestens 500 Millisekunden lang keine weiteren Aktualisierungen anliegen
903 						// Maximale Gesamtverzögerung auf 10000 Millisekunden begrenzt um den Thread nicht verhungern zu lassen
904 						long time0 = System.currentTimeMillis();
905 						int lastSize = _notificationQueue.size();
906 						long timeMax = time0 + 10000;
907 						while(true) {
908 							_notificationQueue.wait(500);
909 							int newSize = _notificationQueue.size();
910 							if(lastSize == newSize) break;
911 							lastSize = newSize;
912 							if(System.currentTimeMillis() > timeMax) break;
913 						}
914 
915 						// Aktualisierungen auslesen
916 						while(!_notificationQueue.isEmpty()) {
917 							final NotificationObject notificationObject = _notificationQueue.removeLast();
918 							switch(notificationObject.getType()) {
919 								case DYNAMIC_CHANGE:
920 									// Aktualisierung eines dynamischen Objekts oder einer dynamischen Menge
921 									gotDynamicUpdate = true;
922 									break;
923 								case NEW_DATA:
924 									// Neues Data wurde empfangen => nur das letzte verarbeiten
925 									gotNewData = true;
926 									lastDataGot = notificationObject.getData();
927 									break;
928 								case TERMINATION:
929 									// SettingsManager wurde gestoppt => Thread beenden
930 									return;
931 							}
932 						}
933 					}
934 					if(gotNewData) {
935 						_lastSettingsData = lastDataGot;
936 					}
937 					if(gotDynamicUpdate || gotNewData) {
938 						final HashMap<DataIdentification, Data> newSettings = new HashMap<DataIdentification, Data>();
939 						extractSettings(_lastSettingsData, newSettings);
940 						activateSettings(newSettings);
941 					}
942 				} 
943 				_debug.fine("SettingsManager.ChangeNotifier beendet sich");
944 			}
945 			catch(InterruptedException e) {
946 				_debug.fine("SettingsManager.ChangeNotifier wurde unterbrochen und beendet sich");
947 			}
948 		}
949 
950 	}
