29  /**
30   * Klasse zum Zugriff auf die Aufrufargumente einer Applikation.
31   * Eine Applikation bekommt die Aufrufargumente von der Laufzeitumgebung in einem String-Array als Parameter
32   * der <code>main</code>-Funktion übergeben. Dieses String-Array wird dem Konstruktor dieser Klasse übergeben.
33   * Mit der Methode {@link #fetchArgument} kann auf die einzelnen Argumente zugegriffen werden.
34   * Beim Zugriff auf ein Argument wird der entsprechende Eintrag im String-Array auf <code>null</code> gesetzt.
35   * Nach dem Zugriff auf alle von einer Applikation
36   * unterstützten Argumente kann mit der Methode {@link #ensureAllArgumentsUsed} sichergestellt werden, daß alle
37   * angegebenen Argumente verwendet wurden.
38   *
39   * @author Kappich Systemberatung
40   * @version $Revision: 5005 $
41   */
42  public class ArgumentList {
43  	/**
44  	 * Speichert eine Kopie der ursprünglichen Aufrufargumente der Applikation, die dem Konstruktor übergeben wurden.
45  	 */
46  	private final String[] _initialArgumentStrings;
47  
48  	/**
49  	 * Speichert eine Referenz auf die Aufrufargumente der Applikation, die dem Konstruktor übergeben wurden.
50  	 * Nachdem ein Argument interpretiert wurde, wird es im übergebenen Array auf <code>null</code> gesetzt.
51  	 *
52  	 */
53  	private final String[] _argumentStrings;
54  
55  	/**
56  	 * Erzeugt eine neue Argumentliste und initialisiert diese mit den übergebenen Aufrufargumenten der Applikation.
57  	 * Einzelne Argumente, die von der Applikation bereits interpretiert wurden, sollten vorher auf den Wert <code>null</code>
58  	 * gesetzt werden, damit sie nicht nochmal interpretiert werden.
59  	 *
60  	 * @param argumentStrings  String-Array, das die Aufrufargumente enthält, die der Applikation beim Aufruf der
61  	 *                         main-Funktion übergeben werden.
62   	 */
63  	public ArgumentList(String[] argumentStrings) {
64  		_argumentStrings= argumentStrings;
65  		_initialArgumentStrings = new String[argumentStrings.length];
66  		System.arraycopy(_argumentStrings, 0, _initialArgumentStrings, 0, _argumentStrings.length);
67  	}
68  
69  	/**
70  	 * Liefert ein bestimmtes Argument zurück und setzt es im String-Array, das beim Konstruktor übergeben wurde,
71  	 * auf <code>null</code>. Im übergebenen Parameter der Funktion wird der Name des Arguments und bei Bedarf
72  	 * (durch ein Gleich-Zeichen getrennt) ein Default-Wert angegeben.
73  	 * Die Aufrufargumente müssen dem Format "argumentName=argumentWert" entsprechen:
74  	 * Beim Zugriff auf ein Argument muss der Argument-Name angegeben werden. Ergebnis des Zugriffs ist ein
75  	 * Objekt der Klasse {@link Argument} über das der Wert des Arguments abgefragt werden kann.
76  	 * Wenn das gewünschte Argument in der Argumentliste gefunden wurde, wird der entsprechende Eintrag im
77  	 * String-Array, das dem {@link #ArgumentList Konstruktor} übergeben wurde, auf den Wert <code>null</code> gesetzt.
78  	 * Wenn auf ein Argument zugegriffen wird, das in der Argumentlist nicht mehr vorhanden ist, weil es bereits vorher
79  	 * interpretiert wurde, dann wird der entsprechende Wert aus der initialen Argumentliste erneut zurückgegeben.
80  	 * Wenn auf ein Argument zugegriffen wird, das in der initialen Argumentliste nicht vorhanden ist,
81  	 * wird als Wert der Default-Wert benutzt, wenn dieser im Parameter der Methode (durch ein Gleichzeichen vom
82  	 * Argumentnamen getrennt) angegeben wurde. Wenn das gewünschte Argument nicht in der Argumentliste enthalten
83  	 * ist und kein Default-Wert angegeben wurde, wird eine Ausnahme generiert.
84  	 * @param nameAndOptionalDefault  Name des gewünschten Arguments und optional durch ein Gleichzeichen getrennt
85  	 *                                der Default-Wert des Arguments.
86  	 * @return  Ein Argument-Objekt über das mit verschiedenen Methoden auf den Wert des Arguments zugegriffen werden kann.
87  	 * @throws IllegalArgumentException  Wenn kein Wert für das gewünschte Argument ermittelt werden konnte.
88  	 */
89  	public Argument fetchArgument(String nameAndOptionalDefault) {
90  		int defaultSeparatorPosition= nameAndOptionalDefault.indexOf("=");
91  		String name= nameAndOptionalDefault;
92  		if(defaultSeparatorPosition >= 0) name= nameAndOptionalDefault.substring(0,defaultSeparatorPosition);
93  		for(int i=0; i<_argumentStrings.length; ++i) {
94  			String argumentString= _argumentStrings[i];
95  			if(argumentString!=null && ( argumentString.equals(name) || argumentString.startsWith(name + "=") )) {
96  				_argumentStrings[i]= null;
97  				return new Argument(argumentString);
98  			}
99  		}
100 		for(int i=0; i<_initialArgumentStrings.length; ++i) {
101 			String argumentString= _initialArgumentStrings[i];
102 			if(argumentString!=null && ( argumentString.equals(name) || argumentString.startsWith(name + "=") )) {
103 				return new Argument(argumentString);
104 			}
105 		}
106 		if(defaultSeparatorPosition >= 0) return new Argument(nameAndOptionalDefault);
107 		throw new java.lang.IllegalArgumentException("fehlendes Argument: " + name);
108 	}

110 	/**
111 	 * Prüft, ob ein bestimmtes Argument vorhanden ist und noch nicht interpretiert wurde.
112 	 *
113 	 * @param name Name des gesuchten Arguments.
114 	 * @return <code>true</code> Wenn das gesuchte Argument in der Argumentliste enthalten ist und noch nicht interpretiert
115 	 *         wurde, sonst <code>false</code>.
116 	 */
117 	public boolean hasArgument(String name) {
118 		for(int i=0; i<_argumentStrings.length; ++i) {
119 			String argumentString= _argumentStrings[i];
120 			if(argumentString!=null && ( argumentString.equals(name) || argumentString.startsWith(name + "=") )) {
121 				return true;
122 			}
123 		}
124 		return false;
125 	}

127 	/**
128 	 * Diese Methode stellt sicher, daß alle Argumente interpretiert wurden.
129 	 * @throws IllegalStateException  Wenn in der Argumentliste noch nicht ausgewertete Argumente enthalten sind.
130 	 */
131 	public void ensureAllArgumentsUsed() {
132 		if(hasUnusedArguments()) {
133 			StringBuffer message= new StringBuffer("Unbenutzte Argumente: {");
134 			Argument[] unusedArguments= fetchUnusedArguments();
135 			for(int i=0; i< unusedArguments.length; ++i) message.append(unusedArguments[i]).append(", ");
136 			message.append("}");
137 			throw new java.lang.IllegalStateException(message.toString());
138 		}
139 	}

141 	/**
142 	 * Bestimmt, ob in der Argumentliste noch Argumente exististieren, die noch nicht ausgewertet wurden.
143 	 * @return <code>true</code>, falls weitere Argumente existieren; <code>false</code>, falls alle Argumente
144 	 *         ausgewertet wurden.
145 	 */
146 	public boolean hasUnusedArguments() {
147 		for(int i=0; i<_argumentStrings.length; ++i) {
148 			if(_argumentStrings[i] != null) return true;
149 		}
150 		return false;
151 	}

153 	/**
154 	 * Liefert den Namen des nächsten noch nicht interpretierten Arguments zurück.
155 	 * @return  Name des nächsten noch nicht interpretierten Arguments.
156 	 * @throws IllegalStateException  Wenn bereits alle Argumente interpretiert wurden.
157 	 */
158 	public String getNextArgumentName() {
159 		for(int i=0; i<_argumentStrings.length; ++i) {
160 			String argumentString= _argumentStrings[i];
161 			if(argumentString!=null) return new Argument(argumentString).getName();
162 		}
163 		throw new java.lang.IllegalStateException("Zu wenig Argumente");
164 	}

166 	/**
167 	 * Liefert das erste noch nicht interpretierte Argument zurück und setzt es im String-Array, das beim Konstruktor übergeben wurde,
168 	 * auf <code>null</code>. Die Aufrufargumente müssen dem Format "argumentName=argumentWert" entsprechen:
169 	 * Beim Zugriff auf ein Argument muss der Argument-Name angegeben werden. Ergebnis des Zugriffs ist ein
170 	 * Objekt der Klasse {@link Argument} über das der Wert des Arguments abgefragt werden kann. Für Argumente die kein
171 	 * Gleichzeichen mit folgendem argumentWert enthalten, wird als Wert der Text "wahr" angenommen.
172 	 * Der entsprechende Eintrag im
173 	 * String-Array, das dem {@link #ArgumentList Konstruktor} übergeben wurde, wird auf den Wert <code>null</code> gesetzt.
174 	 * Wenn in der Argumentliste kein Argument mehr enthalten war, wird eine Ausnahme generiert.
175 	 * @return  Ein Argument-Objekt über das mit verschiedenen Methoden auf den Wert des Arguments zugegriffen werden kann.
176 	 * @throws IllegalStateException  Wenn bereits alle Argumente interpretiert wurden.
177 	 */
178 	public Argument fetchNextArgument() {
179 		for(int i=0; i<_argumentStrings.length; ++i) {
180 			String argumentString= _argumentStrings[i];
181 			if(argumentString!=null) {
182 				_argumentStrings[i]= null;
183 				return new Argument(argumentString);
184 			}
185 		}
186 		throw new java.lang.IllegalStateException("Zu wenig Argumente");
187 	}

189 	/**
190 	 * Bestimmt die Anzahl der in der Argumentliste noch vorhandenen und nicht ausgewerteten Argumente.
191 	 * @return  Anzahl noch nicht ausgewerteten Argumente in der Argumentliste.
192 	 */
193 	public int getUnusedArgumentCount() {
194 		int unusedCount= 0;
195 		for(int i=0; i<_argumentStrings.length; ++i) {
196 			if(_argumentStrings[i] != null) ++unusedCount;
197 		}
198 		return unusedCount;
199 	}
200 
201 	/**
202 	 * Liefert ein Feld mit den noch nicht ausgewerteten Argumenten der Aufrufliste zurück und setzt die
203 	 * entsprechenden Einträge im String-Array, das beim Konstruktor übergeben wurde,
204 	 * auf <code>null</code>.
205 	 * @return  Feld mit Argument-Objekten der noch nicht ausgewerteten Argumente.
206 	 */
207 	public Argument[] fetchUnusedArguments() {
208 		int unusedCount= getUnusedArgumentCount();
209 		if(unusedCount==0) return null;
210 		Argument[] unusedArguments= new Argument[unusedCount];
211 		unusedCount= 0;
212 		for(int i=0; i<_argumentStrings.length; ++i) {
213 			if(_argumentStrings[i] != null) {
214 				unusedArguments[unusedCount++]= new Argument(_argumentStrings[i]);
215 				_argumentStrings[i]= null;
216 			}
217 		}
218 		return unusedArguments;
219 	}

239 		/**
240 		 * Erzeugt ein neues Argument-Objekt aus dem übergebenen Argument-Text.
241 		 * Der Argument-Text muss folgenden Aufbau haben: "argumentName=argumentWert".
242 		 * Name und Wert des Argument-Objekts werden entsprechend gesetzt.
243 		 * Wenn der Argument-Text kein Gleichzeichen enthält, dann wird der ganze
244 		 * Argument-Text als Name des Arguments interpretiert und es wird vermerkt, daß das
245 		 * Argument keinen Wert hat.
246 		 */
247 		Argument(String argumentString) {
248 			int separatorPosition= argumentString.indexOf("=");
249 			if(separatorPosition >= 0) {
250 				_name= argumentString.substring(0,separatorPosition);
251 				_value= argumentString.substring(separatorPosition+1);
252 			}
253 			else {
254 				_name= argumentString;
255 				_value= null; // Default-Wert, wenn kein Wert angegeben wurde
256 			}
257 		}

259 		/**
260 		 * Bestimmt den Namen des Arguments.
261 		 * @return  Name des Arguments.
262 		 */
263 		public String getName() {
264 			return _name;
265 		}

267 		/**
268 		 * Überprüft, ob das Argument einen Wert hat. Wenn das Argument keinen Wert
269 		 * hat, kann nur noch mit der Methode #{@link #booleanValue} auf den Wert
270 		 * zugegriffen werden, ohne daß eine Ausnahme generiert wird.
271 		 * @return <code>true</code>, falls das Argument einen Wert hat; sonst <code>false</code>.
272 		 */
273 		public boolean hasValue() {
274 			if(_value==null) return false;
275 			return true;
276 		}

278 		/**
279 		 * Bestimmt den Wert des Arguments.
280 		 * @return  Wert des Arguments.
281 		 * @throws IllegalStateException  Wenn das Argument keinen Wert hat.
282 		 */
283 		public String getValue() {
284 			if(hasValue()) return _value;
285 			throw new IllegalArgumentException("Argument " + getName() + " hat keinen Wert");
286 		}

288 		/**
289 		 * Gibt den Wert des Arguments als <code>boolean</code> Wert zurück.
290 		 * Die Argumentwerte "wahr", "ja", "1" werden zum Boolschen Wert <code>true</code> konvertiert;
291 		 * die Argumentwerte "falsch", "nein", "0" werden zum Boolschen Wert <code>false</code> konvertiert.
292 		 * Die Groß-/Kleinschreibung des Argumentwerts hat beim Vergleich keine Relevanz.
293 		 * Wenn das Argument keinen Wert hat, dann wird als Ergebnis der Konvertierung <code>true</code>
294 		 * zurückgegeben.
295 		 * @return  Der konvertierte Argumentwert.
296 		 * @throws IllegalArgumentException  Wenn der Argumentwert nicht konvertiert werden konnte.
297 		 */
298 		public boolean booleanValue() throws IllegalArgumentException{
299 			if(!hasValue()) return true;
300 			String value= _value.toLowerCase();
301 			if(value.equals("wahr") || value.equals("ja") || value.equals("1")) return true;
302 			if(value.equals("falsch") || value.equals("nein") || value.equals("0")) return false;
303 			//englische Varianten werden auch unterstützt:
304 			if(value.equals("true") || value.equals("yes")) return true;
305 			if(value.equals("false") || value.equals("no")) return false;
306 			else throw new java.lang.IllegalArgumentException("Argument " + getName() + " hat keinen boolschen Wert: " + getValue());
307 		}

799 		 * beim Aufruf der Methode aktuelle Zeit interpretiert.

807 			value= value.replace('-',' ');

809 				format= _parseDateFormats[i];
810 				try {
811 					synchronized(format) {

813 					}
814 					return date.getTime();
815 				}
816 				catch(ParseException e) {
817 					//continue with next Format
818 				}
819 			}
820 			for(int i= 0; i < _parseTimeFormats.length; ++i) {

823 					synchronized(format) {

842 		/**
843 		 * Interpretiert den Wert des Arguments als relative Zeitangabe.
844 		 * Das Argument muss aus einer Liste von Zahlen und Einheiten bestehen.
845 		 * Als Einheiten sind "t" und "Tag[e]" für Tage, "h" und "Stunde[n]" für Stunden,
846 		 * "m" und "Minute[n]" für Minuten, "s" und "Sekunde[n]" für Sekunden
847 		 * sowie "ms" und "Millisekunden[e]" für Millisekunden erkannt.
848 		 * Die Einzelnen Werte werden in Millisekunden umgerechnet und aufsummiert.
849 		 * Als spezieller Wert wird der Text "jetzt" erkannt und als "0 Sekunden"
850 		 * interpretiert.
851 		 * @return  Relative Zeitangabe in Millisekunden.
852 		 */
853 		public long asRelativeTime() {
854 			String value= getValue();
855 			if(value.toLowerCase().equals("jetzt")) return 0;
856 			String[] splitted= value.trim().split(_relPattern);
857 			long number= 0;
858 			long millis= 0;
859 			for(int i=0; i< splitted.length; ++i) {
860 				String word= splitted[i];
861 				number= Long.decode(word).longValue();
862 				if(++i<splitted.length) {
863 					word= splitted[i].toLowerCase();
864 					if(word.equals("t") || word.startsWith("tag")) {
865 						millis+= (1000*60*60*24) * number;
866 					}
867 					else if(word.equals("h") || word.startsWith("stunde")) {
868 						millis+= (1000*60*60) * number;
869 					}
870 					else if(word.equals("m") || word.startsWith("minute")) {
871 						millis+= (1000*60) * number;
872 					}
873 					else if(word.equals("s") || word.startsWith("sekunde")) {
874 						millis+= 1000 * number;
875 					}
876 					else if(word.equals("ms") || word.startsWith("milli")) {
877 						millis+= number;
878 					}
879 					else throw new IllegalArgumentException("Ungültige relative Zeitangabe: " + splitted[i]);
880 				}
881 				else if(number!=0) throw new IllegalArgumentException("Einheit bei relativer Zeitangabe fehlt");
882 			}
883 			return millis;
884 		}
