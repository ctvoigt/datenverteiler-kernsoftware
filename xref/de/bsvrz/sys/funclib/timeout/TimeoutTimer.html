21  package de.bsvrz.sys.funclib.timeout;
22  
23  import java.util.Date;
24  
25  /**
26   * Ein Objekt dieser Klasse kann mit einer Uhrzeit oder einer Zeitspanne initialisert werden. Das Objekt stellt eine
27   * Methode {@link TimeoutTimer#isTimeExpired()} zur Verfügung, mit der Angefragt werden kann, ob die Uhrzeit erreicht
28   * wurde oder ob die Zeitspanne abgelaufen ist.
29   *
30   * @author Kappich Systemberatung
31   * @version $Revision: 5007 $
32   */
33  public class TimeoutTimer {
34  	/**
35  	 * Zeitpunkt, an dem das Objekt erzeugt wurde
36  	 */
37  	private final long _startTime;
38  
39  	/**
40  	 * Zeitpunkt, an dem das Objekt bei einem Methodenaufruf melden muss, das die vorgegebene Zeit verstrichen ist
41  	 */
42  	private final long _endTime;
43  
44  	/**
45  	 * Ein Objekt, das mit einer Zeitspanne initialisiert wird. Ist die geforderte Zeitspanne abgelaufen, wird der
46  	 * Methodenaufruf {@link TimeoutTimer#isTimeExpired()} <code>false</code> zurückliefern. Die Zeit wird ab dem Zeitpunkt
47  	 * gemessen, ab dem der Construktor aufgerufen wird.
48  	 *
49  	 * @param time Zeitspanne, die ablaufen muss, bis {@link TimeoutTimer#isTimeExpired()} den Rückgabewert
50  	 *             <code>false</code> liefert
51  	 */
52  	public TimeoutTimer(long time) {
53  
54  		_startTime = System.currentTimeMillis();
55  		// Es wurde eine Zeitspanne eingegeben, diese muss nun in eine Zeitpunkt umgerechnet werden
56  		_endTime = _startTime + time;
57  	}
58  
59  	/**
60  	 * Ein Objekt, das mit einem Zeitpunkt initialisiert wird. Wird der geforderte Zeitpunkt überschritten, wird der
61  	 * Methodenaufruf {@link TimeoutTimer#isTimeExpired()} <code>false</code> zurückliefern. Die Zeit wird ab dem Zeitpunkt
62  	 * gemessen, ab dem der Construktor aufgerufen wird.
63  	 *
64  	 * @param date Zeitpunkt, der überschritten werden muss, damit die Methode {@link TimeoutTimer#isTimeExpired()}
65  	 *             <code>false</code> zurückliefert
66  	 */
67  	public TimeoutTimer(Date date) {
68  		_startTime = System.currentTimeMillis();
69  		_endTime = date.getTime();
70  	}
71  
72  	/**
73  	 * Diese Methode wird aufgerufen, wenn geprüft werden soll ob die festgelegte Zeitspanne abgelaufen ist oder der
74  	 * festgelegte Zeitpunkt erreicht wurde.
75  	 *
76  	 * @return true = die Zeit ist abgelaufen; false = die/der Zeitspanne/Zeitpunkt wurde noch nicht erreicht
77  	 */
78  	public boolean isTimeExpired() {
79  		return System.currentTimeMillis() >= _endTime;
80  	}
81  
82  	/**
83  	 * Diese Methode liefert die verbleibende Zeit, bis die vorgegebene Zeitspanne/Zeitpunkt abgelaufen ist.
84  	 *
85  	 * @return Zeit, die noch verbleibt, bis der vorgegebene Wert erreicht wird (in ms). Ist die Zeit abgelaufen, wird
86  	 *         immer <code>0</code> zurückgegeben.
87  	 */
88  	public long getRemainingTime() {
89  		final long remainingTime = _endTime - System.currentTimeMillis();
90  		if (remainingTime < 0) {
91  			// Die Zeit ist abgelaufen, somit wird 0 zurückgegeben
92  			return 0;
93  		} else {
94  			return remainingTime;
95  		}
96  	}
97  
98  	public String toString() {
99  		return "Startzeitpunkt: " + new Date(_startTime).toString() + " Endzeitpunkt: " + new Date(_endTime).toString() + " Zeitspanne bis zum Endzeitpunkt(ms): " + (_endTime - System.currentTimeMillis()) + " Zeit abgelaufen: " + isTimeExpired();
100 	}
101 }
