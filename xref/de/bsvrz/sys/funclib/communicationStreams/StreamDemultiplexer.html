37  /**
38   * Diese Klasse empfängt Nutzdatenpakete, die vom StreamMultiplexer über Streams versandt wurden. Dabei werden die Daten
39   * in streams aufgeteilt. Jeder stream stellt dabei der Applikation Nutzdaten zur Verfügung, die sie mit einer Methode
40   * abrufen kann. Der StreamMultiplexer wird benachrichtigt falls die Applikation Nutzdaten benötigt. Dieser sorgt dann
41   * dafür, dass neue Daten auf dem dafür vorgesehenen Stream bereitgestellt werden. Das Speichern der Nutzdaten geschieht
42   * in einem Puffer, unterschreitet die Anzahl der vorgehaltenen Daten einen bestimmten Wert, so fordert der
43   * StreamDemultiplexer neue Nutzdaten vom StreamMultiplexer an. Dies gewährleistet einen ständigen Vorrat von Nutzdaten,
44   * auf die die Applikation zugreifen kann.
45   *
46   * @author Kappich Systemberatung
47   * @version $Revision: 5021 $
48   */
49  public class StreamDemultiplexer {

56  	 * Der _blockingFactor  bestimmt die größe des Empfängerpuffers. Ist der Puffer nur noch halb voll, wird der Sender
57  	 * benachrichtigt, dass er neue Nutzdatensätze verschicken soll.
58  	 */
59  	private final int _blockingFactor;

108 	/**
109 	 * @param numberOfStreams Anzahl von Streams, die Datenpakete versenden sollen
110 	 * @param blockingFactor  Puffer des Empfängers (dieser Wert muß größer gleich 1 sein)
111 	 * @param director        Schnittstelle, die eine Methode zum verschicken von Informationen an den Sender bereitstellt
112 	 *                        (siehe Interface Beschreibung)
113 	 * @throws IllegalArgumentException Der blockingFactor war kleiner als 1
114 	 */
115 	public StreamDemultiplexer(int numberOfStreams, int blockingFactor, StreamDemultiplexerDirector director) {
116 		_numberOfStreams = numberOfStreams;
117 
118 		if (blockingFactor < 1) {
119 			throw new IllegalArgumentException("Der blockingFactor muß größer gleich 1 sein");
120 		}
121 
122 		_blockingFactor = blockingFactor;
123 
124 		if (_blockingFactor == 1) {
125 			_ticketBlockingFactor = 1;
126 		} else {
127 			_ticketBlockingFactor = _blockingFactor / 2;
128 		}
129 
130 		_arrayOfStreams = new DemultiplexerStreaminformations[_numberOfStreams];
131 		for (int i = 0; i < _arrayOfStreams.length; i++) {
132 			// i ist der Index des Streams
133 			// Wann muß der Stream die erste Sendebestätigung verschicken (_ticketBlockingFactor)
134 			// Wie viele Pakete kommen beim ersten zustanden kommen der Verbindung vom Sender (_blockingFactor)
135 			DemultiplexerStreaminformations streaminformations = new DemultiplexerStreaminformations(i, _ticketBlockingFactor, _blockingFactor);
136 			// Infos im Array speichern
137 			_arrayOfStreams[i] = streaminformations;
138 		}
139 
140 		_director = director;
141 
142 		// Debug
143 		_numberOfReceivedPackets = new int[_numberOfStreams];
144 		_numberOfTakes = new int[_numberOfStreams];
145 	}

147 	/**
148 	 * Eine Methode zum beenden eines Streams. Die restlichen Daten, die sich im  _arrayOfStreams befinden, werden
149 	 * gelöscht. An den Sender der Daten wird ein "-1" geschickt, dieser wird darauf hin keine Daten mehr auf diesem Stream
150 	 * schicken. Der Stream wird über seinen Index identifiziert.
151 	 *
152 	 * @param indexOfStream Index des Streams, der beendet werden soll
153 	 */
154 	public void abort(int indexOfStream) {
155 		_debug.fine("Abort Stream: " + indexOfStream);
156 
157 		// Es wird nur ein Eintrag des Arrays "gesperrt", nicht das gesamte Array
158 		final DemultiplexerStreaminformations stream = _arrayOfStreams[indexOfStream];
159 
160 		synchronized (stream) {
161 			if ((stream.isEndStream() == false) && (stream.isStreamAborted() == false) && (stream.isStreamTerminated() == false) && (stream.isLostConnectionToSender() == false)) {
162 				// Der Stream darf keinen Nutzdaten mehr empfangen/senden.
163 				// Es kann passieren, dass Threads gerade Nutzdaten anfordern (take Methode: referenceDataPacket = stream.getReferenceDataPacket();), allerdings
164 				// sind keine Nutzdaten vorhanden (deblockingFactor ist 1, ein take wurde aufgerufen und die Empfängerapplikation sendet ein abort).
165 				// Die neuen Daten sind dann zwar unterwegs, werden aber abgelehnt und die Threads warten vergebens auf neue Daten.
166 				// In der "setStreamAbortedTrue()" Methode wird ein künstliches "null" Paket erzeugt. Dadruch werden die Threads befreit.
167 				// In der Take Methode wird dann geprüft ob der Stream mit "Abort" beendet wurde, wenn ja, wird eine Exception geworfen.
168 				stream.setStreamAbortedTrue();
169 				// Dem Sender wird mitgeteilt, dass der Stream nicht mehr benötigt wird. Dies geschieht in dem der maximale Index
170 				// bis zu dem gesendet werden darf auf "-1" gesetzt wird.
171 				try {
172 					sendNewTicketIndexToSender(indexOfStream, -1);
173 				} catch (IOException e) {
174 					e.printStackTrace();
175 					_debug.error("Ein Fehler beim serialisieren/deserialisieren: " + e);
176 				}
177 			}
178 		}
179 	}

181 	/**
182 	 * Diese Methode gibt die Nutzdaten eines bestimmten Streams an die Empfängerapplikation zurück.
183 	 * Wenn alle Nutzdatenpakete von der Senderapplikation mit take angefordert wurden, gibt diese
184 	 * Methode nur noch null zurück, im Fehlerfall werden entsprechende Exceptions geworfen.
185 	 *
186 	 * @param indexOfStream Eindeutiger Index des Streams, der Daten zurückgeben soll
187 	 * @return Die Nutzdaten werden als Byte-Array geliefert
188 	 * @throws InterruptedException   Ein Thread, der Nutzdaten mit take anfordert hat, wird mit Interrupt abgebrochen.
189 	 * @throws IllegalStateException  Ein Stream wurde mit abort durch die Empfängerapplikation beendet und anschließend
190 	 *                                führte die Empfängerapplikation erneut ein take auf diesen Stream aus.
191 	 * @throws ProtocolException      Es wurde ein Paket mit einem falschen Index bearbeitet. Das deutet darauf hin, dass
192 	 *                                das Paket entweder schon einmal empfangen wurde (doppelt vorhanden) oder das ein
193 	 *                                Paket fehlt. Der Stream wird automatisch abgebrochen. Die Applikation kann weiter
194 	 *                                "take" aufrufen, wird aber immer für diesen Stream, diese Exception bekommen.
195 	 * @throws ClosedChannelException Die physische Verbindung zum Sender wurde unterbrochen. Jeder take Aufruf wird für
196 	 *                                alle Streams diese Exception ausgeben, da alle Streams betroffen sind
197 	 */
198 	public byte[] take(int indexOfStream) throws InterruptedException, IllegalStateException, ProtocolException, ClosedChannelException {
199 
200 		// stream anfordern
201 		final DemultiplexerStreaminformations stream = _arrayOfStreams[indexOfStream];
202 
203 		// Alle take-Aufrufe auf einem Stream fordern den Monitor auf dieses Objekt an. Somit kann
204 		// nur ein take eines Threads weiter arbeiten.
205 		// Das verhindert, dass alle Threads <code>stream.sizeOfSmallDataPacketQueue() == 0</code>
206 		// bekommen und somit alle Threads ein großes Paket auspacken können.
207 		synchronized (stream._smallDataPacketQueue) {
208 
209 			// Der Aufruf zum entpacken eines großen Pakets darf auf keinen Fall mit in den synch. Block.
210 			// Denn wenn kein großes Parket vorhanden ist, wird der erste Thread sich mit dem synchronisierten Stream
211 			// schlafen legen (beim anforden des Pakets mit take), somit ist es dann unmöglich dem Stream jemals wieder
212 			// große Pakete zu geben (was aber nötig wäre, damit der erste Thread befreit wird). -> Deadlock
213 			boolean unpackBigPacket = false;
214 
215 			synchronized (stream) {
216 
217 				// Dieser If-Abfagen Block ist nötig, weil in der Zwischenzeit ein Fehler aufgetreten sein kann.
218 				// (als Beipsiel, ein Paketdreher beim Empfang)
219 				if (stream.isEndStream() == true) {
220 					// Die Sendeapplikation hatte keine Nutzdaten mehr (Normalfall).
221 
222 					// Dieser Aufruf befreit einen weiteren Thread. Dieser wird auch zu dieser Abfrage kommen und
223 					// einen anderen wartenden Thread befreien, usw. . Alle Threads geben also "null" zurück
224 					stream.createUnlockPacket();
225 					// throw new IllegalStateException("Fehler Stream (Index: " + indexOfStream + "): Der Sender hat keine Nutzdaten mehr für den Empfänger. Der Stream wurde bereits beidseitig beendet, aber von der Empfängerapplikation wurde erneut 'take' augerufen");
226 					return null;
227 				} else if (stream.isStreamAborted() == true) {
228 					// wartende Threads befreien
229 					stream.createUnlockPacket();
230 					// Der Stream wurde mit Abort von der Empfängerapplikation beendet, somit darf kein Take mehr ausgeführt werden.
231 					throw new IllegalStateException("Fehler Stream (Index: " + indexOfStream + "): Der Stream wurde mit 'abort'von der Empfängerapplikation abgebrochen und dann erneut mit 'take' aufgerufen");
232 				} else if (stream.isStreamTerminated() == true) {
233 					// wartende Threads befreien
234 					stream.createUnlockPacket();
235 					// Ein Nutzdatenpaket des Stream ist entweder doppelt vorhanden oder vorschwunden. Der Empfänger hat die Verbindung abgebrochen
236 					throw new ProtocolException("Fehler Stream (Index: " + indexOfStream + "): Ein Nutzdatenpaket wurde entweder doppelt empfangen oder ist verschwunden. Der Empfänger(StreamDemultplexer) hat den Stream abgebrochen und den Sender (StreamMultiplexer) benachrichtigt");
237 				} else if (stream.isLostConnectionToSender() == true) {
238 					// wartende Threads befreien
239 					stream.createUnlockPacket();
240 					// Die physische Verbindung zum Empfänger ist unterbrochen. Die Empfängerapplikation informiert den Empfänger darüber.
241 					
242 					throw new ClosedChannelException();
243 				}
244 
245 				if (stream.sizeOfSmallDataPacketQueue() == 0) {
246 					// Es stehen "keine" Nutzdaten mehr zur Verfügung. Aber vielleicht gibt es noch
247 					// große Pakete, die ausgepackt werden können und Nutzdaten enthalten.
248 					unpackBigPacket = true;
249 				}
250 
251 			} // synchronized (stream)
252 
253 			if (unpackBigPacket == true) {
254 				// Es muss ein großes Paket entpackt werden um Nutzdaten zu erhalten
255 				unpackBigPacket(stream);
256 			}
257 
258 			synchronized (stream) {
259 
260 				// Auch hier muss wieder geprüft werden, ob der Stream noch Nutzdaten schicken darf.
261 				// Es kann zu Paketdrehern gekommen sein (als Beispiel).
262 				if (stream.isEndStream() == true) {
263 					// Die Sendeapplikation hatte keine Nutzdaten mehr (Normalfall).
264 
265 					// Dieser Aufruf befreit einen weiteren Thread. Dieser wird auch zu dieser Abfrage kommen und
266 					// einen anderen wartenden Thread befreien, usw. . Alle Threads geben also "null" zurück
267 					stream.createUnlockPacket();
268 					// throw new IllegalStateException("Fehler Stream (Index: " + indexOfStream + "): Der Sender hat keine Nutzdaten mehr für den Empfänger. Der Stream wurde bereits beidseitig beendet, aber von der Empfängerapplikation wurde erneut 'take' augerufen");
269 					return null;
270 				} else if (stream.isStreamAborted() == true) {
271 					// wartende Threads befreien
272 					stream.createUnlockPacket();
273 					// Der Stream wurde mit Abort von der Empfängerapplikation beendet, somit darf kein Take mehr ausgeführt werden.
274 					throw new IllegalStateException("Fehler Stream (Index: " + indexOfStream + "): Der Stream wurde mit 'abort'von der Empfängerapplikation abgebrochen und dann erneut mit 'take' aufgerufen");
275 				} else if (stream.isStreamTerminated() == true) {
276 					// wartende Threads befreien
277 					stream.createUnlockPacket();
278 					// Ein Nutzdatenpaket des Stream ist entweder doppelt vorhanden oder vorschwunden. Der Empfänger hat die Verbindung abgebrochen
279 					throw new ProtocolException("Fehler Stream (Index: " + indexOfStream + "): Ein Nutzdatenpaket wurde entweder doppelt empfangen oder ist verschwunden. Der Empfänger(StreamDemultplexer) hat den Stream abgebrochen und den Sender (StreamMultiplexer) benachrichtigt");
280 				} else if (stream.isLostConnectionToSender() == true) {
281 					// wartende Threads befreien
282 					stream.createUnlockPacket();
283 					// Die physische Verbindung zum Empfänger ist unterbrochen. Die Empfängerapplikation informiert den Empfänger darüber.
284 					throw new ClosedChannelException();
285 				}
286 
287 				byte[] data = stream.getData();
288 				if (data == null) {
289 					stream.setEndStreamTrue();
290 				}
291 
292 				// Diese Zeilen dienen nur Debugzwecken
293 //				System.out.println("StreamDemultiplexer Thread hält an");
294 //				Thread.sleep(5000);
295 
296 				return data;
297 			} // synchronized (stream) (Teil 2, das ist nicht das erste synch auf dem stream)
298 		} // synchronized (stream._smallDataPacketQueue)
299 	}

301 	private void unpackBigPacket(DemultiplexerStreaminformations stream) throws InterruptedException {
302 
303 		// Nutzdatenpakete (das können mehrere sein) und der Paketindex(des großen Pakets) stehen nun zur Verfügung.
304 		// Genau an dieser Stelle greift der Thread auf eine Queue zu, die ihn schlafen legt, wenn KEIN Paket vorhanden ist.
305 		// Um aber auf diese Schlange ein Paket zu legen, muss man sich auf die <code>DemultiplexerStreaminformations</code>
306 		// aber synchronisieren. Hat aber der wartende Thread den Monitor noch, dann gibt es einen Deadlock.
307 		// Aus diesem Grund wird dieser Zugriff aus jeden <code>synchronized(stream)</code> rausgehalten.
308 		final ReferenceDataPacket referenceDataPacket = stream.getReferenceDataPacket();
309 
310 		// Der neue maximale Paketindex, dieser wird hier deklariert, weil er außerhalb des synch. Blocks benötigt wird.
311 		// Er wird aber innerhalb des synch. Blocks gesetzt, dies stellt aber kein Problem dar (siehe Kommentar zum versenden des Tickets).
312 		int newMaxStreamPacketIndex = 0;
313 
314 		// Wenn im synch. Block ein neuer max Index Wert berechnet wird, dann muß ein Ticket versandt werden (aber nur dann).
315 		// Ist diese Variable auf true, dann muß ein Ticket versandt werden.
316 		boolean sendNewTicket = false;
317 
318 		final int indexOfStream;
319 
320 		// War das null-Paket in dem großen Paket
321 		boolean nullPacket = false;
322 
323 		synchronized (stream) {
324 			// Nutzdaten und der Paketindex stehen nun zur Verfügung
325 
326 			// Index des Stream
327 			indexOfStream = stream.getIndexOfStream();
328 
329 			// Falls der Paketindex mit dem Index an dem eine neue Sendebestätigung gesendet werden muß überein stimmt, dann wird ein Ticket verschickt
330 			if (stream.getPacketIndexToSendNextMaxTicketIndex() == referenceDataPacket.getStreamPacketIndex()) {
331 				// Dieses Packet muß den Sender benachrichtigen, dass er neue Pakete verschicken muß.
332 				// Es werden _blockingFactor viele Pakete angefordert.
333 
334 				// Den neuen Index berechnen, bis zu dem der Sender Pakete schicken darf
335 				newMaxStreamPacketIndex = stream.getPacketIndexToSendNextMaxTicketIndex() + _blockingFactor;
336 
337 				// Den neuen Index berechnen, an dem eine neue Sendeerlaubnis an den Sender verschickt werden darf. Dies entspricht
338 				// _blockingFactor/2 oder aber der Variablen _ticketBlockingFactor. Diese wird bei einem
339 				// _blockingFactor von 1 automatisch auf 1 gesetzt.
340 				final int newPacketIndexToSendNextMaxTicketIndex = stream.getPacketIndexToSendNextMaxTicketIndex() + _ticketBlockingFactor;
341 
342 				stream.setMaxStreamPacketIndex(newMaxStreamPacketIndex);
343 				stream.setPacketIndexToSendNextMaxTicketIndex(newPacketIndexToSendNextMaxTicketIndex);
344 				sendNewTicket = true;
345 			}
346 
347 			// Hier steht nun das große Paket zur Verfügung, alle Berechnungen ob ein Ticket verschickt werden muss
348 			// (wenn ja, was für eins) wurden beendet. Nun können die Nutzdatenpakete (die kleinen Pakete) erzeugt werden.
349 			// Dabei kann gleich geprüft werden, ob ein null-Paket dabei war, falls ja, muß das Ticket (wenn denn eins verschickt
350 			// werden muss) gar nicht mehr versandt werden.
351 
352 			// Das ganze findet auf einem synchronisierten Stream statt, da auf die Queue für die Nutzdaten zugegriffen werden
353 			// muss. Ist dieser nicht synchronisiert, kann die Reihenfolge der Nutzdaten durcheinander kommen.
354 
355 			// Wurden alle Pakete aus dem großen Paket ausgepackt
356 			boolean unpackedAllPackets = false;
357 			// Hier sind alle kleinen Nutzdatenpakete(die für die Empfängerapplikation bestimmt sind) verpackt
358 			final byte[] bigDataPacket = referenceDataPacket.getData();
359 
360 			final InputStream in = new ByteArrayInputStream(bigDataPacket);
361 			//deserialisieren
362 			final Deserializer deserializer = SerializingFactory.createDeserializer(in);
363 
364 			while (unpackedAllPackets == false) {
365 				try {
366 					// Größe des Nutzdatenpakets
367 					int sizeOfData = deserializer.readInt();
368 
369 					byte[] data;
370 					if (sizeOfData >= 0) {
371 						// Es sind Nutzdaten vorhanden, das null-Paket hat einen negativen Index
372 						data = deserializer.readBytes(sizeOfData);
373 
374 						// Das gerade ausgepackte Nutzdatenpaket kann nun im Stream gespeichert werden.
375 						// _debug.fine("Ein normales Nutzdatenpaket wird in die kleine Queue gespeichert");
376 						stream.putDataSmallDataPacketQueue(data);
377 
378 					} else {
379 						// Die Größe des Nutzdatenpakets ist negativ, somit wurde entweder ein null-Paket empfangen oder
380 						// das große Paket hat keine kleinen Nutzdatenpakete mehr.
381 
382 						if (sizeOfData == -1) {
383 							// Das bedeutet, dass die Senderapplikation keine Nutzdaten mehr für die
384 							// Empfängerapplikation hat. Der Stream wurde auf der Senderseite bereits als
385 							// "beendet" markiert.
386 							// Es wird nicht weiter geguckt ob die -2 auch noch versandt wurde (was der Fall ist), weil
387 							// die -1 bereits eindeutig ist.
388 							data = null;
389 							nullPacket = true;
390 							unpackedAllPackets = true;
391 
392 							// Das gerade ausgepackte Nutzdatenpaket kann nun im Stream gespeichert werden.
393 							stream.putDataSmallDataPacketQueue(data);
394 
395 							_debug.fine("Beim auspacken eines großen Pakets, Stream(" + indexOfStream + ") wurde ein null-Paket mit Index(Index des großen Pakets) " + referenceDataPacket.getStreamPacketIndex() + " gefunden");
396 						} else {
397 
398 							// Es wurden alle bytes des byte-Arrays ausgelesen, also ist das große Paket "leer"
399 							unpackedAllPackets = true;
400 							_debug.finer("Ein großes Paket wurde ausgepackt, es enthielt nur Nutzdaten. Stream:" + indexOfStream + " Index des großen Pakets: " + referenceDataPacket.getStreamPacketIndex());
401 						}
402 					}
403 				} catch (IOException e) {
404 					e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
405 				}
406 			}
407 
408 		} // synchronized(stream)
409 
410 		// Das versenden darf außerhalb des synch. Blocks stehen, da auf der Gegenseite (Sender) geprüft wird, ob
411 		// der neue maximale Index größer ist als der der gerade empfangen wird.
412 		// Es ist also nicht schlimm, wenn der Empfänger einen "alten" Index an den Sender schickt.
413 		if ((sendNewTicket == true) && (nullPacket == false)) {
414 			// Ein neues Ticket muß an den Sender verschickt werden. Dies muß aber nur geschehen, wenn der Sender noch Nutzdaten
415 			// für den Empfänger hat. Dies erkennt man daran, dass die Nutzdaten ungleich null sind. Sind sie null, dann hat
416 			// der Sender keine Nutzdaten mehr und hat seinerseits den Stream schon beendet (falls noch Tickets für
417 			// den Stream unterwegs sind, wird der Sender diese ignorieren)
418 			try {
419 				sendNewTicketIndexToSender(indexOfStream, newMaxStreamPacketIndex);
420 			} catch (IOException e) {
421 				e.printStackTrace();
422 				_debug.error("Ein Fehler beim serialisieren/deserialisieren: " + e);
423 			}
424 		}

Paketindex (" + referenceDataPacket.getStreamPacketIndex() + ") _blockingFactor (" + _blockingFactor + ") Größe Dataqueue (" + stream.sizeOfDataQueue() + ")";
