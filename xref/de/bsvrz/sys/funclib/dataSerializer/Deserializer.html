33  /**
34   * Schnittstelle zum Deserialisieren von Datensätzen. Konkrete Objekte zum Deserialisieren können mit den verschiedenen
35   * Methoden der Klasse {@link SerializingFactory} erzeugt werden. Dabei kann der Eingabe-Stream für die Deserialisierung
36   * vorgegeben werden.
37   * <p/>
38   * Mit den verschiedenen <code>read</code>-Methoden können primitive Datentypen, Referenzen auf {@link SystemObject
39   * System-Objekte} und ganze {@link Data Datensätze} von einem {@link InputStream} eingelesen und deserialisiert
40   * werden.
41   * <p/>
42   * Über die {@link #getVersion} Methode kann die Version eines konkreten Deserialisieres abgefragt werden.
43   *
44   * @author Kappich Systemberatung
45   * @version $Revision: 5058 $
46   * @see Serializer
47   */
48  public interface Deserializer {
49  
50  	/**
51  	 * Bestimmt die Version des konkreten Deserialisierers.
52  	 *
53  	 * @return Version des Deserialisierers.
54  	 */
55  	int getVersion();
56  
57  	/**
58  	 * Bestimmt den bei der Deserialisierung zu verwendenden Eingabe-Stream.
59  	 *
60  	 * @return Bei der Deserialisierung zu verwendender Eingabe-Stream.
61  	 */
62  	InputStream getInputStream();
63  
64  	/**
65  	 * Setzt den zu verwendenden Eingabe-Stream.
66  	 * @param inputStream Zu verwendender InputStream
67  	 */
68  	void setInputStream(InputStream inputStream);
69  
70  	/**
71  	 * Liest und deserialisiert einen Datensatz aus dem Eingabe-Stream dieses Deserialisierers.
72  	 *
73  	 * @param data Neuer Datensatz, der mit der Attributgruppe der erwarteten Daten initialisiert wurde.
74  	 * @throws IOException Wenn beim Lesen vom Eingabe-Stream Fehler aufgetreten sind.
75  	 * @see de.bsvrz.dav.daf.main.ClientDavConnection#createData
76  	 */
77  	void readData(Data data) throws IOException;
78  
79  	/**
80  	 * Liest und deserialisiert einen Datensatz aus dem Eingabe-Stream dieses Deserialisierers.
81  	 *
82  	 * @param atg Attributgruppe des einzulesenden Datensatzes.
83  	 * @return Eingelesener Datensatz
84  	 * @throws IOException Wenn beim Lesen vom Eingabe-Stream Fehler aufgetreten sind.
85  	 */
86  	Data readData(AttributeGroup atg) throws IOException;
87  
88  	/**
89  	 * Liest und deserialisiert einen Datensatz aus dem Eingabe-Stream dieses Deserialisierers.
90  	 *
91  	 * @param data Neuer Datensatz, der mit der Attributgruppe der erwarteten Daten initialisiert wurde.
92  	 * @param dataModel Datenmodell mit dessen Hilfe Objektreferenzen aufgelöst werden.
93  	 * @throws IOException Wenn beim Lesen vom Eingabe-Stream Fehler aufgetreten sind.
94  	 * @see de.bsvrz.dav.daf.main.ClientDavConnection#createData
95  	 */
96  	void readData(Data data, ObjectLookup dataModel) throws IOException;
97  
98  	/**

100 	 *
101 	 * @param atg Attributgruppe des einzulesenden Datensatzes.
102 	 * @param dataModel Datenmodell mit dessen Hilfe Objektreferenzen aufgelöst werden.
103 	 * @return Eingelesener Datensatz
104 	 * @throws IOException Wenn beim Lesen vom Eingabe-Stream Fehler aufgetreten sind.
105 	 */
106 	Data readData(AttributeGroup atg, ObjectLookup dataModel) throws IOException;

108 	/**

110 	 *
111 	 * @return Der eingelesene Wert.
112 	 * @throws IOException Wenn beim Lesen vom Eingabe-Stream Fehler aufgetreten sind.

114 	public boolean readBoolean() throws IOException;

197 	 * @throws IOException Wenn beim Lesen vom Eingabe-Stream Fehler aufgetreten sind.

226 	 * @param length Länge des einzulesenden Byte-Arrays
