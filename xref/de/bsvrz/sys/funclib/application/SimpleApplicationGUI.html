271 	 * Diese Methode erzeugt einen Prozess, der ausgeführt wird. Alle Ausgaben
272 	 * werden in der Oberfläche dargestellt. Falls bereits ein Prozess existiert,
273 	 * wird dieser abgebrochen {@link #endProcess}  und ein neuer Prozess erzeugt.
274 	 * Der neue Prozess benutzt die übergebenen Parameter.
275 	 *
276 	 * @param arguments        Übergabeparameter, die benutzt werden sollen oder
277 	 *                         null
278 	 * @param environment      Umgebungsvariablen oder null
279 	 * @param workingDirectory Arbeitsverzeichnis oder null
280 	 *
281 	 * @throws IOException siehe {@link Process} Fehlerbeschreibung
282 	 */
283 	public final void processScript(String[] arguments, String[] environment, File workingDirectory) throws IOException {
284 
285 		// Die Klasse/Argumente/Umgebungsvariablen/Arbeitsverzeichnis werden gespeichert, falls der Prozess mit dem Standardpanel
286 		// neu gestartet werden soll. Wird ein Panel des Benutzers benutzt, werden diese Variablen nicht benötigt, da
287 		// das Panel des Benutzers dafür sorgen muss, dass die richtigen Argumente übergeben werden.
288 		_processArguments = arguments;
289 		_processEnvironment = environment;
290 		_processWorkingDirectory = workingDirectory;
291 
292 		String fileSeparator = System.getProperty("file.separator");
293 		String javaHome = System.getProperty("java.home");
294 		String classPath = System.getProperty("java.class.path");
295 
296 		List commandList = new LinkedList();
297 		commandList.add(javaHome + fileSeparator + "bin" + fileSeparator + "java");
298 		commandList.add("-Dfile.encoding=ISO-8859-1");
299 		commandList.add("-Xms32m");
300 		commandList.add("-Xmx712m");
301 		commandList.add("-cp");
302 		commandList.add(classPath);
303 		commandList.add(_className);
304 		if (arguments != null) commandList.addAll(Arrays.asList(arguments));
305 
306 		final String[] commandArray = (String[]) commandList.toArray(new String[0]);
307 
308 		//		StringBuffer attributeString = new StringBuffer();
309 		//
310 		//		attributeString.append("Der Prozess wird gestartet:" + "\n");
311 		//		attributeString.append("Klasse: " + _className + "\n");
312 		//
313 		//		if (arguments != null) {
314 		//			attributeString.append("Liste der übergebenen Argumente: " + "\n");
315 		//			for (int nr = 0; nr < commandArray.length; nr++) {
316 		//				attributeString.append(commandArray[nr] + "\n");
317 		//			}
318 		//		} else {
319 		//			attributeString.append("Liste der übergebenen Argumente ist leer" + "\n");
320 		//		}
321 		//
322 		//		if (environment != null) {
323 		//			attributeString.append("Liste der Umgebungsargumente: " + "\n");
324 		//			for (int nr = 0; nr < environment.length; nr++) {
325 		//				attributeString.append(environment[nr] + "\n");
326 		//			}
327 		//
328 		//		} else {
329 		//			attributeString.append("Liste der Umgebungsargumente ist leer" + "\n");
330 		//		}
331 		//
332 		//		if (workingDirectory != null) {
333 		//			attributeString.append("Arbeitsverzeichnis: " + workingDirectory.getAbsolutePath());
334 		//		} else {
335 		//			attributeString.append("Ein Arbeitsverzeichnis wurde nicht festgelegt" + "\n");
336 		//		}
337 		//
338 		//		// Zeilenumbruch
339 		//		attributeString.append("\n");
340 		//
341 		//		_debugTextArea.append(attributeString.toString());
342 
343 		if (_process != null) {
344 			// Derzeit läuft bereits ein Prozess, dieser wird abgebrochen
345 			endProcess();
346 			_errorThread.interrupt();
347 			_debugThread.interrupt();
348 
349 			_process = Runtime.getRuntime().exec(commandArray, environment, workingDirectory);
350 			_errorThread = new Thread(new StreamReaderThread(_errorTextArea, _process.getErrorStream()));
351 			_debugThread = new Thread(new StreamReaderThread(_debugTextArea, _process.getInputStream()));
352 
353 			// Die Threads für die Ausgabe werden gestartet (dies ist jetzt nötig, da die Oberfläche bereits

355 			_errorThread.start();
356 			_debugThread.start();
357 		}
358 		else {
359 			_process = Runtime.getRuntime().exec(commandArray, environment, workingDirectory);

361 			_debugThread = new Thread(new StreamReaderThread(_debugTextArea, _process.getInputStream()));
362 			// Die Threads für die Ausgabe werden mit der Erstellung der Oberfläche gestartet
363 			_debugThread.start();
364 			_errorThread.start();
365 		}
366 
367 		// Falls das Arbeitsverzeichnis zum speichern der Protokolle noch nicht gesetzt wurde,

374 	/**
375 	 * Diese Methode wertet die übergebenen Argumente aus und zeigt falls gewünscht
376 	 * die Oberfläche an. Der zu startende Prozess kann entweder sofort gestartet
377 	 * werden oder aber erst später. Falls der Prozess nicht sofort gestartet wird,
378 	 * weil zum Beispiel auf eine Eingabe des Benutzer gewartet wird, muss der
379 	 * Prozess durch den Aufruf {@link #processScript} gestartet werden.
380 	 *
381 	 * @param args         Argumente, mit denen der Prozess gestartet werden soll.
382 	 *                     Die Argumente enthalten ausserdem die Information, ob
383 	 *                     die Oberfläche angezeigt werden soll oder nicht
384 	 * @param startProcess true = Der in den Argumenten festgelegte Prozess wird
385 	 *                     sofort gestartet; false = Der in den Argumenten
386 	 *                     festgelegte Prozess wird nicht gestartet und muss mit
387 	 *                     dem Aufruf {@link #processScript} gestartet werden
388 	 */
389 	public final void start(String[] args, boolean startProcess) {
390 		ArgumentList argumentList = new ArgumentList(args);
391 
392 		// Ist der Parameter vorhanden
393 		if (argumentList.hasArgument("-gui") == false) {
394 			final String errorString = "Der Parameter -gui= wurde nicht angegeben";
395 			JOptionPane.showMessageDialog(null, errorString, "Parameter -gui= fehlt", JOptionPane.ERROR_MESSAGE);
396 
397 			// Es macht keinen Sinn weiterzumachen
398 			return;
399 		}
400 
401 		// Wurde eine Klasse angegeben ? Falls ja, wird diese Ausgelesen
402 		if (argumentList.hasArgument("-prozessname") == false) {
403 			// Der Prozessname wurde nicht angegeben
404 			final String errorString = "Der Parameter -prozessname= wurde nicht angegeben";
405 			JOptionPane.showMessageDialog(null, errorString, "Parameter -prozessname= fehlt", JOptionPane.ERROR_MESSAGE);
406 
407 			// Es macht keinen Sinn weiterzumachen
408 			return;
409 		}
410 		_className = argumentList.fetchArgument("-prozessname=").asString();
411 
412 		// WorkingDirecotry auslesen
413 		File workingDirectory = null;
414 
415 		if (argumentList.hasArgument("-arbeitsverzeichnis")) {
416 			// Es wurde ein Arbeitsverzeichnis angegeben
417 			workingDirectory = new File(argumentList.fetchArgument("-arbeitsverzeichnis").asString());
418 		}
419 
420 		// Soll die Oberfläche angezeigt werden (dies wird so gemacht da ein if (argumentList.fetchArgument("-gui=").booleanValue()))
421 		// zwar funktioniert hat, aber das Argument nicht aus der Liste entfernt wurde !
422 		final boolean showGUI = argumentList.fetchArgument("-gui=").booleanValue();
423 
424 		// Es wurden alle Daten ausgelesen, die benötigt werden um einen Prozess zu starten.
425 		// Da beim rausfiltern der Argumente die ursprüngeliche Argumenteliste verändert wurde (es gibt
426 		// nun Einträge die <code>null</code> sind, wird diese Liste bereinigt.
427 		_modifiedArgumentlist = cleanedArgumentArray(args);
428 
429 		if (showGUI) {
430 			// Die Oberfläche soll angezeigt werden
431 
432 			if (startProcess) {
433 				// Den Prozess starten
434 				try {
435 					processScript(_modifiedArgumentlist, null, workingDirectory);
436 				}
437 				catch (IOException e) {
438 					e.printStackTrace();
439 				}
440 				showGUI();
441 			}
442 			else {
443 				// Es soll nur die GUI angezeigt werden, der Prozess wird nicht gestartet
444 				showGUI();
445 			}
446 
447 		}
448 		else {
449 			// Die Oberfläche soll nicht angezeigt werden
450 
451 			if (startProcess) {
452 				// Der Prozess soll sofort gestartet werden
453 				try {
454 					processScript(_modifiedArgumentlist, null, workingDirectory);
455 				}
456 				catch (IOException e) {
457 					e.printStackTrace();
458 				}
459 			}
460 			else {
461 				final String infoMessage = "Es soll keine Oberfläche angezeigt werden und der in den Parametern " +
462 				        "festgelegte Prozess wird nicht gestartet. Es besteht keine Möglichkeit den Prozess nachträglich " +
463 				        "zu starten ! Bitte starten Sie den Prozess direkt, falls sie keine GUI sehen möchten.";
464 				JOptionPane.showMessageDialog(null, infoMessage, "Prozess wird auf Wunsch des Benutzers nicht gestartet", JOptionPane.INFORMATION_MESSAGE);
465 			}
466 		}
467 	}

500 		userPanelLogPanelSplit.setDividerLocation(_userPanel.getPreferredSize().height);

508 
509 		_mainFrame.getContentPane().add(legendPanelAndRest);
510

512 		_mainFrame.setSize(new Dimension(800, 600));
513 		_mainFrame.setVisible(true);
514 	}

544 		final JPanel standardPanel = new JPanel(new BorderLayout());
545 
546 		final Image k2sImage = new ImageIcon(SimpleApplicationGUI.class.getResource("K2S-LogoGross.png")).getImage();
547 		// Das neue Seitenverhältnis des Logos berechnen

845 		private void append(final String string) {
846 			invokeAndWait(new Runnable() {
847 				public void run() {
848 					int textLength = _textPane.getDocument().getLength();
849 					int selectionStart = _textPane.getSelectionStart();
850 					int selectionEnd = _textPane.getSelectionEnd();
851 					if (textLength > MAX_TEXT_LENGTH) {
852 						// Zuviel Text => obere Hälfte löschen
853 						final int deleteLength = textLength / 2;
854 						_textPane.select(0, deleteLength);
855 						_textPane.replaceSelection("(Anfang gelöscht)...");
856 						final int newLength = _textPane.getDocument().getLength();
857 						final int deletedLength = textLength - newLength;
858 						selectionStart -= deletedLength;
859 						selectionEnd -= deletedLength;
860 						textLength = newLength;
861 						// wenn Selektion im gelöschten Bereich lag, dann Selektion auf Ende setzen
862 						if (selectionStart < 0 || selectionEnd < 0) {
863 							selectionStart = textLength;
864 							selectionEnd = textLength;
865 						}
866 					}
867 					//	                System.out.println("textLength = " + textLength);
868 					//	                System.out.println("selectionStart = " + selectionStart);
869 					//	                System.out.println("selectionEnd = " + selectionEnd);
870 					_textPane.select(textLength, textLength);
871 					_textPane.replaceSelection(string);
872 					if (textLength != selectionStart || textLength != selectionEnd) {
873 						_textPane.select(selectionStart, selectionEnd);
874 					}
875 				}
