21  package de.bsvrz.sys.funclib.debug.gui;
22  
23  import javax.swing.*;
24  import javax.swing.tree.DefaultMutableTreeNode;
25  import javax.swing.tree.DefaultTreeCellRenderer;
26  import java.awt.*;
27  import java.net.URL;
28  import java.util.logging.Level;
29  
30  public class LoggerTreeCellRenderer extends DefaultTreeCellRenderer {
31  
32      Icon loggerNoLevel;
33      Icon loggerWithLevel;
34      Icon noLoggerIcon;
35  
36      public LoggerTreeCellRenderer() {
37          super();
38          //ClassLoader loader = this.getClass().getClassLoader();
39          URL url = getClass().getResource("notset.gif");
40          loggerNoLevel = new ImageIcon(url);
41          url = getClass().getResource("set.gif");
42          loggerWithLevel = new ImageIcon(url);
43          noLoggerIcon = super.getDefaultOpenIcon();
44      }
45      public Component getTreeCellRendererComponent(JTree tree, Object value, boolean sel, boolean expanded, boolean leaf, int row, boolean hasFocus) {
46          DefaultMutableTreeNode node = (DefaultMutableTreeNode)value;
47          TreeNodeObject nodeObject = (TreeNodeObject)node.getUserObject();
48          if (nodeObject.isLogger()) {
49              Level level = nodeObject.getLogger().getLevel();
50              if (level == null) {
51                  super.setClosedIcon(loggerNoLevel);
52                  super.setOpenIcon(loggerNoLevel);
53                  super.setLeafIcon(loggerNoLevel);
54              }
55              else {
56                  super.setClosedIcon(loggerWithLevel);
57                  super.setOpenIcon(loggerWithLevel);
58                  super.setLeafIcon(loggerWithLevel);
59              }
60          }
61          else {
62              super.setClosedIcon(noLoggerIcon);
63              super.setOpenIcon(noLoggerIcon);
64          }
65          return super.getTreeCellRendererComponent(tree, value, sel, expanded,  leaf, row, hasFocus);
66      }
67  }
