33  /**
34   * Gibt die DebugMeldungen als Text aus. Die Ausgabe erfolgt in der folgenden Form (Bespiel für Ausgaben in Verschiedenen Leveln und mit mehrzeiligen
35   * Meldungstexten): <PRE><CODE> ------- 18.09.2003 16:58:22,960:+0200(TID:......) ---------------------- STATUS Ausgabedatei angelegt. #000000 18.09.2003
36   * 16:58:22,960:+0200(TID:000010) ---------------------- INFO LOGGER :ApplikationName MELDUNG: Aktuelle Debugeinstellungen
37   * ---------------------------------------------- Registrierte DebugLogger: global ApplikationName
38   * <p/>
39   * Basiseinstellung des Wurzel-DebugLoggers ---------------------------------------------- Name :ApplikationName DebugLevel      :ALL Filter          :null
40   * ParentName      :java.util.logging.LogManager$RootLogger@e7b241 RCS-Bundle      :null RCS-Bundle-Name :null #000001 18.09.2003 16:58:22,970:+0200(TID:000010)
41   * ###################### FEHLER LOGGER :ApplikationName.hck.debug.TestLogger MELDUNG: hck.debug.TestLogger :Testausgabe in SEVERE #000002 18.09.2003
42   * 16:58:22,970:+0200(TID:000010) ====================== WARNUNG LOGGER :ApplikationName.hck.debug.TestLogger MELDUNG: hck.debug.TestLogger :Testausgabe in
43   * WARNING #000003 18.09.2003 16:58:22,970:+0200(TID:000010) ---------------------- INFO LOGGER :ApplikationName.hck.debug.TestLogger MELDUNG:
44   * hck.debug.TestLogger :Testausgabe in INFO #000004 18.09.2003 16:58:23,070:+0200(TID:000010) ...................... KONFIG LOGGER
45   * :ApplikationName.hck.debug.TestLogger MELDUNG: hck.debug.TestLogger :Testausgabe in CONFIG #000005 18.09.2003 16:58:23,070:+0200(TID:000010) .  .  .  .  .  .
46   *  .  . FEIN LOGGER :ApplikationName.hck.debug.TestLogger MELDUNG: hck.debug.TestLogger :Testausgabe in FINE #000006 18.09.2003 16:58:23,070:+0200(TID:000010)
47   * .  .  .  .  .  .  .  . FEINER LOGGER :ApplikationName.hck.debug.TestLogger MELDUNG: hck.debug.TestLogger :Testausgabe in FINER #000007 18.09.2003
48   * 16:58:23,070:+0200(TID:000010) .  .  .  .  .  .  .  . DETAIL LOGGER :ApplikationName.hck.debug.TestLogger MELDUNG: hck.debug.TestLogger :Testausgabe in
49   * FINEST </CODE></PRE>
50   *
51   * @author Hans Christian Kniß (HCK)
52   * @version $Revision: 6566 $ / $Date: 2009-04-20 17:21:33 +0200 (Mon, 20 Apr 2009) $
53   */
54  public class DebugFormatterStdErrText extends Formatter {
55  
56  	/** Formatstring für das Ausgabeformat des Zeitstempels. Ausgabe erfolgt mit Datum, Uhrzeit, Millisekunden und Zeitoffset zur Zeitangabe in UMT. */
57  	private static final DateFormat _absoluteMillisecondsFormat = new SimpleDateFormat("dd.MM.yyyy HH:mm:ss,SSS:Z");
58  
59  	/** Formatstring mit Angabe für die Formatierung der Zahlenausgaben für die lfd. Nr. der Meldung und die ThreadId. */
60  	private static final DecimalFormat _numberFormat = new DecimalFormat("000000");
61  
62  	/** Systemabhängig zur Laufzeit ermittelter String für den Zeilenumbruch. Daduch wird idie Meldungsausgabe plattformunabhängig. */
63  	private static final String NEWLINE = System.getProperty("line.separator");
64  
65  	/** Linientyp wird bei der Ausgabe vom Level FEHLER verwendet */
66  	private static final String HASH_LINE = " ###################### ";
67  
68  	/** Linientyp wird bei der Ausgabe vom Level WARNUNG verwendet */
69  	private static final String DOUBLE_LINE = " ====================== ";
70  
71  	/** Linientyp wird bei der Ausgabe vom Level INFO verwendet */
72  	private static final String SINGLE_LINE = " ---------------------- ";
73  
74  	/** Linientyp wird bei der Ausgabe vom Level KONFIGURATION verwendet */
75  	private static final String DOT_LINE = " ...................... ";
76  
77  	/** Linientyp wird bei der Ausgabe vom Level FEIN, FEINER und DETAIL verwendet */
78  	private static final String SIMPLE_LINE = " .  .  .  .  .  .  .  . ";
79  
80  	private final boolean _debugStdErrWithStackTraces;
81  
82  	public DebugFormatterStdErrText() {
83  		_debugStdErrWithStackTraces = true;
84  	}
85  
86  	public DebugFormatterStdErrText(final boolean debugStdErrWithStackTraces) {
87  		_debugStdErrWithStackTraces = debugStdErrWithStackTraces;
88  	}
89  
90  	/**
91  	 * Gibt dei DebungMeldung aus
92  	 *
93  	 * @param lr LogRecord mit den Informationen einer Meldung.
94  	 *
95  	 * @return Gibt den als Text formatierten Meldungstext mit den im LogRecord übergebenen Informationen aus.
96  	 */
97  	public String format(LogRecord lr) {
98  		Date date = new Date(lr.getMillis());
99  
100 		StringBuffer sb = new StringBuffer().append(NEWLINE);
101 		sb.append("#").append(_numberFormat.format(lr.getSequenceNumber()));			// Kopfzeile: LfdNr...
102 		sb.append(" ");
103 		sb.append(_absoluteMillisecondsFormat.format(date));							// ... Datum ...
104 		sb.append(" (TID:").append(_numberFormat.format(lr.getThreadID())).append(")");  // ...Thread ID ...
105 
106 		if(lr.getLevel() == Debug.ERROR) {
107 			sb.append(HASH_LINE);														// ... Linie abhängig ...
108 		}																				// ... vom Level ...
109 		else if(lr.getLevel() == Debug.WARNING) {
110 			sb.append(DOUBLE_LINE);
111 		}
112 		else if(lr.getLevel() == Debug.INFO) {
113 			sb.append(SINGLE_LINE);
114 		}
115 		else if(lr.getLevel() == Debug.CONFIG) {
116 			sb.append(DOT_LINE);
117 		}
118 		else {
119 			sb.append(SIMPLE_LINE);
120 		}
121 		sb.append(NEWLINE);
122 
123 		Level l = lr.getLevel();														// ... und Level im Klartext.
124 		if(l == Debug.ERROR) {
125 			sb.append("FEHLER : ");
126 		}
127 		else if(l == Debug.WARNING) {
128 			sb.append("WARNUNG: ");
129 		}
130 		else if(l == Debug.INFO) {
131 			sb.append("INFO   : ");
132 		}
133 		else if(l == Debug.CONFIG) {
134 			sb.append("KONFIG : ");
135 		}
136 		else if(l == Debug.FINE) {
137 			sb.append("FEIN   : ");
138 		}
139 		else if(l == Debug.FINER) {
140 			sb.append("FEINER : ");
141 		}
142 		else if(l == Debug.FINEST) sb.append("DETAIL : ");
143 		sb.append(lr.getLoggerName()).append(NEWLINE);				                    // 2. Zeile: Wurzel+Klassenname
144 
145 		
146 
147 		sb.append(lr.getMessage());
148 
149 		Object[] objects = lr.getParameters();
150 
151 		if((objects != null) && (objects.length == 1)) {
152 			final Object object = objects[0];
153 			sb.append(": ").append(object == null ? "null" : object.toString());
154 		}
155 
156 		if((objects != null) && (objects.length > 1)) {
157 			sb.append(':').append(NEWLINE);
158 			for(int i = 0; i < objects.length; i++) {
159 				final Object object = objects[i];
160 				sb.append("[").append(i).append("] ").append(object == null ? "null" : object.toString()).append(NEWLINE);
161 			}
162 		}
163 
164 		if(lr.getThrown() != null) {
165 			sb.append(":").append(NEWLINE);
166 			if(_debugStdErrWithStackTraces) {
167 				try {
168 					StringWriter sw = new StringWriter();
169 					PrintWriter pw = new PrintWriter(sw);
170 					lr.getThrown().printStackTrace(pw);
171 					pw.close();
172 					sb.append("      ").append(sw.toString());
173 				}
174 				catch(Exception ex) {
175 				}
176 			}
177 			else {
178 				sb.append(lr.getThrown());
179 			}
180 		}
181 
182 		sb.append(NEWLINE);
183 		return sb.toString();
184 	}
