37  /**
38   * Stellt Methoden für die Debugausgabe zur Vefügung.<br> Unterstützt werden neben unterschiedlichen DebugLeveln auch hierarisch verwaltete DebugBereiche, die
39   * je Klasse angelegt werden können. Für die Ausgabe stehen verschiedene Ausgabekänale mit jeweils spezifischen Formatierungen der Ausgabe zur Verfügung ({@link
40   * DebugFormatterXML}, {@link DebugFormatterFileText}, {@link DebugFormatterExcel}, {@link DebugFormatterHTML}).
41   * <p/>
42   * Die Ausgabe (über den DebugLevel) kann zur Laufzeit sowohl für die verfügbaren Ausgabekanäle als auch die DebugBereiche geändert werden.
43   * <p/>
44   * Für jeden DebugBereich (jeder Klasse, in denen DebugAusgben vorgenommen werden) kann der Level ({@link #ERROR}, {@link #WARNING}, {@link #INFO}, {@link
45   * #CONFIG}, {@link #FINE}, {@link #FINER}, {@link #FINEST}, {@link #ALL}, {@link #OFF}) unabhängig vom Level des jeweiligen Ausgabekanals ({@link
46   * DebugFormatterXML}, {@link DebugFormatterStdErrText}, {@link DebugFormatterFileText}, {@link DebugFormatterExcel}, {@link DebugFormatterHTML}) eingestellt
47   * werden. Damit nicht bei einer Applikation mit hunderten von Klassen (und damit entsprechend vielen DebugBereichen) nicht jeden DebugBereich einzeln
48   * einstellen zu müssen, werden die DebugBereiche entsprechend ihrer Hierachie verwaltet. Wird der Level eines DebugBereichs geändert, so werden auch alle Level
49   * für die untergeordneten Bereiche mit umgestellt. Setzt man also den Level des WurzelLoggers auf FEHLER, so werden alle Logger auf diesen Level gesetzt und
50   * nur noch Fehlermeldungen protokolliert. Anschließend kann dann der Logger einer speziel zu untersuchenden Klasse auf z.B. den Level FEINER gesetzt werden, um
51   * so genauere Informationen über einen speziellen Programmteil zu protokollieren.
52   *
53   * @author Kappich+Kniß Systemberatung Aachen (K2S)
54   * @author Hans Christian Kniß (HCK)
55   * @version $Revision: 6566 $ / $Date: 2009-04-20 17:21:33 +0200 (Mon, 20 Apr 2009) $ / ($Author: rs $)
56   */

202 	/**
203 	 * Initialisiert den DebugLogger.
204 	 * <p/>
205 	 * Folgende Initialisierungen werden durchgeführt: <<ul> <li>Alle registrierten DebugLogger werden entfernt</li> <li>Es wird der Wurzel-DebugLogger
206 	 * initialisiert, der den Namen des Aufrufparameters erhält.</li> <li>Es wird der Dateipfad für alle dateibasierten Debugausgaben festgelegt</li> <li>Es werden
207 	 * folgende Ausgabekanäle und Ausgabeformate angemeldet und die Ausgabelevel auf Default eingestellt:</li> <ul> <li>Datei im XML-Format (*.xml),
208 	 * DebugLevel:INFO</li> <li>Datei mit Ausgaben als formatierter Text (*.txt), DebugLevel:INFO</li> <li>Datei im EXCEL-Format (*.csv), DebugLevel:OFF</li>
209 	 * <li>Datei im HTML-Format(*.htm), DebugLevel:OFF</li> <li>Ausgabe auf stderr als Text, DebugLevel:INFO</li> </ul> Die Ausgabekanäle/Ausgabeformate stehen
210 	 * anschliessend allen DebugLoggern automatisch zur Verfügung. <li>Auswertung der Aufrufparameter der Applikation</li> <li>Anmeldung der Telegramme zur
211 	 * Steuerung des DebugLoggers über den Datenverteiler</li> <li>Alle registrierten DebugLogger werden entfernt</li> </ul>
212 	 *
213 	 * @param argumentList Liste der bei Applikationsstart übergebenen Parameter zur Voreinstellung.
214 	 * @param rootName     Wurzel-Name aller verwalteter DebugLogger
215 	 *
216 	 * @since V 1.0
217 	 */
218 	public static void init(String rootName, ArgumentList argumentList) {
219 		// Wurzel-Logger anmelden
220 		_rootName = rootName;
221 		_rootLogger = new Debug(Logger.getLogger(_rootName));
222 		_rootLogger._logger.setUseParentHandlers(false);
223 		_rootLogger._logger.setLevel(ALL);
224 
225 		int debugStdErrWithStackTraces = -1;
226 		try {
227 			// Parameter -debugHelp bzw. -? auswerten
228 
229 			// Parameter auswerten
230 			_debugFileSize = argumentList.fetchArgument("-debugFileSize=10000000").intValue();
231 			_debugFileCount = argumentList.fetchArgument("-debugFileCount=5").intValue();
232 			_debugLevelStdErrText = string2Level(argumentList.fetchArgument("-debugLevelStdErrText=WARNING").asString());
233 			_debugLevelFileText = string2Level(argumentList.fetchArgument("-debugLevelFileText=OFF").asString());
234 			_debugLevelFileXML = string2Level(argumentList.fetchArgument("-debugLevelFileXML=OFF").asString());
235 			_debugLevelFileExcel = string2Level(argumentList.fetchArgument("-debugLevelFileExcel=OFF").asString());
236 			_debugLevelFileHTML = string2Level(argumentList.fetchArgument("-debugLevelFileHTML=OFF").asString());
237 			_debugPath = argumentList.fetchArgument("-debugFilePath=.").asDirectory();
238 			final ArgumentList.ValueSelection stackTracesSelection = new ArgumentList.ValueSelection();
239 			stackTracesSelection.add("auto").ignoreCase().convertTo(-1);
240 			stackTracesSelection.add("ja").alias("wahr").alias("an").alias("true").alias("yes").alias("on").ignoreCase().convertTo(1);
241 			stackTracesSelection.add("nein").alias("falsch").alias("aus").alias("false").alias("no").alias("off").ignoreCase().convertTo(1);
242 			debugStdErrWithStackTraces = argumentList.fetchArgument("-debugStdErrTextStacktraces=auto").asValueCase(stackTracesSelection).intValue();
243 			_debugFilePath = _debugPath.getAbsolutePath();
244 
245 			if(argumentList.fetchArgument("-debugGui=nein").booleanValue()) {
246 				// Das GUI zur Verwaltung der Logger anzeigen
247 				new LoggingFrame().setVisible(true);
248 			}
249 			try {
250 				while(argumentList.hasArgument("-debugSetLoggerAndLevel")) {
251 					String arg = argumentList.fetchArgument("-debugSetLoggerAndLevel").asString();
252 					String[] args = arg.split(":");
253 					String loggerName;
254 					if(args[0].trim().equals("")) {
255 						loggerName = _rootName;
256 					}
257 					else {
258 						loggerName = _rootName + "." + args[0];
259 					}
260 					String level = args[1];
261 
262 					new Debug(Logger.getLogger(loggerName));
263 
264 					Logger.getLogger(loggerName).setLevel(string2Level(level));
265 				}
266 			}
267 			catch(Exception e) {
268 				System.err.println("Fehler bei der Auswertung von -debugSetLoggerAndLevel");
269 			}
270 		}
271 		catch(IllegalArgumentException e) {
272 			System.err.println(e);
273 		}
274 		catch(Exception e) {
275 			e.printStackTrace();
276 			System.err.println("Debug Initialisierung ohne Parameter. Es wird der Standard verwendet. " + e);
277 		}
278 
279 		// Handler und Formatter erzeugen und registrieren
280 		try {
281 			File directory = new File(_debugPath, "debug");
282 			String fileName = directory.getAbsolutePath() + "/" + _rootName + _debugFileNamePattern;
283 			//Handler und Formatter erzeugen, aber nur, wenn mindestens ein FileFormater ungleich OFF
284 			if(!((_debugLevelFileXML == OFF) && (_debugLevelFileText == OFF) && (_debugLevelFileHTML == OFF) && (_debugLevelFileExcel == OFF))) {
285 				// System.err.println("Debug-Ordner anlegen: " + directory.getAbsolutePath());
286 				// System.err.println("filename: " + fileName);
287 				directory.mkdir();
288 			}
289 
290 			// Jeweils Handler und Formatter holen und verknüpfen, Handlerlevel setzen und beim
291 			// Rootlogger registrieren.
292 			// Registrierung erfolgt nur, wenn Level ungleich OFF ist (Ausnahme: Ausgabe auf StdErr erfolgt immer!
293 
294 			if(debugStdErrWithStackTraces == -1) {
295 				if(_debugLevelFileText.equals(OFF)) {
296 					debugStdErrWithStackTraces = 1;
297 				}
298 				else {
299 					debugStdErrWithStackTraces = 0;
300 				}
301 			}
302 			_formatterStdErrText = new DebugFormatterStdErrText(debugStdErrWithStackTraces == 1);  // Formatierer für Textausgabe auf StdErr holen
303 			reinstallConsoleHandler();
304 
305 			if(!_debugLevelFileXML.equals(OFF)) {
306 				_handlerFileXML = new FileHandler(fileName + _fileExtensionXML, _debugFileSize, _debugFileCount, _debugFileAppend);
307 				_formatterXML = new DebugFormatterXML();		        // Formatierer für XML-Formatierung holen
308 				_handlerFileXML.setFormatter(_formatterXML);
309 				_handlerFileXML.setLevel(_debugLevelFileXML);
310 				_rootLogger._logger.addHandler(_handlerFileXML);
311 			}
312 			if(!_debugLevelFileText.equals(OFF)) {
313 				_handlerFileText = new FileHandler(fileName + _fileExtensionText, _debugFileSize, _debugFileCount, _debugFileAppend);
314 				_formatterFileText = new DebugFormatterFileText();      // Formatierer für Textausgabe auf Datei holen
315 				_handlerFileText.setFormatter(_formatterFileText);
316 				_handlerFileText.setLevel(_debugLevelFileText);
317 				_rootLogger._logger.addHandler(_handlerFileText);
318 			}
319 			if(!_debugLevelFileHTML.equals(OFF)) {
320 				_handlerFileHTML = new FileHandler(fileName + _fileExtensionHTML, _debugFileSize, _debugFileCount, _debugFileAppend);
321 				_formatterHTML = new DebugFormatterHTML();		        // Formatierer für HTML-Formatierung holen
322 				_handlerFileHTML.setFormatter(_formatterHTML);
323 				_handlerFileHTML.setLevel(_debugLevelFileHTML);
324 				_rootLogger._logger.addHandler(_handlerFileHTML);
325 			}
326 			if(!_debugLevelFileExcel.equals(OFF)) {
327 				_handlerFileExcel = new FileHandler(fileName + _fileExtensionExcel, _debugFileSize, _debugFileCount, _debugFileAppend);
328 				_formatterExcel = new DebugFormatterExcel();	        // Formatierer für Excel-Formatierung holen
329 				_handlerFileExcel.setFormatter(_formatterExcel);
330 				_handlerFileExcel.setLevel(_debugLevelFileExcel);
331 				_rootLogger._logger.addHandler(_handlerFileExcel);
332 			}
333 		}
334 		catch(SecurityException e) {
335 			System.err.println("Fehler bei Konfiguration der Handler des DebugLoggers:");
336 		}
337 		catch(IOException e) {
338 			System.err.println("Fehler bei Konfiguration der Handler des DebugLoggers:");
339 		}
340 
341 		if(!rootName.equals("DEFAULT-DEBUG")) {
342 			// Aktuellen DebugLogger-Zustand auf Debug-Kanäle ausgeben
343 			_rootLogger.status(debugInfo());
344 
345 			// Argumentliste ausgeben
346 			_rootLogger.info("Aufrufargumente von " + rootName, argumentList.toString());
347 		}
348 	}

350 	public static void reinstallConsoleHandler() {
351 		final ConsoleHandler consoleHandler = new ConsoleHandler();
352 		consoleHandler.setFormatter(_formatterStdErrText);
353 		consoleHandler.setLevel(_debugLevelStdErrText);
354 		synchronized(_rootLogger) {
355 			if(_handlerStderrText != null) {
356 				_rootLogger._logger.removeHandler(_handlerStderrText);
357 			}
358 			_rootLogger._logger.addHandler(consoleHandler);
359 			_handlerStderrText = consoleHandler;
360 		}
361 	}

363 	/**
364 	 * Konvertiert den Debuglevel aus den Aufrufparametern von String in Level.
365 	 *
366 	 * @param s Zu konvertierender String.
367 	 *
368 	 * @return Konvertierter Level.
369 	 */
370 	private static Level string2Level(String s) {
371 		if(s.trim().toUpperCase().equals("ERROR")) return ERROR;
372 		if(s.trim().toUpperCase().equals("WARNING")) return WARNING;
373 		if(s.trim().toUpperCase().equals("CONFIG")) return CONFIG;
374 		if(s.trim().toUpperCase().equals("INFO")) return INFO;
375 		if(s.trim().toUpperCase().equals("FINE")) return FINE;
376 		if(s.trim().toUpperCase().equals("FINER")) return FINER;
377 		if(s.trim().toUpperCase().equals("FINEST")) return FINEST;
378 		if(s.trim().toUpperCase().equals("ALL")) return ALL;
379 		//Wenn kein gültiger Wert übergeben wird, Kanal ausschalten.
380 
381 		return OFF;
382 	}

386 	/**
387 	 * Debugausgabe interner Art (auf Level INFO) für Ausgabe der aktuellen Einstellungen oder bei Änderung der Einstellungen. Bei Ausgabe über
388 	 * <source>status</source> wird der DebugLevel nur für diese Ausgabe auf INFO gesetzt
389 	 *
390 	 * @param msg Auszugebender Debugtext. Der Text kann auch mehrzeilig formatiert sein. Neben dem Text werden noch folgende Ausgaben automatisch mit ausgegeben,
391 	 *            so dass diese nicht im Text enthalten sein sollten. (Datum, Zeit, Fehlerlevel (FEHLER, WARNUNG,...), Klasse des Aufrufers, lfdNr der
392 	 *            Meldung,...)
393 	 */
394 	private void status(String msg) {
395 		Level _aktLevel = _logger.getLevel();		// aktuellen Level merken...
396 		_logger.setLevel(INFO);						// ...für die nächste Ausgabe Level auf INFO setzen...
397 		_logger.log(INFO, msg);						// ...Meldung als Info ausgeben...
398 		_logger.setLevel(_aktLevel);					// ...und Level wieder zurücksetzen.
399 	}

430 	/**
431 	 * Gibt eine DebugLogger zurück, der automatisch in der Klassenhierachie unter dem WurzelLogger einsortiert wurde. Existiert der DebugLogger noch nicht, wird
432 	 * er erzeugt, ansonsten wird der bereits existierende DebugLogger mit dem angeforderten Namen zurückgegeben. Als Name wird automatisch gebildet aus
433 	 * "rootName.voll qualifizierter ClassName" z.B. "root.sys.funclib.Debug". Für jeden Logger kann der DebugLevel per DaV-Telegramm geändert werden.
434 	 *
435 	 * @return Logger Objekt, welches für die Debugausgeben und zur Steuerung des Debuglevels verwendet wird.
436 	 */
437 	public static Debug getLogger() {
438 		String debugClassName = "Klassenname nicht ermittelbar!";
439 		Throwable t = new Throwable();
440 		if(t.getStackTrace().length > 1) {
441 			debugClassName = t.getStackTrace()[1].getClassName();
442 		}
443 		String _loggerName = _rootName + "." + debugClassName;
444 		return new Debug(Logger.getLogger(_loggerName));
445 	}

447 	/**
448 	 * TESTMETHODE: Setzt DebugLevel des entsprechenden Loggers. Setzt den Level des angebenen Loggers UND aller untergeordneten Logger auf den eingestellten
449 	 * Level.
450 	 * <p/>
451 	 * Es wird dazu der Name des Loggers verwendet. Als untergeordnet gelten aller Logger, deren Namensanfang mit dem des angegebenen Loggers übereinstimmt.
452 	 *
453 	 * @param logger Name des mit {@link #getLogger} erzeugter DebugLogger, dessen Level geändert werden soll
454 	 * @param level  Neuer DebugLevel ({@link #ERROR}, {@link #WARNING}, {@link #INFO}, {@link #CONFIG}, {@link #FINE}, {@link #FINER}, {@link #FINEST}, {@link
455 	 *               #ALL}, {@link #OFF}).
456 	 */
457 	public static void setLoggerLevel(String logger, Level level) {
458 		try {
459 			for(Enumeration e = LogManager.getLogManager().getLoggerNames(); e.hasMoreElements();) {
460 				String loggerName = e.nextElement().toString();
461 				if(loggerName.startsWith(logger)) {
462 					Logger.getLogger(loggerName).setLevel(level);
463 				}
464 			}
465 		}
466 		catch(SecurityException e) {
467 			_rootLogger.error("Fehler bei Konfiguration der Handler des DebugLoggers:" + e);
468 		}
469 	}

471 	/**
472 	 * TESTMETHODE: Setzt DebugLevel des entsprechenden Ausgabekanals. Damit werden alle nur noch Ausgaben auf diesem Kanal mit dem eingestellten Level oder
473 	 * darüber ausgegeben, unabhängig davon ob einzelne Logger einen feineren Level eingestellt haben. Die Leveleinstellungen der einzelnen Logger werden aber
474 	 * nicht verändert.
475 	 *
476 	 * @param s     Ausgabekanal, dessem Level geändert werden soll (StdErr, FileText, FileXML, FileHTML, FileExcel)
477 	 * @param level Neuer DebugLevel ({@link #ERROR}, {@link #WARNING}, {@link #INFO}, {@link #CONFIG}, {@link #FINE}, {@link #FINER}, {@link #FINEST}, {@link
478 	 *              #ALL}, {@link #OFF}).
479 	 */
480 
481 	public static void setHandlerLevel(String s, Level level) {
482 		if(s == "StdErr") {
483 			_handlerStderrText.setLevel(level);
484 		}
485 		else if(s == "FileText") {
486 			_handlerFileText.setLevel(level);
487 		}
488 		else if(s == "FileXML") {
489 			_handlerFileXML.setLevel(level);
490 		}
491 		else if(s == "FileHTML") {
492 			_handlerFileHTML.setLevel(level);
493 		}
494 		else if(s == "FileExcel") {
495 			_handlerFileExcel.setLevel(level);
496 		}
497 		else {
498 			_rootLogger.error(
499 					"Levelumschaltung des Ausgabekanals [" + s + "] auf Level " + level + " nicht möglich, da Level [" + s + "] nicht zulässig!"
500 			);
501 		}
502 	}
