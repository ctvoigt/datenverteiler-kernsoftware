285 	public final String parseToString() {
286 		String str = "Anwendungsdatentelegramm : \n";
287 		str += "Index Dieses Telegramms: " + telegramNumber + "\n";
288 		str += "Gesamt Anzahl der Telegramme: " + totalTelegramCount + "\n";
289 		if(baseSubscriptionInfo != null) {
290 			str += baseSubscriptionInfo.toString() + "\n";
291 		}
292 		str += "Sendedaten Nummer: " + dataNumber + "\n";
293 		str += "Nachgelieferte Daten: " + delayedDataFlag + "\n";
294 		if(telegramNumber == 0) {
295 			str += "Datensaetzezeit: " + new java.util.Date(dataTime) + "\n";
296 			str += "Fehler code: " + errorFlag + "\n";
297 			if(attributesIndicator != null) {
298 				str += "Attributsindikator: \n";
299 				for(int i = 0; i < attributesIndicator.length; ++i) {
300 					final byte b = attributesIndicator[i];
301 					final int j = i * 8;
302 					if((b & 0x01) == 0x01) {
303 						str += "Attributsbit an der Position " + (j) + " ist gesetzt.\n";
304 					}
305 					if((b & 0x02) == 0x02) {
306 						str += "Attributsbit an der Position " + (j + 1) + " ist gesetzt.\n";
307 					}
308 					if((b & 0x04) == 0x04) {
309 						str += "Attributsbit an der Position " + (j + 2) + " ist gesetzt.\n";
310 					}
311 					if((b & 0x08) == 0x08) {
312 						str += "Attributsbit an der Position " + (j + 3) + " ist gesetzt.\n";
313 					}
314 					if((b & 0x10) == 0x10) {
315 						str += "Attributsbit an der Position " + (j + 4) + " ist gesetzt.\n";
316 					}
317 					if((b & 0x20) == 0x20) {
318 						str += "Attributsbit an der Position " + (j + 5) + " ist gesetzt.\n";
319 					}
320 					if((b & 0x40) == 0x40) {
321 						str += "Attributsbit an der Position " + (j + 6) + " ist gesetzt.\n";
322 					}
323 					if((b & 0x80) == 0x80) {
324 						str += "Attributsbit an der Position " + (j + 7) + " ist gesetzt.\n";
325 					}
326 				}
327 			}
328 		}
329 		//str += "Daten: " + new String(data) + "\n";

333 	public final void write(DataOutputStream out) throws IOException {
334 		out.writeShort(length);
335 		out.writeShort(telegramNumber);
336 		out.writeShort(totalTelegramCount);
337 		if(baseSubscriptionInfo != null) {
338 			baseSubscriptionInfo.write(out);
339 		}
340 		out.writeLong(dataNumber);
341 		out.writeBoolean(delayedDataFlag);
342 
343 		if(telegramNumber == 0) {
344 			out.writeLong(dataTime);
345 			out.writeByte(errorFlag);
346 			if(attributesIndicator == null) {
347 				out.writeByte(0);
348 			}
349 			else {
350 				out.writeByte(attributesIndicator.length);
351 				for(int i = 0; i < attributesIndicator.length; ++i) {
352 					out.writeByte(attributesIndicator[i]);
353 				}
354 			}
355 		}
356 
357 		if(data == null) {
358 			out.writeInt(0);
359 		}
360 		else {
361 			out.writeInt(data.length);
362 			for(int i = 0; i < data.length; i++) {
363 				byte aData = data[i];
364 				out.writeByte(aData);
365 			}
366 		}
367 	}

369 	public final void read(DataInputStream in) throws IOException {
370 		int _length = in.readShort();
371 		telegramNumber = in.readShort();
372 		totalTelegramCount = in.readShort();
373 		baseSubscriptionInfo = new BaseSubscriptionInfo();
374 		baseSubscriptionInfo.read(in);
375 		dataNumber = in.readLong();
376 		delayedDataFlag = in.readBoolean();
377 		length = 31;
378 		int size;
379 		if(telegramNumber == 0) {
380 			dataTime = in.readLong();
381 			errorFlag = in.readByte();
382 			size = in.readByte();
383 			length += 10;
384 			if(size > 0) {
385 				attributesIndicator = new byte[size];
386 				for(int i = 0; i < size; ++i) {
387 					attributesIndicator[i] = in.readByte();
388 				}
389 				length += attributesIndicator.length;
390 			}
391 		}
392 		size = in.readInt();
393 		if(size > 0) {
394 			data = new byte[size];
395 			for(int i = 0; i < size; ++i) {
396 				data[i] = in.readByte();
397 			}
398 			length += data.length;
399 		}
400 		if(length != _length) {
401 			throw new IOException("Falsche Telegrammlänge");
402 		}
403 		priority = TelegramUtility.getPriority(this);
404 		checkConsistency();
405 	}

407 	private void checkConsistency() {
408 		if(data == null && errorFlag == 0) {
409 			String message = "Inkonsistentes Applikationstelegramm: " + telegramNumber + "/" + totalTelegramCount + ", " +
410 			                 " dataNumber: " + (dataNumber >>> 32) + "#" + ((dataNumber & 0xffffffffL) >> 2) + "#" + (dataNumber & 3) + ", " +
411 			                 baseSubscriptionInfo.toString() + ", errorFlag: " + errorFlag + ", data: " + (data == null ? "null" : String.valueOf(data.length) + " Bytes");
412 			_debug.warning(message, new Exception("Inkonsistentes Applikationstelegramm"));
413 		}
