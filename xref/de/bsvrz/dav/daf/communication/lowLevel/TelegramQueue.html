21  package de.bsvrz.dav.daf.communication.lowLevel;
22  
23  import java.util.*;
24  
25  /**
26   * Klasse, die zum gepufferten Austausch von Telegrammen zwischen verschiedenen Threads verwendet werden kann. Die Gesamtgröße der gepufferten Telegramme ist
27   * beschränkt. Es werden verschiedene Telegrammprioritäten unterstützt.
28   * <p/>
29   * Telegramme können mit der Methode {@link #put} gespeichert werden und mit der Methode {@link #take} wieder ausgelesen werden. Die Methoden blockieren, wenn
30   * beim Speichern nicht genügend Platz vorhanden ist, bzw., wenn beim Auslesen kein Telegramm mehr zur Verfügung steht. Der Methode {@link #close} dient zum
31   * Schließen der Queue. blockiert keine der beiden Methoden mehr.
32   *
33   * @author Kappich Systemberatung
34   * @version $Revision: 5047 $
35   */
36  public class TelegramQueue<Telegram extends QueueableTelegram> {
37  
38  	/** Maximale Gesamtgröße für zwischengespeicherte Telegramme. */
39  	final private int _capacity;
40  
41  	/** Gesamtgröße der aktuell zwischengespeicherten Telegramme. */
42  	private int _size;
43  
44  	/**
45  	 * Array, das je mögliche Priorität eine verkettete Liste mit den zwischengespeicherten Telegrammen enthält. Es dient außerdem der Synchronisation von Threads
46  	 * beim lesenden und schreibenden Zugriff.
47  	 */
48  	final private LinkedList<Telegram>[] _priorityLists;
49  
50  	private boolean _closed = false;
51  
52  	/**
53  	 * Erzeugt eine neue Queue mit den angegebenen Eigenschaften.
54  	 *
55  	 * @param capacity        Maximale Gesamtgröße der gepufferten Telegramme.
56  	 * @param maximumPriority Maximale von Telegrammen verwendete Priorität.
57  	 */
58  	public TelegramQueue(int capacity, int maximumPriority) {
59  		if(capacity <= 0) throw new IllegalArgumentException("capacity muss positiv sein: " + capacity);
60  		if(maximumPriority < 0) throw new IllegalArgumentException("maximumPriority darf nicht negativ sein: " + maximumPriority);
61  		if(maximumPriority > 127) throw new IllegalArgumentException("maximumPriority darf nicht größer als 127 sein: " + maximumPriority);
62  		_capacity = capacity;
63  		_size = 0;
64  		_priorityLists = (LinkedList<Telegram>[])new LinkedList[maximumPriority + 1]; // Compiler-Warnung nicht vermeidbar
65  		for(int i = 0; i < _priorityLists.length; i++) {
66  			_priorityLists[i] = new LinkedList<Telegram>();
67  		}
68  	}
69  
70  	/**
71  	 * Gibt das älteste in der Queue gespeicherte Telegramm mit der höchsten Priorität zurück. Wenn die Queue noch nicht geschlossen wurde, wartet diese Methode,
72  	 * bis ein Telegramm in der Queue zur Verfügung steht.
73  	 *
74  	 * @return Nächstes gespeicherte Telegramm mit der höchsten Priorität. Wenn die Queue geschlossen wurde und kein gespeichertes Telegramm mehr verfügbar ist
75  	 *         wird <code>null</code> zurückgegeben.
76  	 *
77  	 * @throws InterruptedException Wenn der Thread während des Wartens unterbrochen wurde.
78  	 */
79  	public Telegram take() throws InterruptedException {
80  		synchronized(this) {
81  			while(_size == 0) {
82  				// Wenn die Queue leer ist und geschlossen wurde, wird null zurückgegeben
83  				if(_closed) return null;
84  				// Wenn die Queue leer ist und nicht geschlossen wurde, wird gewartet
85  				wait();
86  			}
87  			for(int i = _priorityLists.length - 1; i >= 0; i--) {
88  				LinkedList<Telegram> priorityList = _priorityLists[i];
89  				if(!priorityList.isEmpty()) {
90  					final Telegram telegram = priorityList.removeFirst();
91  					_size -= telegram.getSize();
92  					notifyAll();
93  					return telegram;
94  				}
95  			}
96  		}
97  		throw new IllegalStateException("Interner Fehler: Es wurde kein Telegramm gefunden, obwohl die Gesamtgröße " + _size + " ist");
98  	}
99  
100 	/**
101 	 * Speichert das angegebene Telegramm in der Queue. Bei Bedarf wartet diese Methode bis genügend Platz in der Queue für das zu speichernde Telegramm zur
102 	 * Verfügung steht.
103 	 *
104 	 * @param telegram Das zu speichernde Telegramm
105 	 *
106 	 * @throws InterruptedException Wenn der Thread während des Wartens unterbrochen wurde.
107 	 */
108 	public void put(Telegram telegram) throws InterruptedException {
109 		if(_closed) return;
110 		final int length = telegram.getSize();
111 		if(length <= 0) throw new IllegalArgumentException("Telegrammlänge muss größer 0 sein, ist aber " + length + ": " + telegram);
112 		final byte priority = telegram.getPriority();
113 		synchronized(this) {
114 			while(!_closed && _size + length > _capacity) {
115 				wait();
116 			}
117 			if(_closed) return;
118 			_priorityLists[priority].add(telegram);
119 			_size += length;
120 			notifyAll();
121 		}
122 	}
123 
124 	/**
125 	 * Bestimmt die maximale Gesamtgröße für zwischengespeicherte Telegramme.
126 	 *
127 	 * @return Maximale Gesamtgröße für zwischengespeicherte Telegramme.
128 	 */
129 	public int getCapacity() {
130 		return _capacity;
131 	}
132 
133 	/**
134 	 * Bestimmt die Gesamtgröße der aktuell zwischengespeicherten Telegramme.
135 	 *
136 	 * @return Gesamtgröße der aktuell zwischengespeicherten Telegramme.
137 	 */
138 	public int getSize() {
139 		synchronized(this) {
140 			return _size;
141 		}
142 	}
143 
144 
145 	/**
146 	 * Diese Methode schließt die Verbindung. Danach ignoriert die Methode {@link #put} sämtliche weitere zu speichernde Telegramme und die Methode {@link #take}
147 	 * liefert noch alle bisher gespeicherten Telegramme und danach <code>null</code> zurück. Eventuell blockierte Threads werden geweckt.
148 	 */
149 	public void close() {
150 		synchronized(this) {
151 			_closed = true;
152 			notifyAll();
153 		}
154 	}
155 
156 	/**
157 	 * Diese Methode schließt die Verbindung und löscht alle noch gespeicherten Telegramme. Danach ignoriert die Methode {@link #put} sämtliche weitere zu
158 	 * speichernde Telegramme und die Methode {@link #take} liefert anschließend immer <code>null</code> zurück. Eventuell blockierte Threads werden geweckt.
159 	 */
160 	public void abort() {
161 		synchronized(this) {
162 			_closed = true;
163 			notifyAll();
164 			try {
165 				while(take() != null) ;
166 			}
167 			catch(InterruptedException ignored) {
168 			}
169 		}
170 	}
