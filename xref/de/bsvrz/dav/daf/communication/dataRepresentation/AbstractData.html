204 			return false;

208 			throw new UnsupportedOperationException("Attributliste " + getName() + " kann nicht in einem Textwert dargestellt werden");

211 		public boolean isDefined() {
212 			// Es handelt sich um eine Liste oder ein Array, alle Elemente durchlaufen und
213 			// ebenfalls prüfen.
214 			final Iterator iterator = iterator();
215 
216 			while(iterator.hasNext()) {
217 				final Data data = (Data)iterator.next();
218 				if(!data.isDefined()) {
219 					return false;
220 				}
221 			}
222 			return true;
223 		}
224 
225 		public void setToDefault() {
226 			// Ist das Objekt ein Array, so muss:
227 			// - bei Arrays mit variabler Länge die Länge auf 0 gesetzt werden
228 			// - Arrays mit fester Länge werden auf die Länge gesetzt und die Elemente dann initialisiert
229 			if(isArray()) {
230 				final Data.Array array = asArray();
231 				if(array.isCountVariable()) {
232 					array.setLength(0);
233 				}
234 				else {
235 					array.setLength(array.getMaxCount());
236 				}
237 			}
238 
239 			// Sowohl bei Arrays als auch bei Listen müssen alle Elemente mit ihrem Defaultwert (oder undefiniert)
240 			// initialisiert werden.
241 
242 			final Iterator iterator = iterator();
243 			while(iterator.hasNext()) {
244 				((Data)iterator.next()).setToDefault();
245 			}
246 		}
247 	}

426 
427 		protected abstract DataModel getDataModel();
428 
429 		/**
430 		 * Liefert den Wert dieses Referenzattributs als Text zurück. Wenn das referenzierte Objekt eine Pid hat wird diese zurückgegeben, ansonsten wird die Id des
431 		 * Objekts zurückgegeben.
432 		 *
433 		 * @return pid oder id des referenzierten Objekts als Text.
434 		 *
435 		 * @see #getSuffixText
436 		 * @see #getText
437 		 * @see SystemObject#getPid
438 		 * @see SystemObject#getId
439 		 */
440 		public String getValueText() {
441 			try {
442 				SystemObject object = getSystemObject();
443 				if(object == null) return "undefiniert";
444 				String pid = object.getPid();
445 				if(pid != null && !pid.equals("")) return pid;
446 			}
447 			catch(Exception e) {
448 				//Fehler beim Lesen der pid -> weiter mit Rückgabe der id
449 			}
450 			try {
451 				return String.valueOf(getId());
452 			}
453 			catch(Exception ee) {
454 				return "<<" + ee.getMessage() + ">>";
455 			}
456 		}
457 
458 
459 		/**
460 		 * Liefert Zusatzinformationen zum Wert dieses Referenzattributs. Der zurückgelieferte Text ist als Ergänzung zum Rückgabewert der Methode {@link
461 		 * #getValueText} zu verstehen. Das Ergebnis der Methode enthält abhängig vom Ergebnis der Methode {@link #getValueText} den konstanten Text "id" bzw. "pid"
462 		 * und zusätzlich den Namen des referenzierten Objekts (wenn vorhanden).
463 		 *
464 		 * @return Text mit Zusatzinformation zum Wert dieses Referenzattributs.
465 		 *
466 		 * @see #getSuffixText
467 		 * @see #getText
468 		 */
469 		public String getSuffixText() {
470 			try {
471 				String name = null;
472 				String pid = null;
473 				String exceptionMessage = null;
474 				StringBuffer suffix = new StringBuffer();
475 				try {
476 					SystemObject object = getSystemObject();
477 					if(object == null) return "";
478 					name = object.getName();
479 					pid = object.getPid();
480 				}
481 				catch(Exception e) {
482 					exceptionMessage = " " + e.getLocalizedMessage();
483 				}
484 				if(pid == null || pid.equals("")) {
485 					suffix.append("id");
486 				}
487 				else {
488 					suffix.append("pid");
489 				}
490 				if(name != null && !name.equals("")) suffix.append(" (Name: ").append(name).append(")");
491 				if(exceptionMessage != null) suffix.append(" ").append(exceptionMessage);
492 				return suffix.toString();
493 			}
494 			catch(Exception ee) {
495 				return "<<" + ee.getMessage() + ">>";
496 			}
497 		}
498 
499 		public void setText(String text) {
500 			int startIndex;
501 			boolean tryPid = true;
502 			boolean tryId = true;
503 			String lowercaseText = text.toLowerCase();
504 			startIndex = lowercaseText.lastIndexOf("pid:");
505 			if(startIndex >= 0) {
506 				startIndex += 4;
507 				tryId = false;
508 			}
509 			else {
510 				startIndex = lowercaseText.lastIndexOf("id:");
511 				if(startIndex >= 0) {
512 					startIndex += 3;
513 					tryPid = false;
514 				}
515 				else {
516 					startIndex = 0;
517 				}
518 			}
519 			text = text.substring(startIndex).trim();
520 			if(tryId) {
521 				String numberText = text.split("\\D", 2)[0];
522 				if(numberText.length() > 0) {
523 					long id = Long.parseLong(numberText);
524 					if(id == 0) {
525 						setSystemObject(null);
526 						return;
527 					}
528 					SystemObject object;
529 					try {
530 						object = getDataModel().getObject(id);
531 					}
532 					catch(Exception e) {
533 						object = null;
534 					}
535 					if(object != null) {
536 						setSystemObject(object);
537 						return;
538 					}
539 				}
540 			}
541 			if(tryPid) {
542 				String pid = text.split("[\\s\\Q[]{}():\\E]", 2)[0];
543 				if(pid.equals("null") || pid.equals("undefiniert")) {
544 					setSystemObject(null);
545 					return;
546 				}
547 				SystemObject object;
548 				try {
549 					object = getDataModel().getObject(pid);
550 				}
551 				catch(Exception e) {
552 					object = null;
553 				}
554 				if(object != null) {
555 					setSystemObject(object);
556 					return;
557 				}
558 			}
559 			throw new IllegalArgumentException("Der Text '" + text + "' kann nicht als Objektreferenz interpretiert werden.");
560 		}
561 
562 		public SystemObject getSystemObject() {
563 			try {
564 				long id = getId();
565 				if(id == 0) return null;
566 				SystemObject object = getDataModel().getObject(id);
567 				if(object == null) throw new IllegalStateException("Ungültiges Objekt mit id " + id);
568 				return object;
569 			}
570 			catch(Exception e) {
571 				throw new RuntimeException(e);
572 			}
573 		}

575 		public void setSystemObjectPid(String objectPid, ObjectLookup datamodel) {
576 			final SystemObject systemObject;
577 			if(objectPid.length() == 0) systemObject = null;
578 			else {
579 				systemObject = datamodel.getObject(objectPid);
580 				if(systemObject == null) {
581 					if(tryToStorePid(objectPid)) {
582 						_debug.warning("Eine optionale Referenz auf das Objekt mit der Pid '" + objectPid + "' konnte nicht aufgelöst werden");
583 						return;
584 					}
585 					else {
586 						throw new IllegalArgumentException("Das referenzierte Objekt '" + objectPid + "' wurde nicht gefunden");
587 					}
588 				}
589 			}
590 			setSystemObject(systemObject);
591 		}

618 	public abstract static class RelativeTimeValue extends AbstractData.TimeValue {
619 
620 		public String getValueText() {
621 			try {
622 				StringBuffer text = new StringBuffer();
623 				long val = getMillis();
624 				//Beispiel= "234 Tage 12 Stunden 34 Minuten 33 Sekunden 443 Millisekunden"
625 				int millis = (int)(val % 1000);
626 				val /= 1000;
627 				int seconds = (int)(val % 60);
628 				val /= 60;
629 				int minutes = (int)(val % 60);
630 				val /= 60;
631 				int hours = (int)(val % 24);
632 				val /= 24;
633 				long days = val;
634 				if(days != 0) {
635 					if(days == 1) {
636 						text.append("1 Tag ");
637 					}
638 					else if(days == -1) {
639 						text.append("-1 Tag ");
640 					}
641 					else {
642 						text.append(days).append(" Tage ");
643 					}
644 				}
645 				if(hours != 0) {
646 					if(hours == 1) {
647 						text.append("1 Stunde ");
648 					}
649 					else if(hours == -1) {
650 						text.append("-1 Stunde ");
651 					}
652 					else {
653 						text.append(hours).append(" Stunden ");
654 					}
655 				}
656 				if(minutes != 0) {
657 					if(minutes == 1) {
