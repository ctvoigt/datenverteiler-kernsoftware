63  	public static void checkValue(final AttributeType attributeType, final String textValue, final ObjectLookup objectLookup) {
64  		// Defaultwert vorhanden, ist er OK?
65  		if(attributeType instanceof DoubleAttributeType) {
66  			DoubleAttributeType doubleAttributeType = (DoubleAttributeType)attributeType;
67  			textToDouble(doubleAttributeType, textValue);
68  		}
69  		else if(attributeType instanceof IntegerAttributeType) {
70  			IntegerAttributeType integerAttributeType = (IntegerAttributeType)attributeType;
71  			textToUnscaled(integerAttributeType, textValue);
72  		}
73  		else if(attributeType instanceof ReferenceAttributeType) {
74  			ReferenceAttributeType referenceAttributeType = (ReferenceAttributeType)attributeType;
75  			textToReference(referenceAttributeType, textValue, objectLookup);
76  		}
77  		else if(attributeType instanceof StringAttributeType) {
78  			StringAttributeType stringAttributeType = (StringAttributeType)attributeType;
79  			checkText(stringAttributeType, textValue);
80  		}
81  		else if(attributeType instanceof TimeAttributeType) {
82  			TimeAttributeType timeAttributeType = (TimeAttributeType)attributeType;
83  			textToMillis(timeAttributeType, textValue);
84  		}
85  	}

106 	private static long relativeTimeToMillis(final String text) {
107 		String[] splitted = text.trim().split(_relPattern);
108 		long number = 0;
109 		long millis = 0;
110 		for(int i = 0; i < splitted.length; ++i) {
111 			String word = splitted[i];
112 			number = Long.decode(word).longValue();
113 			if(++i < splitted.length) {
114 				word = splitted[i].toLowerCase();
115 				if(word.equals("t") || word.startsWith("tag")) {
116 					millis += (1000 * 60 * 60 * 24) * number;
117 				}
118 				else if(word.equals("h") || word.startsWith("stunde")) {
119 					millis += (1000 * 60 * 60) * number;
120 				}
121 				else if(word.equals("m") || word.startsWith("minute")) {
122 					millis += (1000 * 60) * number;
123 				}
124 				else if(word.equals("s") || word.startsWith("sekunde")) {
125 					millis += 1000 * number;
126 				}
127 				else if(word.equals("ms") || word.startsWith("milli")) {
128 					millis += number;
129 				}
130 				else {
131 					throw new IllegalArgumentException("Ungültige relative Zeitangabe: " + splitted[i]);
132 				}
133 			}
134 			else {
135 				throw new IllegalArgumentException("Fehlende Einheit bei relativer Zeitangabe: " + text);
136 			}
137 		}
138 		return millis;
139 	}

145 	private static long absoluteTimeToMillis(final String text) {
146 		DateFormat format;
147 		Date date;
148 		for(int i = 0; i < _parseDateFormats.length; ++i) {
149 			format = _parseDateFormats[i];
150 			try {
151 				synchronized(format) {
152 					date = format.parse(text);
153 				}
154 				return date.getTime();
155 			}
156 			catch(ParseException e) {
157 				//continue with next Format
158 			}
159 		}
160 		throw new IllegalArgumentException(
161 				"Ungültig Zeitangabe '" + text + "' (Unterstützte Formate: 'dd.MM.yy HH:mm:ss,SSS', 'dd.MM.yy HH:mm:ss', 'dd.MM.yy HH:mm', 'dd.MM.yy')"
162 		);
163 	}

165 	private static SystemObject textToReference(final ReferenceAttributeType referenceAttributeType, final String textValue, final ObjectLookup lookup) {
166 		int startIndex;
167 		boolean tryPid = true;
168 		boolean tryId = true;
169 		String text = textValue;
170 		String lowercaseText = text.toLowerCase();
171 		startIndex = lowercaseText.lastIndexOf("pid:");
172 		if(startIndex >= 0) {
173 			startIndex += 4;
174 			tryId = false;
175 		}
176 		else {
177 			startIndex = lowercaseText.lastIndexOf("id:");
178 			if(startIndex >= 0) {
179 				startIndex += 3;
180 				tryPid = false;
181 			}
182 			else {
183 				// Wenn weder "Pid:" noch "Id:" gefunden wurde, dann wird von vorne gesucht
184 				startIndex = 0;
185 			}
186 		}
187 		text = text.substring(startIndex).trim();
188 		if(tryId) {
189 			String numberText = text.split("\\D", 2)[0];
190 			if(numberText.length() > 0) {
191 				long id = Long.parseLong(numberText);
192 				if(id == 0) {
193 					if(referenceAttributeType.isUndefinedAllowed()) {
194 						return null;
195 					}
196 					else {
197 						throw new IllegalArgumentException(
198 								"Attributwert '" + textValue + "' wird als undefinierte Referenz interpretiert, diese sind allerdings am Attributtyp '"
199 								+ referenceAttributeType.getPid() + "' nicht zugelassen."
200 						);
201 					}
202 				}
203 				try {
204 					SystemObject object;
205 					object = lookup.getObject(id);
206 					if(object != null) return object;
207 				}
208 				catch(Exception e) {
209 					// Ignorieren
210 				}
211 			}
212 		}
213 		if(tryPid) {
214 			String pid = text.split("[\\s\\Q[]{}():\\E]", 2)[0];
215 			if(pid.equals("null") || pid.equals("undefiniert")) {
216 				if(referenceAttributeType.isUndefinedAllowed()) {
217 					return null;
218 				}
219 				else {
220 					throw new IllegalArgumentException(
221 							"Attributwert '" + textValue + "' wird als undefinierte Referenz interpretiert, diese sind allerdings am Attributtyp '"
222 							+ referenceAttributeType.getPid() + "' nicht zugelassen."
223 					);
224 				}
225 			}
226 			try {
227 				SystemObject object;
228 				object = lookup.getObject(pid);
229 				if(object != null) return object;
230 			}
231 			catch(Exception e) {
232 				// Ignorieren
233 			}
234 		}
235 		throw new IllegalArgumentException("Der Text '" + text + "' kann nicht als Objektreferenz interpretiert werden.");
236 	}

245 	private static long textToUnscaled(final IntegerAttributeType type, String text) {
246 		IntegerValueState bestMatch = stateTextToUnscaled(type, text);
247 		if(bestMatch != null) return bestMatch.getValue();
248 		text = text.trim();
249 
250 		final long unscaledValue;
251 
252 		// Ist der Text im location-distance Format?
253 		if(_locationDistancePattern.matcher(text).matches()) {
254 			unscaledValue = locationDistanceToUnscaled(text);
255 		}
256 		else {
257 			unscaledValue = scaledToUnscaled(text, type);
258 		}
259 		checkRange(type, unscaledValue);
260 		return unscaledValue;
261 	}

263 	private static long scaledToUnscaled(final String text, final IntegerAttributeType type) {
264 		final long unscaledValue;
265 		Number number;
266 		ParsePosition parsePosition = new ParsePosition(0);
267 		synchronized(_parseNumberFormat) {
268 			number = _parseNumberFormat.parse(text.replace('.', ','), parsePosition);
269 		}
270 		if(number == null) throw new IllegalArgumentException("Text " + text + " kann nicht in eine Zahl konvertiert werden");
271 		unscaledValue = scaledToUnscaled(type, number);
272 		return unscaledValue;
273 	}

275 	private static long locationDistanceToUnscaled(final String text) {
276 		String[] locationDistance = text.split("\\s*-\\s*");
277 		int location = Integer.parseInt(locationDistance[0]);
278 		int distance = Integer.parseInt(locationDistance[1]);
279 		if(location > 65535) {
280 			throw new IllegalArgumentException("Wert im Location-Distance Format aber Location zu groß: " + location);
281 		}
282 		if(distance > 255) {
283 			throw new IllegalArgumentException("Wert im Location-Distance Format aber Distance zu groß: " + distance);
284 		}
285 		return (location * 256) + distance;
286 	}

288 	private static IntegerValueState stateTextToUnscaled(final IntegerAttributeType type, final String text) {
289 		List<IntegerValueState> states = type.getStates();
290 		Iterator<IntegerValueState> i = states.iterator();
291 		int matchLength = 0;
292 		IntegerValueState bestMatch = null;
293 		while(i.hasNext()) {
294 			IntegerValueState state = i.next();
295 			if(text.equals(state.getName())) {
296 				bestMatch = state;
297 				break;
298 			}
299 			if(text.startsWith(state.getName())) {
300 				if(state.getName().length() > matchLength) {
301 					bestMatch = state;
302 					matchLength = state.getName().length();
303 				}
304 			}
305 		}
306 		return bestMatch;
307 	}

309 	private static long scaledToUnscaled(final IntegerAttributeType type, final Number number) {
310 		IntegerValueRange range = type.getRange();
311 		if(range == null) {
312 			throw new IllegalArgumentException("Kein Wertebereich definiert");
313 		}
314 		double conversionFactor = range.getConversionFactor();
315 		if(conversionFactor == 1.0) {
316 			return number.longValue();
317 		}
318 		else {
319 			//Folgende Fallunterscheidung ist erforderlich um Rundungsfehler zu vermeiden
320 			//Beispielsweise führt Math.round(0.95 / 0.1) zum falschen Ergebnis 9
321 			//aber Math.round(0.95 * (1 / 0.1)) führt zum richtigen Ergebnis 10
322 			if(conversionFactor < 1.0) {
323 				return Math.round(number.doubleValue() * (1 / conversionFactor));
324 			}
325 			else {
326 				return Math.round(number.doubleValue() / conversionFactor);
327 			}
328 		}
329 	}

337 			throw new IllegalArgumentException(
338 					"Unskalierter Wert " + unscaledValue + " liegt nicht im erlaubten Wertebereich [" + range.getMinimum() + ", " + range.getMaximum() + "]"

343 	private static double textToDouble(final DoubleAttributeType doubleAttributeType, final String textValue) {
344 		Number number;
345 		ParsePosition parsePosition = new ParsePosition(0);
346 		synchronized(_parseNumberFormat) {
347 			number = _parseNumberFormat.parse(textValue.replace('.', ','), parsePosition);
348 		}
349 		if(number == null) throw new IllegalArgumentException("Text " + textValue + " kann nicht in eine Fließkommazahl konvertiert werden");
350 		double doubleValue = number.doubleValue();
351 		if(doubleAttributeType.getAccuracy() == DoubleAttributeType.FLOAT) {
352 			if(doubleValue < -(Float.MAX_VALUE) || Float.MAX_VALUE < doubleValue) {
353 				throw new IllegalArgumentException("Fließkommazahl " + number + " liegt außerhalb der Grenzen des 32 Bit Single-Formats");
354 			}
355 		}
356 		return doubleValue;
357 	}
