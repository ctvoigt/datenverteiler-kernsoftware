144 	/**
145 	 * Prüft ob der Wert eines Attributes vom Typ "ReferenceAttributeType" undefiniert ist und somit nicht verschickt werden kann.
146 	 *
147 	 * @param data                   Attribut, dessen Wert geprüft werden soll
148 	 * @param referenceAttributeType Bestimmt, ob die Referenz "0" als undefiniert Wert gilt, oder als normale Referenz
149 	 *
150 	 * @return true = Das Attribut ist definiert und kann verschickt werden; false = Der Wert des Attributes ist gleich dem "undefiniert Wert" und kann somit nicht
151 	 *         verschickt werden
152 	 */
153 	public boolean isDefinedReference(Data data, ReferenceAttributeType referenceAttributeType) {
154 		if(data.getAttributeType() instanceof ReferenceAttributeType) {
155 			if(referenceAttributeType.isUndefinedAllowed()) {
156 				// Undefinierte Referenzen sind erlaubt, also kann die "0" nicht zum erkennen des "undefiniert Wert" benutzt werden.
157 				// Also ist jeder Wert erlaubt (da jeder Wert mindestens eine undefinierte Referenz darstellt).
158 				return true;
159 			}
160 			else {
161 				// Undefinierte Referenzen sind nicht erlaubt. Also ist die 0 der "undefiniert Wert
162 				if(_undefinedValueReference == data.asReferenceValue().getId()) {
163 					// Die Id ist gleich dem "undefiniert Wert", also ist das Attribut nicht definiert
164 					return false;
165 				}
166 				else {
167 					return true;
168 				}
169 			}
170 		}
171 		else {
172 			throw new IllegalArgumentException("ReferenceAttributeType, Klasse des übergebenen Objekts: " + data.getAttributeType());
173 		}
174 	}

176 	/**
177 	 * Gibt eine Zahl zurück, die entweder den größten Wert darstellt, der mit den angegebenen Bytes darzustellen ist oder aber die kleinste Zahl. Die Zahl ist
178 	 * entweder größer als requiredMaxValue oder kleiner als requiredMinValue.
179 	 *
180 	 * @param integerAttributeType Enthält den ByteCount, die States und die Min/Max
181 	 *
182 	 * @return Zahl die größer als requiredMaxValue ist oder aber kleiner als requiredMinValue. Kann diese Zahl nicht gefunden werden, wird <code>null</code>
183 	 *         zurückgegeben.
184 	 */
185 	public Long getUndefinedValueInteger(IntegerAttributeType integerAttributeType) {
186 
187 		synchronized(_integerUndefinedValues) {
188 			// Prüfen, ob der Wert vielleicht schon berechnet wurde
189 
190 			// Dies kann entweder ein undefinert Wert (Long) sein oder ein Marker(Object) oder <code>null</code>
191 			final Long undefinedValueOrMarkerOrNull = _integerUndefinedValues.get(integerAttributeType);
192 
193 			if(undefinedValueOrMarkerOrNull != null) {
194 				// Entweder ein Marker oder ein undefiniert Wert
195 				if(undefinedValueOrMarkerOrNull == _marker) {
196 					// Das Objekt ist der "Marker", also wurde der Wert bereits berechnet und nicht gefunden.
197 					// Es kann kein undefiniert Wert berechnet werden.
198 					return null;
199 				}
200 				else {
201 					// Es ist ein undefiniert berechnet worden
202 					return undefinedValueOrMarkerOrNull;
203 				}
204 			}
205 			else {
206 				// Es gibt noch keinen Wert oder Marker
207 
208 				// Der Wert muss erst berechnet werden
209 				List<IntegerValueState> integerValueStates = integerAttributeType.getStates();
210 				final int byteCount = integerAttributeType.getByteCount();
211 				final long rangeMinimum;
212 				final long rangeMaximum;
213 				final IntegerValueRange integerValueRange = integerAttributeType.getRange();
214 				if(integerValueRange != null) {
215 					rangeMinimum = integerValueRange.getMinimum();
216 					rangeMaximum = integerValueRange.getMaximum();
217 				}
218 				else {
219 					// Es wurde kein Wertebereich angegeben. Also müssen
220 					// nur die Zustände berücksichtigt werden.
221 					rangeMinimum = 0;
222 					rangeMaximum = 0;
223 				}
224 
225 				// Der undefiniert Wert wie folgt gefunden:
226 
227 				// Start: Der kleinste darstellbare Wert wird als "undefiniert Wert" genommen
228 				// 1) Prüfen ob der undefiniert Wert nicht innerhalb des geforderten Wertebereichs liegt (wenn ja, dann über den Wert hinweg setzen).
229 				//    Ist der undefiniert Wert größer als das darstellebare Maximum -> Abbruch und Exception werfen
230 				// 2) Ist der undefiniert Wert als Zustand nicht vorhanden -> Wert gefunden und Ende
231 				// 3) undefiniert Wert um eins erhöhen und bei 1) weitermachen
232 
233 				// Speichert jeden Zustand als Long. Dies ermöglicht die Frage "Ist der ungültig Wert gleich einem Zustand" in O(1) zu beantworten.
234 				Set<Long> states = new HashSet<Long>(integerValueStates.size());
235 				for(IntegerValueState integerValueState : integerValueStates) {
236 					states.add(new Long(integerValueState.getValue()));
237 				}
238 
239 				// Größer positiver Wert, der dargestellt werden kann
240 				final long max;
241 				// kleinste Wert, der dargestellt werden kann
242 				final long min;
243 				switch(byteCount) {
244 					case IntegerAttributeType.BYTE:
245 						max = Byte.MAX_VALUE;
246 						min = Byte.MIN_VALUE;
247 						break;
248 					case IntegerAttributeType.SHORT:
249 						max = Short.MAX_VALUE;
250 						min = Short.MIN_VALUE;
251 						break;
252 					case IntegerAttributeType.INT:
253 						max = Integer.MAX_VALUE;
254 						min = Integer.MIN_VALUE;
255 						break;
256 					case IntegerAttributeType.LONG:
257 						max = Long.MAX_VALUE;
258 						min = Long.MIN_VALUE;
259 						break;
260 					default:
261 						throw new IllegalStateException("Ganzzahlattributtyp " + integerAttributeType + " mit nicht unterstützter Anzahl Bytes: " + byteCount);
262 				}
263 
264 				// undefiniert Wert, der gesucht wird
265 				long undefinedValue = min;
266 
267 				while(true) {
268 					if(integerValueRange != null && undefinedValue == rangeMinimum) {
269 						// Stossen von unten an die untere Grenze, also über den geforderten Bereich (rangeMinimum bis rangeMaximum) hinwegsetzen
270 						undefinedValue = rangeMaximum;
271 					}
272 					else if(!states.contains(new Long(undefinedValue))) {
273 						// Es gibt den Wert nicht im Zustand und er ist bestimmt ausserhalb
274 						// des angegebenen Wertebereichs. Also wurde der undefiniert Wert gefunden.
275 						_integerUndefinedValues.put(integerAttributeType, new Long(undefinedValue));
276 						return undefinedValue;
277 					}
278 					if(undefinedValue >= max) {
279 						// Es konnte kein Wert berechnet werden, damit die Berechung nicht erneut durchgeführt werden muss,
280 						// wird dieses "Ergebnis" gespeichert.
281 						_integerUndefinedValues.put(integerAttributeType, _marker);
282 						return null;
283 					}
284 					undefinedValue++;
285 				}
286 			}
287 		} // synch map
288 	}

290 	/**
291 	 * @param data                 Attribut, in das der "undefiniert Wert" eingetragen wird
292 	 * @param undefinedValue       Undefiniert Wert des Attributtyps. <code>null</code> wird als "es gibt keinen undefiniert Wert" interpretiert.
293 	 * @param integerAttributeType Dient nur dazu, um bei einem Fehler die Pid des fehlehaften Typs anzugeben
294 	 *
295 	 * @throws IllegalStateException Es kann kein "undefiniert Wert" ermittelt werden, da alle Byte-Kombinationen gebraucht werden um die angegebenen Werte
296 	 *                               darzustellen. Dieser Fall darf eigentlich nicht auftreten, da in diesen Fällen ein default-Wert definiert sein muss.
297 	 */
298 	public void setToUndefinedInteger(Data data, Long undefinedValue, IntegerAttributeType integerAttributeType) {
299 		if(data.getAttributeType() instanceof IntegerAttributeType) {
300 			// Als unskaliert schreiben. Bei unskaliert wird geprüft ob der zu schreibende Wert der undefiniert Wert ist
301 
302 			if(undefinedValue != null) {
303 				data.asUnscaledValue().set(undefinedValue);
304 			}
305 			else {
306 				// Der Wert konnte nicht berechnet werden
307 				throw new IllegalStateException(
308 						"Für ein Attribut konnte kein undefiniert-Wert berechnet werden: Pid des Typs " + integerAttributeType.getPidOrNameOrId()
309 				);
310 			}
311 		}
312 		else {
313 			throw new IllegalArgumentException("IntegerAttributeType, Klasse des übergebenen Objekts: " + data.getAttributeType());
314 		}
315 	}

317 	/**
318 	 * Prüft ob ein Attribut vom Typ IntegerAttributeType definiert ist und damit verschickt werden kann.
319 	 *
320 	 * @param attributeType  Ganzzahl-Attributtyp
321 	 * @param data           Attribut, dessen Wert geprüft werden soll
322 	 * @param undefinedValue enthält den undefiniert Wert des Attributtyps. Wird <code>null</code> übergeben, so wird dies als "kein undefiniert festgelegt"
323 	 *                       interpretiert.
324 	 *
325 	 * @return true = Das Attribut ist definiert und kann verschickt werden; false = Der Wert des Attributes ist gleich dem "undefiniert Wert" und darf somit nicht
326 	 *         verschickt werden
327 	 */
328 	public boolean isDefinedInteger(final IntegerAttributeType attributeType, Data data, Long undefinedValue) {
329 		if(!(data.getAttributeType() instanceof IntegerAttributeType)) {
330 			throw new IllegalArgumentException("Attributtyp des übergebenen Data-Objekts ist kein IntegerAttributeType sondern: " + data.getAttributeType());
331 		}
332 		final long unscaledValue = data.asUnscaledValue().longValue();
333 		// Unabhängig vom Undefined-Wert wird hier geprüft, ob der Wert des Attributs im Wertebereich liegt oder ein Aufzählungszustand ist.
334 		final IntegerValueRange range = attributeType.getRange();
335 		if(range != null) {
336 			// Wertebereich angegeben, liegt der Wert drin?
337 			if(range.getMinimum() <= unscaledValue && unscaledValue <= range.getMaximum()) return true;
338 		}
339 		final List<IntegerValueState> states = attributeType.getStates();
340 		for(IntegerValueState state : states) {
341 			if(state.getValue() == unscaledValue) return true;
342 		}
343 		return false;
344 	}
