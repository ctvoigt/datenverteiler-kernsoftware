21  package de.bsvrz.dav.daf.main.archive;
22  
23  /**
24   * Eine Implementation dieses Interface enthält Informationen darüber, welche Zeitbereiche/Indexbereiche sich im
25   * direkten Zugriff des Archivsystems befinden. Das Objekt bezieht sich sich dabei auf eine Datenidentifikation {@link
26   * ArchiveDataSpecification}. Zu jedem Zeitbereich/Indexbereich wird außerdem das Speichermedium vom Typ B
27   * angegeben, auf dem die Informationen persistent gespeichert sind. In den Fällen wo das Archivsystem den
28   * Zeitbereich/Indexbereich nicht identifizieren kann, wird das Flag "Datenlücke" gesetzt.
29   * <p/>
30   * Zu dem oben beschriebenen Zeitbereich/Indexbereich wird außerdem noch ein Objekt vom Typ {@link
31   * ArchiveDataSpecification} gespeichert.
32   *
33   * @author Kappich Systemberatung
34   * @version $Revision: 5084 $
35   */
36  public interface ArchiveInformationResult {
37  
38  	/**
39  	 * Gibt den Zeit/Indexbereich zurück, auf den sich die Informationen beziehen (obere Grenze).
40  	 *
41  	 * @return obere Grenze des Intervalls
42  	 */
43  	long getIntervalStart();
44  
45  	/**
46  	 * Gibt den Zeit/Indexbereich zurück, auf den sich die Informationen beziehen (untere Grenze).
47  	 *
48  	 * @return untere Grenze des Intervalls
49  	 */
50  	long getIntervalEnd();
51  
52  	/**
53  	 * Gibt die Art des Intervalls (Datenindex, Datenzeit oder Archivzeit) zurück.
54  	 *
55  	 * @return Datenindex, Datenzeit oder Archivzeit
56  	 */
57  	TimingType getTimingType();
58  
59  	/**
60  	 * Der Rückgabewert dieser Methode bestimmt, ob eine Datenlücke vorhanden ist.
61  	 *
62  	 * @return true = potentielle Datenlücke (Daten befinden sich nicht im direkten Zugriff und sind auch auf einem
63  	 *         Speichermedium Typ B nicht vorhanden); false = alle Daten des Intervalls sind verfügbar
64  	 */
65  	boolean isDataGap();
66  
67  	/**
68  	 * Der Rückgabewert dieser Methode bestimmt, ob sich der Zeitbereich im direkten Zugriff des Archivsystems befindet.
69  	 *
70  	 * @return true = Der Zeitbereich ist im direkten Zugriff des Archivsystems; false = Der Zeitbereich ist nicht im
71  	 *         direkten Zugriff des Archivsystems
72  	 */
73  	boolean directAccess();
74  
75  	/**
76  	 * Diese Methode gibt die eindeutige Identifikation des Speichermediums Tyb B zurück, auf dem das angegebene Intervall
77  	 * gesichert wurde.
78  	 *
79  	 * @return eindeutige Identifikation des Speichermediums Tyb B oder <code>null</code> falls die Identifikation
80  	 *         unbekannt ist.
81  	 */
82  	int getVolumeIdTypB();
83  
84  	/**
85  	 * Gibt die Archivanfrage zurück, die diese Informationen erzeugt hat.
86  	 *
87  	 * @return Archivanfrage
88  	 */
89  	ArchiveDataSpecification getArchiveDataSpecification();
90  }
