43  /**
44   * Klasse zum Zugriff auf den Zeitfluß einer Simulation. Nach dem Erzeugen eines Objekts dieser Klasse kann mit der Methode {@link #getTime()} auf die
45   * simulierte Zeit der jeweiligen Simulation zugegriffen werden und mit den Methoden {@link #sleep} und {@link #sleepUntil} kann der aufrufende Thread für eine
46   * bestimmte Zeit im Zeitfluß der Simulation blockiert werden. Die Methode {@link #close()} sollte aufgerufen werden, wenn das Objekt zum Zugriff auf den
47   * Zeitfluß einer Simulation nicht mehr benötigt wird.
48   *
49   * @author Kappich Systemberatung
50   * @version $Revision: 6200 $
51   */
52  public class SimulationTime {

63  
64  	private Aspect _simAspect;
65  
66  	private long _time = -1;
67  
68  	private final Integer _timeNotification = new Integer(hashCode());

72  	/**
73  	 * Erzeugt ein neues Objekt zum Zugriff auf den Zeitfluß einer bestimmten Simulation. Im Falle einer Offline-Simulation findet eine Anmeldung auf die
74  	 * Simulationszeit der entsprechenden Simulationsvariante statt, die vom Simulationsdatengenerator zur Verfügung gestellt wird.
75  	 *
76  	 * @param simulationVariant Simulationsvariante der zu betrachtenden Simulation.
77  	 * @param connection        Datenverteilerverbindung über die die Kommunikation mit dem Simulationsdatengenerator durchgeführt wird.
78  	 *
79  	 * @throws ConfigurationException Wenn bei der Kommunikation mit der Konfiguration Fehler aufgetreten sind.
80  	 */
81  	public SimulationTime(
82  			short simulationVariant, ClientDavConnection connection
83  	) throws ConfigurationException {
84  
85  		if(connection == null) {
86  			throw new IllegalArgumentException("Inkonsistente Parameter");
87  		}
88  		_simulationVariant = simulationVariant;
89  		_connection = connection;
90  		DataModel dataModel = connection.getDataModel();
91  		if(dataModel == null) {
92  			throw new InitialisationNotCompleteException("Keine fertig initialisierte Verbindung zum Datenverteiler.");
93  		}
94  		_internalReceiver = new InternalReceiver();
95  		if(simulationVariant > 0) {
96  			String pid = null;
97  			if(simulationVariant < 10) {
98  				pid = "simulation.00" + simulationVariant;
99  			}
100 			else if(simulationVariant < 100) {
101 				pid = "simulation.0" + simulationVariant;
102 			}
103 			else {
104 				pid = "simulation." + simulationVariant;
105 			}
106 			_simObject = dataModel.getObject(pid);
107 			if(_simObject != null) {
108 				SystemObjectType simObjectType = _simObject.getType();
109 				if("typ.offlineSimulation".equals(simObjectType.getPid())) {
110 					_onlineModus = false;
111 				}
112 			}
113 		}
114 		if(!_onlineModus) {
115 			_simAttributeGroup = (AttributeGroup)dataModel.getObject("atg.simulationsZeit");
116 			_simAspect = (Aspect)dataModel.getObject("asp.standard");
117 			if((_simAttributeGroup == null) || (_simAspect == null)) {
118 				throw new ConfigurationException("Fehlerhafte Konfigurationsdaten.");
119 			}
120 			// atg.simulationsZeit:asp.standard immer mit Simulationsvariante 0 verwenden 
121 			connection.subscribeReceiver(
122 					_internalReceiver, _simObject,
123 					new DataDescription(_simAttributeGroup, _simAspect, (short)0),
124 					ReceiveOptions.normal(),
125 					ReceiverRole.receiver()
126 			);
127 		}
128 	}

130 	/**
131 	 * Bestimmt die Zeit einer Simulation. Bei einer Online-Simulation wird die aktuelle Zeit zurückgegeben und bei einer Offline-Simulation die simulierte Zeit.
132 	 *
133 	 * @return Zeitpunkt in Millisekunden seit 1970.
134 	 *
135 	 * @throws IllegalStateException Wenn der simulierte Zeitfluss im Falle einer Offline-Simulation nicht bestimmt werden kann.
136 	 * @see de.bsvrz.dav.daf.main.ClientDavInterface#getTime
137 	 */
138 	public final long getTime() {
139 		if(_onlineModus) {
140 			return System.currentTimeMillis();
141 		}
142 		if(_time == -1) {
143 			long waitingTime = 0, startTime = System.currentTimeMillis();
144 			long sleepTime = 10;
145 			while(waitingTime < CommunicationConstant.MAX_WAITING_TIME_FOR_SYNC_RESPONCE) {
146 				try {
147 					synchronized(_timeNotification) {
148 						_timeNotification.wait(sleepTime);
149 						if(sleepTime < 1000) sleepTime *= 2;
150 					}
151 					if(_time != -1) {
152 						break;
153 					}
154 					waitingTime = System.currentTimeMillis() - startTime;
155 				}
156 				catch(InterruptedException e) {
157 					e.printStackTrace();
158 					break;
159 				}
160 			}
161 		}
162 		if(_time == -1) {
163 			throw new IllegalStateException("Die Zeit kann nicht bestimmt werden.");
164 		}
165 		return _time;
166 	}

168 	/**
169 	 * Blockiert den aufrufenden Thread für die spezifizierte Zeit. Die angegebene Dauer der Pause wird im Falle einer Online-Simulation in Realzeit und im Falle
170 	 * einer Offline-Simulation im Zeitfluss der Simulation berücksichtigt.
171 	 *
172 	 * @param timeToSleep Wartezeit in Millisekunden seit 1970.
173 	 *
174 	 * @throws IllegalStateException Wenn der simulierte Zeitfluss im Falle einer Offline-Simulation nicht bestimmt werden kann.
175 	 * @see de.bsvrz.dav.daf.main.ClientDavInterface#sleep
176 	 */
177 	public final void sleep(long timeToSleep) {
178 		if(_onlineModus) {
179 			try {
180 				Thread.currentThread().sleep(timeToSleep);
181 			}
182 			catch(InterruptedException ex) {
183 			}
184 		}
185 		else {
186 			long timeToWakeUp = getTime() + timeToSleep;
187 			synchronized(_timeNotification) {
188 				while(getTime() < timeToWakeUp) {
189 					try {
190 						_timeNotification.wait();
191 					}
192 					catch(InterruptedException ex) {
193 						break;
194 					}
195 				}
196 			}
197 		}
198 	}

200 	/**
201 	 * Blockiert den aufrufenden Thread bis die spezifizierte Zeit erreicht ist. Der angegebene Zeitpunkt wird im Falle einer Online-Simulation in Realzeit und im
202 	 * Falle einer Offline-Simulation im Zeitfluss der Simulation berücksichtigt.
203 	 *
204 	 * @param absoluteTime Abzuwartender Zeitpunkt in Millisekunden seit 1970.
205 	 *
206 	 * @throws IllegalStateException Wenn der simulierte Zeitfluss im Falle einer Offline-Simulation nicht bestimmt werden kann.
207 	 * @see de.bsvrz.dav.daf.main.ClientDavInterface#sleepUntil
208 	 */
209 	public final void sleepUntil(long absoluteTime) {
210 		if(_onlineModus) {
211 			while(System.currentTimeMillis() < absoluteTime) {
212 				try {
213 					Thread.currentThread().sleep(100);
214 				}
215 				catch(InterruptedException ex) {
216 				}
217 			}
218 		}
219 		else {
220 			synchronized(_timeNotification) {
221 				while(getTime() < absoluteTime) {
222 					try {
223 						_timeNotification.wait();
224 					}
225 					catch(InterruptedException ex) {
226 						break;
227 					}
228 				}
229 			}
230 		}
231 	}

233 	/**
234 	 * Terminiert dieses Objekt und veranlasst bei einer Offline-Simulation die notwendige Abmeldung der Simulationszeiten.
235 	 *
236 	 * @throws ConfigurationException Wenn bei der Kommunikation mit der Konfiguration Fehler aufgetreten sind.
237 	 */
238 	public final void close() throws ConfigurationException {
239 		if(!_onlineModus) {
240 			// atg.simulationsZeit:asp.standard immer mit Simulationsvariante 0 verwenden
241 			_connection.unsubscribeReceiver(
242 					_internalReceiver, _simObject, new DataDescription(_simAttributeGroup, _simAspect, (short)0)
243 			);
244 		}
245 	}

247 	private class InternalReceiver implements ClientReceiverInterface {
248 
249 		InternalReceiver() {
250 		}
251 
252 		/**
253 		 * Aktualisierungsmethode, die nach Empfang eines angemeldeten Datensatzes von den Datenverteiler-Applikationsfunktionen aufgerufen wird. Diese Methode muss
254 		 * von der Applikation zur Verarbeitung der empfangenen Datensätze implementiert werden.
255 		 *
256 		 * @param results Feld mit den empfangenen Ergebnisdatensätzen.
257 		 */
258 		public final void update(ResultData results[]) {
259 			if(results == null) {
260 				return;
261 			}
262 			if(!_onlineModus) {
263 				for(int i = 0; i < results.length; ++i) {
264 					if(results[i] != null) {
265 						DataDescription dataDescription = results[i].getDataDescription();
266 						if(dataDescription != null) {
267 							if(!_simAttributeGroup.equals(dataDescription.getAttributeGroup())) {
268 								continue;
269 							}
270 							if(!_simAspect.equals(dataDescription.getAspect())) {
271 								continue;
272 							}
273 							final Data data = results[i].getData();
274 							if(data != null) {
275 								_time = data.getTimeValue("Zeit").getMillis();
276 							}
277 						}
278 					}
279 				}
280 				synchronized(_timeNotification) {
281 					_timeNotification.notifyAll();
282 				}
283 			}
284 		}
285 	}
