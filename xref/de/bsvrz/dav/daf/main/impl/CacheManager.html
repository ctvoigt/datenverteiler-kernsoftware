128 
129 //				DataInputStream in = new DataInputStream(new BufferedInputStream(new ByteArrayInputStream(data)));

148 		byte errorFlag = newData.getErrorFlag();
149 		long dataIndex = newData.getDataNumber();
150 		boolean dataInside = ((errorFlag == 0x00) && ((dataIndex & 0x0000000000000003) == 0) && (data != null));
151 		LinkedList list = (LinkedList)cache.get(baseSubscriptionInfo);
152 //		System.out.println("++++++++++++list:" + (list == null ? "null" : list.size() + " entries") + "+++++++++++");
153 		CachedObject cachedObject = null;

156

158 			cachedObject = new CachedObject(
159 					baseSubscriptionInfo, delayedDataFlag, dataIndex, newData.getDataTime(), errorFlag, _dataModel

167 		}

169 			// list enthält die schon im Cache gespeicherten Datensätze dieser Datenidentifikation
170 			synchronized(list) {
171 				ListIterator _iterator = list.listIterator(list.size());
172 				while(_iterator.hasPrevious()) {
173 					CachedObject _cachedObject = (CachedObject)_iterator.previous();
174 					long result = _cachedObject.getDataNumber() - dataIndex;
175 					if(result == 0) {
176 						// Der letzte Datensatz im Cache hatte die gleiche Telegrammnummer
177 
178 						cachedObject = new CachedObject(
179 								baseSubscriptionInfo, delayedDataFlag, dataIndex, newData.getDataTime(), errorFlag, _dataModel
180 						);
181 						if(dataInside) {
182 							cachedObject.update(attributesIndicator, data, delayedDataFlag);
183 						}
184 						cachedObject.setActionTime(System.currentTimeMillis());
185 						_iterator.set(cachedObject);
186 						break;
187 					}
188 					else if(result < 0) {
189 						// Der letzte Datensatz im Cache hatte eine kleinere Telegrammnummer als der neue Datensatz
190 
191 						cachedObject = new CachedObject(
192 								baseSubscriptionInfo, delayedDataFlag, dataIndex, newData.getDataTime(), errorFlag, _dataModel
193 						);
194 						if(dataInside) {
195 							cachedObject.update(attributesIndicator, data, delayedDataFlag);
196 						}
197 						cachedObject.setActionTime(System.currentTimeMillis());
198 
199 						// Wenn der Vorhaltezeitraum der Daten 0 ist und der Datensatz nicht nachgeliefert ist, dann
200 						// wird der bisher aktuelle  Datensatz im Cache durch den gerade empfangenen ersetzt.
201 						if(!delayedDataFlag && (subscriptionManager.getTimeInCache(baseSubscriptionInfo) == 0)) {
202 							_iterator.set(cachedObject);
203 						}
204 						else {
205 							_iterator.next();
206 							_iterator.add(cachedObject);
207 						}
208 						break;
209 					}
210 					else {
211 						// Ein Rücksprung des Datensatzindex ist zugelassen, wenn die Verbindung zur Quelle zeitweise unterbrochen war und nach dem leeren
212 						// Datensatz (zur Markierung des Ausfalls mit einem um 1 erhöhten Datensatzindex) nach erneutem Verbindungsaufbau wieder der (i.a.)
213 						// bereits vorher empfangene aktuelle Datensatz mit einem um 1 erniedrigten Datensatzindex empfangen wird.
214 						if(((dataIndex & 0x0000000000000003L) == 0) && (_cachedObject.getDataNumber() != (dataIndex + 1))) {

221 
222 							_debug.error(
223 									"Empfangener Datensatz hat ungültigen Datensatzindex, Objekt: " + objectName + ", Attributgruppenverwendung: "
224 									+ (atgUsage == null ? String.valueOf(usageIdentification) : atgUsage.getPid()) + ", letzter Index: " + (oldIndex >>> 32)
225 									+ "#" + ((oldIndex & 0xffffffffL) >> 2) + "#" + (oldIndex & 3) + ", aktueller Index: " + (dataIndex >>> 32) + "#"
226 									+ ((dataIndex & 0xffffffffL) >> 2) + "#" + (dataIndex & 3)
227 							);
228 							//_cachedObject.getData()
229 						}
230 						else {
231 							cachedObject = new CachedObject(
232 									baseSubscriptionInfo, delayedDataFlag, dataIndex, newData.getDataTime(), errorFlag, _dataModel
233 							);
234 							cachedObject.setActionTime(System.currentTimeMillis());
235 						}
236 						break;
237 					}
238 				}
239 			}
240 		}
241 		if(cachedObject != null) subscriptionManager.actualDataUpdate(cachedObject);

303 	/**
304 	 * Diese Methode iteriert durch den Cache-Inhalt und sammelt die letzen n Datensätze, vom aktuellen Datensatz startend in die Vergangenheit. Wenn keine
305 	 * nachgelieferten erwünscht sind, so werden alle nachgelieferte Daten übersprungen. Ist kein passender Datensatz vorhanden, so wird null zurückgegeben. Sind
306 	 * weniger als n Datensätze vorhanden, so werden diese zurückgegeben.
307 	 *
308 	 * @param baseSubscriptionInfo Anmeldeinformation eines Datensatzes
309 	 * @param delayedDataFlag      Nachgelieferte Daten
310 	 * @param n                    Anzahl der gültigen Versionen eines Datensatzes
311 	 *
312 	 * @return Datensätze/satz oder <code>null</code>, falls kein Datensatz vorhanden ist. Der Datensatz, der als letztes gecasht wurde steht an Index 0.
313 	 */
314 	public final List getCachedData(
315 			BaseSubscriptionInfo baseSubscriptionInfo, boolean delayedDataFlag, int n
316 	) {
317 		if(baseSubscriptionInfo == null) {
318 			return null;
319 		}
320 		LinkedList list = (LinkedList)cache.get(baseSubscriptionInfo);
321 		if(list == null) {
322 			return null;
323 		}
324 		ArrayList arrayList = new ArrayList();
325 		int index = 0;
326 		synchronized(list) {
327 			ListIterator _iterator = list.listIterator(list.size());
328 			while(_iterator.hasPrevious()) {
329 				CachedObject _cachedObject = (CachedObject)_iterator.previous();
330 				long number = _cachedObject.getDataNumber();
331 				byte error = (byte)(number & 0x0000000000000003);
332 				if(error > 0) {
333 					arrayList.add(_cachedObject);
334 					_cachedObject.setActionTime(System.currentTimeMillis());
335 					++index;
336 				}
337 				else {
338 					if(!delayedDataFlag && _cachedObject.getDelayedDataFlag()) {
339 						continue;
340 					}
341 					else {
342 						arrayList.add(_cachedObject);
343 						_cachedObject.setActionTime(System.currentTimeMillis());
344 						++index;
345 					}
346 				}
347 				if(index == n) {
348 					break;
349 				}
350 			}
351 		}
352 		return arrayList;
353 	}

356 	 * Diese Methode iteriert durch den Cache-Inhalt und sammelt die Datensätze, deren Zeitstempel zwischen den spezifizierten Zeiten liegt. Wenn keine
357 	 * nachgelieferten erwünscht sind, so werden alle nachgelieferte Daten übersprungen. Ist kein passender Datensatz vorhanden, so wird <code>null</code>
358 	 * zurückgegeben.
359 	 *
360 	 * @param baseSubscriptionInfo Anmeldeinformation eines Datensatzes
361 	 * @param delayedDataFlag      Nachgelieferte Daten
362 	 * @param fromTime             Start Zeitintervall. Wird der Wert -1 übergeben, so wird die Datenzeit des zuletzt gecachten Objekts benutzt.
363 	 * @param toTime               Ende Zeitintervall. Wird der Wert -1 übergeben, so wird die Datenzeit des zuletzt gecachten Objekts benutzt.
364 	 *
365 	 * @return Datensätze/Datensatz für den die übergebenen Parameter erfüllt sind oder <code>null</code> wenn kein Datensatz vorhanden ist
366 	 */
367 	public final List getCachedData(
368 			BaseSubscriptionInfo baseSubscriptionInfo, boolean delayedDataFlag, long fromTime, long toTime
369 	) {
370 		if(baseSubscriptionInfo == null) {
371 			return null;
372 		}
373 		LinkedList list = (LinkedList)cache.get(baseSubscriptionInfo);
374 		if(list == null) {
375 			return null;
376 		}
377 		ArrayList arrayList = new ArrayList();
378 		synchronized(list) {
379 			CachedObject lastCachedObject = (CachedObject)list.getLast();
380 			if(lastCachedObject == null) {
381 				return null;
382 			}
383 			if(fromTime == -1) {
384 				fromTime = lastCachedObject.getDataTime();
385 			}
386 			if(toTime == -1) {
387 				toTime = lastCachedObject.getDataTime();
388 			}
389 			if(toTime < fromTime) {
390 				long t = fromTime;
391 				fromTime = toTime;
392 				toTime = fromTime;
393 			}
394 			ListIterator _iterator = list.listIterator(0);
395 			while(_iterator.hasNext()) {
396 				CachedObject _cachedObject = (CachedObject)_iterator.next();
397 				long number = _cachedObject.getDataNumber();
398 				byte error = (byte)(number & 0x0000000000000003);
399 				if(error > 0) {
400 					long time = _cachedObject.getDataTime();
401 					if((time >= fromTime) && (time <= toTime)) {
402 						arrayList.add(_cachedObject);
403 						_cachedObject.setActionTime(System.currentTimeMillis());
404 					}
405 				}
406 				else {
407 					if(!delayedDataFlag && _cachedObject.getDelayedDataFlag()) {
408 						continue;
409 					}
410 					else {
411 						long time = _cachedObject.getDataTime();
412 						if((time >= fromTime) && (time <= toTime)) {
413 							arrayList.add(_cachedObject);
414 							_cachedObject.setActionTime(System.currentTimeMillis());
415 						}
416 					}
417 				}

429 		public void run() {
430 			int loopCount = 0;
431 			while(!interrupted()) {
432 				try {
433 					sleep(10000);
434 					// Alle Daten, die sich im Cache befinden (Jedes Element der Liste entspricht den gecachten Daten(Liste) einer BaseSubscriptionInfo)
435 					ArrayList list = new ArrayList(cache.values());
436 					for(int i = list.size() - 1; i > -1; --i) {
437 						// Alle Daten im Cache, die zu einer BaseSubscriptionInfo gehören
438 						LinkedList allCachedDataOfABaseSubscriptionInfo = (LinkedList)list.get(i);
439 						if(allCachedDataOfABaseSubscriptionInfo != null) {
440 							synchronized(allCachedDataOfABaseSubscriptionInfo) {
441 
442 								// Mit diesem Iterator wird zuerst das letzte Element genommen. Dann wird der Iterator erneut initialisiert, diesmal
443 								// wird vorwärts über die Liste gelaufen.
444 								ListIterator _iterator = allCachedDataOfABaseSubscriptionInfo.listIterator(allCachedDataOfABaseSubscriptionInfo.size());
445 								if(_iterator.hasPrevious()) {
446 									CachedObject _cachedObject = (CachedObject)_iterator.previous();
447 									if(_cachedObject != null) {
448 										// Wie lange darf sich ein Objekte im Cache befinden, bevor es gelöscht wird

453 										// Der Iterator wird vorwärts durchlaufen
