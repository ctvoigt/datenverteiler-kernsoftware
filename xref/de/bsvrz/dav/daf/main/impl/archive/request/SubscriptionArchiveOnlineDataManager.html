95  	/**
96  	 * Meldet sich als Empfänger von Datensätzen an und stellt über das Objekt <code>receiver</code> die Daten
97  	 * (Archivdaten+Online) zur Verfügung.
98  	 *
99  	 * @param receiver
100 	 * @param object
101 	 * @param dataDescription
102 	 * @param options
103 	 * @param historyType
104 	 * @param history
105 	 * @throws de.bsvrz.dav.daf.main.config.ConfigurationException
106 	 */
107 	public void subscribe(
108 			DatasetReceiverInterface receiver,
109 						  SystemObject object,
110 						  DataDescription dataDescription,
111 						  ReceiveOptions options,
112 						  HistoryTypeParameter historyType,
113 						  long history) throws ConfigurationException {
114 
115 		// Objekt erzeugen, das Onlinedaten entgegen nimmt und die Archivdaten vor die Onlinedaten mischt.
116 		
117 
118 
119 		final Receiver newReceiver = new Receiver(_archive, _connection, receiver, object, dataDescription, options, historyType, history);
120 		// damit das Receiverobjekt in die Map eingefügt werden kann
121 		final ReceiverKey key = new ReceiverKey(receiver, object, dataDescription);
122 
123 		// Anmelden
124 		_connection.subscribeReceiver(newReceiver, object, dataDescription, options, ReceiverRole.receiver());
125 
126 		// In einer Map ablegen, damit der Empfänger sich wieder abmelden kann
127 		synchronized (_receiverList) {
128 			_receiverList.put(key, newReceiver);
129 		}
130 	}

132 	public void unsubscribe(DatasetReceiverInterface receiver, SystemObject object, DataDescription dataDescription) throws ConfigurationException {
133 		final ReceiverKey key = new ReceiverKey(receiver, object, dataDescription);
134 
135 		synchronized (_receiverList) {
136 			final Receiver abortReceiver = _receiverList.remove(key);
137 			_connection.unsubscribeReceiver(abortReceiver, object, dataDescription);
138 			abortReceiver.unsubscribeReceiver();
139 		}
140 	}

234 		public Receiver(ArchiveRequestManager archive,
235 						ClientDavInterface connection,
236 						DatasetReceiverInterface receiverArchiveOnlineStream,
237 						SystemObject systemObject,
238 						DataDescription dataDescription,
239 						ReceiveOptions receiveOptions,
240 						HistoryTypeParameter historyType,
241 						long history) {
242 			_archive = archive;
243 			_connection = connection;
244 			_receiverArchiveOnlineStream = receiverArchiveOnlineStream;
245 			_systemObject = systemObject;
246 			_dataDescription = dataDescription;
247 			_receiveOptions = receiveOptions;
248 			_historyType = historyType;
249 
250 			// Wenn XX Datensätze vor dem aktuellen Datensatz gefordert werden muss +1 gerechnet werden,
251 			// weil der letzte Archivdatensatz gleich dem Onlinedatensatz ist.
252 			// Wenn 5 Archivdatensätze gefordert werden würden, würden 5 vom Archivsystem zurückgegeben,
253 			// aber es würden nur 4 der update-Methode übergeben, der fünfte wäre der erste Onlinedatensatz.
254 			// also wird ein Datensatz "mehr" angefordert, weil der letzte Archivdatensatz und der Onlinedatensatz
255 			// identisch sind.
256 			if (_historyType == HistoryTypeParameter.INDEX) {
257 				_history = history + 1;
258 			} else {
259 				_history = history;
260 			}
261 			// Thread mischt Archivdaten unter den Onlinestrom und verschickt diese
262 			_archiveThread = new ArchiveDataRequester(this, _archive, _systemObject, _dataDescription, _receiveOptions, _historyType, _history);
263 			Thread thread = new Thread(_archiveThread, "HistoryThread");
264 			thread.start();
265 		}

267 		public void update(ResultData results[]) {
268 			// Für Debugausgaben
269 			final String format = "dd.MM.yyyy HH:mm:ss,SSS";
270 			final DateFormat timeFormat = new SimpleDateFormat(format);
271 			// Sobald der erste Datensatz empfangen wird, wird dieser Zeitstempel als
272 			// Endzeitpunkt für die Archivanfrage benutzt. Onlinedaten, die zwischendurch empfangen werden, werden
273 			// gepuffert.
274 			// Wurden alle Archivdaten vor den Onlinestrom gemisch, werden die Datensätze nur noch durchgereicht.
275 			synchronized (_dataReceived) {
276 				if (_handOverOnline) {
277 					// Es wurden alle Archivdaten verschickt, der Puffer mit den zwischengespeicherten Online Datensätzen
278 					// wurde auch verschickt, also können die gerade empfangenden Daten durchgreicht werden.
279 					sendData(results);
280 				} else {
281 					// Wurde der erste Datensatz schon empfangen ?
282 					if (_receivedFirstOnlineDataSet) {
283 						// Der erste Datensatz wurde schon empfangen, also müssen diese Daten
284 						// gepuffert werden
285 						_onlineBuffer.add(results);
286 					} else {
287 						// Der erste Datensatz wurde noch nicht empfangen, also befindet er sich unter
288 						// den gerade empfangenen Daten. Dieser wird gespeichert und alle Datensätze
289 						//, mit dem ersten, werden in den Puffer abgelegt.
290 
291 						// Alles in den Puffer
292 						_onlineBuffer.add(results);
293 
294 						if (results[0].getData() != null) {
295 							// Der erste Onlinedatensatz steht zur Verfügung
296 							_firstOnlineDataSet = results[0];
297 							_receivedFirstOnlineDataSet = true;
298 
299 							_debug.finest("Datenzeit erster Onlinedatensatz: " + timeFormat.format(new Date(_firstOnlineDataSet.getDataTime())));
300 							// System.out.println("**** Datenzeit erster Onlinedatensatz: " + timeFormat.format(new Date(_firstOnlineDataSet.getDataTime())));
301 
302 							// Falls jemand auf den ersten Datensatz wartet, wird er benachrichtigt
303 							_dataReceived.notifyAll();
304 
305 						} else {
306 							
307 							_debug.info("Der erste Datensatz war null, der dafür benutzt werden sollte um den Andzeitpunkt der Archivanfrage zu bestimmen. Es wird auf den nächsten Datensatz gewartet");
308 							return;
309 						}
310 					}
311 				}
312 			} // synch
313 		}
314 
315 		/**
316 		 * Diese Methode wird aufgerufen, wenn der Empfänger abgemeldet wird. Weitere Datensätze, die zu dem Empfänger
317 		 * geschickt werden sollen, werden nicht weitergereicht. Falls der Thread, der Archivdaten anfordert, noch Archivdaten
318 		 * verschicken will, werden diese ebenfalls nicht durchgereicht.
319 		 */
320 		public void unsubscribeReceiver() {
321 			// Der Zugriff ist nicht synchronisiert, weil der Empfänger abgemeldet wurde und es somit völlig egal
322 			// ist, wer derzeit Daten senden dürfte, es sollen keine Daten mehr an den Empfänger durchgereicht werden.
323 			_unsubcribeReceiver = true;
324 			// Falls noch Archivdaten unterwegs sind, werden diese jetzt abgebrochen
325 			_archiveThread.cancelThread();
326 		}

340 		/**
341 		 * Verschickt Datensätze an den Vorgegebenen Empfänger, falls dieser die Verbindung noch nicht abgemeldet hat. Wurde
342 		 * die Verbindung abgemeldet, dann werden die Daten verworfen.
343 		 *
344 		 * @param datasetResult Archivdaten oder ResultData, die verschickt werden sollen
345 		 */
346 		public void sendData(Dataset[] datasetResult) {
347 			if (_unsubcribeReceiver == false) {
348 				// Es dürfen noch Daten verschickt werden, falls nicht, werden die Daten einfach verworfen
349 				_receiverArchiveOnlineStream.update(datasetResult);
350 			}
351 		}
352 
353 		/**
354 		 * Diese Methode wird aufgerufen, sobald alle Archivdaten vor den Onlinestrom gemischt wurden. Der Puffer, der die
355 		 * Onlinedaten zwischengespeichert hat, wird dann übertragen. Die Methode sperrt die update-Methode und setzt danach
356 		 * eine Variable, so dass alle folgenden Onlinedaten direkt durchgereicht werden, ohne gepuffert zu werden.
357 		 */
358 		private void sendOnlineBuffer() {
359 			synchronized (_dataReceived) {
360 //				System.out.println("*** Puffer leeren *** " + _onlineBuffer.size());
361 				for (int nr = 0; nr < _onlineBuffer.size(); nr++) {
362 					// Die Datensätze in der Reihenfolge ihres eintreffens verschicken
363 					final ResultData bufferdOnlineData[] = _onlineBuffer.get(nr);
364 					sendData(bufferdOnlineData);
365 				}
366 				// Die folgenden update-Aufrufe reichen die Onlinedaten direkt durch, ohne diese
367 				// zwischenzuspeichern
368 				_handOverOnline = true;
369 			}
370 		}
371 	}

373 	/**
374 	 * Klasse, die Archivdaten anfordert und diese vor einen Onlinedatensatz einmischt
375 	 */
376 	private final static class ArchiveDataRequester implements Runnable {
377 
378 		/**
379 		 * Enthält alle Daten und Methoden, die der Thread braucht um Archivdaten anzufodern und diese vor die aktuellen Daten
380 		 * zu mischen
381 		 */
382 		private final Receiver _receiver;
383 
384 		/**
385 		 * Erster Onlinedatensatz, dieser Datensatz dient als Endzeitpunkt für die Archivanfrage
386 		 */
387 		private ResultData _firstOnlineDataSet;
388 
389 		/**
390 		 * Speichert den letzten Datensatz einer Archivanfrage. Dieser wird benötigt, wenn bei einer Archivanfrage nicht alle
391 		 * benötigten Werte übertragen wurden. Es fehlen zum Beispiel die letzten drei Datensätze, weil diese gerade
392 		 * archiviert werden und das Archivsystem sie deshalb nicht mitgeschickt hat. Um an alle Daten zu kommen, würde erneut
393 		 * angefragt werden, ein paar Archivdatensätze dieser Anfrage könnte aber verworfen werden, weil sie bereits in der
394 		 * ersten Anfrage vorhanden waren. Damit diese Datensätze erkannt werden können, wird der letzte Datensatz
395 		 * gespeichert.
396 		 */
397 		private ArchiveData _lastReceivedDataSet = null;
398 
399 		/**
400 		 * Bezieht sich der Wert in der Variablen _history auf einen Zeitbereich oder einen Datenindex.
401 		 */
402 		private final HistoryTypeParameter _historyType;
403 
404 		/**
405 		 * Zeitbereich oder Index, der aus dem aus Archivsystem angefordert werden soll. Diese Variable wird im laufe der
406 		 * Archivanfrage angepaßt. Wenn 10 Archivdaten benötigt werden, aber es werden nur 5 empfangen, müssen in der nächsten
407 		 * Archivanfrage nur noch 5 Archivdaten angefordert werden. Das selbe passiert mit einer Anfrage, die sich auf Zeit
408 		 * bezieht, wenn 4 Stunden benötigt werden, es werden aber nur 3 geliefert, wird die nächste Archivanfrage auf 1
409 		 * Stunde relativ gestellt.
410 		 */
411 		private long _history;
412 
413 		/**
414 		 * Wird benötigt um zu entscheiden ob die Archidaten delta oder normal sein sollen
415 		 */
416 		private final ReceiveOptions _receiveOptions;
417 		/**
418 		 * Wird für die Archivanfrage gebraucht
419 		 */
420 		private final SystemObject _systemObject;
421 		/**
422 		 * Wird für die Archivanfrage gebraucht
423 		 */
424 		private final DataDescription _dataDescription;
425 		/**
426 		 * Archivsystem, an das die Anfrage gestellt werden soll
427 		 */
428 		private final ArchiveRequestManager _archive;

430 		/**
431 		 * Wurden schon einmal Archivdaten verschickt. Wird nur EOA vom Archiv empfangen und es wurden schon einmal
432 		 * Archivdaten verschickt, dann wird auf Daten gewartet. Wird hingegen EOA empfangen und es wurden noch nie
433 		 * Archivdaten verschickt, hat das Archiv keine Daten für diese Datenidentifikation.
434 		 */
435 		private boolean _archiveDataSend = false;
436 
437 		/**
438 		 * true = der Thread stellt seine Arbeit ein und meldet sich beim Archiv ab.
439 		 */
440 		private boolean _cancel = false;

442 		public ArchiveDataRequester(Receiver receiver,
443 									ArchiveRequestManager archive,
444 									SystemObject systemObject,
445 									DataDescription dataDescription,
446 									ReceiveOptions receiveOptions,
447 									HistoryTypeParameter historyType,
448 									long history) {
449 			_receiver = receiver;
450 			_archive = archive;
451 			_systemObject = systemObject;
452 			_dataDescription = dataDescription;
453 			_historyType = historyType;
454 			_history = history;
455 			_receiveOptions = receiveOptions;
456 		}

458 		public void run() {
459 			// warten, bis Onlinedaten zur Verfügung stehen
460 			Object dataReceivedLockObject = _receiver.getDataReceived();
461 			synchronized (dataReceivedLockObject) {
462 				while (_receiver.isReceivedFirstOnlineDataSet() == false) {
463 					// Es steht noch kein Datensatz zur Verfügung, also kann der Thread schlafen gelegt werden.
464 					// Wird er geweckt und der erste Datensatz steht zur Verfügung, kann er die Archivdaten anfordern
465 					try {
466 						dataReceivedLockObject.wait();
467 					} catch (InterruptedException e) {
468 					}
469 				}
470 
471 				// Der erste Datensatz steht zur Verfügung
472 				_firstOnlineDataSet = _receiver.getFirstOnlineDataSet();
473 				// Der Thread hat nun alle Daten die er braucht um arbeiten zu können. Der Synchblock kann verlassen werden
474 			}
475 
476 			// Der erste Onlinedatensatz wurde gefunden, also können jetzt die Archivdaten angefordert werden
477 
478 			// Wenn das Archiv keine Daten mehr zur Verfügung stelle kann, die vor den Onlinestrom geschrieben werden
479 			// können, wird diese Variable true.
480 			// Es müssen nicht zwangsläufig Daten empfangen worden sein, wenn es keine Archivdaten gibt, wird diese
481 			// Variable ebenfalls true.
482 			boolean allPossibleArchiveDataReceived = false;
483 
484 			while (allPossibleArchiveDataReceived == false) {
485 				// Es müssen zwei Fälle unterschieden werden.
486 				// Fall 1, "Es wurden noch keine Archivdaten angefordert": Bei XXX Datensätze vor
487 				// dem aktuellen Datensatz wird der Index benutzt, bei XXX ms vor dem aktuellen Datensatz wird der Zeitpunkt berechnet.
488 				// Fall 2, "Es wurden bereits Archivdaten angefordert": Der letzte empfangende Datensatz des Archivs ist bekannt
489 				// (_lastReceivedDataSet) und kann als Startindex benutzt werden
490 				final ArchiveTimeSpecification archiveTimeSpecification;
491 
492 				if (_lastReceivedDataSet != null) {
493 					// Der letzte Datensatz einer Archivanfrage ist bekannt.
494 					// Dieser Fall tritt ein, wenn bereits eine Archivanfrage gestellt wurde, aber nicht alle
495 					// Datensätze aus dem Archiv übertragen wurden. Dann wird erneut angefragt, diesmal wird
496 					// aber eine Archivanfrage auf den Datenindex gestellt. Als Anfangswert wird der Datenindex
497 					// des _lastReceivedDataSet benutzt, als Endindex wird der Datenindex des _firstOnlineDataSet
498 					// benutzt.
499 					archiveTimeSpecification = new ArchiveTimeSpecification(TimingType.DATA_INDEX, false, _lastReceivedDataSet.getDataIndex(), _firstOnlineDataSet.getDataIndex());
500 
501 				} else if (_historyType == HistoryTypeParameter.TIME) {
502 					final long archiveDataEndTime = _firstOnlineDataSet.getDataTime();
503 					final long archiveDataStartTime = archiveDataEndTime - _history;
504 					// Es sollen _history viele ms vor dem ersten Onlinedatensatz zurückgegeben werden
505 					archiveTimeSpecification = new ArchiveTimeSpecification(TimingType.DATA_TIME, false, archiveDataStartTime, archiveDataEndTime);
506 				} else {
