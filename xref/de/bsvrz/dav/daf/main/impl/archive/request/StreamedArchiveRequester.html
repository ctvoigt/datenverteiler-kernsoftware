114 	/**
115 	 * Dieser Konstruktor erzeugt ein StreamedArchiveRequester Objekt und meldet sich gleichzeitig auf der übergebenen Verbindung als Sender für Anfragen an. Die
116 	 * Anfragen werden an das Archiv geschickt, das ebenfalls übergeben wird. Die Anfragen enthalten einmal die Archivanfragen und als zweites Tickets, die dem
117 	 * Archiv erlauben weitere Datensätze zu verschicken.
118 	 *
119 	 * @param archiveConnection Eine Verbindung auf der Datensätze verschickt werden sollen
120 	 * @param archiveSystem     Das Archiv, für das die Datensätze bestimmt sind
121 	 */
122 	public StreamedArchiveRequester(
123 			ClientDavInterface archiveConnection, int timeOutArchiveRequest, SystemObject archiveSystem, short defaultSimulationVariant) {
124 		_connection = archiveConnection;
125 		_archiveSystem = archiveSystem;
126 		_requests = Collections.synchronizedMap(new HashMap());
127 
128 		// Legt den Zeitpunkt fest, bis zu dem maximal auf eine Verbidnung zum Archivsystem gewartet wird.
129 		// Wird dieser Zeitpunkt überschritten, so liefert die Methode isConnectionOk immer false zurück, wenn
130 		// keine Verbidnung besteht, es wird nicht gewartet.
131 		_timeOutArchiveRequest = new Date(System.currentTimeMillis() + timeOutArchiveRequest);
132 		_defaultSimulationVariant = defaultSimulationVariant;
133 
134 		// Konfiguration, alle Infos ueber das Objekt werden vom "Konfigurator" angefordert.
135 		final DataModel configuration = _connection.getDataModel();
136 
137 		try {
138 			// Verbindung zum Archiv aufbauen
139 
140 			// Auf die Attributgruppe anmelden
141 			final AttributeGroup attributeGroup;
142 			attributeGroup = configuration.getAttributeGroup("atg.archivAnfrageSchnittstelle");
143 
144 			// In diesem Aspekt stehen die Anfragen/Tickets der Empfängerapplikation, diese müssen gesendet werden
145 			final Aspect aspectSender = configuration.getAspect("asp.anfrage");
146 
147 			final short simulationVariant = 0;
148 
149 			// final DataDescription dataDescriptionSender = new DataDescription(attributeGroup, aspectSender);
150 			_dataDescriptionSender = new DataDescription(attributeGroup, aspectSender, simulationVariant);
151 
152 			// Objekt, das ein ClientSenderInterface implementiert
153 			_clientSender = new ClientSender();
154 
155 			// Auf der connection als Sender anmelden
156 			_connection.subscribeSender(_clientSender, _archiveSystem, _dataDescriptionSender, SenderRole.sender());
157 		}
158 		catch(OneSubscriptionPerSendData oneSubscriptionPerSendData) {
159 			oneSubscriptionPerSendData.printStackTrace();
160 		}
161 
162 		// Objekt, das Anmeldungen auf Online+Archivdaten zuläßt erzeugen
163 		_subscriptionArchiveOnlineDataManager = new SubscriptionArchiveOnlineDataManager(this, _connection);
164 	}

166 	/**
167 	 * Diese Methode stellt eine Anfrage an das Archiv. Da eine Liste mit Objekten, die angefragt werden sollen, zur Verfügung gestellt wird, werden auch soviele
168 	 * Streams bereitgestellt, wie die Liste Einträge hat.
169 	 *
170 	 * @param priority Mit welcher Priorität soll die Anfrage beantwortet werden (hohe, mittlere, niedrige Priorität)
171 	 * @param specs    Eine Liste von Objekten, die alle Informationen enthalten, die zur Bearbeitung der Archivanfrage nötig sind. Für jedes Objekt der Liste wird
172 	 *                 ein Stream bereitgestellt, der die geforderten Informationen enthält.Wird eine leere Liste übergeben, wird das Objekt, das die Antwort auf
173 	 *                 diese Anfrage enthält, ein Array mit Streams zurückgegbene, das ebenfalls leer ist.
174 	 *
175 	 * @return Ein Objekt, das die Möglichkeit bietet zu prüfen ob die Anfrage erfolgreich war oder ob die Anfrage fehlgeschlagen ist. War die Anfrage erfolgreich,
176 	 *         so kann das Objekt die Streams zur weiteren Bearbeitung weitergeben. War die Liste <code>specs</code> leer, ist das Array, das die Streams
177 	 *         darstellt, ebenfalls leer.
178 	 *
179 	 * @throws IllegalStateException Das Archiv, an das die Anfrage gestellt wurde, kann nicht erreicht werden, die Anfrage wird verworfen.
180 	 */
181 	public ArchiveDataQueryResult request(ArchiveQueryPriority priority, List<ArchiveDataSpecification> specs) throws IllegalStateException {
182 
183 		if(specs != null && specs.size() > 0) {
184 			// Am Anfang wird geprüft, ob das Archivsystem verfügbar ist, nur wenn eine Verbidnung besteht wird die
185 			// Anfrage überhaupt bearbeitet
186 			try {
187 				if(isConnectionOk(_timeOutArchiveRequest) == false) {
188 					_debug.warning(
189 							"Die Applikation: " + _connection.getLocalApplicationObject().getNameOrPidOrId() + " will eine Archivanfrage beim Archivsystem: "
190 							+ _archiveSystem.getNameOrPidOrId() + " stellen, bekommt aber keine Rückmeldung vom Archivsystem. Die Archivanfrage wird verworfen."
191 					);
192 					throw new IllegalStateException("Das Archivsystem " + _archiveSystem.getNameOrPidOrId() + " kann nicht erreicht werden");
193 				}
194 			}
195 			catch(InterruptedException e) {
196 				e.printStackTrace();
197 			}
198 
199 			// über diese ID ist jede Archivanfrage eindeutig zu identifizieren
200 			// Als Key wird ein laufender Index und das Archiv (für das die Anfrage ist) genommen
201 			final int indexOfRequest;
202 			synchronized(this) {
203 				indexOfRequest = _indexOfRequest;
204 				_indexOfRequest++;
205 			}
206 
207 			ArchiveQueryID archiveQueryID = new ArchiveQueryID(indexOfRequest, _archiveSystem);
208 
209 			Query archiveQuery = new Query(archiveQueryID, priority, specs, _receiveBufferSize, this, _defaultSimulationVariant);
210 			// Die Anfrage speichern, sobald Daten für diese Anfrage kommen, kann die Archivanfrage über ihren Index
211 			// identifiziert werden. Der Index wird mit der Nachricht versandt.
212 			_requests.put(archiveQuery.getArchiveRequestID(), archiveQuery);
213 
214 			archiveQuery.initiateArchiveRequest();
215 
216 			return archiveQuery;
217 		}
218 		else {
219 			if(specs == null) {
220 				// Es wurde <code>null</code> übergeben
221 				throw new IllegalArgumentException("Die Liste, die eine Archivanfrage spezifiziert, war null");
222 			}
223 			else {
224 				// Die Liste enthält keine Elemente, also ein Array der Größe 0 zurückgeben
225 				ArchiveDataQueryResult queryResult = new ArchiveDataQueryResult() {
226 					public ArchiveDataStream[] getStreams() throws InterruptedException, IllegalStateException {
227 						return new ArchiveDataStream[0];  //To change body of implemented methods use File | Settings | File Templates.
228 					}
229 
230 					public boolean isRequestSuccessful() throws InterruptedException {
231 						return true;
232 					}
233 
234 					public String getErrorMessage() throws InterruptedException {
235 						return "";
236 					}
237 				};
238 				return queryResult;
239 			}
240 		}
241 	}

331 	/**
332 	 * Diese Methode analisiert die Daten und ruft die passenden Methoden auf, die die Daten weiter reichen oder verarbeiten um sie dann weiter zu reichen. Falls
333 	 * ein Paket vom falschen Typ empfangen wird, wird eine entsprechende Exception geworfen.
334 	 *
335 	 * @param data Datensatz eines Archivs
336 	 */
337 	public void dataReceiver(Data data) {
338 		// Von welchem Typ ist das Paket
339 		int dataType = data.getUnscaledValue("nachrichtenTyp").intValue();
340 		_debug.fine("Ein StreamedArchiveRequester hat einen Datensatz empfangen, nachrichtenTyp: " + dataType);
341 
342 		switch(dataType) {
343 			case 1: {
344 				// Das Paket ist vom Typ "Anfrage". Eigentlich sollte so ein Paket niemals ankommen
345 				throw new IllegalArgumentException("Ein StreamedArchiveRequester hat ein Paket mit dem Typ 'Anfrage' erhalten.");
346 			}
347 			case 2: {
348 				// Ein Paket vom Typ "AnfrageErgebnis". Dies ist ein Antwortpaket nach dem ein "Anfragepaket" verschickt wurde
349 				intitiateArchiveQueryResponse(data);
350 				break;
351 			}
352 			case 3: {
353 				// Ein Paket vom Typ "StreamDaten". Dies sind Nutzdaten, die für eine bestimmte Archivanfrage gedacht sind.
354 				archiveDataResponse(data);
355 				break;
356 			}
357 			case 4: {
358 				// Ein Paket vom Typ "StreamSteuerung" wurde empfangen. Diese Pakete enthalten für gewöhnlich
359 				// Ticktes, die für den StreamMultiplexer gedacht sind. Also sind diese für
360 				// für Archivanfragen gar nicht zu gebrauchen, da diese Tickets verschicken.
361 				throw new IllegalArgumentException(
362 						"Ein StreamedArchiveRequester hat ein Ticket empfangen. In einer Archivanfrage"
363 						+ "arbeitet aber ein StreamDemultiplexer und kein StreamMultiplexer."
364 				);
365 			}
366 			case 6: {
367 				requestInfoResponse(data);
368 				break;
369 			}
370 			case 8: {
371 				deletePersistentDataResponse(data);
372 				break;
373 			}
374 			case 10: {
375 				// Eine Archivantwort auf einen Speicherauftrag
376 				savePersistentDataResponse(data);
377 				break;
378 			}
379 			case 12: {
380 				// Eine Archivantwort auf den Auftrag Daten aus der Sicherung in den direkten Zugriff
381 				// des Archivsystems zu bringen
382 				restoreDataResponse(data);
383 				break;
384 			}
385 			case 14: {
386 
387 				// Die Antwort Nummer 14 wird für 2 Auftragstypen benötigt.
388 				// Einmal als Antwort für einen Auftrag, der den Löschzeipunkt verschieben soll
389 				// und einmal als Antwort für einen Löschauftrag, der sich auf verschiedene
390 				// Zeitbereiche bezieht.
391 
392 				// Für welche Archivanfrage ist das Paket
393 				final int indexOfRequest = data.getUnscaledValue("anfrageIndex").intValue();
394 
395 				// Welches Archiv hat die Daten verschickt
396 				final SystemObject archiveReference = data.getReferenceValue("absender").getSystemObject();
397 
398 				// Der Schlüssel zu dem Objekt
399 				ArchiveQueryID hashObject = new ArchiveQueryID(indexOfRequest, archiveReference);
400 
401 				if(_requests.get(hashObject) instanceof IncreaseDeleteTime) {
402 					increaseDeleteTimeResponse(data);
403 					break;
404 				}
405 				else {
406 					deleteDataResponse(data);
407 					break;
408 				}
409 			}
410 			case 16: {
411 				// Eine Archivantwort auf einen Auftrag zum Abgleich der Archivverwaltungsinformationen
412 				archiveFileSaverAlignmentResponse(data);
413 				break;
414 			}
415 			case 18: {
416 				requestDataResponse(data);
417 				break;
418 			}
419 			case 20: {
420 				requestDataResponse(data);
421 				break;
422 			}
423 		}
424 	}

426 	/**
427 	 * Das Archiv Antwortet auf die gestellt Archivanfrage. Somit muss die Archivanfrage, die auf diese Antwort wartet, benachrichtigt werden, dass das Archiv
428 	 * geantwortet hat.
429 	 *
430 	 * @param data Datensatz
431 	 */
432 	private void intitiateArchiveQueryResponse(Data data) {
433 
434 		// Für welche Archivanfrage ist das Paket
435 		final int indexOfRequest = data.getUnscaledValue("anfrageIndex").intValue();
436 
437 		// Welches Archiv hat die Daten verschickt
438 		final SystemObject archiveReference = data.getReferenceValue("absender").getSystemObject();
439 
440 		// Als erstes muss die Archivanfrage "geholt" werden
441 		Query query = getQuery(indexOfRequest, archiveReference);
442 		if(query == null) {
443 			_debug.warning("Unerwartete Antwort (bzgl. einer Archivanfrage) vom Archivsystem empfangen", data);
444 			return;
445 		}
446 
447 		assert query != null : "Index der Anfrage: " + indexOfRequest + " ;Objekt, das zu der Anfrage gehört: " + archiveReference;
448 
449 		// aus den Daten das byte-Array anfordern. In dem Array sind die Informationen, ob die Anfrage geklappt,
450 		// kodiert.
451 		byte[] queryResponse = data.getUnscaledArray("daten").getByteArray();
452 
453 		// Die Antwort wird an die entsprechende Anfrage weitergeleitet
454 		query.initiateArchiveResponse(queryResponse);
455 	}

473 	/**
474 	 * Diese Methode leitet die Nutzdaten, die für eine bestimmte Archivanfrage gedacht sind, weiter.
475 	 *
476 	 * @param data Nutzdaten, die vom Archiv geschickt wurden
477 	 */
478 	void archiveDataResponse(Data data) {
479 
480 		// Für welche Archivanfrage ist das Paket
481 		final int indexOfRequest = data.getUnscaledValue("anfrageIndex").intValue();
482 		// Von welchem Archiv sind die Daten
483 		final SystemObject archiveReference = data.getReferenceValue("absender").getSystemObject();
484 
485 		Query query = getQuery(indexOfRequest, archiveReference);
486 		if(query != null) {
487 			// Die Nutzdaten an die Archivanfrage weiter geben
488 			query.archiveDataResponse(data.getUnscaledArray("daten").getByteArray());
489 		}

510 	 * @throws SendSubscriptionNotConfirmed Es liegt keine positive Sendesteuerung vom Datenverteiler für die zu versendenden Daten vor
511 	 * @throws IllegalStateException        Die Verbindng zum Archive wurde abgebrochen, gleichzeitig wird versucht Datenpakete an das Archiv zu verschicken

513 	void createArchivRequestResultData(ArchiveQueryID archiveRequest, int messageType, byte[] dataArray)

516 		DataModel configuration = _connection.getDataModel();

521 		Aspect aspectTicket = configuration.getAspect("asp.anfrage");
522 
523 		DataDescription dataDescription = new DataDescription(atg, aspectTicket);
524 
525 		// Dieses Ticket wird physich über den DaV verschickt und auf der "Empfänger"seite ausgepackt
526 		// (der Empfänger ist das Archiv)
527 		Data requestData = _connection.createData(dataDescription.getAttributeGroup());
528 
529 		// Der Index der Anfrage
530 		requestData.getUnscaledValue("anfrageIndex").set(archiveRequest.getIndexOfRequest());
531 
532 		// Welche Applikation verschickt diesen Datensatz
533 
534 		// requestData.getReferenceValue("absender").setSystemObject(archiveRequest.getObjectReference());
535 		requestData.getReferenceValue("absender").setSystemObject(_connection.getLocalApplicationObject());
536 
537 		// von welchem Typ ist das Paket
538 		requestData.getUnscaledValue("nachrichtenTyp").set(messageType);
539 
540 		// byte-Array kopieren, falls es mitgegeben wurde, TBD was ist , wenn das array leer ist, null geht nicht, gibt exception !
541 		if(dataArray != null) {
542 			// Es gibt ein Array
543 
544 			// Die Länge des Arrays setzen
545 			final int lengthDataArray = dataArray.length;
546 
547 			requestData.getUnscaledArray("daten").setLength(lengthDataArray);
548 			// Byteweise kopieren
549 
550 			for(int nr = 0; nr < lengthDataArray; nr++) {
551 				requestData.getUnscaledArray("daten").getValue(nr).set(dataArray[nr]);
552 			}
553 		}

555 		ResultData result = new ResultData(_archiveSystem, dataDescription, System.currentTimeMillis(), requestData);
556 
557 		synchronized(_lockState) {
558 			if(_state != 0) {
559 				// Es gibt keinen Empfänger für den Datensatz, somit ist das Archiv nicht mehr zu erreichen.

580 	/**
581 	 * Diese Methode prüft, ob gesendet werden darf. Ist die Sendesteuerung negativ, wird eine Zeitspanne gewartet, ist die Sendesteuerung dann noch immer negativ,
582 	 * wird false zurückgeliefert.
583 	 *
584 	 * @param timeOut Zeitpunkt, bis zu dem gewartet wird, um eine Verbindung aufzubauen
585 	 *
586 	 * @return true = positive Sendesteuerung vorhanden; false = auch nach Ablauf der vorgegebenen Zeitspanne ist keine positive Sendesteuerung vorhanden
587 	 */
588 	private boolean isConnectionOk(Date timeOut) throws InterruptedException {
589 		synchronized(_lockState) {
590 			final TimeoutTimer timer = new TimeoutTimer(timeOut);
591 			// Solange die Sendesteuerung nicht auf positiv geändert wurde oder aber der Timer ist
592 			// abgelaufen, wird nichts gemacht.
593 			while(_state != ClientSenderInterface.START_SENDING && timer.isTimeExpired() == false) {
594 				_debug.fine(
595 						"Die Applikation wartet, bis das Archiv bereit ist Anfragen/Aufträge . Applikation: " + _connection.getLocalApplicationObject()
596 						+ " Archiv: " + _archiveSystem
597 				);
598 				// Falls "0" zurückgegeben wird, würde der Thread für immer schlafen
599 				_lockState.wait(timer.getRemainingTime() + 1);
600 			}
601 
602 			if(_state == ClientSenderInterface.START_SENDING) {
603 				// Es darf nun gesendet werden
604 				_debug.finest(
605 						"Die Applikation kann Archivanfragen stellen . Applikation: " + _connection.getLocalApplicationObject() + " Archiv: " + _archiveSystem
606 				);
607 				return true;
608 			}
609 			else {
610 				_debug.info(
611 						"Die Applikation kann keine Archivanfragen stellen, da die Sendesteuerung weiterhin negativ ist . Applikation: "
612 						+ _connection.getLocalApplicationObject() + " Archiv: " + _archiveSystem
613 				);
614 				return false;
615 			}
616 		}

644 		 * @param dataDescription Welche DataDescription

648 			synchronized(_lockState) {
