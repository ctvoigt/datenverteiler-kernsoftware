178 	/** Mit dieser Methode wird die initiale Archivanfrage verschickt. */
179 	public void initiateArchiveRequest() {
180 		// Beim Archiv Archivanfrage stellen. Das Archiv muss auf die erste Nachricht erst Antworten, solange wird
181 		// jede Methode dieser Klasse blockieren
182 		try {
183 			// Alle Parameter müssen in einen Datensatz verpackt werden.
184 			// Die Priorität und die "spec" werden dabei in einem byte-Array des Datensatzes übertragen.
185 			// Die Daten werden mit einem Serializer verpackt und entsprechend auf der Empfangsseite mit einem
186 			// Deserializer entpackt. Damit die Daten mit der richtigen Version des Deserializers ausgepackt werden,
187 			// wird diese mitgeschickt. Die ersten 4 Bytes des byte-Arrays sind dabei die Versionsnummer
188 			// des Serializers und die entsprechende Version muss der Deserializer benutzen.
189 			// Die ersten 4 Bytes werden "von Hand" ein/ausgepackt, ohne den Serializer/Deserializer zu benutzen.
190 			// (somit ist die Versionsnummer immer unabhängig verfügbar, egal welche Version des Serializer/Deserializer
191 			// die Applikation zur Verfügung hat)
192 
193 			// Ein byte-Array erzeugen, in dem <code>priority</code> und <code>spec</code> gespeichert sind.
194 
195 			ByteArrayOutputStream out = new ByteArrayOutputStream();
196 			Serializer serializer = SerializingFactory.createSerializer(out);
197 
198 			try {
199 
200 				// Das erste int spiegelt die Priorität der Anfrage wieder
201 				serializer.writeInt(_priority.getCode());
202 
203 				// Als nächstes wird der receiveBufferSize gespeichert, dies ist ein int
204 				serializer.writeInt(_receiveBufferSize);
205 
206 				// Anzahl Einträge der Liste speichern, beim auspacken ist dann bekannt, wie viele Einträge
207 				// entpackt werden müssen.
208 				serializer.writeInt(_spec.size());
209 
210 				// Jede Anfrage, die in der spec-Liste steht, in das byte-Array speichern
211 				for(int i = 0; i < _spec.size(); i++) {
212 
213 					ArchiveDataSpecification archiveDataSpecification = (ArchiveDataSpecification)_spec.get(i);
214 
215 					{
216 						// ArchiveTimeSpecification anfordern und einzeln speichern
217 						ArchiveTimeSpecification archiveTimeSpecification = archiveDataSpecification.getTimeSpec();
218 
219 						// TimingType als Byte speichern, dieser String kann beim entkodieren
220 						// benutzt werden, um das alte Objekt wieder zu erzeugen.
221 						// 1 = DATA_TIME
222 						// 2 = ARCHIVE_TIME
223 						// 3 = DATA_INDEX
224 
225 						TimingType timingType = archiveTimeSpecification.getTimingType();
226 
227 						if(timingType == TimingType.DATA_TIME) {
228 							serializer.writeByte(1);
229 						}
230 						else if(timingType == TimingType.ARCHIVE_TIME) {
231 							serializer.writeByte(2);
232 						}
233 						else if(timingType == TimingType.DATA_INDEX) {
234 							serializer.writeByte(3);
235 						}
236 
237 						// boolean startRelative speichern. 0 = false, 1 = true, wird als Byte gespeichert
238 						if(archiveTimeSpecification.isStartRelative() == true) {
239 							serializer.writeByte(1);
240 						}
241 						else {
242 							serializer.writeByte(0);
243 						}
244 
245 						// long intervalStart speichern
246 						serializer.writeLong(archiveTimeSpecification.getIntervalStart());
247 
248 						// long intervalEnd speichern
249 						serializer.writeLong(archiveTimeSpecification.getIntervalEnd());
250 					}
251 
252 					// ArchiveDataKindCombination speichern
253 					ArchiveDataKindCombination archiveDataKindCombination = archiveDataSpecification.getDataKinds();
254 
255 					{
256 						// Jede boolean Variable wird als Byte gespeichert (0 = false; 1 = true)
257 
258 						if(archiveDataKindCombination.isOnline() == true) {
259 							serializer.writeByte(1);
260 						}
261 						else {
262 							serializer.writeByte(0);
263 						}
264 
265 						if(archiveDataKindCombination.isOnlineDelayed() == true) {
266 							serializer.writeByte(1);
267 						}
268 						else {
269 							serializer.writeByte(0);
270 						}
271 
272 						if(archiveDataKindCombination.isRequested() == true) {
273 							serializer.writeByte(1);
274 						}
275 						else {
276 							serializer.writeByte(0);
277 						}
278 
279 						if(archiveDataKindCombination.isRequestedDelayed() == true) {
280 							serializer.writeByte(1);
281 						}
282 						else {
283 							serializer.writeByte(0);
284 						}
285 					}
286 
287 					// ArchiveOrder speichern
288 					ArchiveOrder archiveOrder = archiveDataSpecification.getSortOrder();
289 
290 					{
291 						// Der int Wert kann benutzt werden, um das Objekt wieder herzustellen
292 						serializer.writeInt(archiveOrder.getCode());
293 					}
294 
295 					// ArchiveRequestOption speichern
296 					ArchiveRequestOption archiveRequestOption = archiveDataSpecification.getRequestOption();
297 
298 					{
299 						// Der int Wert kann benutzt werden, um das Objekt wieder herzustellen
300 						serializer.writeInt(archiveRequestOption.getCode());
301 					}
302 
303 					// DataDescription speichern
304 					DataDescription dataDescription = archiveDataSpecification.getDataDescription();
305 
306 					{
307 						if(dataDescription.getAttributeGroup() == null || dataDescription.getAspect() == null) {
308 							throw new IllegalArgumentException("Ein Element der Datenidentifikation ist null. " + dataDescription);
309 						}
310 						// Attributgruppe speichern
311 						serializer.writeObjectReference(dataDescription.getAttributeGroup());
312 
313 						// Aspekt speichern
314 						serializer.writeObjectReference(dataDescription.getAspect());
315 
316 						// short simulationVariant speichern.
317 						// Dabei tritt ein Sonderfall auf, wenn die Simulationsvariante auf
318 						// <code>NO_SIMULATION_VARIANT_SET</code> , dies entspricht einer
319 						// <code>-1</code>, gesetzt wurde. In diesem Fall muss der default Wert
320 						// für die Simulationsvariante gewählt werden. Der default-Wert wurde im Konstruktor dieser
321 						// Klasse übergeben.
322 						if(dataDescription.getSimulationVariant() != -1) {
323 							serializer.writeShort(dataDescription.getSimulationVariant());
324 						}
325 						else {
326 							// Die Simulationsvariante wurde auf -1 gesetzt, somit muss der default-Wert benutzt werden.
327 							serializer.writeShort(_defaultSimulationVariant);
328 						}
329 					}
330 
331 					// SystemObject speichern
332 					SystemObject systemObject = archiveDataSpecification.getObject();
333 
334 					{
335 						// SystemObject speichern
336 						serializer.writeObjectReference(systemObject);
337 					}
338 				}
339 			}
340 			catch(Exception e) {
341 				throw new IllegalStateException("Fehler beim verschicken der initialen Archivanfrage: " + e);
342 			}
343 
344 			// byte-Array erzeugen
345 			final byte[] data = out.toByteArray();
346 
347 			// Die Versionsnummer des Serializer speichern
348 			final int serializerVersion = serializer.getVersion();
349 
350 			// Ein neues byte-Array anlegen, dieses ist 4 Bytes größer als das alte, da die Version des Serializer
351 			// gespeichert werden muss
352 			final byte[] dataAndVersion = new byte[data.length + 4];
353 
354 			// Das höherwärtigste Byte steht in Zelle 0
355 			dataAndVersion[0] = (byte)((serializerVersion & 0xff000000) >>> 24);
356 			dataAndVersion[1] = (byte)((serializerVersion & 0x00ff0000) >>> 16);
357 			dataAndVersion[2] = (byte)((serializerVersion & 0x0000ff00) >>> 8);
358 			dataAndVersion[3] = (byte)(serializerVersion & 0x000000ff);
359 
360 			// Das data-Array hinter die Versionsnummer kopieren
361 			System.arraycopy(data, 0, dataAndVersion, 4, data.length);
362 
363 			// Datensatz erzeugen und verschicken
364 			_streamedArchiveRequester.createArchivRequestResultData(_archiveRequestID, 1, dataAndVersion);
365 		}
366 		catch(DataNotSubscribedException e) {
367 			e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
368 		}
369 		catch(SendSubscriptionNotConfirmed sendSubscriptionNotConfirmed) {
370 			sendSubscriptionNotConfirmed.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
371 		}
372 		catch(ConfigurationException e) {
373 			e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
374 		}
375 	}

377 	/**
378 	 * Diese Methode blockiert solange, bis das Archiv auf die erste Archivanfrage geantwortet hat. In der Archivantwort wird dann geklärt, ob das Archiv die
379 	 * geforderten Nutzdaten zur Verfügung stellen kann (falls nicht, warum nicht).
380 	 *
381 	 * @return true, falls das Archiv die Anfrage bearbeiten kann. false = das Archiv kann die Anfrage nicht bearbeiten, somit steht eine errorMessage zur
382 	 *         Verfügung.
383 	 */
384 	public boolean isRequestSuccessful() throws InterruptedException {
385 
386 		// Diese Frage darf erst beantwortet werden, wenn diese Archivanfrage auch vom Archiv beantwortet wurde.
387 		// Also legen wir diese Anfrage solange schlafen, bis diese Antwort kommt.
388 		// Die Antwort empfängt der StreamedArchiveRequester, dieser kann auch wieder Zugriff auf dieses Objekt
389 		// bekommen und somit alle wartenden aufwecken (also auch diesen Thread, der auf dieses Ereignis wartet).
390 
391 		synchronized(this) {
392 			while(_blocking) {
393 				this.wait();
394 			}
395 
396 			return _requestSuccessful;
397 		} // synchronized (this)
398 	}

400 	public String getErrorMessage() throws InterruptedException {
401 
402 		// Dieser Methodenaufruf läßt die Anfrage nach den Streams solange warten, bis
403 		// es zu einer Antwort des Archivs gekommen ist, dass die angeforderten Nutzdaten
404 		// vorhanden sind oder das ein Fehler aufgetreten ist.
405 		if(isRequestSuccessful() == true) {
406 			return "Die Archivanfrage (" + _archiveRequestID.getIndexOfRequest() + ") war erfolgreich";
407 		}
408 		else {
409 			return _errorMessage;
410 		}
411 	}

414 	 * Diese Methode gibt alle Streams, die zu einer Archivanfrage gehören, zurück. Die geforderten Archivdaten liegen in den Stream bereit und können mit der
415 	 * entsprechenden Methode angefordert werden.
416 	 *
417 	 * @return Alle Streams, die zu einer Archivanfrage gehören
418 	 *

420 	 */
421 	public ArchiveDataStream[] getStreams() throws IllegalStateException, InterruptedException {
422 
423 		// Dieser Methodenaufruf läßt die Anfrage nach den Streams solange warten, bis
424 		// es zu einer Antwort des Archivs gekommen ist. Entweder sind die angeforderten Nutzdaten
425 		// vorhanden oder es ist ein Fehler aufgetreten ist.
426 
427 		if(isRequestSuccessful() == true) {
428 			return _arrayOfStreams;
429 		}
430 		else {
431 			throw new IllegalStateException("Es kam zu einem Fehler bei der Archivanfrage(" + _archiveRequestID.getIndexOfRequest() + "): " + _errorMessage);
432 		}
433 	}

435 	/**
436 	 * Diese Methode wird vom StreamedArchivRequester aufgerufen, sobald das Archiv auf die erste Archivanfrage eines Auftrags antwortet. Diese Antwort bedeutet,
437 	 * dass die Anfrage bearbeitet werden kann oder das ein Fehler aufgetreten ist.
438 	 *
439 	 * @param queryResponse Die Antwort des Archivs, die Informationen im byte-Array codiert
440 	 */
441 	public void initiateArchiveResponse(byte[] queryResponse) {
442 
443 		synchronized(this) {
444 
445 			try {
446 				InputStream in = new ByteArrayInputStream(queryResponse);
447 
448 				//deserialisieren
449 				Deserializer deserializer = SerializingFactory.createDeserializer(in);
450 
451 				// Eine 0 bedeutet, dass die Archivanfrage nicht bearbeitet werden kann, bei einer 1 ist alles in Ordnung
452 				final byte requestSuccessfulFlag = deserializer.readByte();
453 
454 				final boolean requestSuccessful;
455 				if(requestSuccessfulFlag == 0) {
456 					requestSuccessful = false;
457 				}
458 				else {
459 					requestSuccessful = true;
460 				}
461 
462 				_debug.finer("erste Archivantwort: requestSuccessful = " + requestSuccessful);
463 
464 				// Die errorMessage einlesen. War alles in Ordnung, ist der String leer ("")
465 				final String errorMessage = deserializer.readString();
466 				_debug.finer("erste Archivantwort: errorMessage = " + errorMessage);
467 
468 				// Den errechneten blockingFactor einlesen
469 				final int blockingFactor = deserializer.readInt();
470 				_debug.finer("erste Archivantwort: blockingFactor = " + blockingFactor);
471 
472 
473 				_requestSuccessful = requestSuccessful;
474 				_errorMessage = errorMessage;
475 				_blockingFactor = blockingFactor;
476 
477 				// wartende Threads "befreien" und verhindern, dass andere Threads anfangen zu warten
478 				_blocking = false;
479 
480 				// Nur wenn die Archivanfrage erfolgreich war werden die Streams angelegt und können
481 				// angefordert werden.
482 				if(_requestSuccessful == true) {
483 
484 					// StreamDemultiplexer und StreamDemultiplexerDirector anlegen
485 					DirectorStreamDemultiplexer directorStreamDemultiplexer = new DirectorStreamDemultiplexer(_archiveRequestID, _streamedArchiveRequester);
486 					_streamDemultiplexer = new StreamDemultiplexer(_numberOfStreams, _blockingFactor, directorStreamDemultiplexer);
487 
488 					_arrayOfStreams = new DataStream[_numberOfStreams];
489 
490 					// Der Index des Arrays identifiziert auch den Stream
491 					for(int indexOfStream = 0; indexOfStream < _arrayOfStreams.length; indexOfStream++) {
492 						ArchiveDataSpecification archiveDataSpecification = (ArchiveDataSpecification)_spec.get(indexOfStream);
493 						DataStream dataStream = new DataStream(indexOfStream, _streamDemultiplexer, archiveDataSpecification, this);
494 						_arrayOfStreams[indexOfStream] = dataStream;
495 					}
496 				}
497 
498 				// Alle Threads, die auf diese Nachricht warten aufwecken. Da _blocking gesetzt wurde, kann nun
499 				// kein Thread mehr in den status "schlafend" gesetzt werden.
500 				this.notifyAll();
501 			}
502 			catch(IOException e) {
503 				e.printStackTrace();
504 			}
505 		} // synchronized(this)
506 	}

523 	/**
524 	 * Diese Methode wird von einem Objekt der Klasse Query aufgerufen sobald der Stream des Objekts ein null-Paket empfängt oder die Empfängerapplikation abort
525 	 * aufruft. Das null-Paket bedeutet, dass das Archiv für diesen Stream keine Archivdaten mehr zur Verfügung hat. Der Aufruf von abort bedeutet, dass die
526 	 * Empfängeraplikation keine Archivdaten für diesen Stream mehr benötigt.
