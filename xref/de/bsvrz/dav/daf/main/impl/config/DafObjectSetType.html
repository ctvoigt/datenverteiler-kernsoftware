140 	/**
141 	 * Lädt den konfigurierenden Datensatz mit den Eigenschaften dieses Objekts aus der Konfiguration und speichert ihn für weitere Aufrufe dieser Methode.

145 	private synchronized Data getProperties() {
146 		if(_data == null) {
147 			_data = getConfigurationData(getDataModel().getAttributeGroup("atg.mengenTypEigenschaften"));
148 			if(_data == null) {
149 				throw new IllegalStateException(
150 						"Konfigurierender Datensatz 'atg.mengenTypEigenschaften' am " + "MengenTyp " + getPidOrNameOrId() + " konnte nicht geladen werden"
151 				);
152 			}
153 		}
154 		return _data;
155 	}

157 	public ReferenceType getReferenceType() {
158 		Data properties = getProperties();
159 		String referenceType = properties.getTextValue("referenzierungsart").getValueText();
160 		if(referenceType.equals("Assoziation")) {
161 			return ReferenceType.ASSOCIATION;
162 		}
163 		else if(referenceType.equals("Aggregation")) {
164 			return ReferenceType.AGGREGATION;
165 		}
166 		else if(referenceType.equals("Komposition")) {
167 			return ReferenceType.COMPOSITION;
168 		}
169 		else {
170 			throw new IllegalStateException("unbekannte Referenzierungsart am " + getPidOrNameOrId() + ": " + referenceType);
171 		}
172 	}

175 	public final List<SystemObjectType> getObjectTypes() {
176 		if(_objectTypes == null) {
177 			final ArrayList<SystemObjectType> objectTypes = new ArrayList<SystemObjectType>();
178 			ObjectSet objectTypeSet = getObjectSet("ObjektTypen");
179 			List<SystemObject> objectTypeElements = objectTypeSet.getElements();
180 			for(SystemObject objectTypeElement : objectTypeElements) {
181 				if(objectTypeElement instanceof SystemObjectType) {
182 					SystemObjectType systemObjectType = (SystemObjectType)objectTypeElement;
183 					objectTypes.add(systemObjectType);
184 				}
185 			}
186 			_objectTypes = Collections.unmodifiableList(objectTypes);
187 		}
188 		return _objectTypes;
189 	}
