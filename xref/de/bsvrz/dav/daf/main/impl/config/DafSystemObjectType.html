103 	public String parseToString() {

106 		str = str + super.parseToString();
107 		str += "Konfigurierend: " + _isConfigurating + "\n";
108 		str += "Name ist permanent: " + _isNameOfObjectsPermanent + "\n";
109 		final List attributeGroups = getAttributeGroups();

136 	public final List<AttributeGroup> getAttributeGroups() {
137 		if(_attributeGroups == null) {
138 			List<AttributeGroup> directAttributeGroups = getDirectAttributeGroups();
139 			List<AttributeGroup> attributeGroups = new ArrayList<AttributeGroup>();
140 			attributeGroups.addAll(directAttributeGroups);
141 
142 			List list = getSuperTypes();
143 			if((list != null) && (list.size() > 0)) {
144 				for(int i = 0; i < list.size(); ++i) {
145 					DafSystemObjectType _systemObjectType = (DafSystemObjectType)list.get(i);
146 					List tmpList = _systemObjectType.getAttributeGroups();
147 					if(tmpList != null) {
148 						for(int j = 0; j < tmpList.size(); ++j) {
149 							DafAttributeGroup attributeGroup = (DafAttributeGroup)tmpList.get(j);
150 							long attributeGroupId = attributeGroup.getId();
151 							int k = 0;
152 							for(; k < attributeGroups.size(); ++k) {
153 								DafAttributeGroup _attributeGroup = (DafAttributeGroup)attributeGroups.get(k);
154 								if(_attributeGroup.getId() == attributeGroupId) {
155 									break;
156 								}
157 							}
158 							if(k == attributeGroups.size()) {
159 								attributeGroups.add(attributeGroup);
160 							}
161 						}
162 					}
163 				}
164 			}

282 	public final boolean inheritsFrom(SystemObjectType other) {
283 		// direkte Vererbung
284 		final List<SystemObjectType> superTypes = getSuperTypes();
285 		if(superTypes.contains(other)) return true;
286 		// indirekte Vererbung
287 		for(SystemObjectType superType : superTypes) {
288 			if(superType.inheritsFrom(other)) return true;
289 		}
290 		return false;
291 	}
