36  /**
37   * Klasse zum Zugriff auf die Eigenschaften von Referenz-Attributtypen. Attribute von diesem Attributtyp referenzieren andere Objekte. Der Typ der
38   * referenzierten Objekte wird durch den Attributtyp festgelegt.
39   *
40   * @author Kappich Systemberatung
41   * @version $Revision: 5055 $
42   */
43  public class DafReferenceAttributeType extends DafAttributeType implements ReferenceAttributeType {
44  
45  	/** Die ID des Typs der referenzierbaren Objekte */
46  	private long _referencedTypeId;
47  
48  	/** Typ der referenzierbaren Objekte */
49  	private DafSystemObjectType _systemObjectType;
50  
51  	/** Konfigurierender Datensatz mit den wesentlichen Eigenschaften dieses Attributtyps oder <code>null</code>, wenn der Datensatz noch nicht geladen wurde. */
52  	private Data _data = null;
53  
54  	/** Kennung, die gesetzt ist, wenn bei Referenzattributen dieses Typs undefinierte Referenzen erlaubt sind. */
55  	private boolean _isUndefinedAllowed;
56  
57  	/** Referenzierungsart von Referenzen dieses Typs */
58  	private ReferenceType _referenceType;
59  
60  	/**
61  	 * Erzeugt ein neues Objekt dessen Eigenschaften im Anschluss mit der read-Methode eingelesen werden sollten.
62  	 *
63  	 * @param dataModel DataModel Implementierung, der das neue Objekt zugeordnet ist.
64  	 */
65  	public DafReferenceAttributeType(DafDataModel dataModel) {
66  		super(dataModel);
67  		_internType = REFERENCE_ATTRIBUTE_TYPE;
68  		_dataValueType = DataValue.LONG_TYPE;
69  	}

71  	/** Erzeugt ein neues Objekt mit den angegebenen Eigenschaften */
72  	public DafReferenceAttributeType(
73  			long id,
74  			String pid,
75  			String name,
76  			long typeId,
77  			byte state,
78  			String error,
79  			DafDataModel dataModel,
80  			short validFromVersionNumber,
81  			short validToVersionNumber,
82  			long responsibleObjectId,
83  			long[] setIds,
84  			long referenceId,
85  			String defaultAttributeValue,
86  			boolean isUndefinedAllowed,
87  			ReferenceType referenceType
88  	) {
89  		super(
90  				id, pid, name, typeId, state, error, dataModel, validFromVersionNumber, validToVersionNumber, responsibleObjectId, setIds, defaultAttributeValue
91  		);
92  		_internType = REFERENCE_ATTRIBUTE_TYPE;
93  		_dataValueType = DataValue.LONG_TYPE;
94  		_referencedTypeId = referenceId;
95  		_referenceType = referenceType;
96  		_isUndefinedAllowed = isUndefinedAllowed;
97  	}

99  	public SystemObjectType getReferencedObjectType() {
100 		if(_referencedTypeId == 0) return null;
101 		if(_systemObjectType == null) {
102 			_systemObjectType = (DafSystemObjectType)_dataModel.getObject(_referencedTypeId);
103 		}
104 		return _systemObjectType;
105 	}
106 
107 	public boolean isUndefinedAllowed() {
108 		return _isUndefinedAllowed;
109 	}
110 
111 	public ReferenceType getReferenceType() {
112 		return _referenceType;
113 	}

115 	public final String parseToString() {
116 		String str = "Objektreferenz: \n";
117 		str += super.parseToString();
118 		str += "ID des Typs: " + _referencedTypeId + "\n";
119 		return str;
120 	}

122 	public final void write(DataOutputStream out) throws IOException {
123 		super.write(out);
124 		out.writeLong(_referencedTypeId);
125 		out.writeBoolean(_isUndefinedAllowed);
126 		final int referenceTypeCode;
127 		switch(_referenceType) {
128 			case ASSOCIATION:
129 				referenceTypeCode = 0;
130 				break;
131 			case AGGREGATION:
132 				referenceTypeCode = 1;
133 				break;
134 			case COMPOSITION:
135 				referenceTypeCode = 2;
136 				break;
137 			default:
138 				throw new IllegalStateException("Unbekannte Referenzierungsart beim Attributtyp " + this + ": " + _referenceType);
139 		}
140 		out.writeByte(referenceTypeCode);
141 	}
142 
143 	public final void read(DataInputStream in) throws IOException {
144 		super.read(in);
145 		_referencedTypeId = in.readLong();
146 		_isUndefinedAllowed = in.readBoolean();
147 		final int referenceTypeCode = in.readByte();
148 		switch(referenceTypeCode) {
149 			case 0:
150 				_referenceType = ReferenceType.ASSOCIATION;
151 				break;
152 			case 1:
153 				_referenceType = ReferenceType.AGGREGATION;
154 				break;
155 			case 2:
156 				_referenceType = ReferenceType.COMPOSITION;
157 				break;
158 			default:
159 				throw new IllegalStateException("Unbekannte Referenzierungsart beim lesen des Attributtyps " + getPid() + ": " + referenceTypeCode);
160 		}
161 	}

165 	}
