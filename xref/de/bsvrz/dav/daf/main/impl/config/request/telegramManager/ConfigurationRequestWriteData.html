31   * Ermöglicht es, Anfragen an die Konfiguration zu stellen. Die Anfragen sind schreibender Natur und
32   * verändern die Daten der Konfiguration.

34   * @author Achim Wullenkord (AW), Kappich Systemberatung
35   * @version $Revision: 5060 $ / $Date: 2007-09-01 15:04:35 +0200 (Sat, 01 Sep 2007) $ / ($Author: rs $)
36   */
37  public class ConfigurationRequestWriteData extends AbstractSenderReceiverCommunication implements SenderReceiverCommunication {
38  		private final ClientDavInterface _connection;
39  	private final Aspect _requestAspect;
40  	private final AttributeGroup _requestAtg;
41  	private final AttributeGroup _responseAtg;
42  	private final Aspect _responseAspect;
43  
44  	public ConfigurationRequestWriteData(ClientDavInterface connection, ConfigurationAuthority configurationAuthority, SystemObject localApplication) throws OneSubscriptionPerSendData {
45  		super(connection, configurationAuthority, localApplication);
46  		_connection = connection;
47  		final DataModel dataModel = connection.getDataModel();
48  
49  		_requestAspect = dataModel.getAspect("asp.anfrage");
50  		_requestAtg = dataModel.getAttributeGroup("atg.konfigurationsAnfrageSchnittstelleSchreibend");
51  
52  		_responseAspect = dataModel.getAspect("asp.antwort");
53  		_responseAtg = dataModel.getAttributeGroup("atg.konfigurationsAnfrageSchnittstelleSchreibend");
54  
55  		// Sender und Empfänger anmelden
56  		init(_requestAtg, _requestAspect, _responseAtg, _responseAspect, null);
57  	}
