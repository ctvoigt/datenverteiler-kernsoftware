97  	/**
98  	 * Lieferte eine kurze beschreibende Information des jeweiligen Systemobjekts zurück, die <code>HTML</code> konform ist.
99  	 *
100 	 * @return <code>HTML</code> konformer Text, der die Kurzbeschreibung enthält. Ist keine Beschreibung vorhanden, wird ein leerer String <code>""</code>
101 	 *         zurückgegeben.
102 	 */
103 	public String getShortInfo() {
104 		if (_shortInfoAsHTML == null) {
105 			synchronized (_lock) {
106 				if (_shortInfoAsXML.startsWith("<html>") || "".equals(_shortInfoAsXML)) {
107 					// Kurzinfo ist bereits in HTML umgewandelt, braucht also nicht mehr
108 					_shortInfoAsHTML = _shortInfoAsXML;
109 				} else {
110 					// Kurzinfo muss umgewandelt werden
111 					_shortInfoAsHTML = getHTMLText("<kurzinfo>" + _shortInfoAsXML + "</kurzinfo>");
112 				}
113 			}
114 		}
115 		return _shortInfoAsHTML;
116 	}

128 					// Beschreibung ist bereits in HTML umgewandelt, braucht also nicht mehr umgewandelt werden

133 				}

136 		return _descriptionAsHTML;

169 	/**
170 	 * Wandelt Text im XML-Format in einen Text im HTML-Format um.
171 	 *
172 	 * @param text umzuwandelnder Text im XML-Format
173 	 * @return Text im HTML-Format
174 	 */
175 	private String getHTMLText(String text) {
176 		try {
177 			final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
178 			DocumentBuilder builder = factory.newDocumentBuilder();
179 			StringReader reader = new StringReader(text);
180 			InputSource source = new InputSource(reader);
181 			Document xmlDocument = builder.parse(source);
182 			Element element = xmlDocument.getDocumentElement();
183 			NodeList childs = element.getChildNodes();
184 			if (childs.getLength() <= 0) return "";	// Tag zwar da, aber ohne Inhalt
185 			if (childs.getLength() == 1 && (childs.item(0) instanceof Text)) {
186 				// enthält der Tag nur Text, dann ...
187 				return ((Text) childs.item(0)).getNodeValue();
188 			} else {
189 				// Tags werden durch HTML konforme Tags ersetzt
190 				StringBuilder resultText = new StringBuilder();
191 				resultText.append("<html><body>");
192 				transferXml(childs, resultText);
193 				resultText.append("</body></html>");
194 				return resultText.toString();
195 			}
196 		} catch (Exception ex) {
197 			ex.printStackTrace();
198 			return text;
199 		}
200 	}

202 	/**
203 	 * Ersetzt einzelne XML-Tags durch passende HTML-Tags
204 	 *
205 	 * @param node	   XML-Tag
206 	 * @param resultText umgewandelter Text in HTML-Format
207 	 */
208 	private void transferXml(Node node, StringBuilder resultText) {
209 		if (node instanceof Element) {
210 			Element element = (Element) node;
211 			String tag = element.getTagName();
212 			if (tag.equals("verweis")) {
213 				resultText.append(element.getAttribute("txt"));
214 			} else if (tag.equals("absatz")) {
215 				resultText.append("<p>");
216 				transferXml(element.getChildNodes(), resultText);
217 				resultText.append("</p>");
218 			} else if (tag.equals("titel")) {
219 				resultText.append("<p><b>");
220 				transferXml(element.getChildNodes(), resultText);
221 				resultText.append("</b></p>");
222 			} else if (tag.equals("wichtig")) {
223 				resultText.append("<b>");
224 				transferXml(element.getChildNodes(), resultText);
225 				resultText.append("</b>");
226 			} else if (tag.equals("liste")) {
227 				resultText.append("<ul>");
228 				transferXml(element.getChildNodes(), resultText);
229 				resultText.append("</ul>");
230 			} else if (tag.equals("numListe")) {
231 				resultText.append("<ol>");
232 				transferXml(element.getChildNodes(), resultText);
233 				resultText.append("</ol>");
234 			} else if (tag.equals("listenPunkt")) {
235 				resultText.append("<li>");
236 				transferXml(element.getChildNodes(), resultText);
237 				resultText.append("</li>");
238 			} else if (tag.equals("code") || tag.equals("pre")) {
239 				resultText.append("<code>");
240 				transferXml(element.getChildNodes(), resultText);
241 				resultText.append("</code>");
242 			} else if (tag.equals("kapitel") || tag.equals("sektion")) {
243 				transferXml(element.getChildNodes(), resultText);
244 			} else {
245 				_debug.finer("XML-Tag " + element.getTagName() + " wird noch nicht in HTML umgesetzt!");
246 			}
247 		} else if (node instanceof Text) {
248 			Text textNode = (Text) node;
249 			String text = textNode.getNodeValue();
250 			text = Pattern.compile("\\s+").matcher(text).replaceAll(" ");
251 			text = Pattern.compile("&").matcher(text).replaceAll("&amp;");
252 			text = Pattern.compile("<").matcher(text).replaceAll("&lt;");
253 			text = Pattern.compile(">").matcher(text).replaceAll("&gt;");
254 			text = Pattern.compile("ä").matcher(text).replaceAll("&auml;");
255 			text = Pattern.compile("ö").matcher(text).replaceAll("&ouml;");
256 			text = Pattern.compile("ü").matcher(text).replaceAll("&uuml;");
257 			text = Pattern.compile("ß").matcher(text).replaceAll("&szlig;");
258 			text = Pattern.compile("Ä").matcher(text).replaceAll("&Auml;");
259 			text = Pattern.compile("Ö").matcher(text).replaceAll("&Ouml;");
260 			text = Pattern.compile("Ü").matcher(text).replaceAll("&Uuml;");
261 			resultText.append(text);
262 		} else if (node instanceof CDATASection) {
263 			CDATASection cdataSection = (CDATASection) node;
264 			resultText.append(cdataSection.getNodeValue());
265 		}
266 	}

268 	/**
269 	 * Parst den Text nach XML-Tags und gibt sie zur weiteren Verarbeitung an die Methoden {@link #transferXml(org.w3c.dom.Node, StringBuilder)} weiter.
270 	 *
271 	 * @param nodes	  alle im Text enthaltenen Tags
272 	 * @param resultText den im HTML-Format umgewandelten Text
273 	 */
274 	private void transferXml(NodeList nodes, StringBuilder resultText) {
275 		int nodeCount = nodes.getLength();
276 		for (int i = 0; i < nodeCount; ++i) {
277 			transferXml(nodes.item(i), resultText);
278 		}
279 	}
