21  package de.bsvrz.dav.daf.main.config;
22  
23  /**
24   * Schnittstellenklasse zum Zugriff auf die Eigenschaften von Referenz-Attributtypen. Attribute von diesem Attributtyp
25   * referenzieren andere Objekte. Der Typ der referenzierten Objekte wird durch den Attributtyp festgelegt.
26   *
27   * @author Kappich+Kniß Systemberatung Aachen (K2S)
28   * @author Roland Schmitz (rs)
29   * @author Stephan Homeyer (sth)
30   * @version $Revision: 5052 $ / $Date: 2007-08-31 20:02:55 +0200 (Fri, 31 Aug 2007) $ / ($Author: rs $)
31   */
32  public interface ReferenceAttributeType extends AttributeType, UndefinedAttributeValueAccess {
33  	/**
34  	 * Bestimmt den Typ von Objekten, die von Attributen dieses Attribut-Typs referenziert werden können.
35  	 *
36  	 * @return Typ der Objekte, die von Attributen dieses Typs referenziert werden können oder <code>null</code> falls
37  	 *         beliebige Objekte referenziert werden können.
38  	 */
39  	public SystemObjectType getReferencedObjectType();
40  
41  	/**
42  	 * Bestimmt, ob undefinierte Referenzen in Attributwerten dieses Attributtyps zugelassen werden.
43  	 *
44  	 * @return ob undefinierte Referenzen in Attributwerten dieses Attributtyps zugelassen werden
45  	 */
46  	public boolean isUndefinedAllowed();
47  
48  	/**
49  	 * Bestimmt, ob die in Attributwerten dieses Attributtyps enthaltenen Referenzen als gerichtete Assoziation, als
50  	 * Aggregation oder als Komposition realisiert werden sollen.
51  	 *
52  	 * @return die Referenzierungsart
53  	 */
54  	public ReferenceType getReferenceType();
55  }
