102 	/** Aufzählung für die verschiedenen Persistenzmodi der dynamischen Objekte eines Typs */
103 	enum PersistenceMode {
104 
105 		/** Dynamische Objekte werden nicht persistent gespeichert. */
106 		TRANSIENT_OBJECTS(1, "transient"),
107 
108 		/** Dynamische Objekte werden persistent gespeichert. */
109 		PERSISTENT_OBJECTS(2, "persistent"),
110 
111 		/** Dynamische Objekte werden persistent gespeichert und beim Neustart auf ungültig gesetzt. */
112 		PERSISTENT_AND_INVALID_ON_RESTART(3, "persistentUndUngültigNachNeustart");
113 
114 		private final int _value;
115 
116 		private final String _name;
117 
118 		/**
119 		 * Gibt den Integer-Wert zurück, durch den die PersistenceMode im Datenmodell repräsentiert wird.
120 		 * @return Unskalierter Integer-Wert des Attributs im Datenmodell
121 		 */
122 		public int getIntValue() {
123 			return _value;
124 		}
125 
126 		/**
127 		 * Gibt den Text-Wert zurück, durch den die PersistenceMode im Datenmodell repräsentiert wird.
128 		 * @return Text-Wert des Attributes im Datenmodell
129 		 */
130 		public String getStatusName() {
131 			return _name;
132 		}
133 
134 		PersistenceMode(final int value, final String name) {
135 			_value = value;
136 			_name = name;
137 		}
138 
139 		/**
140 		 * Erstellt ein PersistenceMode aus einem String.
141 		 * @param persistenceMode Eingabe-String
142 		 * @return PersistenceMode
143 		 * @throws IllegalArgumentException Falls der String keinem Statuswert entspricht.
144 		 */
145 		public static PersistenceMode parse(final String persistenceMode) {
146 			if(persistenceMode.equals(TRANSIENT_OBJECTS.getStatusName())) {
147 				return TRANSIENT_OBJECTS;
148 			}
149 			else if(persistenceMode.equals(PERSISTENT_OBJECTS.getStatusName())) {
150 				return PERSISTENT_OBJECTS;
151 			}
152 			else if(persistenceMode.equals(PERSISTENT_AND_INVALID_ON_RESTART.getStatusName())) {
153 				return PERSISTENT_AND_INVALID_ON_RESTART;
154 			}
155 			else {
156 				throw new IllegalArgumentException("persistenceMode hat einen ungültigen Wert: " + persistenceMode);
157 			}
158 		}
159 
160 		/**
161 		 * Erstellt ein PersistenceMode aus einem Integer-Wert.
162 		 * @param persistenceMode Eingabe-Wert
163 		 * @return PersistenceMode
164 		 * @throws IllegalArgumentException Falls der eingabewert nicht zugeordnet werden kann
165 		 */
166 		public static PersistenceMode parse(final int persistenceMode) {
167 			switch(persistenceMode) {
168 				case 1:
169 					return PersistenceMode.TRANSIENT_OBJECTS;
170 				case 2:
171 					return PersistenceMode.PERSISTENT_OBJECTS;
172 				case 3:
173 					return PersistenceMode.PERSISTENT_AND_INVALID_ON_RESTART;
174 				default:
175 					throw new IllegalArgumentException("persistenceMode hat einen ungültigen Wert: " + persistenceMode);
176 			}
177 		}
178 	}
