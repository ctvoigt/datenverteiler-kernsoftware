22  package de.bsvrz.dav.daf.main.config.management.consistenycheck;

24  import java.util.*;
25  
26  /**
27   * Dieses Interface wird von Klassen implementiert, die das Ergebnis einer Konsistenzprüfung enthalten (siehe TPuK1-138).
28   *
29   * @author Kappich Systemberatung
30   * @version $Revision: 5060 $
31   */
32  public interface ConsistencyCheckResultInterface {
33  
34  	/**
35  	 * Die Implementierung dieser Methode gibt an, ob es bei der Konsistenzprüfung zu lokalen Fehlern gekommen ist.
36  	 *
37  	 * @return true = Es ist zu lokalen Fehlern gekommen; false = Es ist zu keinem lokalen Fehler gekommen
38  	 */
39  	boolean localError();
40  
41  	/**
42  	 * Die Implementierung dieser Methode gibt an, ob es bei der Konsistenzprüfung zu Interferenzfehlern gekommen ist.
43  	 *
44  	 * @return true = Es ist zu Interferenzfehlern gekommen; false = Es ist zu keinem Interferenzfehler gekommen
45  	 */
46  	boolean interferenceErrors();
47  
48  	/**
49  	 * Die Implementierung dieser Methode gibt an, ob es bei der Konsistenzprüfung zu Warnungen gekommen ist.
50  	 *
51  	 * @return true = Es gab Warnungen; false = Es gab keine Warnungen
52  	 */
53  	boolean warnings();
54  
55  	/**
56  	 * Die Implementierung dieser Methode gibt alle aufgetretenen lokalen Fehler zurück.
57  	 *
58  	 * @return Liste mit lokalen Fehlern. Sind keine lokalen Fehler aufgetreten, ist die Liste leer.
59  	 */
60  	List<ConsistencyCheckResultEntry> getLocalErrors();
61  
62  	/**
63  	 * Die Implementierung dieser Methode gibt alle aufgetretenen Interferenzfehler zurück.
64  	 *
65  	 * @return Liste mit allen Interferenzfehlern. Sind keine Interferenzfehler aufgetreten ist die Liste leer.
66  	 */
67  	List<ConsistencyCheckResultEntry> getInterferenceErrors();
68  
69  	/**
70  	 * Die Implementierung dieser Methode gibt alle aufgetretenen Warnungen zurück.
71  	 *
72  	 * @return Warnungen, die bei der Konsistenzprüfung aufgetreten sind. Die Liste ist leer, falls es keine Warnungen gegeben hat.
73  	 */
74  	List<ConsistencyCheckResultEntry> getWarnings();
75  }
