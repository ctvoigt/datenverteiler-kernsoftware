91  	/**
92  	 * Serialisiert einen Datensatz in einen Bytestrom und schreibt diesen auf den angegebenen Ausgabe-Stream.
93  	 * AttributListen und Arrays werden durch Serialisierung der enthalten Attribute serialisiert. Bei Arrays variabler
94  	 * Länge wird die Länge vorweg serialisiert, und zwar je nach maximaler Anzahl der Elemente in 1, 2 oder 4 Bytes als
95  	 * vorzeichenloser Wert.
96  	 *
97  	 * @param data Der zu serialisierende Datensatz.
98  	 * @throws IOException Wenn ein I/O Fehler bei Schreiben auf den Ausgabe-Stream auftritt.
99  	 */
100 	public void writeData(final Data data) throws IOException {
101 		if (data.isPlain()) {
102 			try {
103 				final AttributeType att = data.getAttributeType();
104 				if (att instanceof IntegerAttributeType) {
105 					final IntegerAttributeType integerAtt = (IntegerAttributeType) att;
106 					final Data.NumberValue unscaledValue = data.asUnscaledValue();
107 					switch (integerAtt.getByteCount()) {
108 						case 1:
109 							writeByte(unscaledValue.byteValue());
110 							break;
111 						case 2:
112 							writeShort(data.asUnscaledValue().shortValue());
113 							break;
114 						case 4:
115 							writeInt(data.asUnscaledValue().intValue());
116 							break;
117 						case 8:
118 							writeLong(data.asUnscaledValue().longValue());
119 							break;
120 						default:
121 							throw new RuntimeException("Ganzzahlattribut mit ungültiger Byte-Anzahl: " + integerAtt.getNameOrPidOrId());
122 					}
123 				} else if (att instanceof ReferenceAttributeType) {
124 					final ReferenceAttributeType referenceAttributeType = (ReferenceAttributeType)att;
125 					if(_version >= 3 && referenceAttributeType.getReferenceType() == ReferenceType.ASSOCIATION) {
126 						final String pid;
127 						final SystemObject systemObject = data.asReferenceValue().getSystemObject();
128 						if(systemObject != null) {
129 							pid = systemObject.getPid();
130 							if(pid.length() == 0) {
131 								throw new IllegalArgumentException(
132 										"Serialisierung des Attributs " + data.getName() + " kann nicht durchgeführt werden, weil"
133 										+ "das referenzierte Objekt keine Pid hat und als Referenzierungsart Assoziation festgelegt ist"
134 								);
135 							}
136 						}
137 						else {
138 							// Bei nicht auflösbaren Referenzen wird die ursprüngliche Pid (falls vorhanden) eingetragen, sonst ein Leerstring für undefiniert
139 							pid = data.asReferenceValue().getSystemObjectPid();
140 						}
141 						writeString(pid, 255);
142 					}
143 					else {
144 						writeObjectReference(data.asReferenceValue().getSystemObject());
145 					}
146 				} else if (att instanceof TimeAttributeType) {
147 					final TimeAttributeType timeAtt = (TimeAttributeType) att;
148 					if (timeAtt.getAccuracy() == TimeAttributeType.MILLISECONDS) {
149 						writeLong(data.asTimeValue().getMillis());
150 					} else {
151 						writeInt((int) data.asTimeValue().getSeconds());
152 					}
153 				} else if (att instanceof StringAttributeType) {
154 					final StringAttributeType stringAtt = (StringAttributeType) att;
155 					writeString(data.asTextValue().getValueText(), stringAtt.getMaxLength());
156 				} else if (att instanceof DoubleAttributeType) {
157 					final DoubleAttributeType doubleAtt = (DoubleAttributeType) att;
158 					if (doubleAtt.getAccuracy() == DoubleAttributeType.DOUBLE) {
159 						writeDouble(data.asUnscaledValue().doubleValue());
160 					} else {
161 						writeFloat(data.asUnscaledValue().floatValue());
162 					}
163 				} else {
164 					throw new RuntimeException("Serialisierung einer unbekannten Attributart nicht möglich");
165 				}
166 			} catch (ConfigurationException e) {
167 				throw new RuntimeException(e);
168 			}
169 		} else {
170 			if (data.isArray()) {
171 				final Data.Array array = data.asArray();
172 				final int arrayLength = array.getLength();
173 				final int maxCount = array.getMaxCount();
174 				if (array.isCountVariable()) {
175 					if (maxCount <= 0 || maxCount > 65535) {
176 						writeInt(arrayLength);
177 					} else if (maxCount > 255) {
178 						writeShort(arrayLength);
179 					} else {
180 						writeByte(arrayLength);
181 					}
182 				} else {
183 					if(arrayLength != maxCount) {
184 						throw new RuntimeException(
185 								"Länge des Arrays im Attribut " + data.getName() + " ist " + arrayLength + ", " +
186 								"aber es sollte die Länge " + maxCount + " haben"
187 						);
188 					}
189 				}
190 			}
191 			final Iterator iterator = data.iterator();
192 			while (iterator.hasNext()) {
193 				final Data subData = (Data) iterator.next();
194 				writeData(subData);
195 			}
196 		}
197 	}

301 	/**
302 	 * Serialisiert einen <code>String</code>-Wert in einen Bytestrom und schreibt diesen auf den Ausgabe-Stream. Strings werden in ISO-8859-1 kodiert und vorweg
303 	 * wird die Länge des Strings serialisiert und zwar je nach maximaler Stringlänge in 1, 2, oder 4 Bytes als vorzeichenloser Wert.
304 	 *
305 	 * @param value     Der zu serialisierende Wert.
306 	 * @param maxLength Maximale Länge des zu serialisierenden Strings oder <code>0</code> wenn keine Begrenzung vorgegeben werden kann.
307 	 *
308 	 * @throws IOException              Wenn ein I/O Fehler bei Schreiben auf den Ausgabe-Stream auftritt.
309 	 * @throws IllegalArgumentException Wenn die Länge des Strings größer als die angegebene Maximallänge ist.
310 	 */
311 	public void writeString(final String value, final int maxLength) throws IOException {
312 		final int length = value.length();
313 		if(maxLength <= 0 || maxLength > 65535) {
314 			writeInt(length);
315 		}
316 		else {
317 			if(length > maxLength) {
318 				throw new IllegalArgumentException(
319 						"Länge " + length + " des zu serialisierenden Strings '" + value + "' ist größer als das zulässige Maximum " + maxLength
320 				);
321 			}
322 			if(maxLength > 255) {
323 				writeShort(length);
324 			}
325 			else {
326 				writeByte(length);
327 			}
328 		}
329 		final byte[] bytes = value.getBytes("ISO-8859-1");
330 		if(length != bytes.length) {
331 			throw new RuntimeException("Stringlänge ungleich kodierter Stringlänge: " + value);
332 		}
333 		_outputStream.write(bytes);
334 	}
