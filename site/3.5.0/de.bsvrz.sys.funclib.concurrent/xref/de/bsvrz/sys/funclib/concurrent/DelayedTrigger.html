21  package de.bsvrz.sys.funclib.concurrent;
22  
23  import java.util.*;
24  import java.util.concurrent.CopyOnWriteArrayList;
25  
26  /**
27   * Definiert einen Trigger mit verzögerter Auslösung.
28   *
29   * @author Kappich Systemberatung
30   * @version $Revision: 5897 $
31   */
32  public class DelayedTrigger implements Trigger {
33  
34  	/** Listener, die beim Auslösen und Schließen des Triggers benachrichtigt werden sollen. */
35  	private final List<TriggerTarget> _triggerTargets = new CopyOnWriteArrayList<TriggerTarget>();
36  
37  	/** Anzahl der Trigger-Aufrufe, nach der die sofortige Auslösung angestoßen wird. */
38  	private int _maximumDelayedTriggerCount;
39  
40  	/** Aktuelle Anzahl der Trigger-Aufrufe. */
41  	private int _delayedTriggerCount = 0;
42  
43  	/** Verzögerungszeit in der nach einem Trigger-Aufrufe auf weitere Trigger-Aufrufe gewartet wird, bevor die sofortige Auslösung angestoßen wird. */
44  	private long _delayDuration;
45  
46  	/**
47  	 * Zeitpunkt an dem die sofortige Auslösung angestoßen wird, wenn nicht vorher ein weiterer Trigger-Aufruf stattfindet. Mit jedem neuen Trigger-Aufruf wird
48  	 * dieses Feld auf die aktuelle Zeit plus <code>_delayDuration</code> gesetzt.
49  	 */
50  	private long _triggeringTime;
51  
52  	/**
53  	 * Maximale Verzögerungszeit in der nach dem jeweils ersten Trigger-Aufruf auf weitere Trigger-Aufrufe gewartet wird, bevor die sofortige Auslösung angestoßen
54  	 * wird.
55  	 */
56  	private long _maximumDelayDuration;
57  
58  	/**
59  	 * Zeitpunkt an dem die sofortige Auslösung angestoßen wird. Mit dem jeweils ersten verzögerten Anstoß wird dieses Feld auf die aktuelle Zeit plus
60  	 * <code>_maximumDelayDuration</code> gesetzt.
61  	 */
62  	private long _maximumTriggeringTime;
63  
64  	/** <code>true</code>, wenn die close-Methode aufgerufen wurde; sonst <code>false</code>. */
65  	private boolean _closed = false;
66  
67  	/**
68  	 * Erzeugt ein neues Triggerobjekt mit den angegebenen Eigenschaften
69  	 *
70  	 * @param threadName                 Name des Threads für die asynchrone Auslösung des Triggers.
71  	 * @param maximumDelayedTriggerCount Anzahl der Trigger-Aufrufe, nach der die sofortige Auslösung angestoßen wird.
72  	 * @param delayDuration              Verzögerungszeit in der nach einem Trigger-Aufrufe auf weitere Trigger-Aufrufe gewartet wird, bevor die sofortige
73  	 *                                   Auslösung angestoßen wird.
74  	 * @param maximumDelayDuration       Maximale Verzögerungszeit in der nach dem jeweils ersten Trigger-Aufruf auf weitere Trigger-Aufrufe gewartet wird, bevor
75  	 *                                   die sofortige Auslösung angestoßen wird.
76  	 */
77  	public DelayedTrigger(final String threadName, final int maximumDelayedTriggerCount, final long delayDuration, final long maximumDelayDuration) {
78  		_maximumDelayDuration = maximumDelayDuration;
79  		_delayDuration = delayDuration;
80  		_maximumDelayedTriggerCount = maximumDelayedTriggerCount;
81  		final Thread asyncTriggerThread = new Thread(new AsyncTriggerThreadRunnable(), threadName);
82  		asyncTriggerThread.setDaemon(true);
83  		asyncTriggerThread.start();
84  	}
85  
86  	/** Führt zu einer verzögerten Auslösung des Triggers mit einer asynchronen Benachrichtigung aller angemeldeten TriggerTargets */
87  	public void trigger() {
88  		synchronized(this) {
89  			if(_closed) throw new IllegalStateException("Trigger wurde bereits geschlossen");
90  			long now = System.currentTimeMillis();
91  			if((_delayedTriggerCount++) == 0) {
92  				_maximumTriggeringTime = now + _maximumDelayDuration;
93  				this.notifyAll();
94  			}
95  			_triggeringTime = now + _delayDuration;
96  			if(_delayedTriggerCount >= _maximumDelayedTriggerCount) this.notifyAll();
97  		}
98  	}
99  
100 	/** Führt zu einer sofortigen Auslösung des Triggers mit einer asynchronen Benachrichtigung aller angemeldeten TriggerTargets */
101 	public void shoot() {
102 		synchronized(this) {
103 			trigger();
104 			_maximumTriggeringTime = System.currentTimeMillis();
105 			this.notifyAll();
106 		}
107 	}
108 
109 	/**
110 	 * Schließt den Trigger. Die sofortige Auslösung des Triggers wird angestoßen, falls vorherige verzögerte Trigger-Aufrufe noch nicht zu einer Auslösung des
111 	 * Triggers geführt haben. Alle angemeldeten TriggerTargets werden anschließend über das Schließen des Triggers benachrichtigt. Der Thread zur asynchronen
112 	 * Benachrichtigung der TriggerTargets wird beendet.
113 	 */
114 	public void close() {
115 		synchronized(this) {
116 			_closed = true;
117 			this.notifyAll();
118 		}
119 	}
120 
121 	/**
122 	 * Wartet bis der Trigger ausgelöst werden soll oder der Trigger geschlossen wurde.
123 	 *
124 	 * @return <code>false</code>, wenn der Trigger ausgelöst werden kann; <code>true</code>, wenn der Trigger geschlossen wurde.
125 	 *
126 	 * @throws InterruptedException Wenn der Thread vor oder während des Wartens unterbrochen wurde.
127 	 */
128 	private boolean awaitTrigger() throws InterruptedException {
129 		synchronized(this) {
130 			while(_delayedTriggerCount == 0 && !_closed) wait();
131 			if(_delayedTriggerCount == 0 && _closed) return true;
132 			while(true) {
133 				long now = System.currentTimeMillis();
134 				if(_closed || _delayedTriggerCount >= _maximumDelayedTriggerCount || now >= _triggeringTime || now >= _maximumTriggeringTime) {
135 					_delayedTriggerCount = 0;
136 					return false;
137 				}
138 				long delay;
139 				if(_maximumTriggeringTime < _triggeringTime) {
140 					delay = _maximumTriggeringTime - now;
141 				}
142 				else {
143 					delay = _triggeringTime - now;
144 				}
145 				if(delay>0)wait(delay);
146 			}
147 		}
148 	}
149 
150 	public void addTriggerTarget(TriggerTarget triggerTarget) {
151 		_triggerTargets.add(triggerTarget);
152 	}
153 
154 	public void removeTriggerTarget(TriggerTarget triggerTarget) {
155 		_triggerTargets.remove(triggerTarget);
156 	}
157 
158 	/** Benachrichtigt alle TriggerTargets über das Auslösen des Triggers */
159 	private void shootTriggerTargets() {
160 		for(TriggerTarget triggerTarget : _triggerTargets) {
161 			triggerTarget.shot();
162 		}
163 	}
164 
165 	/** Benachrichtigt alle TriggerTargets über das Auslösen des Triggers */
166 	private void closeTriggerTargets() {
167 		for(TriggerTarget triggerTarget : _triggerTargets) {
168 			triggerTarget.close();
169 		}
170 	}
171 
172 	/** Runnable mit der run-Methode, die in einem eigenen Thread ausgeführt wird und die TriggerTargets asynchron benachrichtigt. */
173 	private class AsyncTriggerThreadRunnable implements Runnable {
174 
175 		public void run() {
176 			try {
177 				while(!Thread.interrupted()) {
178 					final boolean closed = awaitTrigger();
179 					if(closed) return;
180 					shootTriggerTargets();
181 				}
182 			}
183 			catch(InterruptedException e) {
184 				// Thread beendet sich
185 			}
186 			finally {
187 				closeTriggerTargets();
188 			}
189 		}
190 	}
