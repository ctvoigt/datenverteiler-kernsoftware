21  package de.bsvrz.sys.funclib.concurrent;
22  
23  /**
24   * Diese Klasse implementiert ein Semaphor, das zur Synchronisation von Threads verwendet werden kann.
25   *
26   * @author Kappich Systemberatung
27   * @version $Revision: 5009 $
28   */
29  public class Semaphore {
30  
31  	/**
32  	 * Diese Variable gibt an, wie oft eine Sperre geholt werden darf (wie oft ein kritischer Abschnitt betreten werden
33  	 * kann). Ist die untere Grenze (0) erreicht legt jeder Thread sich automatsich schlafen(wait), der versucht auf dieses
34  	 * Objekt ein <code> acquire </code> auszuführen.
35  	 */
36  	private int _permits;
37  
38  	/**
39  	 * Wie viele Threads dürfen maximal in den kritischen Abschnitt. Diese Variable verhindert, dass das die Variable
40  	 * _permits beliebig mit <code> release </code> erhöht werden kann. Diese Variable kann mit set-Methode erhöht werden.
41  	 */
42  	private int _upperBoundPermit;
43  
44  	/**
45  	 * Einen Semaphor erzeugen, der <code> permit </code> viele <code> acquire </code> Zugriffe zuläßt, bis er jeden
46  	 * weiteren Thread, der <code> acquire </code> aufruft, mit wait "schlafen" legt.
47  	 *
48  	 * @param permits Wie viele Threads dürfen in den kritischen Abschnitt
49  	 */
50  
51  	public Semaphore(int permits) {
52  		_permits = permits;
53  		_upperBoundPermit = permits;
54  	}
55  
56  	/**
57  	 * Eine Sperre des Semaphores anfordern, sind keine Sperren mehr zu vergeben, dann wird der Thread mit wait()
58  	 * angehalten. Ein anderer Thread, der seine Sperre wieder aufgibt, kann den wartenden Thread wieder befreien.
59  	 */
60  	public void acquire() {
61  		// Es wird über das gesamte Objekt synchronisiert
62  		synchronized (this) {
63  			while (_permits == 0) {
64  				// Solange schlafen, bis ein permit (Sperre) frei ist
65  				try {
66  					this.wait();
67  				} catch (InterruptedException e) {
68  					e.printStackTrace();
69  				}
70  			}
71  			// einmal sperren
72  			_permits--;
73  		}
74  	}
75  
76  	/**
77  	 * Eine Sperre des Semaphores freigeben, dies befreit andere Threads, die auf eine Sperre warten.
78  	 */
79  	public void release() {
80  		synchronized (this) {
81  			assert _permits < _upperBoundPermit : "_permits = " + (_permits + 1) + " _upperBoundPermit = " + _upperBoundPermit;
82  			// Einen anderen Thread Zugriff auf einen kritischen Abschnitt erlauben
83  			if (_permits < _upperBoundPermit) {
84  				_permits++;
85  			}
86  			// Einen Thread, der mit wait warten könnte, freilassen.
87  			this.notifyAll();
88  		}
89  	}
90  
91  	/**
92  	 * Diese Methode setzt die Anzahl der Sperren neu. Es ist somit zur Laufzeit möglich, dem Semaphor neue Sperren zu
93  	 * geben (entweder Sperren entziehen oder neue Sperren freischalten). Gibt es neue Sperren, dann werden sofort alle
94  	 * Threads, die warten, mit notifyAll darauf hingewiesen.
95  	 * <p/>
96  	 * TBD Dies ist keine "normale" Funktion, die auf Semaphoren angewandt wird. Ist diese Methode hier in Ordnung ?
97  	 *
98  	 * @param upperBoundPermit Neue Obergrenze, soviele Sperren stehen nun zur Verfügung
99  	 */
100 	public void setUpperBoundPermit(int upperBoundPermit) {
101 
102 		synchronized (this) {
103 			if (upperBoundPermit > _upperBoundPermit) {
104 				// Es wurden neue Sperren vergeben, somit könnten wartende Threads aufgeweckt werden.
105 				_upperBoundPermit = upperBoundPermit;
106 				this.notifyAll();
107 			} else {
108 				// die neue Grenze liegt niedriger, somit können keine Threads aufgeweckt werden.
109 				_upperBoundPermit = upperBoundPermit;
110 			}
111 		}
112 	}
113 }
