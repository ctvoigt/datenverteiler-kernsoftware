21  package de.bsvrz.dav.daf.main.archive;
22  
23  import java.io.IOException;
24  
25  /**
26   * Schnittstelle zum Zugriff auf einen einzelnen Ergebnisdatenstrom einer Archivanfrage. Über die hier definierten
27   * Methoden können die Datensätze dieses Archivdatenstroms abgefragt werden, es kann signalisiert werden, dass keine
28   * weiteren Datensätze benötigt werden und es kann die mit diesem Ergebnisdatenstrom korrespondierende
29   * Archivdatenspezifikation, die in der Archivanfrage angegeben wurde, abgefragt werden. Die einzelnen zu einer
30   * Archivanfrage gehörenden Ergebnisdatenströme können über die Methode {@link ArchiveDataQueryResult#getStreams} abgefragt
31   * werden.
32   *
33   * @author Kappich Systemberatung
34   * @version $Revision: 5084 $
35   * @see ArchiveRequestManager#request(ArchiveQueryPriority,ArchiveDataSpecification)
36   * @see ArchiveRequestManager#request(ArchiveQueryPriority, java.util.List)
37   * @see ArchiveDataQueryResult#getStreams
38   */
39  public interface ArchiveDataStream {
40  
41  	/**
42  	 * Bestimmt die Archivdatenspezifikation, die zu diesem Ergebnisdatenstrom geführt hat.
43  	 *
44  	 * @return Archivdatenspezifikation, die zu diesem Ergebnisdatenstrom geführt hat.
45  	 */
46  	ArchiveDataSpecification getDataSpecification();
47  
48  	/**
49  	 * Entfernt einen Datensatz vom Ergebnisdatenstrom und gibt ihn zurück. Diese Methode wird von einer Applikation
50  	 * aufgerufen um die vom Archivsystem auf diesem Ergebnisdatenstrom zur Verfügung gestellten Datensätze abzurufen. Die
51  	 * einzelnen Datensätze werden mit wiederholten Aufrufen dieser Methode sukzessiv in der gleichen Reihenfolge wie sie
52  	 * im Archivsystem erzeugt werden zur Verfügung gestellt. Sind alle Datensätze so übergeben worden, dann muss dies
53  	 * durch Rückgabe von <code>null</code> signalisiert werden.
54  	 *
55  	 * @return Der nächste Archivdatensatz oder <code>null</code>, wenn alle Datensätze dieses Ergeebnisdatenstroms
56  	 *         abgefragt wurden.
57  	 * @throws IllegalStateException Falls der Ergebnisdatenstrom mit der Methode {@link #abort} abgebrochen wurde.
58  	 * @throws InterruptedException  Falls der aufrufende Thread unterbrochen wurde, während auf den nächsten Datensatz
59  	 *                               gewartet wurde.
60  	 * @throws IOException           Falls Probleme in der Kommunikation mit dem Archivsystem aufgetreten sind und noch
61  	 *                               nicht alle Datensätze übertragen wurden.
62  	 */
63  	ArchiveData take() throws InterruptedException, IOException, IllegalStateException;
64  
65  	/**
66  	 * Bricht die Übertragung von Datensätzen für diesen Ergebnisdatenstrom ab. Diese Methode kann von einer Applikation
67  	 * aufgerufen werden, um zu signalisieren, dass keine weiteren Datensätze mehr von diesem Ergebnisdatenstrom benötigt
68  	 * werden. Anschließende Aufrufe der Methode {@link #take} werden mit einer entsprechenden Exception quittiert.
69  	 */
70  	void abort();
71  
72  
73  }
