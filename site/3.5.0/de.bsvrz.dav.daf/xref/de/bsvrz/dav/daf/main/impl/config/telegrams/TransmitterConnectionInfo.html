219 	/**
220 	 * Gibt ein String zurück, der diesen Datensatz beschreibt
221 	 *
222 	 * @return Beschreibung des Datensatzes
223 	 */
224 	public final String parseToString() {
225 		String str = "Datenverteiler Verbindungs Informationen: \n";
226 		str += "Von Datenverteiler: ";
227 		str += _transmitter_1.parseToString();
228 		str += "Zum Datenverteiler: ";
229 		str += _transmitter_2.parseToString();
230 		str += "Gewicht: " + _connectionWeight + "\n";
231 		str += "Benutzername1: " + _userName + "\n";
232 		str += "Benutzername2: " + _remoteUserName+ "\n";
233 		if(_connectionType == 0) {
234 			str += "Ersatzverbindung.\n";
235 		}
236 		else if(_connectionType == 1) {
237 			str += "Normale Verbindung.\n";
238 		}
239 		else {
240 			str += "Doppelte Verbindung (" + (isActiveConnection() ? "Aktiv" : "Passiv") + ").\n";
241 		}
242 		str += "Verbindungstimeout: " + _connectionTimeThreshold + "\n";
243 		if(_autoExchangeTransmitterDetection) {
244 			str += "Automatische Ermittlung der nicht mehr erreichbaren Datenverteiler ist eingeschaltet.\n";
245 		}
246 		else {
247 			str += "Automatische Ermittlung der nicht mehr erreichbaren Datenverteiler ist ausgeschaltet.\n";
248 		}
249 		if(_exchangeTransmitterList != null) {
250 			str += "Ersatzdatenverteiler: \n";
251 			for(int i = 0; i < _exchangeTransmitterList.length; ++i) {
252 				str += _exchangeTransmitterList[i].parseToString();
253 			}
254 		}
255 		return str;
256 	}

258 	/**
259 	 * Schreibt einen Datensatz in den gegebenen DataOutputStream.
260 	 *
261 	 * @param out     DataOutputStream
262 	 * @param version Version des übergeordneten Antworttelegramms. Der Wert 0 kennzeichnet, dass nur die ursprüngliche Version ohne Benutzernamen für die
263 	 *                Authentifizierung übertragen wird; der Wert 1 kennzeichnet, dass die neue Version mit Benutzernamen für die Authentifizierung übertragen
264 	 *                wird.
265 	 *
266 	 * @throws IOException muss geworfen werden.
267 	 * @see TransmitterConnectionInfoAnswer
268 	 */
269 	public final void write(DataOutputStream out, final long version) throws IOException {
270 		_transmitter_1.write(out);
271 		_transmitter_2.write(out);
272 		out.writeShort(_connectionWeight);
273 		out.writeByte(_connectionType);
274 		out.writeLong(_connectionTimeThreshold);
275 		if(version >= 1 ) {
276 			out.writeUTF(_userName);
277 		}
278 		if(version >= 2 ) {
279 			out.writeUTF(_remoteUserName);
280 		}
281 		out.writeBoolean(_autoExchangeTransmitterDetection);
282 		if(_exchangeTransmitterList == null) {
283 			out.writeShort(0);
284 		}
285 		else {
286 			out.writeShort(_exchangeTransmitterList.length);
287 			for(int i = 0; i < _exchangeTransmitterList.length; ++i) {
288 				_exchangeTransmitterList[i].write(out);
289 			}
290 		}
291 	}

293 	/**
294 	 * Liest einen Datensatz aus dem gegebenen DataInputStream
295 	 *
296 	 * @param in      DataInputStream
297 	 * @param version Version des übergeordneten Antworttelegramms. Der Wert 0 kennzeichnet, dass nur die ursprüngliche Version ohne Benutzernamen für die
298 	 *                Authentifizierung übertragen wird; der Wert 1 kennzeichnet, dass die neue Version mit Benutzernamen für die Authentifizierung übertragen
299 	 *                wird.
300 	 *
301 	 * @throws IOException muss geworfen werden
302 	 * @see TransmitterConnectionInfoAnswer
303 	 */
304 	public final void read(DataInputStream in, final long version) throws IOException {
305 		_transmitter_1 = new TransmitterInfo();
306 		_transmitter_1.read(in);
307 		_transmitter_2 = new TransmitterInfo();
308 		_transmitter_2.read(in);
309 		_connectionWeight = in.readShort();
310 		_connectionType = in.readByte();
311 		_connectionTimeThreshold = in.readLong();
312 		if(version >= 1) {
313 			_userName = in.readUTF();
314 		}
315 		else {
316 			_userName = "";
317 		}
318 		if(version >= 2) {
319 			_remoteUserName = in.readUTF();
320 		}
321 		else {
322 			_remoteUserName = "";
323 		}
324 		_autoExchangeTransmitterDetection = in.readBoolean();
325 		int size = in.readShort();
326 		if(size > 0) {
327 			_exchangeTransmitterList = new TransmitterInfo[size];
328 			for(int i = 0; i < size; ++i) {
329 				TransmitterInfo tmp = new TransmitterInfo();
330 				tmp.read(in);
331 				_exchangeTransmitterList[i] = tmp;
332 			}
333 		}
