34  public class ArrayAttributeInfo extends AbstractAttributeInfo {
35  	private final boolean _sizeFixed;
36  	private final int _fixedSize;
37  	private final AttributeInfo _elementInfo;
38  
39  	public ArrayAttributeInfo(final Attribute attribute, int offset, AttributeInfo offsetReferral, AttributeDefinitionInfo definitionInfo) {
40  		super(attribute, offset, offsetReferral, definitionInfo);
41  		_elementInfo = new ArrayElementAttributeInfo(attribute, definitionInfo);
42  		try {
43  			_sizeFixed = definitionInfo.isSizeFixed() && !attribute.isCountVariable();
44  			if(_sizeFixed) {
45  				_fixedSize = 4 + getDefinitionInfo().getFixedSize() * attribute.getMaxCount();
46  			}
47  			else {
48  				_fixedSize = 0;
49  			}
50  		}
51  		catch(ConfigurationException e) {
52  			e.printStackTrace();
53  			throw new RuntimeException(e);
54  		}
55  	}
56  
57  	public boolean isArray() {
58  		return true;
59  	}
60  
61  	public boolean isSizeFixed() {
62  		return _sizeFixed;
63  	}
64  
65  	public int getFixedSize() {
66  		return _fixedSize;
67  	}

69  	public int getSize(byte[] bytes, int offset) {
70  //		System.out.println(getName() + ".getSize(..," +offset + ")");
71  		final int elementCount = getElementCount(bytes, offset);
72  		final int relativeOffset = getRelativeOffset(bytes, offset, elementCount);
73  //		System.out.println("- returning " + getName() + ".getSize(..," +offset + "): " + relativeOffset);
74  		return relativeOffset;
75  	}
76  
77  	public int getElementCount(final byte[] bytes, final int offset) {
78  		final int b0 = bytes[offset + 0] & 0xff;
79  		final int b1 = bytes[offset + 1] & 0xff;
80  		final int b2 = bytes[offset + 2] & 0xff;
81  		final int b3 = bytes[offset + 3] & 0xff;
82  //		System.out.println(getName() + ": bx = " + b0);
83  //		System.out.println(getName() + ": bx = " + b1);
84  //		System.out.println(getName() + ": bx = " + b2);
85  //		System.out.println(getName() + ": bx = " + b3);
86  		final int arraySize = (b0 << 24) | (b1 << 16) | (b2 << 8) | (b3 << 0);
87  //		System.out.println(getName() + ": arraySize = " + arraySize);
88  		return arraySize;
89  	}
90  
91  	public int getAbsoluteElementOffset(byte[] bytes, int offset, int elementIndex) {
92  		final int elementCount = getElementCount(bytes, offset);
93  		if(elementIndex < 0 || elementIndex >= elementCount) {
94  			throw new ArrayIndexOutOfBoundsException("Ung¸ltiger Index " + elementIndex + " beim Zugriff auf Array " + getName() + ", Arraygrˆﬂe: " + elementCount);
95  		}
96  		return offset + getRelativeOffset(bytes, offset, elementIndex);
97  
98  
99  	}

101 	public AttributeInfo getElementInfo() {
102 		return _elementInfo;
103 	}

105 	private int getRelativeOffset(byte[] bytes, int offset, int elementIndex) {
106 		final AttributeDefinitionInfo definitionInfo = getDefinitionInfo();
107 		if(definitionInfo.isSizeFixed()) {
108 			return 4 + elementIndex * definitionInfo.getFixedSize();
109 		}
110 		int size = 4;
111 		for(int i = 0; i < elementIndex; ++i) {
112 			size += definitionInfo.getSize(bytes, offset + size);
113 		}
114 		return size;
115 	}
