21  package de.bsvrz.dav.daf.main.config;
22  
23  /**
24   * Beschreibt die Verwendung einer Menge im Kontext eines bestimmten Objekt-Typs. Zu jeder Mengen-Verwendung,
25   * die bei einem Objekt-Typ konfiguriert ist, wird angegeben, welcher Name die Menge haben muss, von welchem
26   * Mengen-Typ sie sein muss und ob sie an jedem Objekt des jeweiligen Objekt-Typs vorhanden sein muss bzw.
27   * darf.
28   *
29   * @author Kappich+Kniß Systemberatung Aachen (K2S)
30   * @author Roland Schmitz (rs)
31   * @author Stephan Homeyer (sth)
32   * @version $Revision: 5052 $ / $Date: 2007-08-31 20:02:55 +0200 (Fri, 31 Aug 2007) $ / ($Author: rs $)
33   */
34  public interface ObjectSetUse extends ConfigurationObject {
35  	/**
36  	 * Liefert den persistenten Namen, den eine Menge dieser Verwendung haben muss.
37  	 *
38  	 * @return Name einer Menge dieser Verwendung.
39  	 */
40  	public String getObjectSetName();
41  
42  	/**
43  	 * Bestimmt den Typ dieser Menge. Der Mengentyp enthält Beschränkungen über den
44  	 * Typ und die Anzahl der in einer Menge enthaltenen Objekte und eine
45  	 * Information darüber, ob Elemente online hinzugefügt oder entfernt werden
46  	 * dürfen. Die Methode entspricht mit Ausnahme des Rückgabetyps der Methode
47  	 * {@link SystemObject#getType}.
48  	 *
49  	 * @return Mengentyp einer Menge dieser Verwendung.
50  	 */
51  	public ObjectSetType getObjectSetType();
52  
53  	/**
54  	 * Liefert eine Information darüber, ob die Verwendung der Menge bei einem Objekt des jeweiligen Objekt-Typs
55  	 * erforderlich ist.
56  	 *
57  	 * @return <code>true</code>, wenn die Menge vorhanden sein muss;<br> <code>false</code>, wenn die Menge
58  	 *         vorhanden sein darf.
59  	 */
60  	public boolean isRequired();
61  }
