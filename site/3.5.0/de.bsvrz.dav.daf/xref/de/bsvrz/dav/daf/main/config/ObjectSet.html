21  package de.bsvrz.dav.daf.main.config;
22  
23  /**
24   * Schnittstelle zum Zugriff auf die Eigenschaften von Objekt-Mengen. Einem Konfigurationsobjekt können
25   * mehrere Mengen zugeordnet sein. Die verschiedenen Mengen werden mit ihrem permanenten Mengen-Namen
26   * unterschieden. Eine Menge enthält Verweise auf andere System-Objekte. Bei Objekt-Mengen wird unterschieden
27   * zwischen online änderbaren und nicht online änderbaren Mengen. Bei online änderbaren Mengen können zur
28   * Laufzeit Objekte hinzugefügt und entfernt werden. Änderungen an nicht online änderbaren Mengen werden erst
29   * mit Aktivierung der nächsten Konfigurationsversion aktiv. Der Zugriff auf die speziellen Methoden von
30   * online änderbaren Mengen ist mit der Schnittstelleklasse {@link MutableSet} möglich. Der Zugriff auf die
31   * speziellen Eigenschaften und Funktionen von nicht online änderbaren Mengen ist mit der Schnittstellenklasse
32   * {@link NonMutableSet} möglich. In der vorliegenden Schnittstelle sind die Gemeinsamkeiten beider
33   * Mengenarten zusammengefasst.
34   *
35   * @author Kappich+Kniß Systemberatung Aachen (K2S)
36   * @author Roland Schmitz (rs)
37   * @author Fouad
38   * @author Stephan Homeyer (sth)
39   * @version $Revision: 5084 $ / $Date: 2007-09-03 10:42:50 +0200 (Mon, 03 Sep 2007) $ / ($Author: rs $)
40   * @see MutableSet
41   * @see NonMutableSet
42   */
43  public interface ObjectSet extends ConfigurationObject, SystemObjectCollection {
44  	/**
45  	 * Bestimmt den Typ dieser Menge. Der Mengentyp enthält Beschränkungen über den Typ und die Anzahl der in
46  	 * einer Menge enthaltenen Objekte und eine Information darüber, ob Elemente online hinzugefügt oder entfernt
47  	 * werden dürfen. Die Methode entspricht mit Ausnahme des Rückgabetyps der Methode {@link
48  	 * SystemObject#getType}.
49  	 *
50  	 * @return Typ der Menge
51  	 */
52  	public ObjectSetType getObjectSetType();
53  
54  	/**
55  	 * Erweitert die Menge um ein weiteres Element. Wenn das angegebene Element schon in der Menge enthalten ist,
56  	 * dann wird die Menge nicht verändert. Wenn der Typ des angegebenen System-Objekts in der Menge nicht
57  	 * erlaubt ist, wird die Menge nicht verändert und eine Ausnahme generiert. Wenn bei online änderbaren Mengen
58  	 * die maximale Anzahl von Objekten bereits erreicht ist, wird die Menge nicht verändert und eine Ausnahme
59  	 * generiert. Wenn die Menge nicht online änderbar ist, dann wird das neue Element erst mit Aktivierung der
60  	 * nächsten Konfigurationsversion in die Menge aufgenommen.
61  	 *
62  	 * @param object Das System-Objekt, das der Menge hinzugefügt werden soll.
63  	 * @throws ConfigurationChangeException Wenn das übergebene Objekt nicht in die Menge aufgenommen werden
64  	 *                                      konnte und noch nicht in der Menge enthalten war.
65  	 */
66  	public void add(SystemObject object) throws ConfigurationChangeException;
67  
68  	/**
69  	 * Erweitert die Menge um beliebig viele Elemente. Sind angegebene Elemente bereits in der Menge, so werden
70  	 * sie ignoriert. Ausnahmen werden generiert, u.a. wenn der Typ eines angegebenen System-Objekts in der Menge
71  	 * nicht erlaubt ist und wenn bei online änderbaren Mengen die maximale Anzahl von Objekten überschritten
72  	 * wird. Bei Ausnahmen wird die Menge nicht verändert. Wenn die Menge nicht online änderbar ist, dann werden
73  	 * die neuen Elemente erst mit Aktivierung der nächsten Konfigurationsversion in die Menge aufgenommen.
74  	 *
75  	 * @param objects Die System-Objekte, die der Menge hinzugefügt werden sollen.
76  	 * @throws ConfigurationChangeException Wenn eines der übergebenen Objekte nicht in die Menge aufgenommen
77  	 *                                      werden konnte und noch nicht in der Menge enthalten war.
78  	 */
79  	public void add(SystemObject[] objects) throws ConfigurationChangeException;
80  
81  	/**
82  	 * Entfernt ein Element der Menge. Wenn das Element nicht in der Menge enthalten ist, wird eine Ausnahme
83  	 * generiert. Wenn bei online änderbaren Mengen die minimale Anzahl von Objekten bereits erreicht ist, wird
84  	 * das Element nicht aus der Menge entfernt und eine Ausnahme generiert. Bei nicht online änderbaren Mengen
85  	 * wird das gegebene Element erst mit Aktivierung der nächsten Konfigurationsversion aus der Menge entfernt.
86  	 *
87  	 * @param object Das System-Objekt, das aus der Menge entfernt werden soll.
88  	 * @throws ConfigurationChangeException Wenn das übergebene Objekt nicht aus der Menge entfernt werden
89  	 *                                      konnte.
90  	 */
91  	public void remove(SystemObject object) throws ConfigurationChangeException;
92  
93  	/**
94  	 * Entfernt beliebige Elemente aus der Menge. Ausnahmen werden generiert, u.a. wenn ein Element nicht in der
95  	 * Menge enthalten ist und wenn bei online änderbaren Mengen die minimale Anzahl von Objekten bereits
96  	 * erreicht ist. Bei Ausnahmen wird die Menge nicht verändert. Bei nicht online änderbaren Mengen werden die
97  	 * angegebenen Elemente erst mit Aktivierung der nächsten Konfigurationsversion aus der Menge entfernt.
98  	 *
99  	 * @param objects Die System-Objekte, die aus der Menge entfernt werden sollen.
100 	 * @throws ConfigurationChangeException Wenn eines der übergebenen Objekte nicht aus der Menge entfernt
101 	 *                                      werden konnte.
102 	 */
103 	public void remove(SystemObject[] objects) throws ConfigurationChangeException;
104 }
