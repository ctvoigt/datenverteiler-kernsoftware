21  package de.bsvrz.dav.daf.main.config;
22  
23  import java.util.List;
24  
25  /**
26   * Schnittstelle zum Zugriff auf die beschreibenden Informationen und Einschränkungen von Mengen-Typen. Zu
27   * jedem Mengen-Typ wird konfiguriert welcher Name eine Menge dieses Typs haben muss, welche Typen von
28   * Objekten enthalten sein dürfen, wieviele Objekte mindestens und höchstens enthalten sein müssen bzw.
29   * dürfen, ob eine Menge an den entsprechenden Objekten vorhanden sein muss oder darf und ob eine Menge dieses
30   * Typs konfigurierend oder dynamisch ist.
31   *
32   * @author Kappich+Kniß Systemberatung Aachen (K2S)
33   * @author Roland Schmitz (rs)
34   * @author Stephan Homeyer (sth)
35   * @version $Revision: 5052 $ / $Date: 2007-08-31 20:02:55 +0200 (Fri, 31 Aug 2007) $ / ($Author: rs $)
36   */
37  public interface ObjectSetType extends ConfigurationObjectType {
38  	/**
39  	 * Liefert eine Liste von {@link SystemObjectType Objekt-Typen} zurück, die in Mengen dieses Typs verwendet
40  	 * werden können.
41  	 *
42  	 * @return Liste von Typ-Objekten.
43  	 */
44  	public List<SystemObjectType> getObjectTypes();
45  
46  	/**
47  	 * Liefert die Anzahl von Objekten, die mindestens in der Menge vorhanden sein müssen.
48  	 *
49  	 * @return Anzahl der mindestens in der Menge geforderten Objekte
50  	 */
51  	public int getMinimumElementCount();
52  
53  	/**
54  	 * Liefert die Anzahl von Objekten, die höchstens in der Menge vorhanden sein dürfen.
55  	 *
56  	 * @return Anzahl der höchstens in der Menge erlaubten Objekte oder <code>0</code> falls die Anzahl
57  	 *         unbegrenzt ist.
58  	 */
59  	public int getMaximumElementCount();
60  
61  	/**
62  	 * Liefert zurück, ob eine Menge dieses Typs online änderbar ist. Mengen, deren Typ vom Typ DynamischeMenge
63  	 * abgeleitet ist, sind online änderbar, d.h. es können Objekte online hinzugefügt und entfernt werden.
64  	 * Mengen, deren Typ vom Typ KonfigurationsMenge abgeleitet ist, sind nicht online änderbar, d.h. nur mit
65  	 * einer neuen Konfigurationsversion können Objekte hinzugefügt und entfernt werden.
66  	 *
67  	 * @return <code>true</code>, wenn die Menge online änderbar ist;<br> <code>false</code>, wenn die Menge
68  	 *         nicht online änderbar ist.
69  	 */
70  	public boolean isMutable();
71  
72  	/**
73  	 * Liefert die Referenzierungsart des Mengen-Typs zurück. Von der Konfiguration werden folgende
74  	 * Referenzierungsarten unterschieden: <ul> <li>Gerichtete Assoziation</li> <li>Aggregation</li>
75  	 * <li>Komposition</li> </ul>
76  	 *
77  	 * @return die Referenzierungsart des Mengen-Typs
78  	 */
79  	public ReferenceType getReferenceType();
80  }
