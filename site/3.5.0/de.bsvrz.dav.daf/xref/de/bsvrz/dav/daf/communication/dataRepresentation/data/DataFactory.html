22  package de.bsvrz.dav.daf.communication.dataRepresentation.data;

24  import de.bsvrz.dav.daf.main.Data;
25  import de.bsvrz.dav.daf.main.config.AttributeGroup;
26  import de.bsvrz.dav.daf.main.config.DataModel;
27  import de.bsvrz.dav.daf.communication.dataRepresentation.data.info.version1.ConcreteDataFactory;

29  /**
30   * Klasse, die zum Erzeugen von Datensätzen aus einem Byte-Array mit verschiedenen Serialisiererversionen verwendet werden kann.
31   *
32   * @author Kappich Systemberatung
33   * @version $Revision: 5055 $
34   */
35  public abstract class DataFactory {
36  
37  	/**
38  	 * Liefert eine konkrete DataFactory-Implementierung für eine vorgegebene Serialisierungsversion zurück.
39  	 * @param version Version der gewünschten Serialisierung. Momentan wird hier nur die Version 1 unterstützt.
40  	 * @return Konkrete DataFactory-Implementierung für die angegebene Serialisierungsversion.
41  	 */
42  	public static DataFactory forVersion(int version) {
43  		switch(version) {
44  			case 1:
45  				return ConcreteDataFactory.getInstance();
46  			default:
47  				throw new IllegalArgumentException("Serialisierungsversion wird nicht unterstützt: " + version);
48  		}
49  	}
50  
51  	/**
52  	 * Löscht die für eine Datenverteilerverbindung bzw. eine DataModel-Implementierung zwischengespeicherten Datenstrukturen. Diese Methode sollte
53  	 * aufgerufen werden, wenn eine Datenverteilerverbindung bzw. eine DataModel-Implementierung nicht mehr zur Verfügung steht und die zwischengespeicherten
54  	 * Datenstrukturen freigegeben werden sollen.
55  	 * @param dataModel DataModel
56  	 */
57  	public static void forget(DataModel dataModel) {
58  		ConcreteDataFactory.forgetDataModel(dataModel);
59  	}
60  
61  	/**
62  	 * Erzeugt ein nicht modifizierbares Data-Objekt einer vorgegebenen Attributgruppe aus einem serialisierten Datensatz.
63  	 * @param atg Attributgruppe des Datensatzes
64  	 * @param bytes Serialisierter Datensatz
65  	 * @return Nicht modifizierbares Data-Objekt zum Zugriff auf den Datensatz.
66  	 */
67  	public abstract Data createUnmodifiableData(AttributeGroup atg, byte[] bytes);
68  
69  	/**
70  	 * Erzeugt ein modifizierbares Data-Objekt einer vorgegebenen Attributgruppe aus einem serialisierten Datensatz.
71  	 * @param atg Attributgruppe des Datensatzes
72  	 * @param bytes Serialisierter Datensatz
73  	 * @return Modifizierbares Data-Objekt zum Zugriff auf den Datensatz.
74  	 */
75  	public abstract Data createModifiableData(AttributeGroup atg, byte[] bytes);
