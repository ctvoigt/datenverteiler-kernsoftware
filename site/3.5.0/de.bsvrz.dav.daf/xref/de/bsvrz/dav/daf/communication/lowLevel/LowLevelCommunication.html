46  /**
47   * Klasse zur Verwaltung der unteren Ebene von Datenverteilerverbindungen.
48   * <p/>
49   * Die Aufgabe dieser Klasse ist es Telegramme zu versenden, zu empfangen und KeepAlive-Mechanismus zu verwalten.
50   * <p/>
51   * Jede dieser Aufgaben wird durch mindestens einen Thread übernommen:
52   * <p/>
53   * Empfang von Daten: ReceivingChannel (nimmt die Daten entgegen und legt diese in _receivingTable ab. Bei Konfigurationsanfragen kann es zur Zerstücklung von
54   * Telegrammen kommen, diese Teiltelgramme werden in _splittedTelegramsTable abgelegt und dort mit jedem empfang eines Teiltelegramms langsam zusammen gebaut
55   * und nach Vollendung an ClientHighLevelCommunication durchgereicht (nicht über den Worker-Thread). Beim empfang von Daten wird der KeepAlive-Thread
56   * benachrichtigt.), WorkerThread (bearbeitet die _receivingTable und reicht empfangene Telegramme an ClientHighLevelCommunication weiter).
57   * <p/>
58   * Versand von Daten: SendingChannel (verschickt Telegramme aus der _sendingTable und benachrichtigt den KeepAlive Thread, sobald Daten verschickt wurden. Prüft
59   * die Laufzeit der einzelnen Telegramme und unterbricht die Verbindung sobald diese einen kritischen Wert unterschreitet).
60   * <p/>
61   * KeepAlive-Mechanismus: KeepAliveThread (Verwaltet ob ein "KeepAlive" Telegramm verschickt werden muss. Dies ist abhängig davon, ob eine Telegramm
62   * versendet/empfangen wurde. Wurde eine bestimmte Anzahl Keepalive Telegramme versendet, so wird die Verbindung abgebaut. Wird nach dem versand eines KeepAlive
63   * Telegramms ein normales Telegramm empfangen/gesendet, so wird der KeepAlive-Telegrammzähler wieder auf 0 gesetzt. Anmerkung: Das KeepAlive Paket wird nicht
64   * über den SendingChannel-Thread verschickt, sondern direkt über den Outputstream.)
65   * <p/>
66   * Alle vollständigen Telegramme werden in zwei TelegramQueue-Objekten gespeichert (eine für den Empfang, eine für den Versand). Die oben beschriebenen Threads (ausser
67   * der KeepAlive-Thread) arbeiten und synchronisieren sich auf den jeweiligen "PriorityTables".
68   * <p/>
69   * <p/>
70   * Begriffsdefinitionen: Protokollsteuerung DaV-DAF (Klasse ClientHighLevelCommunication), Datenrepräsentation (Klassen DataInputStream und DataOutputStream),
71   * TCP-Kommunikationskanal (Klasse TCP_IP_Communication)
72   *
73   * @author Kappich Systemberatung
74   * @version $Revision: 9281 $
75   */

154 	/**
155 	 * @param connection              Verbindungsobjekt über dass die Kommunikation mit dem Kommunikationspartner realisiert wird.
156 	 * @param sendBufferSize          Sendetabellenkapazität (in Byte)
157 	 * @param receiveBufferSize       Empfangstabellenkapazität (in Byte)
158 	 * @param keepAliveSendTimeOut    Zeitspanne in ms. Wird solange kein Telegramm verschickt, wird ein KeepAlive-Telegramm verschickt.
159 	 * @param keepAliveReceiveTimeOut Zeitspanne in ms. Wird solange keine Telegramm empfangen, wird ein Zähler herabgesetzt. Erreicht der Zähler 0 wird die
160 	 *                                Verbindung terminiert. Wird zwischendurch ein Telegramm empfangen, wird der Zähler auf das maximum gesetzt.
161 	 * @param mode                    Modus für das Konfigurationsdatenverhalten. Falls hier der Wert {@link #HANDLE_CONFIG_RESPONCES_MODE} übergeben wird, dann
162 	 *                                werden Konfigurationsantworten ausgewertet und entsprechend an die höhere Kommunikationsebene weitergegeben. Falls hier der
163 	 *                                Wert {@link #NORMAL_MODE} übergeben wird, dann werden Konfigurationsantworten wie normale Datentelegramme verarbeitet.
164 	 * @param connected               Information, ob die Verbindung bereits erfolgt ist oder nicht (connected)
165 	 *
166 	 * @throws de.bsvrz.dav.daf.main.ConnectionException Wenn das Verbindungsobjekt sich nicht im erwarteten Zustand befindet.
167 	 */
168 	public LowLevelCommunication(
169 			ConnectionInterface connection,
170 			int sendBufferSize,
171 			int receiveBufferSize,
172 			long keepAliveSendTimeOut,
173 			long keepAliveReceiveTimeOut,
174 			byte mode,
175 			boolean connected) throws ConnectionException {
176 		if(connection == null) {
177 			throw new ConnectionException("Keine Kommunikationskomponente vorhanden.");
178 		}
179 		_connection = connection;
180 		_mode = mode;
181 		if(_mode == HANDLE_CONFIG_RESPONCES_MODE) {
182 			_splittedTelegramsTable = new SplittedApplicationTelegramsTable();
183 		}
184 		_sendQueue = new TelegramQueue<DataTelegram>(sendBufferSize, CommunicationConstant.MAX_PRIORITY);
185 		_throughputChecker = new ThroughputChecker();
186 		_receiveQueue = new TelegramQueue<DataTelegram>(receiveBufferSize, CommunicationConstant.MAX_PRIORITY);
187 		_keepAliveSendTimeOut = keepAliveSendTimeOut;
188 		_keepAliveReceiveTimeOut = keepAliveReceiveTimeOut;
189 		if(connected) {
190 			OutputStream outputStream = _connection.getOutputStream();
191 			if(outputStream == null) {
192 				throw new ConnectionException("Inkonsistente Kommunikationskomponente.");
193 			}
194 			_outStream = new DataOutputStream(new BufferedOutputStream(outputStream));
195 
196 			InputStream inputStream = _connection.getInputStream();
197 			if(inputStream == null) {
198 				throw new ConnectionException("Inkonsistente Kommunikationskomponente.");
199 			}
200 			_inStream = new DataInputStream(new BufferedInputStream(inputStream));
201 			_disconnected = false;
202 			setRemoteAddress(_connection.getMainAdress(), _connection.getSubAdressNumber());
203 		}
204 		else {
205 			_disconnected = true;
206 		}
207 	}

338 	/**
339 	 * Diese Methode wird von der Protokollschicht DaV-DAF aufgerufen, wenn die Kommunikationskanäle geschlossen werden sollen.
340 	 * <p/>
341 	 * Zunächst wird der Empfangsthread geschlossen; dadurch werden keine Daten mehr empfangen. Danach wird der Keep-alive-Thread geschlossen. Daraufhin wird
342 	 * gewartet, bis der Sendethread alle Telegramme aus der Sendetabelle gesendet hat, damit keine Daten verloren gehen. Anschließend wird der Sendethread
343 	 * beendet. Zuletzt werden die Kommunikationskanäle geschlossen.
344 	 *
345 	 * @param error   Besagt, ob es sich um eine Terminierung mit Fehler handelt. Falls <code>true</code> werden sämtliche noch zum Versand gepufferten Telegramme
346 	 * verworfen; falls <code>false</code> wird versucht, sämtliche zum Versand gepufferten Telegramme zu versenden.
347 	 * @param message Ursache der Terminierung im Fehlerfall.
348 	 * @param terminationTelegram Das Telegramm, dass als letztes Telegramm vor dem Schließen der Verbindung versendet werden soll oder <code>null</code>, falls
349 	 * kein abschließendes Telegramm versendet werden soll.
350 	 */
351 	public final void disconnect(boolean error, String message, final DataTelegram terminationTelegram) {
352 		if(_waitingForSendingChannel) {
353 			_sendQueue.abort();
354 			try {
355 				Thread.sleep(1000);
356 			}
357 			catch(InterruptedException e) {
358 			}
359 			if(_sendingChannel != null) _sendingChannel.interrupt();
360 			return;
361 		}
362 		synchronized(this) {
363 			if(_disconnected) {
364 				return;
365 			}
366 			_disconnected = true;
367 		}
368 		_debug.fine(getRemotePrefix() + "LowLevelCommunication.disconnect: Verbindung wird geschlossen, error=" + error + ", message=" + message);
369 		if(_sendTerminationTelegramWhenClosing) {
370 			_sendTerminationTelegramWhenClosing = false;
371 			_terminationTelegram = terminationTelegram;
372 		}
373 
374 		if(error) {
375 			_sendQueue.abort();
376 		}
377 		else {
378 			_sendQueue.close();
379 		}
380 
381 		try {
382 			_waitingForSendingChannel = true;
383 			while(_sendQueue.getSize() > 0 && _connection.isConnected() && _sendingChannel != null && _sendingChannel.isAlive()) {
384 				_debug.fine("Warte auf den Versand von gepufferten Telegrammen");
385 				Thread.sleep(1000);
386 			}
387 			_waitingForSendingChannel = false;
388 		}
389 		catch(InterruptedException e) {
390 			_debug.info(getRemotePrefix() + Thread.currentThread().getName() + " wurde beim Senden von gepufferten Telegrammen beim Verbindungsabbau unterbrochen");
391 		}
392 
393 		if(_sendingChannel != null) {
394 			try {
395 				_sendingChannel.join(2000);
396 			}
397 			catch(InterruptedException ignored) {
398 			}
399 			_sendingChannel.interrupt();
400 		}
401 
402 		if(_keepAliveThread != null) {
403 			_keepAliveThread.interrupt();
404 		}
405 
406 		if(_receivingChannel != null) {
407 			_receivingChannel.interrupt();
408 		}
409 
410 		if(error) {
411 			_receiveQueue.abort();
412 		}
413 		else {
414 			_receiveQueue.close();
415 		}
416 		try {
417 			while(_receiveQueue.getSize() > 0 && _connection.isConnected()) {
418 				_debug.fine("Warte auf die Verarbeitung von gepufferten empfangenen Telegrammen");
419 				Thread.sleep(1000);
420 			}
421 		}
422 		catch(InterruptedException e) {
423 			_debug.info(getRemotePrefix() + Thread.currentThread().getName() + " wurde beim Verarbeiten von gepufferten Telegrammen beim Verbindungsabbau unterbrochen");
424 		}
425 
426 		_connection.disconnect();
427 
428 		try {
429 			if(_inStream != null) {
430 				_inStream.close();
431 			}
432 		}
433 		catch(IOException ex) {
434 		}
435 		try {
436 			if(_outStream != null) {
437 				_outStream.close();
438 			}
439 		}
440 		catch(IOException ex) {
441 		}
442 	}

446 		if(_highLevelComponent != null) {
447 			final Runnable runnable = new Runnable() {
448 
449 				public void run() {

451 				}
452 			};
453 			Thread disconnectNotifierThread = new Thread(runnable, "LowLevelCommunication-disconnectNotifier");
454 			disconnectNotifierThread.setDaemon(true);
455 			disconnectNotifierThread.start();
456 		}
457 	}

487 	/**
488 	 * Dieser Thread verschickt Telegramme mittels einer Datenverteilerverbindung (Outputstream der Verbindung). Der Thread synchronisiert sich auf der
489 	 * Datenstruktur <code>_sendingTable</code>, die alle zu versendenden Telegramme enthält.
490 	 */
491 	class SendingChannel extends Thread {
492 
493 		public SendingChannel() {
494 			super("SendingChannel");
495 		}
496 
497 		public final void run() {
498 			_debug.fine(getRemotePrefix() + "Thread LowLevelCommunication.SendingChannel startet");
499 			try {
500 				DataTelegram telegram = null;
501 				while(!interrupted() && (telegram = _sendQueue.take()) != null) {
502 					synchronized(_outStream) {
503 //						_debug.info(">>>>>Telegram wird gesendet", telegram);
504 						_outStream.writeByte(telegram.getType());
505 						telegram.write(_outStream);
506 						_outStream.flush();
507 					}
508 					_keepAliveThread.sentTelegram();
509 					_throughputChecker.sentTelegram(telegram.getSize());
510 				}
511 				if(_terminationTelegram != null) {
512 					_debug.info(getRemotePrefix() + Thread.currentThread().getName() + " sendet ein Terminierungstelegramm, weil die Sende-Queue geschlossen wurde");
513 					_outStream.writeByte(_terminationTelegram.getType());
514 					_terminationTelegram.write(_outStream);
515 					_outStream.flush();
516 				}
517 				_debug.info(getRemotePrefix() + Thread.currentThread().getName() + " beendet sich jetzt weil die Sende-Queue geschlossen wurde");
518 			}
519 			catch(InterruptedException e) {
520 				_debug.info(getRemotePrefix() + Thread.currentThread().getName() + " wurde unterbrochen und beendet sich jetzt");
521 				_sendQueue.abort();
522 			}
523 			catch(IOException ex) {
524 				handleAbnormalBehaviour(
525 						false, "Verbindung wird wegen eines Kommunikationsfehlers beim Senden terminiert: " + ex.getMessage()
526 				);
527 			}
528 			finally {
529 				_debug.fine(getRemotePrefix() + "Thread LowLevelCommunication.SendingChannel beendet sich");
530 			}
531 			return;
532 		}
533 	}

541 		/**

637 						return;
638 					}
639 				}

683 	private class ThroughputChecker {
684 
685 		/** Anzahl Bytes im Sendepuffer, ab dem die Durchsatzprüfung gestartet wird. */
686 		private int _buffersizeThreshold;
687 
688 		/** Die Zeit zwischen zwei Messungen der Durchsatzprüfung in Millisekunden */
689 		private long _controlInterval;
690 
691 		/** Minimaler Sendedurchsatz für die Durchsatzprüfung in Bytes pro Sekunde */
692 		private int _minimumThroughput;
693 
694 		ThroughputCheckerState _state;
695 
696 		private long _stateChangeTime;
697 
698 
699 		private int _numberOfBytesSent;
700 
701 		private long _lastCheckedThroughput;
702 
703 		public ThroughputChecker() {
704 			setThroughputParameters(
705 					CommunicationConstant.FLOW_CONTROL_FACTOR, CommunicationConstant.THROUGHPUT_CONTROL_INTERVAL, CommunicationConstant.MINIMUM_THROUGHPUT
706 			);
707 			setState(ThroughputCheckerState.EMPTY_BUFFER);
708 		}
709 
710 		/**
711 		 * Diese Methode setzt die Parameter für die Durchsatzprüfung.
712 		 *
713 		 * @param throughputControlSendBufferFactor
714 		 *                                  Füllungsgrad des Sendepuffers als Faktor zwischen 0 und 1, ab dem die Durchsatzprüfung anfängt zu arbeiten.
715 		 * @param throughputControlInterval Zeit zwischen zwei Durchsatzprüfungen in Millisekunden
716 		 * @param minimumThroughput         Minimal zulässiger Verbindungsdurchsatz in Bytes pro Sekunde
717 		 */
718 		public synchronized final void setThroughputParameters(float throughputControlSendBufferFactor, long throughputControlInterval, int minimumThroughput) {
719 			if(throughputControlInterval <= 0) {
720 				throw new IllegalArgumentException(getRemotePrefix() + "Prüfintervall für Durchsatzprüfung ist zu klein: " + throughputControlInterval + " ms");
721 			}
722 			if(minimumThroughput <= 0) {
723 				throw new IllegalArgumentException(getRemotePrefix() + "Minimal Durchsatz für Durchsatzprüfung ist zu klein: " + _controlInterval + " Byte/s");
724 			}
725 			if(throughputControlSendBufferFactor <= 0.0) {
726 				throw new IllegalArgumentException(getRemotePrefix() + "Pufferfüllgrad für Durchsatzprüfung ist zu klein: " + throughputControlSendBufferFactor);
727 			}
728 			if(throughputControlSendBufferFactor >= 1.0) {
729 				throw new IllegalArgumentException(getRemotePrefix() + "Pufferfüllgrad für Durchsatzprüfung ist zu groß: " + throughputControlSendBufferFactor);
730 			}
731 			_buffersizeThreshold = (int)(throughputControlSendBufferFactor * _sendQueue.getCapacity());
732 			_controlInterval = throughputControlInterval;
733 			_minimumThroughput = minimumThroughput;
734 		}
735 
736 		private void setState(final ThroughputCheckerState state) {
737 			_debug.fine(getRemotePrefix() + "Zustand der Durchsatzprüfung", state);
738 			_debug.fine(getRemotePrefix() + "noch zu versendende Daten ", _sendQueue.getSize() + " Byte, Grenze: " + _buffersizeThreshold + " Byte");
739 			_state = state;
740 			_stateChangeTime = System.currentTimeMillis();
741 			_numberOfBytesSent = 0;
742 			_lastCheckedThroughput = -1;
743 		}
744 
745 		public synchronized void queuedTelegram() {
746 			switch(_state) {
747 				case EMPTY_BUFFER:
748 					if(_sendQueue.getSize() > _buffersizeThreshold) {
749 						setState(ThroughputCheckerState.FULL_BUFFER);
750 					}
751 					break;
752 				case FULL_BUFFER:
753 					// fall through
754 				case CHECKING_THROUGHPUT:
755 					if(_sendQueue.getSize() < _buffersizeThreshold) {
756 						setState(ThroughputCheckerState.EMPTY_BUFFER);
757 					}
758 					break;
759 			}
760 		}
761 
762 		public synchronized void sentTelegram(int telegramSize) {
763 			switch(_state) {
764 				case EMPTY_BUFFER:
765 					break;
766 				case FULL_BUFFER:
767 					if(_sendQueue.getSize() < _buffersizeThreshold) {
768 						setState(ThroughputCheckerState.EMPTY_BUFFER);
769 					}
770 				case CHECKING_THROUGHPUT:
771 					if(_sendQueue.getSize() < _buffersizeThreshold) {
772 						setState(ThroughputCheckerState.EMPTY_BUFFER);
773 					}
774 					else {
775 						_numberOfBytesSent += telegramSize;
776 					}
777 					break;
778 			}
779 		}
780 
781 		/**
782 		 * Prüft den Durchsatz und liefert die Zeit bis zur nächsten Prüfung zurück.
783 		 *
784 		 * @return Zeit bis zur nächsten Prüfung in Millisekunden
785 		 *
786 		 * @throws IllegalStateException wenn ein zu geringer Durchsatz festgestellt wurde.
787 		 */
788 		public synchronized long checkThroughput() {
789 			switch(_state) {
790 				case EMPTY_BUFFER:
791 					break;
792 				case FULL_BUFFER:
793 					long bufferfullTimeout = _controlInterval - (System.currentTimeMillis() - _stateChangeTime);
794 					if(bufferfullTimeout > 0) {
795 						return bufferfullTimeout;
796 					}
797 					setState(ThroughputCheckerState.CHECKING_THROUGHPUT);
798 					break;
799 				case CHECKING_THROUGHPUT:
800 					long checkingTime = System.currentTimeMillis() - _stateChangeTime;
801 					long checkingTimeout = _controlInterval - checkingTime;
802 					if(checkingTimeout > 0) {
803 						return checkingTimeout;
804 					}
805 					final long throughput = (long)_numberOfBytesSent * 1000L / checkingTime;
806 					_debug.info(getRemotePrefix() + "Sendedurchsatz: " + throughput + " Byte/s");
807 					if(throughput < _minimumThroughput) {
808 						_lastCheckedThroughput = throughput;
809 						throw new IllegalStateException(getRemotePrefix() + "Sendedurchsatz war in den letzten " + checkingTime + " ms zu gering: " + throughput + " Byte/s");
810 					}
811 					setState(ThroughputCheckerState.CHECKING_THROUGHPUT);
812 					_lastCheckedThroughput = throughput;
813 					break;
814 			}
815 			return _controlInterval;
816 		}

818 		/**
819 		 * Liefert einen beschreibenden Text mit dem Zustand des Sendepuffers
820 		 * @return Zustand des Sendepuffers
821 		 */
822 		public synchronized String getSendBufferState() {
823 			final StringBuilder text = new StringBuilder();
824 			text.append(_sendQueue.getSize()).append(" Byte");
825 			switch(_state) {
826 				case EMPTY_BUFFER:
827 					break;
828 				case FULL_BUFFER:
829 					text.append(", Puffer voll");
830 					break;
831 				case CHECKING_THROUGHPUT:
832 					text.append(", Druchsatzprüfung");
833 					if(_lastCheckedThroughput >= 0) text.append(" ").append(_lastCheckedThroughput).append("Byte/s"); 
834 					break;
835 			}
836 			return text.toString();
837 		}

841 	 * Dieser Thread verschickt Keepalive Telegramme und baut die Verbindung ab, wenn dreimal nacheinander eine bestimmte Zeit lang keine Daten mehr empfangen
842 	 * wurden.
843 	 * <p/>
844 	 * KeepAlive-Thread Senden: Wurde einen bestimmten Zeitraum kein Telegramm mehr verschickt, wird ein Keep-Alive Telegramm verschickt.
845 	 * <p/>
846 	 * KeepAlive-Thread Empfangen: Wurde dreimal nacheinander eine bestimmte Zeit kein Telegramm empfangen, wird die Verbindung abgebaut.
847 	 */

850 		/** Maximale Anzahl für das Ablaufen des Empfangstimeouts bevor die Verbindung terminiert wird. */
851 		private static final byte MAX_SOULS = 3;
852 
853 		/** Aktuelle noch verbleibende Anzahl für das Ablaufen des Empfangstimeouts bevor die Verbindung terminiert wird. */
854 		private int _souls;

856 		/** Zeitpunkt, an dem zuletzt angefangen wurde auf den Empfang von Daten zu warten */
857 		private long _lastStartReceivingTime;

870 		 */
871 		private static final int NOT_RECEIVING_MULTIPLIER = 3;
872 
873 		private boolean _waitingForData;
874 
875 		public KeepAliveThread() {
876 			super("KeepAlive");
877 			_souls = MAX_SOULS;
878 			final long now = System.currentTimeMillis();
879 			_lastReceivingTime = now;

886 				_lastStartReceivingTime = System.currentTimeMillis();

889 			}

893 			synchronized(this) {
894 				_lastReceivingTime = System.currentTimeMillis();

897 			}

902 				_lastSendingTime = System.currentTimeMillis();
903 			}

907 			synchronized(this) {
908 				// run-Methode s.u. soll das wait vorzeitig verlassen und die neuen Einstellungen berücksichtigen
909 				this.notifyAll();
910 			}

913 		/**
914 		 * Diese Methode dient dazu JUnit-Tests zu unterstützen.
915 		 *
916 		 * @return Anzahl Versuche, bis die Verbindung terminiert wird. Sobald ein Telegramm verschickt wird, wird <code>_souls</code> wieder auf den maximalen Wert
917 		 *         gesetzt.
918 		 */
919 		int getSouls() {
920 			synchronized(this) {
921 				return _souls;
922 			}
923 		}

925 		public final void run() {
926 			_debug.fine(getRemotePrefix(), "Thread LowLevelCommunication.KeepAliveThread startet");
927 			try {
928 				while(!interrupted()) {
929 					synchronized(this) {
930 						try {
931 							long lastSendOrQueuedTime = _lastSendingTime < _lastQueuedKeepAliveTime ? _lastQueuedKeepAliveTime : _lastSendingTime;
932 							long sendingRemainingTime = _keepAliveSendTimeOut - (System.currentTimeMillis() - lastSendOrQueuedTime);
933 							if(sendingRemainingTime <= 0) {
934 								// Wenn noch ein Telegramm in der sendQueue ist, dann wird das KeepAliveTelegramm unterdrückt, weil überflüssig
935 								if(_connection.isConnected() && (_outStream != null) && _sendQueue.getSize() == 0) {
936 									DataTelegram telegram = new KeepAliveTelegram();
937 									_sendQueue.put(telegram);
938 									_throughputChecker.queuedTelegram();
939 								}
940 								_lastQueuedKeepAliveTime = System.currentTimeMillis();
941 								sendingRemainingTime = _keepAliveSendTimeOut;
942 							}
943 
944 							long receivingRemainingTime = _keepAliveReceiveTimeOut - (System.currentTimeMillis() - _lastReceivingTime);
945 
946 							if(receivingRemainingTime <= 0) {
947 								final long now = System.currentTimeMillis();
948 								_lastReceivingTime = now;
949 								receivingRemainingTime = _keepAliveReceiveTimeOut;
950 								--_souls;
951 								if(_waitingForData) {
952 									final long delta = now - _lastReceivingTime;
953 									_debug.fine(
954 											getRemotePrefix() + "Seit " + delta + "ms wurden keine Telegramme mehr empfangen, verbleibende Versuche: " + _souls
955 									);
956 								}
957 								else {
958 									final long delta = now - _lastStartReceivingTime;
959 									_debug.fine(
960 											getRemotePrefix() + "Seit " + delta
961 											+ "ms konnten keine Telegramme mehr empfangen werden, weil die Empfangswarteschlange voll ist, verbleibende Versuche: "
962 											+ _souls
963 									);
964 								}
965 							}
966 							if(_souls <= 0) {
967 								_debug.error(getRemotePrefix() + "Die Verbindung wird terminiert, weil keine Telegramme mehr empfangen werden.");
968 								if(_waitingForData) {
969 									handleAbnormalBehaviour(
970 											false,
971 											"Es wurden " + MAX_SOULS + " mal in Folge für jeweils " + (_keepAliveReceiveTimeOut / 1000)
972 											+ " Sekunden keine KeepAlive- oder sonstige Telegramme empfangen"
973 									);
974 								} else {
975 									handleAbnormalBehaviour(
976 											false,
977 											"Es wurden " + (MAX_SOULS * NOT_RECEIVING_MULTIPLIER) + " mal in Folge für jeweils " + (_keepAliveReceiveTimeOut / 1000)
978 											+ " Sekunden keine KeepAlive- oder sonstige Telegramme empfangen, weil die Empfangswarteschlange voll ist"
979 									);
980 								}
981 								return;
982 							}
983 							long waitTime = sendingRemainingTime < receivingRemainingTime ? sendingRemainingTime : receivingRemainingTime;
984 
985 							// Durchsatzprüfung durchführen und Zeit bis zur nächsten Prüfung ermitteln
986 							try {
987 								final long throughputCheckWaitTime = _throughputChecker.checkThroughput();
988 								if(throughputCheckWaitTime < waitTime) waitTime = throughputCheckWaitTime;
989 							}
990 							catch(IllegalStateException e) {
991 								// Durchsatz zu gering
992 								_debug.error(getRemotePrefix() + "Die Verbindung wird terminiert: " + e.getMessage());
993 								handleAbnormalBehaviour(false, e.getMessage());
994 								return;
995 							}
996 
997 							// Hier wird gewartet, bis eine der Restzeiten abgelaufen ist.
998 							// Wenn die Timeoutparameter sich ändern, wird durch ein notify aus der Methode timeoutsChanged
999 							// das wait() vorzeitig beendet.
1000 							if(waitTime > 0) this.wait(waitTime);
1001 						}
1002 						catch(InterruptedException e) {
1003 							return;
1004 						}
1005 					}
1006 				}
1007 			}
1008 			catch(Exception e) {
1009 				_debug.error(getRemotePrefix() + "unerwartete Ausnahme im KeepAliveThread", e);
1010 			}
1011 			finally {
1012 				_debug.fine(getRemotePrefix() + "Thread LowLevelCommunication.KeepAliveThread beendet sich");
1013 			}
1014 		}
