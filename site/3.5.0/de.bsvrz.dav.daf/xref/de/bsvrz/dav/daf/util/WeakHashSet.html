20  package de.bsvrz.dav.daf.util;
21  
22  import java.lang.ref.Reference;
23  import java.lang.ref.SoftReference;
24  import java.lang.ref.WeakReference;
25  import java.util.AbstractSet;
26  import java.util.Iterator;
27  import java.util.WeakHashMap;
28  
29  /**
30   * Ein WeakHashSet referenziert seine Elemente nur über {@link WeakReference}s. Damit können Elemente im WeakHashSet
31   * jederzeit durch den Garbage-Collector entsorgt werden, sofern sie nirgendwo anders direkt referenziert werden.
32   * <p/>
33   * Das Hinzufügen von null-Elementen wird still ignoriert, da null-Elemente hier bedeuten, dass ein entsprechendes
34   * Objekt vom Garbage-Collector entfernt wurde. Das Hinzufgen von null kann also so interpretiert werden, dass ein
35   * bereits nicht mehr verfügbares Element hinzugefügt werden soll, was in keiner Änderung des Sets resultiert.
36   *
37   * @author Kappich Systemberatung
38   * @version $Revision: 9883 $
39   */
40  public class WeakHashSet<E> extends AbstractSet<E> {
41  	private final WeakHashMap<E, Reference<E>> _backingHashMap;
42  
43  	/**
44  	 * Erstellt ein neues WeakHashSet
45  	 */
46  	public WeakHashSet() {
47  		_backingHashMap = new WeakHashMap<E, Reference<E>>();
48  	}
49  
50  	/**
51  	 * Erstellt ein neues WeakHashSet
52  	 *
53  	 * @param initialCapacity ursprüngliche Kapazität
54  	 */
55  	public WeakHashSet(final int initialCapacity) {
56  		_backingHashMap = new WeakHashMap<E, Reference<E>>(initialCapacity);
57  	}
58  
59  	/**
60  	 * Erstellt ein neues WeakHashSet
61  	 *
62  	 * @param initialCapacity ursprüngliche Kapazität
63  	 * @param loadFactor      load Factor
64  	 */
65  	public WeakHashSet(final int initialCapacity, final float loadFactor) {
66  		_backingHashMap = new WeakHashMap<E, Reference<E>>(initialCapacity, loadFactor);
67  	}
68  
69  	/**
70  	 * Gibt einen iterator über alle verbleibenden Elemente zurück
71  	 *
72  	 * @return einen iterator über alle verbleibenden Elemente
73  	 */
74  	@Override
75  	public Iterator<E> iterator() {
76  		return _backingHashMap.keySet().iterator();
77  	}
78  
79  	/**
80  	 * Gibt die Anzahl der enthaltenen Elemente zurück. Da praktisch zu jeder Zeit Elemente vom Garbage Collector entfernt
81  	 * werden können, kann die zurückgegebene Anzahl sich von folgenden Funktionsaufrufen unterscheiden.
82  	 *
83  	 * @return die Anzahl der enthaltenen Elemente
84  	 */
85  	@Override
86  	public int size() {
87  		return _backingHashMap.size();
88  	}
89  
90  	@Override
91  	public boolean isEmpty() {
92  		return _backingHashMap.isEmpty();
93  	}
94  
95  	@Override
96  	public boolean contains(final Object o) {
97  		if(o == null) return false;
98  		return _backingHashMap.containsKey(o);
99  	}
100 
101 	/**
102 	 * Fügt ein neues Element hinzu, welches über eine {@link WeakReference} referenziert wird.
103 	 *
104 	 * @param e Element
105 	 * @see #addWeakReference(Object)
106 	 * @see #addSoftReference(Object)
107 	 */
108 	@Override
109 	public boolean add(final E e) {
110 		return addWeakReference(e);
111 	}
112 
113 	/**
114 	 * Fügt ein neues Element hinzu, welches über eine {@link WeakReference} referenziert wird. Identisch zu {@link
115 	 * #add(Object)}
116 	 *
117 	 * @param e Element
118 	 */
119 	public boolean addWeakReference(final E e) {
120 		if(e == null) return false;
121 		return _backingHashMap.put(e, new WeakReference<E>(e)) == null;
122 	}
123 
124 	/**
125 	 * Fügt ein neues Element hinzu, welches über eine {@link SoftReference} referenziert wird.
126 	 *
127 	 * @param e Element
128 	 */
129 	public boolean addSoftReference(final E e) {
130 		if(e == null) return false;
131 		return _backingHashMap.put(e, new SoftReference<E>(e)) == null;
132 	}
133 
134 	@Override
135 	public boolean remove(final Object o) {
136 		if(o == null) return false;
137 		return _backingHashMap.remove(o) == null;
138 	}
139 
140 	@Override
141 	public void clear() {
142 		_backingHashMap.clear();
143 	}
144 
145 	/**
146 	 * Gibt einen bereits gespeicherten Eintrag zurück, bei dem equals() mit dem übergebenen Objekt true liefern würde.
147 	 * Durch diese Methode kann das WeakHashSet als ein Cache benutzt werden, da hier mehrere Identische Objekte durch ein
148 	 * einzelnes Objekt mit den gleichen Eigenschaften ersetzt werden kann.
149 	 *
150 	 * @return Eintrag oder null falls kein solches Element (mehr) exstiert
151 	 */
152 	public E get(final E entry) {
153 		Reference<E> reference = _backingHashMap.get(entry);
154 		if(reference == null) return null;
155 		return reference.get();
156 	}
157 }
