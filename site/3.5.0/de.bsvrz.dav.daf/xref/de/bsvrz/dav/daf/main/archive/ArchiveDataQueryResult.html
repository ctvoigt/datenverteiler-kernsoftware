21  package de.bsvrz.dav.daf.main.archive;
22  
23  /**
24   * Schnittstelle zum Zugriff auf Ergebnisse von Archivanfragen. Diese Schnittstelle wird von Applikationen benutzt, um
25   * auf Ergebnisse von Archivdatenanfragen zuzugreifen, die mit den Methoden {@link ArchiveRequestManager#request}
26   * gestellt wurden. Eine Implementierung dieser Schnittstelle stellt neben den Methoden des übergeordneten Interfaces
27   * {@link ArchiveQueryResult} eine Methode zur Verfügung, mit der auf die Ergebnisdatenströme zugegriffen werden kann.
28   *
29   * @author Kappich Systemberatung
30   * @version $Revision: 5064 $
31   * @see ArchiveRequestManager#request(ArchiveQueryPriority,ArchiveDataSpecification)
32   * @see ArchiveRequestManager#request(ArchiveQueryPriority, java.util.List)
33   */
34  public interface ArchiveDataQueryResult extends ArchiveQueryResult {
35  	/**
36  	 * Bestimmt die Ergebnisdatenströme der entsprechenden Archivdatenspezifikationen aus der Archivanfrage. Diese Methode
37  	 * wird von einer Applikation aufgerufen, um auf die Ergebnisdatenströme zuzugreifen. Eine Implementierung dieser
38  	 * Methode muss zu jeder Archivdatenspezifikation aus der Archivanfrage einen entsprechenden Ergebnisdatenstrom
39  	 * bereitstellen.
40  	 *
41  	 * @return Array mit mit Ergebnisdatenströmen entsprechend den Archivdatenspezifikationen aus der Archivanfrage. Zu
42  	 *         jeder Archivdatenspezifikation aus der Archivanfrage wird ein korrespondierender Ergebnisdatenstrom im Array
43  	 *         erzeugt.
44  	 * @throws IllegalStateException Falls die Archivanfrage nicht erfolgreich war und keine Ergebnisdatenströme bestimmt
45  	 *                               werden können.
46  	 * @throws InterruptedException  Falls der aufrufende Thread unterbrochen wurde, während auf die entsprechende
47  	 *                               Antwortnachricht aus dem Archivsystem gewartet wurde.
48  	 * @see ArchiveRequestManager#request(ArchiveQueryPriority,ArchiveDataSpecification)
49  	 * @see ArchiveRequestManager#request(ArchiveQueryPriority, java.util.List)
50  	 */
51  	ArchiveDataStream[] getStreams() throws InterruptedException, IllegalStateException;
52  }
