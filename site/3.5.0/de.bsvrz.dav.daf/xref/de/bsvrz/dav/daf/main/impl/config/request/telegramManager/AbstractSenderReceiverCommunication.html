157 	public void init(
158 			AttributeGroup requestAtg,
159 			Aspect requestAspect,
160 			AttributeGroup responseAtg,
161 			Aspect responseAspect,
162 			short simulationVariant,
163 			DataListener dataListener
164 	) throws OneSubscriptionPerSendData {
165 
166 		// Sender von Aufträgen
167 		_requestDescription = new DataDescription(requestAtg, requestAspect, simulationVariant);
168 		_dataListener = dataListener;
169 
170 		_requester = new RequestSender();
171 		_debug.fine("Anmeldung als Sender Objekt " + _senderObject + " Datenidentifikation " + _requestDescription);
172 
173 		_connection.subscribeSender(_requester, _senderObject, _requestDescription, SenderRole.sender());
174 
175 		if(responseAtg != null && responseAspect != null) {
176 			// Senke für Antworten
177 			_responseDescription = new DataDescription(responseAtg, responseAspect, simulationVariant);
178 			_receiver = new AnswerReceiver();
179 			_connection.subscribeReceiver(_receiver, _receiverObject, _responseDescription, ReceiveOptions.normal(), ReceiverRole.drain());
180 			_subscribeReceiver = true;
181 			_debug.finer("Anmeldung als Senke Objekt " + _receiverObject + " Datenidentifikation " + _responseDescription);
182 		}
183 		else {
184 			_subscribeReceiver = false;
185 		}

199 	public void sendData(String messageType, byte[] data, int queryIndex) throws SendSubscriptionNotConfirmed, IllegalStateException {
200 		if(_dataStateReceiver != DataState.NO_RIGHTS) {
201 
202 			synchronized(_monitor) {
203 				// warten bis die Sendesteuerung eindeutig gesetzt ist
204 				while(_connectionState == ConnectionState.NotYetConnected) {
205 					try {
206 						_monitor.wait();
207 					}
208 					catch(InterruptedException e) {
209 						e.printStackTrace();
210 					}
211 				}
212 			}
213 			if(_connectionState == ConnectionState.Connected) {
214 
215 				final Data requestData = createRequestData(messageType, data, queryIndex);
216 				_debug.finer("Senden", messageType);
217 //				_debug.finest("Senden", requestData);
218 				_connection.sendData(
219 						new ResultData( _senderObject, _requestDescription, System.currentTimeMillis(), requestData)
220 				);
221 			}
222 			else {
223 				throw new SendSubscriptionNotConfirmed(
224 						"Keine Positive Sendesteuerung: " + _connectionState + " Datenidentifikation Anfrage: " + _requestDescription
225 						+ " Datenidentifikation Antwort: " + _responseDescription
226 				);
227 			}
228 		}
229 		else {
230 			throw new IllegalStateException(
231 					"Eine Konfigurationsanfrage mit Datenidentifikation " + _requestDescription
232 					+ " konnte nicht ausgeführt werden, weil für den Empfang der Antwort mit Datenidentifikation " + _responseDescription
233 					+ " keine Rechte zum empfang vorlagen. Systemobjekt Konfiguration: " + _senderObject + " Systemobjekt Applikation " + _receiverObject
234 					+ " Nachrichtentyp, der verschickt werden sollte " + messageType
235 			);
236 		}
237 	}

239 	private Data createRequestData(String messageType, byte[] message, int requestIndex) {
240 		final AttributeGroup attributeGroup = _requestDescription.getAttributeGroup();
241 		Data data = AttributeBaseValueDataFactory.createAdapter(attributeGroup, AttributeHelper.getAttributesValues(_requestDescription.getAttributeGroup()));
242 
243 		// Der Absender des Telegramms ist das Objekt, dass sich als Empfänger auf die Daten angemeldet hat
244 		data.getReferenceValue("absender").setSystemObject(_receiverObject);
245 
246 		data.getScaledValue("anfrageIndex").set(requestIndex);
247 		data.getScaledValue("nachrichtenTyp").setText(messageType);
248 		final Data.NumberArray messageArrayData = data.getUnscaledArray("daten");
249 		messageArrayData.setLength(message.length);
250 		for(int i = 0; i < message.length; i++) {
251 			messageArrayData.getValue(i).set(message[i]);
252 		}
253 		return data;
254 	}

263 	public Data waitForReply(int reqestIndex) throws RequestException {
264 		if(_subscribeReceiver) {
265 			Data reply = null;
266 			synchronized(_replyList) {
267 				try {
268 					while(reply == null) {
269 						if(_closed) throw new RequestException("Verbindung zum Datenverteiler wurde terminiert");
270 						for(Iterator iterator = _replyList.iterator(); iterator.hasNext();) {
271 							Data data = (Data)iterator.next();
272 							if(data.getScaledValue("anfrageIndex").intValue() == reqestIndex) {
273 								reply = data;
274 								break;
275 							}
276 						}
277 						if(reply == null) {
278 							_replyList.wait();
279 						}
280 					}
281 					_replyList.remove(reply);
282 				}
283 				catch(InterruptedException e) {
284 					e.printStackTrace();
285 					throw new RuntimeException(e); 
286 				}
287 			}
288 			return reply;
289 		}
290 		else {
291 			// Es wurde gar kein Empfänger für Nachrichten angemeldet. Also wird nie eine Antwort kommen
292 			throw new IllegalStateException("Es wurde keine Senke für Antworten angemeldet.");
293 		}
294 	}

302 	public void close() {
303 		synchronized(_replyList) {
304 			_closed = true;
305 			_replyList.notifyAll();
306 		}
307 		_connection.unsubscribeSender(_requester, _senderObject, _requestDescription);
308 		if(_receiver != null) {
309 			_connection.unsubscribeReceiver(_receiver, _receiverObject, _responseDescription);
310 		}
311 
312 		if(_dataListener != null) {
313 			_dataListener.close();
314 		}

318 	/** Receiverklasse, die Anworten der Konfiguration verarbeitet */
319 	private class AnswerReceiver implements ClientReceiverInterface, NonQueueingReceiver {
320 
321 		/**
322 		 * Nimmt Antworten der Konfiguration entgegen.
323 		 *
324 		 * @param results Empfangene Datensätze.
325 		 */
326 		public void update(ResultData results[]) {
327 			for(int i = 0; i < results.length; i++) {
328 				ResultData result = results[i];
329 				_debug.finer("Konfigurationsantwort erhalten");
330 				//_debug.finest("Konfigurationsantwort erhalten", result.getData());
331 
332 				// Es dürfen keine Antworten empfangen werden, da keine Rechte gesetzt wurde.
333 				// Also dürfen auch keine Anfragen verschickt werden, da die Antwort der Konfiguration
334 				// niemals empfangen werden könnte.
335 				if(result.getDataState() == DataState.NO_RIGHTS) {
336 					// Es gibt keine Rechte, also versenden von Daten verhindern
337 					_dataStateReceiver = DataState.NO_RIGHTS;
338 				}
339 				else if(_dataStateReceiver == DataState.NO_RIGHTS) {
340 					// Die Variable steht auf "nicht senden", aber ein neuer Datensatz erlaubt das
341 					// versenden von Daten. Also den Zustand ändern.
342 					// Hatte die Variable vorher einen anderen Zustand, muss dieser nicht angepaßt werden.
343 					_dataStateReceiver = result.getDataState();
344 				}
345 
346 				try {
347 					final Data data = result.getData();
348 					if(data == null || !result.hasData()) {
349 						_debug.fine("leerer Datensatz erhalten", data);
350 					}
351 					else {
352 						SystemObject replySender = data.getReferenceValue("absender").getSystemObject();
353 
354 						
355 
356 
357 
358 
359 						final boolean processTelegram;
360 						_debug.finer("Empfangen von", replySender);
361 						if(_dataListener != null) {
362 							// Soll das Telegramm vielleicht woanders bearbeitet werden und nicht durch
363 							// den normalen Mechanismus ?
364 							processTelegram = _dataListener.messageReceived(data);
365 						}
366 						else {
367 							// Es gibt kein Objekt, das Telegramm aus dem Telegramm-Strom herausziehen möchte
368 							processTelegram = true;
369 						}
370 
371 						if(processTelegram) {
372 							handleReply(data);
373 						}
374 					}
375 				}
376 				catch(Exception e) {
377 					e.printStackTrace();
378 					_debug.warning("Antwort auf eine Konfigurationsanfrage konnte nicht interpretiert werden", e);
379 				}
380 			}
381 		}
382 	}

389 		 * bzw. anzuhalten. Der Datenverteiler signalisiert damit einer Quelle oder einem Sender dass mindestens ein Abnehmer bzw. kein Abnehmer mehr für die zuvor

401 		 */
402 		public void dataRequest(SystemObject object, DataDescription dataDescription, byte state) {
403 			_debug.finer("RequestSender Sendesteuerung " + state);
404 			synchronized(_monitor) {
405 				switch(state) {
406 					case START_SENDING:
407 						_connectionState = ConnectionState.Connected;
408 						break;
409 					case STOP_SENDING:
410 						if(_connectionState != ConnectionState.NotYetConnected) _connectionState = ConnectionState.Disconnected;
411 						break;
412 					default:
413 						_connectionState = ConnectionState.Error;
414 				}
415 				_monitor.notifyAll();
416 			}
417 		}

435 	/**
436 	 * Dieses Interface ermöglich es Daten, die durch eine Instanz der Klasse {@link AbstractSenderReceiverCommunication} empfangen wurde, zu verarbeiten, bevor
437 	 * diese durch den normalen Mechanismus bearbeitet werden.
438 	 * <p/>
439 	 * Ein Beispiel wären dynamische Mengen. Die Konfigurations verschickt, sobald sich der Status einer dynamischen Menge ändert, eine Nachricht mit den
440 	 * Änderungen der Menge an alle angemeldeten Objekte, die sich für Änderungen interssieren. Für diese Nachricht gibt es kein Objekt, das aktiv {@link
441 	 * AbstractSenderReceiverCommunication#waitForReply} wartet.
442 	 * <p/>
443 	 * Die Nachricht wird mit der update-Methode empfangen und dann zuerst an ein Objekt, das dieses Interface implementiert, weitergereicht. Dort kann dann
444 	 * entschieden werden, ob das Telegramm anders verarbeitet werden soll (wie es bei dynamischen Menge der Fall ist) oder ob das Telegramm normal weitergeleitet
445 	 * werden soll (an ein Objekt, das aktiv mit  {@link AbstractSenderReceiverCommunication#waitForReply} wartet).
446 	 */
447 	public interface DataListener {
448 
449 		/**
450 		 * Diese Methode wird aufgerufen, sobald eine Instanz von {@link AbstractSenderReceiverCommunication} ein Telegramm in der update-Methode empfängt. Das
451 		 * Telegramm kann normal weiter verarbeitet werden oder aber aus dem Strom der Telegramme entfernt werden.
452 		 *
453 		 * @param data Telegramm, das empfangen wurde
454 		 *
455 		 * @return true = Das Telegramm soll normal weiterverarbeitet werden; false = Das Telegramm wurde aus dem Strom der Telegramm entfernt und soll nicht weiter
456 		 *         beachtet werden
457 		 */
458 		boolean messageReceived(Data data);
459 
460 		/** Diese Methode wird aufgerufen, wenn die Kommunikation abgebrochen werden soll. Alle Sende/Empfangsanmeldungen, Threads, usw. sind zu beenden. */
461 		void close();
462 	}
