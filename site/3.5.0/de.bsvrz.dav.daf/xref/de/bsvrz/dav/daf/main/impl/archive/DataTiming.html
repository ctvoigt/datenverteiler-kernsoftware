21  package de.bsvrz.dav.daf.main.impl.archive;
22  
23  import java.text.DateFormat;
24  import java.text.SimpleDateFormat;
25  import java.util.Date;
26  
27  /**
28   * Datenstruktur, die Datenzeitstempel, Archivzeitstempel und Datensatzindex eines Datensatzes verwaltet.
29   *
30   * @author Kappich+Kniß Systemberatung Aachen (K2S)
31   * @author Roland Schmitz (rs)
32   * @version $Revision: 5064 $ / $Date: 2007-09-01 22:25:35 +0200 (Sat, 01 Sep 2007) $ / ($Author: rs $)
33   */
34  public final class DataTiming {
35  	private final long _dataTime;
36  	private final long _archiveTime;
37  	private final long _dataIndex;
38  
39  	/**
40  	 * Legt eine neue Datenstruktur mit den übergebenen Werten von Datenzeitstempel, Archivzeitstempel und Datensatzindex
41  	 * an.
42  	 *
43  	 * @param dataTime    Datenzeitstempel des Datensatzes in Millisekunden seit 1970.
44  	 * @param archiveTime Archivzeitstempel des Datensatzes in Millisekunden seit 1970.
45  	 * @param dataIndex   Innerhalb des Archivdatencontainers eindeutiger Datensatzindex des Datenverteilers.
46  	 */
47  	public DataTiming(long dataTime, long archiveTime, long dataIndex) {
48  		_dataTime = dataTime;
49  		_archiveTime = archiveTime;
50  		_dataIndex = dataIndex;
51  	}
52  
53  	/**
54  	 * Bestimmt den Datenzeitstempel dieses Datensatzes.
55  	 *
56  	 * @return Datenzeitstempel in Millisekunden seit 1970.
57  	 */
58  	public long getDataTime() {
59  		return _dataTime;
60  	}
61  
62  	/**
63  	 * Bestimmt den Archivzeitstempel dieses Datensatzes.
64  	 *
65  	 * @return Archivzeitstempel in Millisekunden seit 1970.
66  	 */
67  	public long getArchiveTime() {
68  		return _archiveTime;
69  	}
70  
71  	/**
72  	 * Betimmt den je Archivdatenidentifikation eindeutigen Datensatzindex des Datenverteilers.
73  	 *
74  	 * @return Datensatzindex.
75  	 */
76  	public long getDataIndex() {
77  		return _dataIndex;
78  	}
79  
80  	/**
81  	 * Liefert eine textuelle Beschreibung dieses Objekts für Debug-Zwecke.
82  	 *
83  	 * @return Beschreibung dieses Objekts.
84  	 */
85  	public String toString() {
86  		DateFormat timeFormat = new SimpleDateFormat("dd.MM.yyyy HH:mm:ss,SSS");
87  		return "DataTiming{" +
88  		        "_dataTime=" + timeFormat.format(new Date(_dataTime)) +
89  		        ", _archiveTime=" + timeFormat.format(new Date(_archiveTime)) +
90  		        ", _dataIndex=" + (_dataIndex >> 32) + "#" + ((_dataIndex >> 2) & 0x3fffffff) + "#" + (_dataIndex & 0x3) +
91  		        "}";
92  	}
93  
94  
95  }
