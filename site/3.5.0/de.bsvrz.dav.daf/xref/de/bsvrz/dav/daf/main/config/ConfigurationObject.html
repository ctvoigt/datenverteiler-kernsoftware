22  package de.bsvrz.dav.daf.main.config;
23  
24  import java.util.*;

26  /**
27   * Schnittstelle zum Zugriff auf die Eigenschaften eines Konfigurationsobjektes. Konfigurationsobjekte haben neben den Eigenschaften aller System-Objekte eine
28   * verantwortliche Instanz (Zuständiger), eine Version der Konfiguration ab der das Objekt gültig ist und eine Version der Konfiguration ab der das Objekt nicht
29   * mehr gültig ist.<BR/> Die verantwortliche Instanz macht eine Aussage darüber in welcher Umgebung Konfigurationsänderungen für das Objekt durchgeführt werden
30   * können. Dies ist für Anwendungsobjekte üblicherweise eine Verkehrsrechnerzentrale oder eine Unterzentrale und für Objekte des Datenmodells eine globale
31   * Instanz.<BR/> In einer Konfiguration können Konfigurationsobjekte von verschiedenen verantwortlichen Instanzen gespeichert werden. Jede Konfiguration
32   * verwaltet für jede verantwortliche Instanz zu der sie Objekte gespeichert hat eine lokal aktivierte Versionsnummer die nicht unbedingt mit der in der
33   * verantwortlichen Instanz aktivierten Versionsnummer übereinstimmen muss. Beispielsweise können in der Konfiguration einer VRZ die Konfigurationsobjekte einer
34   * UZ übernommen werden. Die VRZ verwaltet dann für die übernommene Konfiguration eine lokale Versionsnummer.<BR/> Die Versionsnummern ab der ein
35   * Konfigurationsobjekt gültig bzw. nicht mehr gültig ist bezieht sich immer auf die lokal aktivierte Versionsnummer der verantwortlichen Instanz des jeweiligen
36   * Objekts. Wenn in der VRZ die lokal aktivierte Version der UZ-Konfiguration kleiner als die in der UZ aktivierten Version ist, dann werden z.B. die neusten
37   * Objekte der UZ in der VRZ als noch nicht gültig angesehen, obwohl sie in der UZ schon gültig sind.<BR/> Die Schnittstelle bietet außerdem Möglichkeiten, um
38   * auf die einem Konfigurationsobjekt zugeordneten benannten Objekt-Mengen zuzugreifen.
39   *
40   * @author Kappich Systemberatung
41   * @version $Revision: 5084 $
42   */
43  public interface ConfigurationObject extends SystemObject {
44  
45  	/**
46  	 * Liefert die Versionsnummer der Konfiguration ab der dieses Konfigurationsobjekt gültig ist.
47  	 *
48  	 * @return Version ab der das Objekt gültig ist.
49  	 */
50  	public short getValidSince();
51  
52  	/**
53  	 * Liefert die Versionsnummer der Konfiguration ab der dieses Konfigurationsobjekt nicht mehr gültig ist.
54  	 *
55  	 * @return Versionsnummer der Konfiguration ab der dieses Konfigurationsobjekt nicht mehr gültig ist. Wenn das Objekt noch nicht mit der Methode {@link
56  	 *         SystemObject#invalidate} ungültig gemacht worden ist, dann wird der Wert <code>0</code> zurückgegeben.
57  	 */
58  	public short getNotValidSince();
59  
60  	/**
61  	 * Liefert eine diesem Konfigurations-Objekt zugeordnete nicht online änderbare Menge zurück.
62  	 *
63  	 * @param name Der Name der gewünschten Menge
64  	 *
65  	 * @return Gewünschte Konfigurationsmenge oder <code>null</code>, wenn die spezifizierte Menge nicht vorhanden ist.
66  	 */
67  	public NonMutableSet getNonMutableSet(String name);
68  
69  	/**
70  	 * Liefert eine diesem Konfigurations-Objekt zugeordnete online änderbare Menge zurück.
71  	 *
72  	 * @param name Der Name der gewünschten Menge
73  	 *
74  	 * @return Gewünschte dynamische Menge oder <code>null</code>, wenn die spezifizierte Menge nicht vorhanden ist.
75  	 */
76  	public MutableSet getMutableSet(String name);
77  
78  	/**
79  	 * Liefert eine diesem Konfigurations-Objekt zugeordnete Objekt-Menge zurück. Wenn die spezifizierte Objekt-Menge nicht online änderbar ist, dann unterstützt
80  	 * die zurückgegebene Menge die {@link NonMutableSet Schnittstelle für nicht online änderbare Mengen}. Wenn die Menge online änderbar ist, dann unterstützt das
81  	 * zurückgegebene Mengenobjekt die {@link MutableSet Schnittstelle für online änderbare Mengen}.
82  	 *
83  	 * @param name Der Name der gewünschten Menge
84  	 *
85  	 * @return Menge von System-Objekten oder <code>null</code>, wenn die spezifizierte Menge nicht vorhanden ist.
86  	 */
87  	public ObjectSet getObjectSet(String name);
88  
89  	/**
90  	 * Liefert die Liste aller diesem Konfigurationsobjekt zugeordneten Mengen zurück.
91  	 *
92  	 * @return Liste aller Mengen dieses Objekts. Die in der Liste enthaltenen Mengen implementieren je nach Art entweder die {@link NonMutableSet Schnittstelle
93  	 *         für nicht online änderbare Mengen} oder die {@link MutableSet Schnittstelle für online änderbare Mengen}.
94  	 */
95  	public List<ObjectSet> getObjectSets();
96  
97  	/**
98  	 * Macht ein bereits als ungültig markiertes Objekt wieder gültig. Wenn ein Konfigurationsobjekt mit der Methode {@link SystemObject#invalidate} für eine
99  	 * zukünftige Konfigurationsversion als ungültig markiert wurde und diese Konfigurationsversion noch nicht aktiviert wurde, dann kann das Objekt durch Aufruf
100 	 * dieser Methode wieder gültig gemacht werden.
101 	 *
102 	 * @throws ConfigurationChangeException Wenn das Objektes nicht wieder gültig gemacht werden konnte.
103 	 */
104 	public void revalidate() throws ConfigurationChangeException;
105 
106 	/**
107 	 * Dupliziert ein Konfigurationsobjekt. Es ist zu beachten, dass Komponenten nicht isoliert dupliziert werden können, sondern im Sinne der Komposition immer
108 	 * nur ganze Objekt-Einheiten zusammen dupliziert werden können, d.h. ausgehend von einem freien Objekt wird das Objekt mit all seinen Komponenten rekursiv
109 	 * dupliziert. Das Duplikat wird mit Aktivierung der in Bearbeitung befindlichen neuen Version des jeweiligen Konfigurationsbereichs gültig. Da die Pids gleich
110 	 * bleiben, muss zuvor das "alte" Objekt {@link #invalidate() gelöscht} werden.
111 	 *
112 	 * @return Das Duplikat dieses Konfigurationsobjekts.
113 	 *
114 	 * @throws ConfigurationChangeException Falls das Objekt kein freies Objekt ist und das Duplizieren nicht erlaubt ist oder das Duplikat nicht erstellt werden
115 	 *                                      konnte.
116 	 * @see #duplicate(java.util.Map)
117 	 */
118 	public SystemObject duplicate() throws ConfigurationChangeException;
119 
120 	/**
121 	 * Dupliziert ein Konfigurationsobjekt. Es ist zu beachten, dass Komponenten nicht isoliert dupliziert werden können, sondern im Sinne der Komposition immer
122 	 * nur ganze Objekt-Einheiten zusammen dupliziert werden können, d.h. ausgehend von einem freien Objekt wird das Objekt mit all seinen Komponenten rekursiv
123 	 * dupliziert. Das Duplikat wird mit Aktivierung der in Bearbeitung befindlichen neuen Version des jeweiligen Konfigurationsbereichs gültig.
124 	 * <p/>
125 	 * Zu ersetzende Pids der Komponenten können in der Map (altePid, neuePid) übergeben werden. Nicht ersetzte Pids, deren Objekte aber durch Komposition an die
126 	 * Objekt-Einheit gebunden sind, müssen vor Aktivierung {@link #invalidate() gelöscht} werden.
127 	 *
128 	 * @param substitudePids Map, die die Wert-Paare (altePid, neuePid) enthält.
129 	 *
130 	 * @return Das Duplikat dieses Konfigurationsobjekts.
131 	 *
132 	 * @throws ConfigurationChangeException Falls das Objekt kein freies Objekt ist und das Duplizieren nicht erlaubt ist oder das Duplikat nicht erstellt werden
133 	 *                                      konnte.
134 	 * @see #duplicate()
135 	 */
136 	public SystemObject duplicate(Map<String, String> substitudePids) throws ConfigurationChangeException;
137 
138 	/**
139 	 * Ordnet dem Konfigurationsobjekt eine weitere Menge zu. Die Zuordnung wird erst mit der nächsten Konfigurationsversion gültig.
140 	 *
141 	 * @param set Menge, die dem Konfigurationsobjekt zugeordnet werden soll.
142 	 *
143 	 * @throws ConfigurationChangeException Wenn die Menge nicht hinzugefügt werden konnte.
144 	 */
145 	public void addSet(ObjectSet set) throws ConfigurationChangeException;
146 
147 	/**
148 	 * Entfernt die Zuordnung von diesem Konfigurationsobjekt zu einer Menge. Die Änderung wird erst mit der nächsten Konfigurationsversion gültig.
149 	 *
150 	 * @param set Menge, die entfernt werden soll.
151 	 *
152 	 * @throws ConfigurationChangeException Wenn die Menge nicht entfernt werden konnte.
153 	 */
154 	public void removeSet(ObjectSet set) throws ConfigurationChangeException;
155 }
