20  package de.bsvrz.dav.daf.main.impl.archive.filesaver;
21  
22  import java.io.IOException;
23  import java.io.File;
24  import java.util.Collection;
25  
26  /**
27   * Dieses Interface stellt einem Archivsystem die Möglichkeit zur Verfügung Dateien dauerhaft zu speichern. Die
28   * Implementierung, die diese Aufgabe übernimmt, wird auch als Sicherung bezeichnet.
29   *
30   * @author Kappich+Kniß Systemberatung Aachen (K2S)
31   * @author Achim Wullenkord (AW)
32   * @version $Revision: 5064 $ / $Date: 2007-09-01 22:25:35 +0200 (Sat, 01 Sep 2007) $ / ($Author: rs $)
33   */
34  public interface ArchiveFileSaver {
35  
36  	/**
37  	 * Die Implementierung dieser Methode speichert eine Datei mit einem festgelegten Dateinamen auf einen Datenträger der
38  	 * Sicherung. Nachdem die Datei dauerhaft auf dem Datenträger gesichert wurde, wird die eindeutige
39  	 * Datenträgeridentifikation des Datenträgers zurückgegeben.
40  	 *
41  	 * @param filename  Dateiname, unter dem die Datei gesichert werden muss
42  	 * @param directory Verzeichnis, unter dem die Datei zu finden ist, die gesichert werden soll
43  	 * @return Eindeutige Datenträgerindentifikation, auf dem die zu sichernde Datei gesichert wurde
44  	 * @throws IOException Fehler beim speichern der Datei auf dem Datenträger der Sicherung oder beim Zugriff auf die zu
45  	 *                     sichernden Datei
46  	 */
47  	public int saveFile(String filename, File directory) throws IOException;
48  
49  	/**
50  	 * Die Implementierung dieser Methode stellt eine Datei zur Verfügung, die mit der Methode <code>saveFile</code> auf
51  	 * einen Datenträger der Sicherung gesichert wurde. Damit die Datei eindeutig zu identifizieren ist, wird sowohl die
52  	 * Identifizierung des Speichermediums, als auch der Name der Datei angegeben. Die Datei wird dem angegebenen
53  	 * Verzeichnis zur Verfügung gestellt.
54  	 *
55  	 * @param filename         Name der Datei, die benötigt wird
56  	 * @param volumeID         Eindeutige Identifizierung des Datenträgers, auf dem die benötigte Datei gespeichert wurde
57  	 *                         (volumeID > 0)
58  	 * @param restoreDirectory Verzeichnis, unter dem die Datei zur Verfügung gestellt werden muss
59  	 * @throws IOException              Der Zugriff auf die Datei über das Speichermedium der Sicherung ist fehlgeschlagen
60  	 * @throws IllegalArgumentException Die angegebene Datei war auf dem angegebenen Datenträger nicht zu finden
61  	 * @throws BadVolumeException       Der angegebene Datenträger ist nicht mehr vorhanden oder nicht mehr im direkten
62  	 *                                  Zugriff der Sicherung
63  	 */
64  	public void restoreFile(String filename, int volumeID, File restoreDirectory) throws IOException, IllegalArgumentException, BadVolumeException;
65  
66  	/**
67  	 * Die Implementierung dieser Methode stellt eine Liste mit den eindeutigen Identifizierungen aller Datenträger
68  	 * zusammen, die sich im direkten Zugriff der Sicherung befinden.
69  	 *
70  	 * @return Eindeutige Datenträgeridentifikation aller Datenträger, die sich im direkten Zugriff befinden
71  	 */
72  	public Collection<Integer> volumesDirectAccess();
73  
74  
75  	/**
76  	 * Die Implementierung dieser Methode gibt alle Dateinamen eines Speichermediums der Sicherung zurück. Die ausgwählten
77  	 * Dateien müssen mit der Methode <code>saveFile</code> der Sicherung übergeben worden sein.
78  	 *
79  	 * @param volumeId Speichermedium, von dem alle Dateinamen angefordert werden
80  	 * @return Alle Dateinamen, die auf dem angegebenen Speichermedium gespeichert sind. Die Dateien müssen mit der Methode
81  	 *         <code>saveFile</code> der Sicherung übergeben worden sein
82  	 * @throws IOException        Der physische Zugriff auf den Datenträger ist fehlgeschlagen
83  	 * @throws BadVolumeException Der angegebene Datenträger ist nicht mehr vorhanden oder nicht mehr im direkten Zugriff
84  	 *                            der Sicherung
85  	 */
86  	public Collection<String> getAllFilenames(int volumeId) throws IOException, BadVolumeException;
87  }
