34  public class SendSubscriptionObject {
35  
36  	/** Der Sendeindex dieser Anmeldung */
37  	private int _sendIndex;
38  
39  	/** Die Sendeanmeldeinformationen */
40  	private SenderSubscription _senderSubscription;
41  
42  	/** Die Zeit des Sendeindexes */
43  	private long _time;
44  
45  	/** DebugLogger für Debug-Ausgaben */
46  	private static final Debug _debug = Debug.getLogger();
47  
48  
49  	/**
50  	 * Falls auf die Sendesteuerung (positiv/negativ) gewartet werden muss, wird zum synchronisieren dieses Objekt benutzt. Falls sich die Sendesterung ändert,
51  	 * wird jeder Thread geweckt, der auf dieses Objekt wartet. Das selbe passiert, wenn eine fest vorgegebene Zeitspanne abläuft, auch in diesem Fall wird jeder
52  	 * benachtichtigt, der auf diesem Objekt wartet.
53  	 */
54  	private final Object _requestLock = new Object();
55  
56  	/** true = Es wurde eine Sendesteuerung gesetzt (positiv oder negativ); false = Es wird noch auf eine Sendesteuerung gewartet. */
57  	private boolean _requestAnswered = false;
58  
59  	/** Zeitraum (in ms), der gewartet wird um eine Sendesteuerung (positiv/negativ) zu erhalten. Es wird derzeit 1 Minute gewartet. */
60  	private final long _timeOut = 1 * 60 * 1000;
61  
62  	/**
63  	 * Diese Variable wird true, falls das erste mal Daten verschickt werden können. Wenn danach die Sendesteuerung negativ wird, wird nicht noch einmal die
64  	 * <code>_timeOut</code> gewartet.
65  	 */
66  	private boolean _checkedConnectionFirstTime = false;
67  
68  	public SendSubscriptionObject(SenderSubscription senderSubscription) {
69  		_senderSubscription = senderSubscription;
70  		_sendIndex = 0;
71  		_time = (((System.currentTimeMillis() / 1000L) << 32) & 0xFFFFFFFF00000000L);
72  	}
73  
74  	/** Gibt den Anmelde-Zeitstempel zurück
75  	 * @return Sekunden seit 1970
76  	 */
77  	public int getTimeStamp() {
78  		return (int)(_time >> 32);
79  	}
80  
81  	/**
82  	 * Gibt die Sendeanmeldeinformationen zurück
83  	 *
84  	 * @return Sendeanmeldeinformationen
85  	 */
86  	public final SenderSubscription getSenderSubscription() {
87  		return _senderSubscription;
88  	}
89  
90  	/**
91  	 * Gibt den Index der Sendung der übergebenen Anmeldungsinformation zurück. Der Index startet immer mit 1 und wird bei jeder Abfrage um 1 erhöht, wenn es
92  	 * 0x3FFFFFFF (2 hoch 29 ) erreicht hat, fängt es wieder bei 1 an.
93  	 *
94  	 * @return der Index der Sendung
95  	 */
96  	public final long getSendDataIndex() {
97  		++_sendIndex;
98  		if(_sendIndex > 0x3FFFFFFF) {
99  			_sendIndex = 1;
100 		}
101 		long dataIndex = ((long)(_sendIndex << 2) & 0x00000000FFFFFFFCL);
102 		long retValue = ((_time | dataIndex) & 0xFFFFFFFFFFFFFFFCL);
103 		return retValue;
104 	}

112 		if(_senderSubscription == null) {

118 	/**
119 	 * Gibt die Information zurük, ob die Applikation Daten senden kann oder nicht.
120 	 *
121 	 * @return true: Applikation kann Daten senden, false: Applikation kann keine Daten senden.
122 	 */
123 	public final boolean canSendData() {
124 		if(_senderSubscription == null) {
125 			return false;
126 		}
127 		synchronized(_requestLock) {
128 			// Falls beim ersten mal eine gar keine Sendesteuerung vorliegt (positiv/negativ), wird eine bestimmte Zeitdauer gewartet.
129 			// In dieser Zeit muss die Sendesteuerung positiv oder negativ werden.
130 			// Wird die Sendesteuerung nicht positiv, wird <code>false</code> zurückgegeben, dies führt
131 			// dann zu einer Exception. Wird die Sendesteuerung vor Ablauf der Zeit positiv, wird
132 			// <code>true</code> zurückgegeben.
133 			// Liegt gar keine Sendesteuerung vor, wird <code>false</code> zurückgegeben.
134 
135 			// Bei Anmeldung als Quelle wird direkt <code>true</code> zurückgegeben
136 			if(_senderSubscription.isSource()) {
137 				return true;
138 			}
139 			else if(_checkedConnectionFirstTime == true && _requestAnswered == true) {
140 				// Es wurde ein bestimmter Zeitraum abgewartet und es liegt eine Antwort vor, dann
141 				// wird dieses Ergebnis benutzt (wenn also auch nach Ablauf der Zeit eine Sendersteuerung vorliegt,
142 				// die nach Ablauf der Zeit noch nicht vorhanden war, wird dieser Zustand zurückgegeben)
143 				return _senderSubscription.isRequestConfirmed();
144 			}
145 			else if(_checkedConnectionFirstTime == true && _requestAnswered == false) {
146 				// Es wurde ein bestimmter Zeitraum abgewartet und es liegt !keine! Antwort vor, dann
147 				// wird immer false zurückgegeben, da der Zustand unbekannt ist.
148 				return false;
149 			}
150 
151 			// Es besteht keine Sendesteuerung (positiv oder negativ) und das "erste warten" hat noch nicht statt gefunden, also warten.
152 			// Dieser Teil wird nur einmal am Anfang durchlaufen, danach verhindert _checkedConnectionFirstTime dies.
153 			final TimeoutTimer timer = new TimeoutTimer(_timeOut);
154 			// Solange die Sendesteuerung nicht auf positiv/negativ geändert wurde oder aber der Timer ist
155 			// abgelaufen, wird nichts gemacht
156 
157 			while((_requestAnswered == false) && (timer.isTimeExpired() == false)) {
158 				_debug.fine("Applikation wartet auf eine positive/negative Sendesteuerung: sendIndex = " + _sendIndex);
159 
160 				// Die Restzeit speichern, ist diese 0 darf der Thread nicht mit Restzeit 0 schlafen gelegt werden.
161 				// Dies ist nötig, denn ein Thread, der mit 0 schlafen gelegt wird, wartet auf ein notify und
162 				// wacht nicht nach 0 Sekunden auf !
163 				// (bei Restzeit 0 wird die while-Schleife von alleine verlassen, da isTimeExpired == true ist)
164 				final long remainingTime = timer.getRemainingTime();
165 				if(remainingTime > 0) {
166 					try {
167 						_requestLock.wait(remainingTime);
168 					}
169 					catch(InterruptedException e) {
170 						_debug.info("Ein Thread, der auf eine positive/negative Sendersteuerung wartet, wurde abgebrochen: ", e);
171 					}
172 				}
173 			}
174 
175 			// Die while-Schleife wurde verlassen, also ist die Sendesteuerung(positiv oder negativ) entweder vorhanden oder die Zeit ist abgelaufen.
176 			_checkedConnectionFirstTime = true;
177 			if(_requestAnswered == true) {
178 				// Es wurde eine positive/negative Sendesteuerung gesetzt
179 				return _senderSubscription.isRequestConfirmed();
180 			}
181 			else {
182 				// Es wurde noch keine Sendesteuerung gesetzt, also eine Excpetion auslösen
183 				return false;
184 			}
185 		}
186 	}
187 
188 	/**
189 	 * Setzt die Flagge, ob Daten gesendet werden können oder nicht
190 	 *
191 	 * @param state Status
192 	 */
193 	public final void confirmSendDataRequest(byte state) {
194 		if(state == 0) {
195 			// Es liegt eine positive Sendesteuerung vor, falls jemand auf diese Nachricht wartet, wird er geweckt
196 			_senderSubscription.confirmRequest(true);
197 			synchronized(_requestLock) {
198 				_requestAnswered = true;
199 				_requestLock.notifyAll();
200 			}
201 		}
202 		else {
203 			// Es liegt keine positive Sendesteuerung vor, falls jemand auf die Antwort wartet, wird er geweckt
204 			_senderSubscription.confirmRequest(false);
205 			synchronized(_requestLock) {
206 				_requestAnswered = true;
207 				_requestLock.notifyAll();
208 			}
209 		}
210 	}
