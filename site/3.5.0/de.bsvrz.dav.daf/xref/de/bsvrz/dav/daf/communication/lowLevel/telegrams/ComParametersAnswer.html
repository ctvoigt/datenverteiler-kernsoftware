26  import java.io.DataInputStream;

30  /**
31   * In diesem Systemtelegramm werden die vom Datenverteiler festgelegten Verbindungsparameter an die Applikationsfunktionen übertragen.
32   *
33   * @author Kappich Systemberatung
34   * @version $Revision: 5059 $
35   */
36  public class ComParametersAnswer extends DataTelegram {
37  
38  	/** Zeit, nach der spätestens ein Keep-Alive-Telegramm versendet wird, wenn keine sonstigen Telegramme während dieser Zeit versendet wurden. */
39  	private long keepAliveSendTimeOut;
40  
41  	/**
42  	 * Timeoutzeit, in der spätestens ein Telegramm empfangen werden muss. Wird diese Zeit ohne Empfang von Telegrammen überschritten, wird die Verbindung zum
43  	 * Kommunikationspartner terminiert.
44  	 */
45  	private long keepAliveReceiveTimeOut;
46  
47  	/** Belegung des Sendepuffers in Prozent ab der zyklisch die Durchsatzprüfung erfolgen soll. */
48  	public byte cacheThresholdPercentage;
49  
50  	/**
51  	 * Messintervall zur Bestimmung des Durchsatzes in Sekunden. Ist der ermittelte Durchsatz kleiner als der MindestDurchsatz wird die Verbindung zum
52  	 * Datenverteiler terminiert.
53  	 */
54  	public short flowControlThresholdTime;
55  
56  	/** Mindestdurchsatz in Bytes pro Sekunde */
57  	public int minConnectionSpeed;
58  
59  	public ComParametersAnswer() {
60  		type = COM_PARAMETER_ANSWER_TYPE;
61  		priority = CommunicationConstant.SYSTEM_TELEGRAM_PRIORITY;
62  	}
63  
64  	/**
65  	 * Creates new CommunicationParametersAnswer
66  	 *
67  	 * @param _keepAliveSendTimeOut     Keepalive-Sendezeit
68  	 * @param _keepAliveReceiveTimeOut  Keepalive-Empfangszeit
69  	 * @param _cacheThresholdPercentage Pufferfüllgrad
70  	 * @param _flowControlThresholdTime Prüfintervall
71  	 * @param _minConnectionSpeed       Mindestdurchsatz
72  	 */
73  	public ComParametersAnswer(
74  			long _keepAliveSendTimeOut, long _keepAliveReceiveTimeOut, byte _cacheThresholdPercentage, short _flowControlThresholdTime, int _minConnectionSpeed
75  	) {
76  		type = COM_PARAMETER_ANSWER_TYPE;
77  		priority = CommunicationConstant.SYSTEM_TELEGRAM_PRIORITY;
78  		keepAliveSendTimeOut = _keepAliveSendTimeOut;
79  		keepAliveReceiveTimeOut = _keepAliveReceiveTimeOut;
80  		cacheThresholdPercentage = _cacheThresholdPercentage;
81  		flowControlThresholdTime = _flowControlThresholdTime;
82  		minConnectionSpeed = _minConnectionSpeed;
83  		length = 23;
84  	}
85  
86  	/**
87  	 * Gibt den prozentualen Anteil der Cache, um die Flusscontrole zu starten, zurück.
88  	 *
89  	 * @return Pufferfüllgrad
90  	 */
91  	public final byte getCacheThresholdPercentage() {
92  		return cacheThresholdPercentage;
93  	}
94  
95  	/**
96  	 * Gibt die Zeit zwichen zwei Durchsatzprüfungen zurück.
97  	 *
98  	 * @return Prüfintervall
99  	 */
100 	public final short getFlowControlThresholdTime() {
101 		return flowControlThresholdTime;
102 	}
103 
104 	/**
105 	 * Gibt den minimum Verbindungsdurchsatz zurück.
106 	 *
107 	 * @return Mindestdurchsatz
108 	 */
109 	public final int getMinConnectionSpeed() {
110 		return minConnectionSpeed;
111 	}
112 
113 	/**
114 	 * Gibt die Zeit zurück, nach der spätestens ein keepalive Telegramm geschickt werden muss, wenn in diese Zeit kein Telegramm empfangen wurde.
115 	 *
116 	 * @return die Keepalive-Sendezeit
117 	 */
118 	public final long getKeepAliveSendTimeOut() {
119 		return keepAliveSendTimeOut;
120 	}
121 
122 	/**
123 	 * Gibt die Zeit zurück, in der spätestens ein Telegramm empfangen werden muss, sonst wird die verbindung terminiert.
124 	 *
125 	 * @return die Keepalive-Empfangszeit
126 	 */
127 	public final long getKeepAliveReceiveTimeOut() {
128 		return keepAliveReceiveTimeOut;
129 	}
130 
131 	public final String parseToString() {
132 		String str = "Systemtelegramm setzen der Kommunikationsparametern Antwort: \n";
133 		str += "Keep Alive Sendezeit    : " + keepAliveSendTimeOut + "\n";
134 		str += "Keep Alive Empfangszeit : " + keepAliveReceiveTimeOut + "\n";
135 		str += "Pufferfüllgrad: " + cacheThresholdPercentage + " % \n";
136 		str += "Prüfintervall: " + flowControlThresholdTime + " s \n";
137 		str += "Mindestdurchsatz: " + minConnectionSpeed + "\n";
138 		return str;
139 	}

141 	public final void write(DataOutputStream out) throws IOException {
142 		out.writeShort(length);
143 		out.writeLong(keepAliveSendTimeOut);
144 		out.writeLong(keepAliveReceiveTimeOut);
145 		out.writeByte(cacheThresholdPercentage);
146 		out.writeShort(flowControlThresholdTime);
147 		out.writeInt(minConnectionSpeed);
148 	}
149 
150 	public final void read(DataInputStream in) throws IOException {
151 		int _length = in.readShort();
152 		keepAliveSendTimeOut = in.readLong();
153 		keepAliveReceiveTimeOut = in.readLong();
154 		cacheThresholdPercentage = in.readByte();
155 		flowControlThresholdTime = in.readShort();
156 		minConnectionSpeed = in.readInt();
157 		length = 23;
158 		if(length != _length) {
159 			throw new IOException("Falsche Telegrammlänge");
160 		}
161 	}
