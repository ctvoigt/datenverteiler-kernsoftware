63  	public ClientConnectionProperties(ClientDavParameters clientDavParameters) throws ConnectionException {
64  		super(
65  				null,
66  				null,
67  				clientDavParameters.getUserName(),
68  				clientDavParameters.getUserPassword(),
69  				clientDavParameters.getCommunicationParameters().getSendKeepAliveTimeout(),
70  				clientDavParameters.getCommunicationParameters().getReceiveKeepAliveTimeout(),
71  				clientDavParameters.getCommunicationOutputBufferSize(),
72  				clientDavParameters.getCommunicationInputBufferSize()
73  		);
74  		try {
75  			String comProtocol = clientDavParameters.getLowLevelCommunicationName();
76  			if(comProtocol == null) {
77  				throw new InitialisationNotCompleteException("Unbekannter Kommunikationsprotokollname.");
78  			}
79  			Class aClass = Class.forName(comProtocol);
80  			if(aClass == null) {
81  				throw new InitialisationNotCompleteException("Unbekannter Kommunikationsprotokollname.");
82  			}
83  			ConnectionInterface connection = (ConnectionInterface)aClass.newInstance();
84  			setLowLevelCommunication(
85  					new LowLevelCommunication(
86  							connection,
87  							clientDavParameters.getCommunicationOutputBufferSize(),
88  							clientDavParameters.getCommunicationInputBufferSize(),
89  							clientDavParameters.getCommunicationParameters().getSendKeepAliveTimeout(),
90  							clientDavParameters.getCommunicationParameters().getReceiveKeepAliveTimeout(),
91  							LowLevelCommunication.HANDLE_CONFIG_RESPONCES_MODE,
92  							false
93  					)
94  			);
95  
96  			String authentificationName = clientDavParameters.getAuthentificationProcessName();
97  			if(authentificationName == null) {
98  				throw new InitialisationNotCompleteException("Unbekanntes Authentifikationsverfahren.");
99  			}
100 			aClass = Class.forName(authentificationName);
101 			if(aClass == null) {
102 				throw new InitialisationNotCompleteException("Unbekanntes Authentifikationsverfahren.");
103 			}
104 			setAuthentificationProcess((AuthentificationProcess)aClass.newInstance());
105 
106 			_applicationName = clientDavParameters.getApplicationName();
107 			_incarnationName = clientDavParameters.getIncarnationName();
108 			_applicationTypePid = clientDavParameters.getApplicationTypePid();
109 			_configurationPid = clientDavParameters.getConfigurationPid();
110 			_address = clientDavParameters.getDavCommunicationAddress();
111 			_subAddress = clientDavParameters.getDavCommunicationSubAddress();
112 			_communicationParameters = clientDavParameters.getCommunicationParameters();
113 		}
114 		catch(ClassNotFoundException ex) {
115 			throw new InitialisationNotCompleteException("Fehler beim Erzeugen der logischen Verbindung zum Datenverteiler.");
116 		}
117 		catch(InstantiationException ex) {
118 			throw new InitialisationNotCompleteException("Fehler beim Erzeugen der logischen Verbindung zum Datenverteiler.");
119 		}
120 		catch(IllegalAccessException ex) {
121 			throw new InitialisationNotCompleteException("Fehler beim Erzeugen der logischen Verbindung zum Datenverteiler.");
122 		}
123 	}
