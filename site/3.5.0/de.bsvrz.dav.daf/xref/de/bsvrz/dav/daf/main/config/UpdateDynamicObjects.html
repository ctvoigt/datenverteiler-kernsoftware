21  package de.bsvrz.dav.daf.main.config;

23  /**
24   * Dieses Interface stellt Methoden zur Verfügung mit denen dynamische Objekte auf Meta-Seite geändert werden können.
25   * <p/>
26   * Dieser Mechanismus wird benötigt, weil sich dynamische Objekte zur Laufzeit ändern. Die Konfiguration wird diese Änderungen an alle Applikationen
27   * propagieren, die sich bei der Konfiguration angemeldet haben.
28   * <p/>
29   * Einige Beispiele für sich ändernde Werte von dynamischen Objekten sind: Name, Gültigkeit des Objekts, Zeitpunkt ab dem das Objekt nicht mehr gültig war,
30   * usw..
31   *
32   * @author Kappich Systemberatung
33   * @version $Revision: 5623 $
34   */
35  public interface UpdateDynamicObjects {
36  
37  	/**
38  	 * Diese Methode wird aufgerufen, wenn die Konfiguration eine Namensänderung eines dynamischen Objekts propagiert hat.
39  	 * <p/>
40  	 * Auf Meta-Seite muss dann der Name des Objekts aktualisiert werden. Des Weiteren muss dem Typ des Objekts die Namensänderung mitgeteilt werden. Dies ist
41  	 * nötig, weil am Typ des Objekts Listener für Namensänderungen vorhanden sind.
42  	 * <p/>
43  	 * Beim setzen des Namens darf nicht die {@link DynamicObject#setName(String)} Methode benutzt werden, weil dadurch erneut eine Anfrage an die Konfiguration
44  	 * verschickt werden würde.
45  	 * <p/>
46  	 * Der beschriebene Mechanismus muss nur durchgeführt werden wenn: Sich das Objekt im Cache des Meta-Datamodells befindet oder ein Listener auf
47  	 * Namensänderungen für den Typ des Objekt angemeldet ist.
48  	 * <p/>
49  	 * Sind beide Bedingungen nicht erfüllt kann die Namensänderung verworfen werden.
50  	 *
51  	 * @param objectId Id des Objekts, dessen Name aktualisiert werden soll
52  	 * @param typeId   Typ des Objekts, der informiert wird, dass sich der Name eines Objekts geändert hat
53  	 * @param newName  Neuer Name des Objekts
54  	 */
55  	public void updateName(final long objectId, final long typeId, final String newName);
56  
57  	/**
58  	 * Diese Methode wird aufgerufen, wenn die Konfiguration propagiert, dass ein dynamisches Objekt ungültig wurde.
59  	 * <p/>
60  	 * Die Methode muss beim dynamischen Objekt den Zeitpunkt aktualisieren, an dem es ungültig wurde und der Methodenaufruf von {@link
61  	 * DynamicObject#isValid()} muss false zurück geben. Der Typ des Objekts muss ebenfalls informatiert werden, dass das
62  	 * Objekt nicht mehr gültig ist. Dies ist nötig, weil eventuell Listener auf diese Änderungen angemeldet sind.
63  	 * <p/>
64  	 * Der beschriebene Mechanismus muss nur durchgeführt werden wenn: Sich das Objekt im Cache des Meta-Datamodells befindet oder ein Listener auf Invalidation
65  	 * für den Typ des Objekt angemeldet ist.
66  	 * <p/>
67  	 * Sind beide Bedingungen nicht erfüllt kann die Änderung verworfen werden.
68  	 *
69  	 * @param objectId  Id des Objekts, dessen UngültigAb-Wert aktualisiert werden soll
70  	 * @param typeId    Typ des Objekts, der informiert wird, dass das Objekt ungültig geworden ist
71  	 * @param notValidTime Zeitpunkt, an dem das Objekt ungültig wurde
72  	 */
73  	public void updateNotValidSince(final long objectId, final long typeId, final long notValidTime);
74  
75  	/**
76  	 * Diese Methode wird aufgerufen, wenn die Konfiguration propagiert, dass ein neues dynamisches Objekt erzeugt wurde.
77  	 * <p/>
78  	 * Diese Information muss an den jeweiligen Typen des Objekt weitergereicht werden. Dies ist nötig, weil vielleicht Listener auf diese Information angemeldet
79  	 * sind. Ist dies der Fall, so muss das vollständige Objekt aus der Konfiguration angefordert werden.
80  	 *
81  	 * @param objectId Objekt, das neu angelegt wurde
82  	 * @param typeId   Typ des neuen Objekts. Dieser Typ wird darüber informiert, dass ein neues Objekt angelegt wurde.
83  	 */
84  	public void newDynamicObjectCreated(final long objectId, final long typeId);
85  }
