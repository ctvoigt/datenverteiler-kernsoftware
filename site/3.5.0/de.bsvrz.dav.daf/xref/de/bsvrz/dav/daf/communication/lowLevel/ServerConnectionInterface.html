22  package de.bsvrz.dav.daf.communication.lowLevel;
23  
24  import de.bsvrz.dav.daf.main.CommunicationError;
25  
26  /**
27   * Eine Implementierung dieses Interfaces legt fest, welches Netzwerkprotokoll die Daten zwischen Softwareeinheiten oder Rechner übertragen werden. Die
28   * Funktionalität wird protokollneutral durch dieses Interface für die Server-seitig notwendigen Methoden nach Außen vertreten.
29   *
30   * @author Kappich Systemberatung
31   * @version $Revision: 5084 $
32   */
33  public interface ServerConnectionInterface {
34  
35  	/**
36  	 * Erstellt eine Verbindung zur lokalen Adresse mit der Nummer der Subadresse. Eine maximal erlaubte Anzahl an Verbindungen kann spezifiziert werden.
37  	 *
38  	 * @param subAdressNumber die Nummer der Subadresse
39  	 *
40  	 * @throws de.bsvrz.dav.daf.main.CommunicationError Wenn die Verbindung nicht erfolgreich erzeugt wurde.
41  	 */
42  	public void connect(int subAdressNumber) throws CommunicationError;
43  
44  	/** Schließt die aktuelle Verbindung. */
45  	public void disconnect();
46  
47  	/**
48  	 * Wartet auf eine Verbindungsanfrage eines Clients (passive Verbindung). Wenn die Anfrage erfolgt, wird der Repräsentant einer Verbindung erzeugt und
49  	 * zurückgegeben.
50  	 *
51  	 * @return Repräsentant einer Verbindung oder <code>null</code>, wenn keine Verbindung erzeugt werden konnte.
52  	 */
53  	public ConnectionInterface accept();
54  
55  	/**
56  	 * Gibt eine Instanz der Kommunikationsklasse vom Typ {@link ConnectionInterface} des gleichen Protokolls.
57  	 *
58  	 * @return eine Instanz der Kommunikationsklasse
59  	 */
60  	public ConnectionInterface getPlainConnection();
61  
62  	/**
63  	 * Liefert den Klassennamen der Kommunikationsklasse vom Typ {@link ConnectionInterface} des gleichen Protokolls.
64  	 *
65  	 * @return Klassenname der Kommunikationsklasse
66  	 */
67  	public String getPlainConnectionName();
68  }
