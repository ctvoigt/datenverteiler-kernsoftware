22  package de.bsvrz.dav.daf.communication.lowLevel;
23  
24  import de.bsvrz.dav.daf.communication.lowLevel.telegrams.DataTelegram;
25  import de.bsvrz.dav.daf.main.ConnectionException;
26  
27  /**
28   * Dieses Interface legt die öffentlichen Methoden der unteren Kommunikationsebene von Datenverteilerverbindungen fest.
29   *
30   * @author Kappich Systemberatung
31   * @version $Revision: 7470 $
32   */
33  public interface LowLevelCommunicationInterface {
34  
35  	/**
36  	 * Diese Methode baut eine Kommunikationsverbindung zu den angegebenen Adressen auf. Sie wird von der Protokollschicht DaV-DAF während der
37  	 * Verbindungsinitialisierung aufgerufen.
38  	 *
39  	 * @param mainAddress Adresse des Kommunikationspartners dieser Verbindung.
40  	 * @param subAddress  Subadresse der Adresse.
41  	 *
42  	 * @throws de.bsvrz.dav.daf.main.ConnectionException Falls die Verbindung nicht hergestellt werden kann.
43  	 */
44  	public void connect(String mainAddress, int subAddress) throws ConnectionException;
45  
46  	/**
47  	 * Diese Methode wird von der Protokollschicht DaV-DAF aufgerufen, wenn die Kommunikationskanäle geschlossen werden sollen.
48  	 *
49  	 * @param error   Besagt, ob es sich um eine Terminierung mit Fehler handelt.
50  	 * @param message der Fehlertext
51  	 * @param terminationTelegram
52  	 */
53  	public void disconnect(boolean error, String message, final DataTelegram terminationTelegram);
54  
55  	/**
56  	 * Gibt als Information zurück, ob die Kommunikationsverbindung unterbrochen ist.
57  	 *
58  	 * @return <code>true</code> = Es besteht keine Verbindung; <code>false</code> = sonst
59  	 */
60  	public boolean isNotConnected();
61  
62  	/**
63  	 * Diese Methode wird von der Protokollschicht DaV-DAF aufgerufen, wenn ein Telegramm gesendet werden soll.
64  	 *
65  	 * @param telegram Das zu versendende Telegramm.
66  	 */
67  	public void send(DataTelegram telegram);
68  
69  	/**
70  	 * Fügt mehrere Telegramme in die Sendetabelle ein.
71  	 *
72  	 * @param telegrams Die zu versendenden Telegramme.
73  	 *
74  	 * @see #send(de.bsvrz.dav.daf.communication.lowLevel.telegrams.DataTelegram)
75  	 */
76  	public void send(DataTelegram telegrams[]);
77  
78  	/**
79  	 * Setzt die Interpretationsschicht dieser Komponente für den internen Datenaustausch. Erst nach Aufruf dieser Methode von der Protokollschicht DaV-DAF werden
80  	 * die Kommunikations-Threads gestartet, weil sonst keine Telegramme interpretiert oder weitergeleitet werden können.
81  	 *
82  	 * @param highLevelComponent Komponente, die benachrichtigt werden muss, wenn neue Nachrichten empfangen werden.
83  	 */
84  	public void setHighLevelComponent(HighLevelCommunicationCallbackInterface highLevelComponent);
85  
86  	/**
87  	 * Diese Methode wird von der Protokollschicht DaV-DAF aufgerufen, wenn die Keepalive-Parameter-Verhandlung erfolgreich abgeschlossen ist und setzt somit die
88  	 * Timeouts des Keepaliveprozesses.
89  	 *
90  	 * @param keepAliveSendTimeOut    Sendekeepalivetimeout
91  	 * @param keepAliveReceiveTimeOut Empfangekeepalivetimeout
92  	 */
93  	public void updateKeepAliveParameters(long keepAliveSendTimeOut, long keepAliveReceiveTimeOut);
94  
95  	/**
96  	 * Diese Methode setzt die Parameter für die Durchsatzprüfung. Sie wird von der Protokollschicht DaV-DAF aufgerufen, wenn die Parameter für die
97  	 * Durchsatzprüfung erfolgreich verhandelt wurden.
98  	 *
99  	 * @param throughputControlSendBufferFactor
100 	 *                                 Füllungsgrad des Sendepuffers als Faktor zwischen 0 und 1, ab dem die Durchsatzprüfung anfängt zu arbeiten.
101 	 * @param throughputControlInterval Zeit zwischen zwei Durchsatzprüfungen in Millisekunden
102 	 * @param minimumThroughput        Minimal zulässiger Verbindungsdurchsatz in Bytes pro Sekunde
103 	 */
104 	public void updateThroughputParameters(float throughputControlSendBufferFactor, long throughputControlInterval, int minimumThroughput);
105 
106 	/**
107 	 * Gibt den Repräsentant der Verbindung zurück
108 	 *
109 	 * @return Repräsentant der Verbindung
110 	 */
111 	public ConnectionInterface getConnectionInterface();
112 
113 	/**
114 	 * Liefert einen beschreibenden Text mit dem Zustand des Sendepuffers
115 	 * @return Zustand des Sendepuffers
116 	 */
117 	public String getSendBufferState();
118 
119 	/**
120 	 * Diese Methode setzt den Namen des Kommunikationspartners, der für Fehlermeldungen etc. verwendet wird.
121 	 * @param name Name oder Identifikation des Kommunikationspartners
122 	 */
123 	void setRemoteName(final String name);
124 }
