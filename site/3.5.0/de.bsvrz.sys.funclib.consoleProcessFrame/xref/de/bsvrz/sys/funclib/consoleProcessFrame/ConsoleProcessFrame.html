22  package de.bsvrz.sys.funclib.consoleProcessFrame;
23  
24  import javax.swing.*;
25  import java.awt.*;
26  import java.awt.event.WindowAdapter;
27  import java.awt.event.WindowEvent;
28  import java.io.File;
29  import java.io.IOException;
30  
31  /**
32   * Implementiert ein Fenster, das mit einem externen Java-Prozess verbunden ist. Der Prozess wird beim erzeugen des Fensters gestartet und beim Schlieﬂen des
33   * Fensters beendet. Die Textausgaben des Prozesses werden im Fenster dargestellt.
34   * @author Kappich Systemberatung
35   * @version $Revision: 5021 $
36   */
37  public class ConsoleProcessFrame extends JFrame {
38  
39  	private final ConsoleProcessPanel _consoleProcessPanel;
40  
41  	protected ConsoleProcessFrame(String title, ConsoleProcessPanel consoleProcessPanel) throws IOException {
42  		super(title);
43  		_consoleProcessPanel = consoleProcessPanel;
44  		addWindowListener(
45  				new WindowAdapter() {
46  					public void windowClosing(WindowEvent e) {
47  						_consoleProcessPanel.killProcess();
48  					}
49  				}
50  		);
51  		getContentPane().add(_consoleProcessPanel, BorderLayout.CENTER);
52  		pack();
53  		setVisible(true);
54  		_consoleProcessPanel.start();
55  	}
56  
57  	public static ConsoleProcessFrame createJavaProcessFrame(String title, String className, String[] arguments, String[] environment, File workingDirectory)
58  			throws IOException {
59  		ConsoleProcessPanel panel = ConsoleProcessPanel.createJavaProcessPanel(className, arguments, environment, workingDirectory);
60  		return new ConsoleProcessFrame(title, panel);
61  	}
62  
63  	public static void main(String[] args) {
64  		try {
65  			createJavaProcessFrame("Test", "rstest.ExecutionTest", null, null, null);
66  		}
67  		catch(Exception e) {
68  			System.err.println("Fehler:" + e);
69  		}
70  	}
71  }
