36  /**
37   * Diese Klasse verschickt Nutzdatenpakete mit Streams an einen StreamDemultiplexer. Die Applikation, die ein Objekt
38   * dieser Klasse erzeugt hat, stellt ihrerseits Nutzdaten für jeden Stream zur Verfügung. Auf der Gegenseite kann der
39   * StreamDemultiplexer Nutzdaten auf jedem Stream anfordern und verarbeiten. Der StreamMultiplexer sendet seinerseits
40   * nur dann Nutzdatenpakete, wenn ihn der StreamDemultiplexer dazu auffordert. Die Nutzdaten werden auch erst dann
41   * erzeugt, wenn diese verschickt werden sollen. Der StreamMultiplexer verschickt die Nutzdatenpakete nicht einzeln,
42   * sondern bündelt diese in einem großen Paket. Diese großen Pakete werden dann vom StreamDemultiplexer entgegen
43   * genommen und ausgepackt. Diese Bündelung findet für jeden Stream einzeln statt, in jedem großen Paket befinden sich
44   * also nur Nutzdaten für diesen einen Stream, nicht die Nutzdaten anderer Streams.
45   *
46   * @author Kappich Systemberatung
47   * @version $Revision: 8223 $
48   */
49  public class StreamMultiplexer {
50  
51  	/**
52  	 * Der blockingFactor bestimmt die Größe des Empfangspuffers. Dieser Wert bestimmt beim Sender nur, wie viele Pakete
53  	 * der Sender ohne Bestätigung des Empfängers, beim ersten Versenden, verschicken darf (danach muß der Empfänger dem
54  	 * Sender eine Sendeerlaubnis(Ticket) schicken). Dieser Wert muß beim Sender und beim Empfänger gleich sein.
55  	 */
56  	private final int _blockingFactor;
57  
58  	/**
59  	 * Diese Variable bestimmt die gesamte Größe des Puffers, der zum StreamMultiplexer gehört. Jeder Stream besitzt einen
60  	 * eigenen Puffer. Sobald dieser gefüllt ist, werden alle Nutzdatenpakete als ein großes Paket verschickt.
61  	 */
62  	private final int _bufferSizeStreamMultiplexer;
63  
64  	/**
65  	 * Diese Variable bestimmt die Größe des Puffers, den jeder Stream zur Verfügung hat. Im Konstruktor wird dieser Wert
66  	 * mit <code>_bufferSizeStreamMultiplexer /_numberOfStreams</code> gesetzt.
67  	 */
68  	private final int _bufferSizeStream;
69  
70  	/**
71  	 * Eine Warteschlange, in ihr werden alle Streams gespeichert, die Nutzdatenpakete verschicken können. In der
72  	 * Warteschlange werden Objekte vom Type "IndexOfStreamAndMaxSendPackets" gespeichert. Diese Objekte enthalten den
73  	 * "indexOfStream", welcher Stream kann Nutzdatenpakete verschicken, und "maxSendPackets". Dieser Wert drückt aus, wie
74  	 * viele Nutzdatenpakete verschickt werden dürfen (für diesen Stream). Die Warteschlange erspart somit das "Suchen"
75  	 * eines sendebereiten Streams.
76  	 */
77  	private final UnboundedQueue _queueWithStreamsPermitedToSendData;
78  
79  	/**
80  	 * Anzahl der Streams auf denen Nutzdatenpakete verschickt werden können
81  	 */
82  	private final int _numberOfStreams;
83  	/**
84  	 * Anzahl der Streams, die mit "Abort" abgebrochen wurden oder die ein Nutzdatenpaket verschickt haben, in dem die
85  	 * Nutzdaten <code>null</code> waren.
86  	 */
87  	private int _numberTerminatedStreams = 0;
88  
89  	/**
90  	 * In diesem Array werden alle Informationen aller Streams gespeichert.
91  	 */
92  	private final MultiplexerStreaminformations[] _streams;

119 	/**
120 	 * @param numberOfStreams             Anzahl von Streams, die Datenpakete versenden sollen
121 	 * @param blockingFactor              Anzahl der Pakete, die initial am Anfang versendet werden
122 	 * @param bufferSizeStreamMultiplexer Diese Variable bestimmt die gesamte Größe des Puffers, der zum StreamMultiplexer
123 	 *                                    gehört
124 	 * @param serializerVersion           Diese Variable legt die Versionsnummer des Deserializer/Serializer fest, der
125 	 *                                    benutzt wird. Sowohl der StreamMultiplexer als auch der StreamDemultiplexer
126 	 *                                    müssen die selbe Version benutzen
127 	 * @param director                    Schnittstelle, die eine Methode zum verschicken von Informationen an den Sender
128 	 *                                    bereitstellt (siehe Interface Beschreibung)
129 	 * @see de.bsvrz.sys.funclib.dataSerializer.Serializer
130 	 * @see de.bsvrz.sys.funclib.dataSerializer.Deserializer
131 	 */
132 	public StreamMultiplexer(int numberOfStreams, int blockingFactor, int bufferSizeStreamMultiplexer, int serializerVersion, StreamMultiplexerDirector director) {
133 		_numberOfStreams = numberOfStreams;
134 		_blockingFactor = blockingFactor;
135 		_bufferSizeStreamMultiplexer = bufferSizeStreamMultiplexer;
136 		_bufferSizeStream = _bufferSizeStreamMultiplexer / _numberOfStreams;
137 		_serializerVersion = serializerVersion;
138 		_director = director;
139 
140 		_streams = new MultiplexerStreaminformations[_numberOfStreams];
141 
142 		// In dieser Warteschlange werden alle Streams eingetragen, die Daten versenden dürfen.
143 		_queueWithStreamsPermitedToSendData = new UnboundedQueue();
144 
145 
146 		for (int i = 0; i < _streams.length; i++) {
147 			// Für jeden Stream ein Objekt für seine Informationen anlegen und im Array aller Streams speichern.
148 			// Auf dieses Objekt kann dann später ein synchronized angewendet werden.
149 			MultiplexerStreaminformations newStreaminformations = new MultiplexerStreaminformations(_blockingFactor, _director);
150 			_streams[i] = newStreaminformations;
151 
152 			// Diese Objekt speichert, wie viele Nutzdatenpakete auf welchem Stream gesendet werden dürfen.
153 			// Beim ersten Senden dürfen aber blockingFactor viele Packete an den Empfänger geschickt werden, ohne das dieser
154 			// bestätigen muß.
155 			IndexOfStreamAndMaxSendPackets paketsForStream = new IndexOfStreamAndMaxSendPackets(i, _blockingFactor);
156 			// Den Sendeauftrag in die Warteschlange einfügen
157 			_queueWithStreamsPermitedToSendData.put(paketsForStream);
158 		}
159 	}

161 	/**
162 	 * Diese Methode verschickt die Nutzdaten über einen bestimmten Stream an den Empfänger.
163 	 *
164 	 * @param indexOfStream     : Der eindeutige Index des Streams auf dem gesendet werden soll
165 	 * @param streamPacketIndex : Jedes Datenpacket bekommt eine laufende Nummer
166 	 * @param data              : Nutzdaten, die versendet werden sollen
167 	 */
168 	private void sendDataToReceiver(int indexOfStream, int streamPacketIndex, byte[] data) {
169 		// Die übergebenen Variablen müßen nun in ein byte-Array verpackt werden. Dieses byte-Array wird dann mit dem
170 		// _director übertragen und auf der Gegenseite (StreamDemultiplexer) wieder "ausgepackt" und in die entsprechenden Variablen
171 		// umgewandelt und die passenden Objekte werden erzeugt.
172 
173 		// Dieses Paket wird physisch über eine Verbindung an einen StreamDemultiplexer verschickt.
174 		// Die größe des Pakets setzt sich aus den drei Integerzahlen zusammen (3 * 4 Bytes) und der Länge
175 		// des Nutzdatenpakets.
176 
177 		ByteArrayOutputStream out = new ByteArrayOutputStream();
178 		//serialisieren
179 		Serializer serializer = null;
180 		try {
181 			serializer = SerializingFactory.createSerializer(_serializerVersion, out);
182 		} catch (NoSuchVersionException e) {
183 			e.printStackTrace();
184 			_debug.error("Die geforderte Serializer-Version kann vom Serializer des StreamMultiplexer nicht benutzt werden, geforderte Version: " + _serializerVersion);
185 		}
186 
187 		try {
188 			// Die Daten werden wie folgt in dem Byte-Array verpackt
189 			// 1) indexOfStream (int)
190 			// 2) streamPacketIndex (int)
191 			// 3) Länge des Byte-Arrays (int)
192 			// 4) Byte-Array (byte[])
193 			serializer.writeInt(indexOfStream);
194 			serializer.writeInt(streamPacketIndex);
195 			serializer.writeInt(data.length);
196 			serializer.writeBytes(data);
197 		} catch (IOException e) {
198 			e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
199 		}
200 
201 		// Das Byte-Array erzeugen und verschicken
202 		_director.sendData(out.toByteArray());
203 
204 		_numberOfPacketsSend++;
205 		_debug.finer("\t\t\t\t\t\t\t\t\t\tSenderpaket: Stream = " + indexOfStream + " PaketIndex = " + streamPacketIndex);
206 	}
207 
208 	/**
209 	 * Diese Methode fordert von einer Application, auf einem bestimmten Stream, neue Nutzdaten an.
210 	 * <p/>
211 	 * Die Applikation wird von dem StreamMultiplexer angesprochen, darum private.
212 	 *
213 	 * @param indexOfStream Index des Streams, auf dem neue Daten angefordert werden sollen
214 	 * @return Nutzdaten
215 	 */
216 	private byte[] take(int indexOfStream) {
217 
218 		// Es werden soviele Nutzdaten angefordert, bis <code>_bufferSizeStream</code> erreicht wird, dann wird
219 		// das große Paket (mit den vielen kleinen Nutzdatenpaketen als Inhalt) verschickt.
220 
221 		// Hier werden alle Informationen für das große Paket gespeichert. Dies wird später in ein byte-Array umgewandelt
222 		ByteArrayOutputStream bigPacket = new ByteArrayOutputStream();
223 		// Serializer vorbereiten
224 		Serializer serializer = null;
225 		try {
226 			serializer = SerializingFactory.createSerializer(_serializerVersion, bigPacket);
227 		} catch (NoSuchVersionException e) {
228 			e.printStackTrace();
229 			_debug.error("Die geforderte Serializer-Version kann vom Serializer des StreamMultiplexer nicht benutzt werden, geforderte Version: " + _serializerVersion);
230 		}
231 
232 		// Wie viele Bytes wurden bereits in dem großen Paket gespeichert, dies wird benötigt um zu prüfen ob das
233 		// Paket schon die passende Größe hat.
234 		int sizeOfBigPacket = 0;
235 
236 		// Es werden solange Nutzdaten in das große Paket gepackt, bis dies die gewünschte Größe hat oder null-Paket
237 		// von der Senderapplikation versandt wurde (null bedeutet, dass keine weiteren Nutzdaten mehr zur Verfügung stehen).
238 
239 		boolean bigPacketReady = false;
240 
241 		// Anmerkung: Es wird mindestens ein Nutzdatenpaket verschickt (auch wenn das eine Paket die Größe des Puffers überschreitet),
242 		// da die größe des aktuellen Pakets erst nach dem einfügen eines Pakets geprüft wird.
243 
244 		while (bigPacketReady == false) {
245 			// Ein Nutzdatum anfordern
246 			byte[] oneData = _director.take(indexOfStream);
247 //			_dataCount++;
248 			if (oneData == null) {
249 				// Die Senderappliaktion hat keine Nutzdaten mehr zur Verfügung, somit muß ein "null"-Paket verpackt werden.
250 				// Dies wird durch einen negativen Index für die Nutzdatenpaketgröße angezeigt
251 
252 				try {
253 					int negativPaketSize = -1;
254 					serializer.writeInt(negativPaketSize);
255 				} catch (IOException e) {
256 					e.printStackTrace();
257 				}
258 
259 				// Nun muß der Stream gekennzeichnet werden, dass er nicht mehr senden braucht.
260 				// Das null-Paket wird in dem großen Paket aber noch verschickt.
261 
262 				synchronized (_streams[indexOfStream]) {
263 
264 					// Somit darf der Stream keine Daten mehr senden, ABER das letzte große Paket darf er noch verschicken.
265 					// Das passiert auch in der <code>sendAllStreamData()</code> Methode.
266 					_streams[indexOfStream].setStreamFinished();
267 					_debug.finer("\t\t\t\t\t\t\t\t\t\tSender verschickt null-Paket. Stream: " + indexOfStream + " und beendet den Stream");
268 					_numberTerminatedStreams++;
269 				}
270 
271 			} else {
272 				// Es wurden Nutzdaten von der Senderapplikation geschickt
273 
274 				// Wie groß ist das Nutzdatum
275 				int sizeOfData = oneData.length;
276 				sizeOfBigPacket = sizeOfBigPacket + sizeOfData;
277 
278 				// Als erstes wird gespeichert wie groß der Nutzdatensatz ist und dann das Array für die Nutzdaten.
279 				// Somit ist beim auspacken des großen Pakets immer klar, wie groß denn nun die eigentlichen
280 				// Nutzdatenpakete sind (die größe steht ja vor den Nutzdatenpaketen).
281 				try {
282 					serializer.writeInt(sizeOfData);
283 					serializer.writeBytes(oneData);
284 				} catch (IOException e) {
285 					e.printStackTrace();
286 				}
287 
288 			}
289 
290 			if ((sizeOfBigPacket > _bufferSizeStream) || (oneData == null)) {
291 				// Das erzeugte Paket ist nun größer als der Puffer des Streams, somit kann das Paket verschickt werden.
292 				// Wenn keine Nutzdaten mehr zur Verfügung stehen (take liefert null), dann ist das Paket auch bereits
293 				// versandt zu werden.
294 				bigPacketReady = true;
295 			}
296 		}
297 
298 		// Das große Paket ist nun fertig, damit der Empfänger weiß wann keine Daten (int, byte[]) mehr kommen, wird eine
299 		// Art EOF gesetzt. dies wird durch eine -2 symbolisiert. Jedes "bigPacket"/großes Paket bekommt diesen Stempel
300 		// am Ende (auch das null-Paket).
301 
302 		try {
303 			serializer.writeInt(-2);
304 		} catch (IOException e) {
305 			e.printStackTrace();
306 		}
307 
308 		// Das bigPaket in ein Byte-Array umwandeln
309 		byte[] data = bigPacket.toByteArray();
310 
311 		// Das große Paket "verschicken"
312 		return data;
313 	}

315 	/**
316 	 * Diese Methode verschickt Nutzdaten, die die Senderapplikation erzeugt hat, an den Empfänger (StreamDemultiplexer).
317 	 * Ein Problem das sich dabei ergibt ist, welcher Stream ist gerade sendebereit ? Das durchsuchen aller Streams nach
318 	 * einem sendebereiten Stream kann dabei sehr ungeschickt sein, da im "worst Case" immer alle Streams betrachtet werden
319 	 * müßten und das für jede Nachricht. Da jeder Stream mindestens eine Nachricht verschickt wäre somit eine quadratische
320 	 * Laufzeit erreicht.
321 	 * <p/>
322 	 * Die Grundidee ist, dass alle Tickets an einer zentralen Stelle gesammelt werden. Dafür wurde als Datenstruktur eine
323 	 * Warteschlange gewählt. Diese synchronisiert sich selbständig und schickt wartende Threads automatisch in den wait
324 	 * Modus, gleichzeitig werden diese Threads wieder aufgeweckt, wenn neue Daten zur Verfügung stehen.
325 	 * <p/>
326 	 * Sobald der Empfänger dem Sender eine Sendeerlaubnis erteilt wird der Stream mit seinem Index und der Anzahl Pakete,
327 	 * die er senden darf, in die Warteschlange eingetragen.
328 	 * <p/>
329 	 * Wenn nun ein Stream gesucht wird, der senden soll, dann wird von der Warteschlange das vorderste Element
330 	 * angefordert. Es steht somit sofort ein Stream zur Verfügung. Gleichzeitig ist bekannt, wieviele Pakete dieser Stream
331 	 * verschicken darf.
332 	 *
333 	 * @throws InterruptedException Ein Thread, der auf ein Objekt in der Warteschlange gewartet hat, wurde mit Interrupt
334 	 *                              unterbrochen.
335 	 */
336 	public void sendAllStreamData() throws InterruptedException {
337 
338 		// In welchem Modus werden die Pakete versandt (jeder Stream verschickt nr ein Paket und reiht sich dann erneut ein, oder
339 		// jeder Stream verschickt soviele Pakete wie er darf).
340 		boolean singlePacketMode;
341 
342 		// Solange es Streams gibt, die Senden dürfen
343 		while (_numberTerminatedStreams < _numberOfStreams) {
344 
345 			// Ein Element aus der Warteschlange anfordern.
346 			// Anmerkung, in der Zwischenzeit kann der StreamMultiplexer unterbrochen worden sein (killAllStreams).
347 			// Dieser Abbruch kann notwenig gewesen sein, weil der StreamDemultiplexer physich nicht mehr zu erreichen war,
348 			// es würden also keine Tickets mehr verschickt und somit würde es auch keine Sendebereiten Streams mehr geben.
349 			// Der Thread der den folgenden Aufruf tätigt würde also nie wieder die Queue verlassen können.
350 			// Der Aufruf <code>killAllStreams</code> legt einen Dummy-Stream in die Queue, der abgebrochen wurde.
351 			// Somit wird der hier wartende Thread befreit und verläßt die while-Schleife.
352 			IndexOfStreamAndMaxSendPackets paketsForStream = (IndexOfStreamAndMaxSendPackets) _queueWithStreamsPermitedToSendData.take();
353 
354 			// Welcher Stream darf Nutzdatenpakete verschicken
355 			int indexOfStream = paketsForStream.getIndexOfStream();
356 
357 			// Falls der Stream noch nicht mit Abort abgebrochen wurde, darf der Stream senden.
358 			// Somit werden alle Streams, die noch in der Warteschleife stehen aber abgebrochen wurden,
359 			// langsam aus der Warteschlange entfernt ohne Daten zu verschicken.
360 
361 			MultiplexerStreaminformations stream = _streams[indexOfStream];
362 			synchronized (stream) {
363 
364 				// Wurde der Stream bereits abgebrochen ?
365 				if (stream.isStreamTerminated() == false) {
366 					_debug.finer("\n\t\t\t\t\t\t\t\t\t\tSender verschickt Nutzdatenpaket(e)");
367 
368 					// Wie viele Pakete sollen auf einmal verschickt werden. Ist der single Modus aktiv, dann nur eins.
369 					// Im anderen Fall soviele wie erlaubt.
370 					final int numberOfPackets;
371 
372 					if (_queueWithStreamsPermitedToSendData.size() == 0) {
373 						// Die Schlange ist leer, somit kann der gerade gewählte Stream alle Nutzdatenpakete verschicken
374 						// die er verschicken darf.
375 						singlePacketMode = false;
376 						// Wie viele Pakete sollen verschickt werden (in diesem Modus so viele wie möglich)
377 						numberOfPackets = paketsForStream._maxSendPackets;
378 					} else {
379 						// Es gibt noch andere Streams die senden möchten. Also verschickt jeder nur ein Nutzdatenpaket
380 						// und reiht sich dann wieder in der Warteschlange ein.
381 						singlePacketMode = true;
382 						// Wie viele Pakete sollen verschickt werden (in diesem Modus nur eins)
383 						numberOfPackets = 1;
384 					}
385 
386 					for (int nr = 0; nr < numberOfPackets; nr++) {
387 
388 						// Wie viele Nutzdatenpackete hat der Stream schon verschickt. Hinweis: Das Array wird mit "0" in jedem Eintrag initialisiert.
389 						int currentStreamPacketIndex = stream.getCurrentStreamPacketIndex();
390 						// Ein Paket verschicken
391 						currentStreamPacketIndex++;
392 
393 						// Nutzdaten von der "Applikation" anfordern, wenn die Applikation eine "null" zurück gibt, dann sind
394 						// keine Nutzdaten mehr vorhanden. Der stream kann also beendet werden.
395 						byte[] data = take(indexOfStream);
396 
397 						// Debug
398 						_debug.finer("\t\t\t\t\t\t\t\t\t\tPaketdaten: Stream(" + indexOfStream + ") Paketindex(" + currentStreamPacketIndex + ") Anzahl Pakete, die zu diesem Satz gehören (" + numberOfPackets + ")" + "Paketnummer des Satzes (Nummer 0 ist das erste Paket)) (" + nr + ")");
399 
400 						// Das mit take angeforderte große Nutzdatenpaket verschicken
401 						sendDataToReceiver(indexOfStream, currentStreamPacketIndex, data);
402 						// Da ein Paket verschickt wurde, muß der neue Index gespeichert werden
403 						stream.setCurrentStreamPacketIndex(currentStreamPacketIndex);
404 
405 						// Wenn bei dem Aufruf der take Methode ein null-Paket verpackt wurde, dann wird der Stream dort
406 						// mit <code>setStreamTerminated</code> beendet. Das angelegte Paket WURDE aber noch verschickt.
407 						// Allerdings werden für den Stream keine weiteren Pakete mehr versandt.
408 						if (stream.isStreamTerminated()) {
409 							// Es gibt für den Stream keine Nutzdaten mehr, der Sender beendet auf seiner Seite den Stream.
410 							// Der Empfänger muß nicht benachrichtigt werden, da er ein Nutzdatenpaket mit leeren
411 							// Nutzdaten empfängt. Er wird daraufhin auf seiner Seite den Stream
412 							// ebenfalls beenden (Der Sender wird davon nicht benachrichtigt, da er den Stream sowieso schon beendet hat).
413 
414 							// Nun brauchen die restlichen Pakete nicht mehr verschickt werden, die For-Schleife kann abgebrochen werden
415 							break;
416 						}
417 					}
418 
419 					if ((singlePacketMode == true) && (stream.isStreamTerminated() == false)) {
420 						// Im Single-Modus wird nur ein Paket verschickt, wenn der Stream noch Daten verschicken darf
421 
422 						// Da ein Paket verschickt wurde, darf der Stream ein Paket weniger verschicken als vorher
423 						paketsForStream.decrementMaxSendPackets();
424 						if (paketsForStream.getMaxSendPackets() > 0) {
425 							// Der Stream darf noch Nutzdaten verschicken, also wird er wieder in die Warteschlange eingefügt.
426 							// Wurden alle Nutzdatenpakete verschickt ("virtueller else Zweig"), dann muß nichts gemacht werden.
427 
428 							_queueWithStreamsPermitedToSendData.put(paketsForStream);
429 						}
430 					}
431 
432 					_debug.finest("\t\t\t\t\t\t\t\t\t\t Anzahl verschickter Pakete(alle Streams) = " + _numberOfPacketsSend);
433 				}
434 			}
435 		}

441 	 * <p/>
442 	 * Diese Methode setzt den "maximumStreamTicketIndex" eines Streams herauf. Dadurch kann der Stream Datenpakete bis zu
443 	 * diesem neuen Index versenden. Wird der "maximumStreamTicketIndex" erreicht, stellt der Stream seine Sendetätigkeiten
444 	 * ein, bis der "maximumStreamTicketIndex" wieder erhöht wird. Verschickt der Empfänger eine "-1", dann will er die
445 	 * Empfangstätigkeiten auf diesem Stream einstellen.
446 	 * <p/>
447 	 * <p/>
448 	 * Die ersten 4 Byte enthalten den Index des Streams. Die letzen vier Bytes enthalten den neuen maximalen Index. Dies
449 	 * ist eine Steuerung des einen Multis(Sender) durch den anderen(Receiver), kein Zugriff von außen.
450 	 *
451 	 * @param streamTicketPacket Dieses Byte-Array enthält verschlüsselt den Index des Streams und den maximalen Index, bis
452 	 *                           zu dem der StreamMultiplexer senden darf.
453 	 * @throws IOException Ein Fehler beim deserialisieren von Daten
454 	 */
455 	public void setMaximumStreamTicketIndexForStream(byte[] streamTicketPacket) throws IOException {
456 
457 		// Das empfangene Byte-Array in die entsprechenden Objekte umwandeln
458 
459 		InputStream in = new ByteArrayInputStream(streamTicketPacket);
460 
461 		//deserialisieren
462 		Deserializer deserializer = null;
463 		try {
464 			deserializer = SerializingFactory.createDeserializer(_serializerVersion, in);
465 		} catch (NoSuchVersionException e) {
466 			e.printStackTrace();
467 			_debug.error("Die geforderte Serializer-Version kann vom Serializer des StreamMultiplexer nicht benutzt werden, geforderte Version: " + _serializerVersion);
468 		}
469 
470 		// Index des Streams, dessen Ticketanzahl erhöht werden soll
471 		int indexOfStream = deserializer.readInt();
472 
473 		// Eine neue Obergrenze bis zu der der Stream Daten senden darf, -1 zeigt an, dass der Stream beendet werden soll
474 		int maximumStreamTicketIndex = deserializer.readInt();
475 
476 		// Wenn ein Stream beendet werden soll, verschickt der Empfänger eine "-1 mit dem ""maximumStreamTicketIndex"
477 		if (maximumStreamTicketIndex > 0) {
478 			// Es wird auf das Objekt MultiplexerStreaminformations synchronisiert, nicht auf das gesamte Array
479 			MultiplexerStreaminformations stream = _streams[indexOfStream];
480 			synchronized (stream) {
481 				// Es kann sein, das der Sender einen Stream beendet (keine Nutzdaten mehr), aber denoch gerade Tickets zu ihm unterwegs sind
482 				if (stream.isStreamTerminated() == false) {
483 					// Der Empfänger kann "maximumStreamTicketIndex" verschicken, die zu alt sind (ihr Wert ist kleiner als der schon empfangene oder sogar 0)
484 					// Siehe Kommentar StreamDemultiplexer take-Methode, Stichwort: verschicken neuer Tickets.
485 					if (stream.getMaxStreamPacketIndex() <= maximumStreamTicketIndex) {
486 						// Es muß nun ein neues Element in die Warteschlange gelegt werden und der maximale Index, bis zu dem gesendet werden darf, muß
487 						// aktualisiert werden.
488 
489 						// Die neue maximale Anzahl von zu verschickenden Pakten wird berechnet
490 						// (dies sind wirklich einzelne Pakete, keine Index Nummer bis zu der geschickt werden darf).
491 						int maxNumberOfPacketsSendByStream = maximumStreamTicketIndex - stream.getMaxStreamPacketIndex();
492 						// Objekt für die Warteschlange anlegen
493 						IndexOfStreamAndMaxSendPackets paketsForStream = new IndexOfStreamAndMaxSendPackets(indexOfStream, maxNumberOfPacketsSendByStream);
494 						// Objekt in die Warteschlange einfügen.
495 						// Der Zugriff auf die Warteschlange muß nicht synchronisiert werden, da dies die Datenstruktur von alleine macht.
496 						_queueWithStreamsPermitedToSendData.put(paketsForStream);
497 						// Den neuen maximalen Index speichern
498 						stream.setMaxStreamPacketIndex(maximumStreamTicketIndex);
499 
500 						// Debug
501 						_debug.finer("\t\t\t\t\t\t\t\t\t\tNeuer Max Ticket Index gesetzt: Stream(" + indexOfStream + ") neue maximaler Ticketindex(" + stream.getMaxStreamPacketIndex() + ")");
502 					} else {
503 						// Debug, wie viele "falsche" Ticktes werden verschickt (der maximale Paketindex war kleiner, als der aktuelle max Index)
504 						_numberOfFalseMaxTickets++;
505 					}
506 				}
507 			}
508 
509 		} else {
510 			// Der Empfänger hat einen Stream beendet
511 			_numberTerminatedStreams++;
512 
513 			// Der Stream darf keine Nutzdatenpakete mehr verschicken
514 			synchronized (_streams[indexOfStream]) {
515 				_streams[indexOfStream].setStreamTerminated(indexOfStream);
516 			}
517 			// Falls der Stream, der gerade von der Empfangsseite beendet wurde, der letzte Stream war,
518 			// wird der Thread in der Warteschlange <code>_queueWithStreamsPermitedToSendData</code>
519 			// festhängen. Da keine Tickets mehr ankommen werden, somit muss der Thread befreit werden.
520 			// Jeder Stream, der abgebrochen wird, wird sich  noch einemal in diese Warteschlange
521 			// eintragen. Da der Stream aber abgebrochen, wird er keine Daten mehr verschicken.
522 			// Der wartende Thread wird dadurch aber erneut gestartet und kann erkennen, dass
523 			// der Stream nicht mehr senden dard und falls alle Streams beendet wurden, wird
524 			// sich der Thread ebenfalls beenden.
525 			// Das Problem ist fast identisch mit der <code>killAllStreams</code> - Methode, auch
526 			// dort konnte es zu einem Deadlock kommen. Die Lösung ist somit fast identisch.
527 
528 			// Als indexOfStream, wird der abgebrochene Stream genommen
529 			// (das ist der unterschied zu <code>killAllStreams</code>, dort wird immer der Stream mit Index 0 genommen
530 			// der Effekt ist identisch)
531 			IndexOfStreamAndMaxSendPackets dummyEntry = new IndexOfStreamAndMaxSendPackets(indexOfStream, 1);
532 			_queueWithStreamsPermitedToSendData.put(dummyEntry);
533 		}
534 	}

536 	/**
537 	 * Alle Streams werden beendet, da die Verbindung zum Empfänger unterbrochen wurde.
538 	 * <p/>
539 	 * Wenn das Objekt, dem der StreamMultiplexer gehört, einen Fehler des DaV gemeldet bekommt (die Leitung zur
540 	 * Empfängerapplikation wurde unterbrochen, als Beispiel) wird mit dieser Methode jeder Stream abgebrochen.
541 	 * Gleichzeitig wird die Sendeapplikation darauf hingewiesen, dass sie alle Nutzdaten für die Streams verwerfen kann.
542 	 * Auf der Gegenseite wird dem StreamDemultiplexer ebenfalls gemeldet, dass etwas mit der Verbindung nicht stimmt (dies
543 	 * übernimmt dort das Objekt, das den StreamDemultiplexer erzeugt hat). Der StreamDemultiplexer wird daraufhin
544 	 * ebenfalls alle Streams beenden. Der beidseitige Abbruch geschieht automatisch.
545 	 */
546 	public void killAllStreams() {
547 		for (int nr = 0; nr < _numberOfStreams; nr++) {
548 			MultiplexerStreaminformations stream = _streams[nr];
549 			synchronized (stream) {
550 				// Der Stream wird abgebrochen/beendet
551 				stream.setStreamTerminated(nr);
552 				_numberTerminatedStreams++;
553 			}
554 		}
555 		// Alle Streams wurden abgebrochen. Es kann sein, das kein Stream senden durfte und somit
556 		// der Thread in der _queueWithStreamsPermitedToSendData festhängt. Es muss somit ein Dummy-Eintrag
557 		// erzeugt werden, dieser befreit den Thread. Nach dem er befreit wurde, wird festgestellt, dass
558 		// alle Streams abgeborchen wurden und der Thread beendet sich wie gewollt.
559 		IndexOfStreamAndMaxSendPackets dummyEntry = new IndexOfStreamAndMaxSendPackets(0, 1);
560 		_queueWithStreamsPermitedToSendData.put(dummyEntry);
561 
562 	}
