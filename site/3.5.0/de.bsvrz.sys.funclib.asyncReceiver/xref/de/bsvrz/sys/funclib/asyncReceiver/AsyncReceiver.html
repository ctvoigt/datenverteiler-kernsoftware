30  /**
31   * Klasse, die beim Empfang von Datens‰tzen zur Entkopplung von Datenverteiler-Applikationsfunktionen und
32   * Verarbeitungsfunktionen eingesetzt werden kann. Empfangene Datens‰tze werden in eine Queue eingetragen und asynchron
33   * von einem eigenen Thread an die eigentlichen Empf‰nger weitergeleitet. <br/> Wenn zum Beispiel statt
34   * <pre>
35   *    ClientReceiverInterface receiver= new ....;
36   *    connection.subscribeReceiver(receiver, ... );
37   * </pre>
38   * der folgende Code bei der Anmeldung auf bestimmte Daten benutzt wird:
39   * <pre>
40   *    ClientReceiverInterface receiver= new ....;
41   *    ClientReceiverInterface asyncReceiver= new AsyncReceiver(receiver);
42   *    connection.subscribeReceiver(asyncReceiver, ... );
43   * </pre>
44   * dann wird jeder empfangene Datensatz im Hintergrund verarbeitet. <br/> Es ist zu beachten, dass die Klasse ein
45   * einzigen Thread verwendet, der asynchron zu den anderen Aktivit‰ten der Applikation und insbesondere asynchron zum
46   * update-Thread der Datenverteiler-Applikationsfunktionen arbeitet. Die in einer Queue zwischengespeicherten Datens‰tze
47   * werden sequentiell an die update-Methode des eigentlichen Empf‰ngers weiterleitet.
48   *
49   * @author Kappich Systemberatung
50   * @version $Revision: 5058 $
51   */
52  public class AsyncReceiver implements ClientReceiverInterface {
53  	private static final Debug _debug = Debug.getLogger();
54  
55  	private static class QueueEntry {
56  		private final ClientReceiverInterface _syncReceiver;
57  		private final ResultData[] _results;
58  
59  		public ClientReceiverInterface getSyncReceiver() {
60  			return _syncReceiver;
61  		}
62  
63  		public ResultData[] getResults() {
64  			return _results;
65  		}
66  
67  		public QueueEntry(ClientReceiverInterface syncReceiver, ResultData[] results) {
68  			_syncReceiver = syncReceiver;
69  			_results = results;
70  		}
71  	}
72  
73  	private static final class QueueWorker implements Runnable {
74  		public void run() {
75  			while(true) {
76  				try {
77  					QueueEntry entry = (QueueEntry)_queueSingleton.take();
78  					if(entry == null) {
79  						// null signalisiert dem Thread, dass er sich beenden soll
80  						break;
81  					}
82  					entry.getSyncReceiver().update(entry.getResults());
83  				}
84  				catch(RuntimeException e) {
85  					_debug.error("Fehler bei der asynchronen Bearbeitung von empfangenen Daten: " + e);
86  					e.printStackTrace();
87  					System.exit(1);
88  				}
89  				catch(InterruptedException e) {
90  					_debug.warning("Unterbrechung bei der asynchronen Bearbeitung von empfangenen Daten");
91  					break;
92  				}
93  
94  			}
95  		}
96  	}
97  
98  	private static final UnboundedQueue _queueSingleton;
99  
100 	static {
101 		_queueSingleton = new UnboundedQueue();
102 		Thread thread = new Thread(new QueueWorker(), "AsyncReceiverQueueWorker");
103 		thread.setDaemon(true);
104 		thread.start();
105 	}
106 
107 	private final ClientReceiverInterface _syncReceiver;
108 
109 	/**
110 	 * Erzeugt ein neues Empf‰ngerobjekt mit asynchroner Weiterleitung empfangener Daten an das ¸bergebene
111 	 * Empf‰ngerobjekt.
112 	 *
113 	 * @param syncReceiver Empf‰ngerobjekt, an das empfangene Daten asynchron zur Verarbeitung weitergeleitet werden
114 	 *                     sollen.
115 	 */
116 
117 	public AsyncReceiver(ClientReceiverInterface syncReceiver) {
118 		_syncReceiver = syncReceiver;
119 	}
120 
121 	/**
122 	 * Aktualisierungsmethode, die nach Empfang eines angemeldeten Datensatzes von den
123 	 * Datenverteiler-Applikationsfunktionen aufgerufen wird.
124 	 * <p/>
125 	 * Empfangene Datens‰tze werden in eine Queue eingetragen und asynchron an das eigentliche Empf‰ngerobjekt zur
126 	 * Verarbeitung weitergeleitet.
127 	 *
128 	 * @param results Feld mit den empfangenen Ergebnisdatens‰tzen.
129 	 */
130 	public void update(ResultData results[]) {
131 		_queueSingleton.put(new QueueEntry(_syncReceiver, results));
132 	}
