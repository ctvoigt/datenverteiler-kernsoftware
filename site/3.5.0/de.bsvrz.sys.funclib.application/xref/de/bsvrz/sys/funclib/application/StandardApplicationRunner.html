66  	/**
67  	 * Diese Methode erstellt eine Verbindung zum Datenverteiler anhand der Standard-Parameter her.
68  	 *
69  	 * @param application Applikation, die eine Verbindung zum Datenverteiler benötigt.
70  	 * @param applicationTypePid Pid des Applikationstyps. Der Datenverteiler erzeugt für die Appliaktion ein Objekt dieses
71  	 *      Typs. Der Applikationstyp sollte "typ.applikation" sein oder davon abgeleitet sein.
72  	 * @param args        Aufrufargumente der Applikation
73  	 */
74  	public static void run(StandardApplication application, String applicationTypePid, String[] args) {
75  		createApplicationLabel(args);
76  		final ArgumentList argumentList = new ArgumentList(args);
77  		initializeDebug(application, argumentList);
78  		Thread.setDefaultUncaughtExceptionHandler(new UncaughtExceptionHandler());
79  		try {
80  			// ArgumentListe wird in ClientDavParameters konvertiert
81  			final ClientDavParameters parameters = new ClientDavParameters(argumentList);
82  			parameters.setApplicationTypePid(applicationTypePid);
83  			parameters.setApplicationName(_applicationName);
84  			// zuerst darf die Applikation die ArgumentListe durcharbeiten
85  			application.parseArguments(argumentList);
86  			argumentList.ensureAllArgumentsUsed();
87  			final ClientDavInterface connection = new ClientDavConnection(parameters);
88  			// Fertigmeldung für Start/Stop wird eplizit selbst übernommen
89  			establishConnection(connection);
90  			_applicationLabel.append(connection.getLocalConfigurationAuthority().getPid());	// ApplikationsKennung
91  			MessageSender.getInstance().init(connection, _applicationName, _applicationLabel.toString());
92  			application.initialize(connection);
93  			// Fertigmeldung wird gesendet
94  			connection.sendApplicationReadyMessage();
95  		}
96  		catch(Exception ex) {
97  			_debug.error("Fehler", ex);
98  			System.exit(1);
99  		}
100 	}

112 	/**
113 	 * Diese Methode startet einen Login-Dialog und meldet sich anhand der eingetragenen IP-Adresse, Portnummer, Benutzername und Passwort beim Datenverteiler an.
114 	 *
115 	 * @param application Applikation, die eine Verbindung zum Datenverteiler benötigt.
116 	 * @param applicationTypePid Pid des Applikationstyps. Der Datenverteiler erzeugt für die Appliaktion ein Objekt dieses
117 	 *      Typs. Der Applikationstyp sollte "typ.applikation" sein oder davon abgeleitet sein.
118 	 * @param args        Aufrufargumente der Applikation
119 	 */
120 	public static void run(GUIApplication application, String applicationTypePid, String[] args) {
121 		createApplicationLabel(args);
122 		final ArgumentList argumentList = new ArgumentList(args);
123 		initializeDebug(application, argumentList);
124 		Thread.setDefaultUncaughtExceptionHandler(new UncaughtExceptionHandler());
125 		try {
126 			// soll der Login-Dialog überhaupt angezeigt werden?
127 			boolean autoLogin = argumentList.fetchArgument("-autologin=false").booleanValue();
128 			if(autoLogin) {
129 				if(!(argumentList.hasArgument("-benutzer") && argumentList.hasArgument("-authentifizierung"))) {
130 					if(argumentList.hasArgument("-benutzer")) argumentList.fetchArgument("-benutzer");
131 					if(argumentList.hasArgument("-authentifizierung")) {
132 						argumentList.fetchArgument("-authentifizierung");
133 					}
134 					_debug.warning(
135 							"Der Login-Dialog wird trotz des Aufrufparameters '-autologin' aufgerufen, da die Parameter '-benutzer' und '-authentifizierung' "
136 							+ "unvollständig oder nicht korrekt angegeben wurden."
137 					);
138 					autoLogin = false;
139 				}
140 			}
141 
142 			// ArgumentListe wird in ClientDavParameters konvertiert
143 			final ClientDavParameters parameters = new ClientDavParameters(argumentList);
144 			// zuerst darf die Applikation die ArgumentListe durcharbeiten
145 			parameters.setApplicationTypePid(applicationTypePid);
146 			parameters.setApplicationName(_applicationName);
147 			application.parseArguments(argumentList);
148 			argumentList.ensureAllArgumentsUsed();
149 
150 //			System.out.println("IP: " + parameters.getDavCommunicationAddress() + " Port: " + parameters.getDavCommunicationSubAddress());
151 			ClientDavInterface connection = null;
152 			if(autoLogin) {
153 				connection = new ClientDavConnection(parameters);
154 				// Verbindung aufbauen (inkl. Anmelden zur Fertigmeldung für Start/Stop)
155 				establishConnection(connection);
156 			}
157 			else {
158 				// Login-Dialog aufrufen, der eine Verbindung zum Datenverteiler aufbaut und zurückgibt.
159 				connection = application.connect(parameters);
160 
161 			}
162 			if(connection != null) {
163 				_applicationLabel.append(connection.getLocalConfigurationAuthority().getPid());	// ApplikationsKennung
164 				MessageSender.getInstance().init(connection, _applicationName, _applicationLabel.toString());
165 				application.initialize(connection);
166 				// Fertigmeldung senden
167 				connection.sendApplicationReadyMessage();
168 			}
169 			else {
170 				throw new RuntimeException("Die Verbindung zum Datenverteiler konnte nicht hergestellt werden.");
171 			}
172 		}
173 		catch(Exception ex) {
174 			_debug.error("Fehler in der Applikation", ex);
175 			System.exit(1);
176 		}
177 	}

180 	 * Diese Methode baut die Verbindung zum Datenverteiler auf und teilt mit, dass sich die Applikation selbst um die Fertigmeldung für Start/Stop kümmert.
181      * @param connection aufzubauende Verbindung
182      * @throws CommunicationError
183      * @throws ConnectionException
184      * @throws InconsistentLoginException

186     private static void establishConnection(final ClientDavInterface connection) throws CommunicationError, ConnectionException, InconsistentLoginException {
187 	    connection.enableExplicitApplicationReadyMessage();
188 	    connection.connect();
189 	    connection.login();
190     }

194 	 *
195 	 * @param application Applikations-Objekt
196 	 * @param argumentList Aufrufargumente

198 	private static void initializeDebug(Object application, ArgumentList argumentList) {
199 		// Der Klassenname wird für die Initialisierung des Debug benötigt.
200 //		System.out.println("application.getClass().getName() = " + application.getClass().getName());
201 		final String[] classNameParts = application.getClass().getName().split("[.]");
202 //		System.out.println("classNameParts.length = " + classNameParts.length);
203 		final int lastPartIndex = classNameParts.length - 1;
204 		final String name;
205 		if(lastPartIndex < 0) {
206 			name = "StandardApplication";
207 		}
208 		else {
209 			name = classNameParts[lastPartIndex];
210 		}
211 		_applicationName = name;
212 		_applicationLabel.append(name);
213 		Debug.init(name, argumentList);
214 		_debug = Debug.getLogger();
215 	}

229 	/**
230 	 * Implementierung eines UncaughtExceptionHandlers, der bei nicht abgefangenen Exceptions und Errors entsprechende Ausgaben macht und im Falle eines Errors den
231 	 * Prozess terminiert.
232 	 */
233 	private static class UncaughtExceptionHandler implements Thread.UncaughtExceptionHandler {
234 
235 		/** Speicherreserve, die freigegeben wird, wenn ein Error auftritt, damit die Ausgaben nach einem OutOfMemoryError funktionieren */
236 		private volatile byte[] _reserve = new byte[20000];
237 
238 		public void uncaughtException(Thread t, Throwable e) {
239 			if(e instanceof Error) {
240 				// Speicherreserve freigeben, damit die Ausgaben nach einem OutOfMemoryError funktionieren
241 				_reserve = null;
242 				try {
243 					System.err.println("Schwerwiegender Laufzeitfehler: Ein Thread hat sich wegen eines Errors beendet, Prozess wird terminiert");
244 					System.err.println(t);
245 					e.printStackTrace(System.err);
246 					_debug.error("Schwerwiegender Laufzeitfehler: " + t + " hat sich wegen eines Errors beendet, Prozess wird terminiert", e);
247 				}
248 				catch(Throwable ignored) {
249 					// Weitere Fehler während der Ausgaben werden ignoriert, damit folgendes exit() auf jeden Fall ausgeführt wird.
250 				}
251 				System.exit(1);
252 			}
253 			else {
254 				System.err.println("Laufzeitfehler: Ein Thread hat sich wegen einer Exception beendet:");
255 				System.err.println(t);
256 				e.printStackTrace(System.err);
257 				_debug.warning("Laufzeitfehler: " + t + " hat sich wegen einer Exception beendet", e);
258 			}
259 		}
260 	}
